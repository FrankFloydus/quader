52TEST(UiFileDialogService, FakeDialogServicePreservesOpenRequestAndResult) {
53 FakeFileDialogService dialogs;
54 QWidget parent;
55 dialogs.next_open_result = QStringLiteral("C:/tmp/scene.fake");
56
57 quader::ui::OpenFileDialogRequest request;
58 request.title = QStringLiteral("Open Scene");
59 request.initial_directory = QStringLiteral("C:/tmp");
60 request.filters.push_back(quader::ui::FileDialogFilter{
61 QStringLiteral("Fake"),
62 { QStringLiteral("*.fake") },
63 });
64
65 const auto kResult = dialogs.choose_open_file(&parent, request);
66 if (!kResult.has_value()) {
67 ADD_FAILURE() << "open dialog did not return the configured path";
68 return;
69 }
70 EXPECT_EQ(*kResult, QStringLiteral("C:/tmp/scene.fake"));
71 EXPECT_EQ(dialogs.open_calls, 1);
72 EXPECT_TRUE(dialogs.last_open_parent == &parent);
73 EXPECT_EQ(dialogs.last_open_request.title, QStringLiteral("Open Scene"));
74 EXPECT_EQ(dialogs.last_open_request.filters.front().patterns.front(), QStringLiteral("*.fake"));
75}
