27[[nodiscard]] quader::foundation::Result<void, IoError> validate_descriptor_extensions(
28 const FileFormatDescriptor &descriptor) {
29 std::set<std::string> normalized_extensions;
30 for (const auto &extension : descriptor.extensions) {
31 const auto kNormalized = normalize_extension(extension);
32 if (kNormalized.empty()) {
33 return quader::foundation::Result<void, IoError>::failure(
34 registry_validation_error("file format extension cannot be empty", descriptor.id));
35 }
36
37 const auto [_, inserted] = normalized_extensions.insert(kNormalized);
38 if (!inserted) {
39 return quader::foundation::Result<void, IoError>::failure(
40 registry_validation_error("file format descriptor repeats extension '" + kNormalized + "'",
41 descriptor.id));
42 }
43 }
44
45 return quader::foundation::Result<void, IoError>::success();
46}

48template <class Interface>
49[[nodiscard]] quader::foundation::Result<void, IoError> validate_unique_registration(
50 const std::vector<std::unique_ptr<Interface>> &existing,
51 const FileFormatDescriptor &descriptor) {
52 if (descriptor.id.empty()) {
53 return quader::foundation::Result<void, IoError>::failure(
54 registry_validation_error("file format id cannot be empty"));
55 }
56
57 auto extension_validation = validate_descriptor_extensions(descriptor);
58 if (!extension_validation) {
59 return extension_validation;
60 }
61
62 for (const auto &registered : existing) {
63 const auto kRegisteredDescriptor = registered->descriptor();
64 if (kRegisteredDescriptor.id == descriptor.id) {
65 return quader::foundation::Result<void, IoError>::failure(
66 registry_validation_error("duplicate file format id '" + descriptor.id + "'",
67 descriptor.id));
68 }
69
70 for (const auto &registered_extension : kRegisteredDescriptor.extensions) {
71 const auto kNormalizedRegistered = normalize_extension(registered_extension);
72 for (const auto &candidate_extension : descriptor.extensions) {
73 const auto kNormalizedCandidate = normalize_extension(candidate_extension);
74 if (kNormalizedRegistered == kNormalizedCandidate) {
75 return quader::foundation::Result<void, IoError>::failure(
76 registry_validation_error(
77 "duplicate file extension claim '" + kNormalizedCandidate + "'",
78 descriptor.id));
79 }
80 }
81 }
82 }
83
84 return quader::foundation::Result<void, IoError>::success();
85}

89quader::foundation::Result<void, IoError> ImportExportRegistry::register_importer(
90 std::unique_ptr<IImporter> importer) {
91 if (!importer) {
92 return quader::foundation::Result<void, IoError>::failure(
93 registry_validation_error("cannot register a null importer"));
94 }
95
96 const auto kDescriptor = importer->descriptor();
97 auto validation = validate_unique_registration(importers_, kDescriptor);
98 if (!validation) {
99 return validation;
100 }
101
102 importers_.push_back(std::move(importer));
103 return quader::foundation::Result<void, IoError>::success();
104}
