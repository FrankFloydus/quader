34[[nodiscard]] const crimson::MaterialParameter *find_parameter(
35 const crimson::MaterialInstance &material,
36 std::string_view name) {
37 for (const crimson::MaterialParameter &parameter : material.parameters) {
38 if (parameter.name == name) {
39 return &parameter;
40 }
41 }
42
43 return nullptr;
44}

46[[nodiscard]] const quader::io::gltf::TextureSlotMapping *find_texture_mapping(
47 const quader::io::gltf::GltfMaterialMapping &mapping,
48 std::string_view slot_name) {
49 for (const quader::io::gltf::TextureSlotMapping &texture : mapping.texture_slots) {
50 if (texture.slot_name == slot_name) {
51 return &texture;
52 }
53 }
54
55 return nullptr;
56}

58template <typename T>
59[[nodiscard]] T parameter_value(const crimson::MaterialInstance &material, std::string_view name) {
60 const crimson::MaterialParameter *parameter = find_parameter(material, name);
61 EXPECT_TRUE(parameter != nullptr);
62 if (parameter == nullptr) {
63 return T{};
64 }
65
66 const auto *value = std::get_if<T>(&parameter->value);
67 EXPECT_TRUE(value != nullptr);
68 return value == nullptr ? T{} : *value;
69}

71TEST(GltfMaterialMapper, AlphaModesMapToV1BaseShaders) {
72 GltfMaterialSource opaque;
73 opaque.name = "OpaqueMaterial";
74 opaque.alpha_mode = GltfAlphaMode::Opaque;
75 auto opaque_mapping = map_material(opaque);
76 EXPECT_TRUE(opaque_mapping);
77 EXPECT_EQ(opaque_mapping.value().material.base_shader_id, crimson::BaseShaderId::OpaquePbr);
78 EXPECT_TRUE(find_parameter(opaque_mapping.value().material, "alpha_cutoff") == nullptr);
79 EXPECT_TRUE(find_parameter(opaque_mapping.value().material, "opacity") == nullptr);
80
81 GltfMaterialSource mask;
82 mask.name = "CutoutMaterial";
83 mask.alpha_mode = GltfAlphaMode::Mask;
84 mask.alpha_cutoff = 0.33F;
85 auto mask_mapping = map_material(mask);
86 EXPECT_TRUE(mask_mapping);
87 EXPECT_EQ(mask_mapping.value().material.base_shader_id, crimson::BaseShaderId::AlphaCutoutPbr);
88 EXPECT_EQ(parameter_value<float>(mask_mapping.value().material, "alpha_cutoff"), 0.33F);
89
90 GltfMaterialSource blend;
91 blend.name = "TransparentMaterial";
92 blend.alpha_mode = GltfAlphaMode::Blend;
93 blend.pbr.base_color_factor = crimson::MaterialColorSrgb{ 0.2F, 0.3F, 0.4F, 0.25F };
94 auto blend_mapping = map_material(blend);
95 EXPECT_TRUE(blend_mapping);
96 EXPECT_EQ(blend_mapping.value().material.base_shader_id, crimson::BaseShaderId::TransparentPbr);
97 EXPECT_EQ(parameter_value<float>(blend_mapping.value().material, "opacity"), 0.25F);
98 EXPECT_TRUE(find_parameter(blend_mapping.value().material, "alpha_cutoff") == nullptr);
99}

101TEST(GltfMaterialMapper, PbrFactorsAndDoubleSidedMapThroughDeclaredParameters) {
102 GltfMaterialSource source;
103 source.name = "PaintedMetal";
104 source.pbr.base_color_factor = crimson::MaterialColorSrgb{ 0.25F, 0.5F, 0.75F, 0.8F };
105 source.pbr.metallic_factor = 0.65F;
106 source.pbr.roughness_factor = 0.35F;
107 source.normal_scale = 2.0F;
108 source.occlusion_strength = 0.45F;
109 source.emissive_factor = crimson::MaterialColorSrgb{ 0.1F, 0.2F, 0.3F, 1.0F };
110 source.double_sided = true;
111

114 if (!mapping) {
115 return;
116 }
117

120 EXPECT_EQ(parameter_value<crimson::MaterialColorSrgb>(material, "base_color").g, 0.5F);
121 EXPECT_EQ(parameter_value<crimson::MaterialColorSrgb>(material, "base_color").b, 0.75F);
122 EXPECT_EQ(parameter_value<float>(material, "metallic"), 0.65F);
123 EXPECT_EQ(parameter_value<float>(material, "roughness"), 0.35F);
124 EXPECT_EQ(parameter_value<float>(material, "normal_strength"), 2.0F);
125 EXPECT_EQ(parameter_value<float>(material, "occlusion_strength"), 0.45F);
126 EXPECT_EQ(parameter_value<crimson::MaterialColorSrgb>(material, "emissive_color").g, 0.2F);
127 EXPECT_EQ(parameter_value<float>(material, "emissive_strength"), 1.0F);
128 EXPECT_EQ(parameter_value<bool>(material, "double_sided"), true);

132TEST(GltfMaterialMapper, TextureSlotMappingsFollowSchemaOrderAndColorSpaces) {
133 GltfMaterialSource source;
134 source.name = "Textured";
135 source.pbr.base_color_texture = GltfTextureRef{ "albedo.png", 0 };
136 source.pbr.metallic_roughness_texture = GltfTextureRef{ "metal_rough.png", 1 };
137 source.normal_texture = GltfTextureRef{ "normal.png", 0 };
138 source.occlusion_texture = GltfTextureRef{ "ao.png", 1 };
139 source.emissive_texture = GltfTextureRef{ "emissive.png", 0 };
140
141 auto mapping = map_material(source);
142 EXPECT_TRUE(mapping);
143 if (!mapping) {
144 return;
145 }
146
147 EXPECT_EQ(mapping.value().texture_slots.size(), 5U);
148 EXPECT_EQ(mapping.value().texture_slots[0].slot_name, std::string("base_color"));
149 EXPECT_EQ(mapping.value().texture_slots[0].expected_color_space, crimson::TextureColorSpace::Srgb);
150 EXPECT_EQ(mapping.value().texture_slots[1].slot_name, std::string("metallic_roughness"));
151 EXPECT_EQ(mapping.value().texture_slots[1].expected_color_space, crimson::TextureColorSpace::Linear);
152 EXPECT_EQ(mapping.value().texture_slots[2].slot_name, std::string("normal"));
153 EXPECT_EQ(mapping.value().texture_slots[2].expected_color_space, crimson::TextureColorSpace::Data);
154 EXPECT_EQ(mapping.value().texture_slots[3].slot_name, std::string("occlusion"));
155 EXPECT_EQ(mapping.value().texture_slots[3].expected_color_space, crimson::TextureColorSpace::Linear);
156 EXPECT_EQ(mapping.value().texture_slots[4].slot_name, std::string("emissive"));
157 EXPECT_EQ(mapping.value().texture_slots[4].expected_color_space, crimson::TextureColorSpace::Srgb);
158 EXPECT_EQ(mapping.value().texture_slots[1].source.texcoord, 1);
159
160 for (const crimson::MaterialTextureBinding &texture : mapping.value().material.textures) {
161 EXPECT_FALSE(crimson::is_valid_handle(texture.texture));
162 }
163}

227TEST(GltfMaterialMapper, SchemaValidationFailureReturnsIoDiagnostic) {
228 const crimson::BaseShaderRegistry kRegistry = crimson::make_v1_base_shader_registry();
229 const crimson::BaseShaderDefinition *opaque = kRegistry.find(crimson::BaseShaderId::OpaquePbr);
230 EXPECT_TRUE(opaque != nullptr);
231 if (opaque == nullptr) {
232 return;
233 }
234
235 crimson::BaseShaderDefinition invalid_definition = *opaque;
236 invalid_definition.parameters[0].kind = crimson::MaterialParameterKind::Float;
237 crimson::BaseShaderRegistry invalid_registry{ std::vector<crimson::BaseShaderDefinition>{ invalid_definition } };
238
239 GltfMaterialSource source;
240 source.name = "InvalidSchema";
241 auto mapping = quader::io::gltf::map_gltf_material_to_crimson(source, invalid_registry);
242 EXPECT_FALSE(mapping);
243 EXPECT_EQ(mapping.error().code, quader::io::IoErrorCode::ValidationFailed);
244 EXPECT_EQ(mapping.error().diagnostic.code, quader::io::IoDiagnosticCode::MaterialMappingFailed);
245 EXPECT_TRUE(mapping.error().related.has_errors());
246}
