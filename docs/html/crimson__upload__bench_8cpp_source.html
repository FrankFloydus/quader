20[[nodiscard]] std::vector<crimson::RenderMeshUploadDesc> make_uploads(
21 std::uint32_t count,
22 std::vector<std::array<float, 18>> &vertices,
23 std::vector<std::array<std::uint32_t, 3>> &indices) {
24 vertices.resize(count);
25 indices.resize(count);
26 std::vector<crimson::RenderMeshUploadDesc> uploads;
27 uploads.reserve(count);
28 for (std::uint32_t index = 0; index < count; ++index) {
29 vertices[index] = {
30 -1.0F,
31 -1.0F,
32 0.0F,
33 0.0F,
34 0.0F,
35 1.0F,
36 1.0F,
37 -1.0F,
38 0.0F,
39 0.0F,
40 0.0F,
41 1.0F,
42 0.0F,
43 1.0F,
44 0.0F,
45 0.0F,
46 0.0F,
47 1.0F,
48 };
49 indices[index] = { 0, 1, 2 };
50 uploads.push_back(crimson::RenderMeshUploadDesc{
51 .handle = crimson::RenderMeshHandle{ index + 1, 1 },
52 .revision = crimson::RenderMeshRevision{ 1, 1, index % 4 },
53 .position_normal_interleaved = vertices[index],
54 .indices = indices[index],
55 .attributes = crimson::VertexAttributePosition | crimson::VertexAttributeNormal,
56 .bounds = quader::math::Aabb{
57 .min = { -1.0F, -1.0F, -1.0F },
58 .max = { 1.0F, 1.0F, 1.0F },
59 },
60 });
61 }
62 return uploads;
63}

67BenchmarkResult run_crimson_upload_benchmark(const BenchmarkRunConfig &config) {
68 std::vector<std::array<float, 18>> vertices;
69 std::vector<std::array<std::uint32_t, 3>> indices;
70 const std::vector<crimson::RenderMeshUploadDesc> kUploads =
71 make_uploads(config.fixture_size, vertices, indices);
72
73 return run_benchmark("crimson_upload_tracker", config, [&]() {
74 crimson::RenderMeshUploadTracker tracker;
75 const auto kFirst = tracker.process_uploads(kUploads);
76 const auto kClean = tracker.process_uploads(kUploads);
77 crimson::FrameUploadStats upload_stats;
78 if (kFirst) {
79 upload_stats.mesh_create_count += kFirst.value().mesh_create_count;
80 upload_stats.uploaded_vertex_bytes += kFirst.value().uploaded_vertex_bytes;
81 upload_stats.uploaded_index_bytes += kFirst.value().uploaded_index_bytes;
82 }
83 if (kClean) {
84 upload_stats.skipped_clean_resource_count += kClean.value().skipped_clean_resource_count;
85 }
86 const crimson::FrameStats kStats = crimson::make_frame_stats(crimson::FrameStatsInput{
87 .uploads = upload_stats,
88 });
89 return crimson::make_renderer_counters(kStats);
90 });
91}
