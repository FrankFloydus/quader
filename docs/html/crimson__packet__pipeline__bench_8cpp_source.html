21[[nodiscard]] quader::math::Aabb bounds_for(std::uint32_t index) {
22 const float kX = static_cast<float>(index % 128) * 0.05F;
23 const float kY = static_cast<float>((index / 128) % 64) * 0.05F;
24 const float kZ = -5.0F - static_cast<float>(index % 256) * 0.01F;
25 return quader::math::Aabb{
26 .min = { kX - 0.25F, kY - 0.25F, kZ - 0.25F },
27 .max = { kX + 0.25F, kY + 0.25F, kZ + 0.25F },
28 };
29}

31[[nodiscard]] std::vector<crimson::RenderObject> make_objects(
32 std::uint32_t count,
33 crimson::RenderMaterialHandle opaque,
34 crimson::RenderMaterialHandle transparent) {
35 std::vector<crimson::RenderObject> objects;
36 objects.reserve(count);
37 for (std::uint32_t index = 0; index < count; ++index) {
38 const bool kTransparentObject = index % 11 == 0;
39 const quader::math::Aabb kBounds = bounds_for(index);
40 crimson::RenderObject object;
41 object.object_id = index + 1;
42 object.mesh = crimson::RenderMeshHandle{ 1 + (index % 8), 1 };
43 object.material = kTransparentObject ? transparent : opaque;
44 object.base_shader = kTransparentObject ? crimson::BaseShaderId::TransparentPbr : crimson::BaseShaderId::OpaquePbr;
45 object.queue = kTransparentObject ? crimson::RenderQueue::Transparent : crimson::RenderQueue::Opaque;
46 object.world_bounds = kBounds;
47 object.world_from_object[12] = quader::math::center(kBounds).x;
48 object.world_from_object[13] = quader::math::center(kBounds).y;
49 object.world_from_object[14] = quader::math::center(kBounds).z;
50 objects.push_back(object);
51 }
52 return objects;
53}

57BenchmarkResult run_crimson_packet_pipeline_benchmark(const BenchmarkRunConfig &config) {
58 crimson::MaterialSystem materials;
59 const crimson::RenderMaterialHandle kOpaque =
60 materials.create_default_material(crimson::BaseShaderId::OpaquePbr).value();
61 const crimson::RenderMaterialHandle kTransparent =
62 materials.create_default_material(crimson::BaseShaderId::TransparentPbr).value();
63 const std::vector<crimson::RenderObject> kObjects = make_objects(config.fixture_size, kOpaque, kTransparent);
64 const crimson::RenderCamera kCamera{
65 .eye = { 0.0F, 0.0F, 0.0F },
66 .target = { 0.0F, 0.0F, -1.0F },
67 .forward = { 0.0F, 0.0F, -1.0F },
68 .far_plane_m = 1000.0F,
69 };
70
71 return run_benchmark("crimson_packet_pipeline", config, [&]() {
72 const crimson::DrawPacketBuildResult kPackets = crimson::build_draw_packets(
73 kObjects,
74 materials.registry(),
75 materials,
76 kCamera,
77 crimson::RenderMeshHandle{ 1, 1 },
78 kOpaque);
79 std::vector<crimson::DrawPacket> lit_packets;
80 lit_packets.reserve(kPackets.draw_packet_count());
81 lit_packets.insert(lit_packets.end(), kPackets.opaque.begin(), kPackets.opaque.end());
82 lit_packets.insert(lit_packets.end(), kPackets.alpha_cutout.begin(), kPackets.alpha_cutout.end());
83 lit_packets.insert(lit_packets.end(), kPackets.transparent.begin(), kPackets.transparent.end());
84 const std::vector<crimson::InstanceBatch> kBatches = crimson::build_instance_batches(lit_packets);
85 const crimson::FrameStats kStats = crimson::make_frame_stats(crimson::FrameStatsInput{
86 .queues = crimson::FrameQueueStats{ .draw_packet_count = static_cast<std::uint32_t>(kPackets.draw_packet_count()) },
87 .culling = kPackets.culling,
88 .packets = kPackets.packets,
89 .instancing = crimson::make_instancing_stats(kBatches),
90 });
91 return crimson::make_renderer_counters(kStats);
92 });
93}
