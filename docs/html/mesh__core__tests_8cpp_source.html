54bool is_same_cyclic_vertex_order(const std::vector<quader::mesh::VertexId> &actual,
55 const std::array<quader::mesh::VertexId, 3> &expected) {
56 if (actual.size() != expected.size()) {
57 return false;
58 }
59
60 for (std::size_t offset = 0; offset < expected.size(); ++offset) {
61 bool matches = true;
62 for (std::size_t index = 0; index < expected.size(); ++index) {
63 if (actual[(offset + index) % expected.size()] != expected[index]) {
64 matches = false;
65 break;
66 }
67 }
68 if (matches) {
69 return true;
70 }
71 }
72 return false;
73}
