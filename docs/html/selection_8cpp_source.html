18[[nodiscard]] DocumentError invalid_component_error(ComponentKind kind) {
19 switch (kind) {
20 case ComponentKind::Vertex:
21 return make_document_error(DocumentErrorCode::InvalidVertex,
22 "selection contains an invalid vertex id");
23 case ComponentKind::Edge:
24 return make_document_error(DocumentErrorCode::InvalidEdge,
25 "selection contains an invalid edge id");
26 case ComponentKind::Face:
27 return make_document_error(DocumentErrorCode::InvalidFace,
28 "selection contains an invalid face id");
29 }
30
31 return make_document_error(DocumentErrorCode::InvalidSelection,
32 "selection contains an invalid component id");
33}

35[[nodiscard]] auto component_key(const ComponentRef &ref) {
36 return std::visit(
37 [](auto id) {
38 return std::tuple{
39 id.index(),
40 id.generation(),
41 };
42 },
43 ref.component);
44}
45
46[[nodiscard]] bool component_less(const ComponentRef &left, const ComponentRef &right) {
47 if (left.object != right.object) {
48 return left.object < right.object;
49 }
50
51 if (left.component.index() != right.component.index()) {
52 return left.component.index() < right.component.index();
53 }
54
55 return component_key(left) < component_key(right);
56}

60ComponentKind component_kind(const ComponentRef &ref) noexcept {
61 switch (ref.component.index()) {
62 case 0U:
63 return ComponentKind::Vertex;
64 case 1U:
65 return ComponentKind::Edge;
66 case 2U:
67 return ComponentKind::Face;
68 default:
69 return ComponentKind::Vertex;
70 }

86quader::foundation::Result<void, DocumentError> Selection::set_objects(std::vector<ObjectId> objects) {
87 for (const auto kObject : objects) {
88 if (!kObject.is_valid()) {
89 return quader::foundation::Result<void, DocumentError>::failure(
90 make_document_error(DocumentErrorCode::InvalidObject,
91 "selection contains an invalid object id"));
92 }
93 }
94
95 std::sort(objects.begin(), objects.end());
96 objects.erase(std::unique(objects.begin(), objects.end()), objects.end());
97
98 objects_ = std::move(objects);
99 components_.clear();
100 return quader::foundation::Result<void, DocumentError>::success();
101}

103quader::foundation::Result<void, DocumentError> Selection::set_components(
104 std::vector<ComponentRef> components) {
105 for (const auto &component : components) {
106 if (!component.object.is_valid()) {
107 return quader::foundation::Result<void, DocumentError>::failure(
108 make_document_error(DocumentErrorCode::InvalidObject,
109 "selection contains an invalid object id"));
110 }
111
112 if (!component_id_is_valid(component)) {
113 return quader::foundation::Result<void, DocumentError>::failure(
114 invalid_component_error(component_kind(component)));
115 }
116 }
117
118 std::sort(components.begin(), components.end(), component_less);
119 components.erase(std::unique(components.begin(), components.end()), components.end());
120
121 objects_.clear();
122 components_ = std::move(components);
123 return quader::foundation::Result<void, DocumentError>::success();
124}

134bool Selection::references_object(ObjectId object) const noexcept {
135 if (std::find(objects_.begin(), objects_.end(), object) != objects_.end()) {
136 return true;
137 }
138
139 return std::any_of(components_.begin(), components_.end(), [object](const ComponentRef &component) {
140 return component.object == object;
141 });
142}

144void Selection::remove_object(ObjectId object) {
145 objects_.erase(std::remove(objects_.begin(), objects_.end(), object), objects_.end());
146 components_.erase(std::remove_if(components_.begin(),
147 components_.end(),
148 [object](const ComponentRef &component) {
149 return component.object == object;
150 }),
151 components_.end());
152}
