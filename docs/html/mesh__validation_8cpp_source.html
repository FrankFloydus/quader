26[[nodiscard]] std::string slot_message(const char *type, std::size_t index, const char *message) {

30template <class Handle>
31[[nodiscard]] int handle_index(Handle handle) noexcept {
32 return handle.is_valid() ? handle.idx() : -1;
33}

35template <class Handle>
36[[nodiscard]] quader::foundation::IdIndex reverse_slot(const std::vector<quader::foundation::IdIndex> &reverse,
37 Handle handle) noexcept {
38 const auto kIndex = handle_index(handle);
39 if (kIndex < 0) {
40 return kNoSlot;
41 }
42
43 const auto kReverseIndex = static_cast<std::size_t>(kIndex);
44 if (kReverseIndex >= reverse.size()) {
45 return kNoSlot;
46 }
47
48 return reverse[kReverseIndex];
49}

51template <class Handle>
52[[nodiscard]] bool live_backend_handle(const detail::QuaderOpenMesh &backend, Handle handle) noexcept {
53 return handle.is_valid() && backend.is_valid_handle(handle) && !backend.status(handle).deleted();
54}

60template <class Slot>
61void validate_reverse_map(MeshValidationReport &report,
62 const char *type,
63 const std::vector<Slot> &slots,
64 const std::vector<quader::foundation::IdIndex> &reverse) {
65 for (quader::foundation::IdIndex handle_index_value = 0; handle_index_value < reverse.size();
66 ++handle_index_value) {
67 const auto kSlotIndex = reverse[handle_index_value];
68 if (kSlotIndex == kNoSlot) {
69 continue;
70 }
71
72 if (kSlotIndex >= slots.size() || !slots[kSlotIndex].alive) {
73 report.add_issue(MeshValidationCode::MissingQuaderId,
74 std::string(type) + " reverse map points at a missing Quader id");
75 continue;
76 }
77
78 if (handle_index(slots[kSlotIndex].handle) != static_cast<int>(handle_index_value)) {
79 report.add_issue(MeshValidationCode::ReverseMappingMismatch,
80 std::string(type) + " reverse map points at a mismatched backend handle");
81 }
82 }
83}

85template <class Slot>
86void validate_backend_slot(MeshValidationReport &report,
87 const detail::QuaderOpenMesh &backend,
88 const char *type,
89 quader::foundation::IdIndex slot_index,
90 const Slot &slot,
91 const std::vector<quader::foundation::IdIndex> &reverse) {
92 if (!slot.alive) {
93 return;
94 }
95
96 if (!slot.handle.is_valid() || !backend.is_valid_handle(slot.handle)) {
97 report.add_issue(MeshValidationCode::MissingBackendHandle,
98 slot_message(type, slot_index, "missing backend handle"));
99 return;
100 }
101
102 if (backend.status(slot.handle).deleted()) {
103 report.add_issue(MeshValidationCode::DeletedElementMapped,
104 slot_message(type, slot_index, "maps to a deleted backend element"));
105 return;
106 }
107
108 if (reverse_slot(reverse, slot.handle) != slot_index) {
109 report.add_issue(MeshValidationCode::ReverseMappingMismatch,
110 slot_message(type, slot_index, "reverse map does not point back to this Quader id"));
111 }
112}

116MeshValidationReport validate_mesh(const Polyhedron &mesh) {
117 MeshValidationReport report;
118 const auto &storage = *mesh.storage_;
119 const auto &backend = storage.backend;
120
121 validate_reverse_map(report, "vertex", storage.vertices, storage.vertex_reverse.handle_to_slot);
122 validate_reverse_map(report, "halfedge", storage.halfedges, storage.halfedge_reverse.handle_to_slot);
123 validate_reverse_map(report, "edge", storage.edges, storage.edge_reverse.handle_to_slot);
124 validate_reverse_map(report, "face", storage.faces, storage.face_reverse.handle_to_slot);
125
126 for (quader::foundation::IdIndex index = 0; index < storage.vertices.size(); ++index) {
127 const auto &vertex = storage.vertices[index];
128 validate_backend_slot(report, backend, "vertex", index, vertex, storage.vertex_reverse.handle_to_slot);
129 if (!vertex.alive || !live_backend_handle(backend, vertex.handle)) {
130 continue;
131 }
132
133 if (!quader::math::is_finite(from_openmesh_point(backend.point(vertex.handle)))) {
134 report.add_issue(MeshValidationCode::NonFinitePosition,
135 slot_message("vertex", index, "position contains non-finite values"));
136 }
137
138 const auto kOutgoing = backend.halfedge_handle(vertex.handle);
139 if (kOutgoing.is_valid()) {
140 const auto kOutgoingSlot = reverse_slot(storage.halfedge_reverse.handle_to_slot, kOutgoing);
141 if (kOutgoingSlot == kNoSlot || kOutgoingSlot >= storage.halfedges.size() || !storage.halfedges[kOutgoingSlot].alive) {
142 report.add_issue(MeshValidationCode::InvalidId,
143 slot_message("vertex", index, "outgoing halfedge has no Quader id"));
144 } else if (backend.from_vertex_handle(kOutgoing) != vertex.handle) {
145 report.add_issue(MeshValidationCode::VertexReferenceMismatch,
146 slot_message("vertex", index, "outgoing halfedge does not originate here"));
147 }
148 }
149 }
150
151 for (quader::foundation::IdIndex index = 0; index < storage.edges.size(); ++index) {
152 const auto &edge = storage.edges[index];
153 validate_backend_slot(report, backend, "edge", index, edge, storage.edge_reverse.handle_to_slot);
154 if (!edge.alive || !live_backend_handle(backend, edge.handle)) {
155 continue;
156 }
157
158 if (!mesh.is_valid(edge.representative_halfedge)) {
159 report.add_issue(MeshValidationCode::InvalidId,
160 slot_message("edge", index, "representative halfedge id is invalid"));
161 } else {
162 const auto kRepresentative = storage.halfedges[edge.representative_halfedge.index()].handle;
163 if (backend.edge_handle(kRepresentative) != edge.handle) {
164 report.add_issue(MeshValidationCode::EdgeReferenceMismatch,
165 slot_message("edge", index, "representative halfedge points at another edge"));
166 }
167 }
168 }
169
170 for (quader::foundation::IdIndex index = 0; index < storage.halfedges.size(); ++index) {
171 const auto &halfedge = storage.halfedges[index];
172 validate_backend_slot(report, backend, "halfedge", index, halfedge, storage.halfedge_reverse.handle_to_slot);
173 if (!halfedge.alive || !live_backend_handle(backend, halfedge.handle)) {
174 continue;
175 }
176
177 const HalfedgeId kHalfedgeId{ index, halfedge.generation };
178 const auto kOriginHandle = backend.from_vertex_handle(halfedge.handle);
179 const auto kOriginSlot = reverse_slot(storage.vertex_reverse.handle_to_slot, kOriginHandle);
180 if (kOriginSlot == kNoSlot || kOriginSlot >= storage.vertices.size() || !storage.vertices[kOriginSlot].alive) {
181 report.add_issue(MeshValidationCode::InvalidId,
182 slot_message("halfedge", index, "origin vertex id is invalid"));
183 }
184
185 const auto kEdgeHandle = backend.edge_handle(halfedge.handle);
186 const auto kEdgeSlot = reverse_slot(storage.edge_reverse.handle_to_slot, kEdgeHandle);
187 if (kEdgeSlot == kNoSlot || kEdgeSlot >= storage.edges.size() || !storage.edges[kEdgeSlot].alive) {
188 report.add_issue(MeshValidationCode::InvalidId,
189 slot_message("halfedge", index, "edge id is invalid"));
190 }
191
192 const auto kOppositeHandle = backend.opposite_halfedge_handle(halfedge.handle);
193 const auto kOppositeSlot = reverse_slot(storage.halfedge_reverse.handle_to_slot, kOppositeHandle);
194 if (kOppositeSlot == kNoSlot || kOppositeSlot >= storage.halfedges.size() || !storage.halfedges[kOppositeSlot].alive) {
195 report.add_issue(MeshValidationCode::BrokenOppositePair,
196 slot_message("halfedge", index, "opposite halfedge id is invalid"));
197 } else {
198 const auto kOpposite = storage.halfedges[kOppositeSlot].handle;
199 if (backend.opposite_halfedge_handle(kOpposite) != halfedge.handle || backend.edge_handle(kOpposite) != kEdgeHandle) {
200 report.add_issue(MeshValidationCode::BrokenOppositePair,
201 slot_message("halfedge", index, "opposite halfedge does not point back"));
202 }
203 }
204
205 if (backend.is_boundary(halfedge.handle)) {
206 if (backend.face_handle(halfedge.handle).is_valid()) {
207 report.add_issue(MeshValidationCode::BoundaryPlaceholderMismatch,
208 slot_message("halfedge", index, "boundary halfedge has a face"));
209 }
210 continue;
211 }
212
213 const auto kFaceHandle = backend.face_handle(halfedge.handle);
214 const auto kFaceSlot = reverse_slot(storage.face_reverse.handle_to_slot, kFaceHandle);
215 if (kFaceSlot == kNoSlot || kFaceSlot >= storage.faces.size() || !storage.faces[kFaceSlot].alive) {
216 report.add_issue(MeshValidationCode::InvalidId,
217 slot_message("halfedge", index, "face id is invalid"));
218 }
219
220 const auto kNextHandle = backend.next_halfedge_handle(halfedge.handle);
221 const auto kPrevHandle = backend.prev_halfedge_handle(halfedge.handle);
222 const auto kNextSlot = reverse_slot(storage.halfedge_reverse.handle_to_slot, kNextHandle);
223 const auto kPrevSlot = reverse_slot(storage.halfedge_reverse.handle_to_slot, kPrevHandle);
224 if (kNextSlot == kNoSlot || kPrevSlot == kNoSlot || kNextSlot >= storage.halfedges.size() || kPrevSlot >= storage.halfedges.size() || !storage.halfedges[kNextSlot].alive || !storage.halfedges[kPrevSlot].alive) {
225 report.add_issue(MeshValidationCode::BrokenNextPrevCycle,
226 slot_message("halfedge", index, "next or prev id is invalid"));
227 continue;
228 }
229
230 const auto kNext = storage.halfedges[kNextSlot].handle;
231 const auto kPrev = storage.halfedges[kPrevSlot].handle;
232 if (backend.prev_halfedge_handle(kNext) != halfedge.handle || backend.next_halfedge_handle(kPrev) != halfedge.handle) {
233 report.add_issue(MeshValidationCode::BrokenNextPrevCycle,
234 slot_message("halfedge", index, "next/prev links are not reciprocal"));
235 }
236 if (backend.face_handle(kNext) != kFaceHandle || backend.face_handle(kPrev) != kFaceHandle) {
237 report.add_issue(MeshValidationCode::FaceCycleMismatch,
238 slot_message("halfedge", index, "next/prev halfedges belong to another face"));
239 }
240
241 (void)kHalfedgeId;
242 }
243
244 for (quader::foundation::IdIndex index = 0; index < storage.faces.size(); ++index) {
245 const auto &face = storage.faces[index];
246 validate_backend_slot(report, backend, "face", index, face, storage.face_reverse.handle_to_slot);
247 if (!face.alive || !live_backend_handle(backend, face.handle)) {
248 continue;
249 }
250
251 std::vector<quader::math::Vec3> positions;
252 for (auto halfedge_it = backend.cfh_iter(face.handle); halfedge_it.is_valid(); ++halfedge_it) {
253 const auto kHalfedgeHandle = *halfedge_it;
254 const auto kHalfedgeSlot = reverse_slot(storage.halfedge_reverse.handle_to_slot, kHalfedgeHandle);
255 if (kHalfedgeSlot == kNoSlot || kHalfedgeSlot >= storage.halfedges.size() || !storage.halfedges[kHalfedgeSlot].alive) {
256 report.add_issue(MeshValidationCode::InvalidId,
257 slot_message("face", index, "cycle references an invalid halfedge"));
258 break;
259 }
260
261 if (backend.is_boundary(kHalfedgeHandle) || backend.face_handle(kHalfedgeHandle) != face.handle) {
262 report.add_issue(MeshValidationCode::FaceCycleMismatch,
263 slot_message("face", index, "cycle contains a boundary or mismatched halfedge"));
264 break;
265 }
266
267 const auto kVertexHandle = backend.from_vertex_handle(kHalfedgeHandle);
268 const auto kVertexSlot = reverse_slot(storage.vertex_reverse.handle_to_slot, kVertexHandle);
269 if (kVertexSlot == kNoSlot || kVertexSlot >= storage.vertices.size() || !storage.vertices[kVertexSlot].alive) {
270 report.add_issue(MeshValidationCode::InvalidId,
271 slot_message("face", index, "cycle contains an invalid vertex"));
272 break;
273 }
274
275 positions.push_back(from_openmesh_point(backend.point(kVertexHandle)));
276 }
277
278 if (positions.size() < 3U || !quader::geometry::all_points_finite(positions)) {
279 report.add_issue(MeshValidationCode::DegenerateFace,
280 slot_message("face", index, "cycle does not contain a finite polygon"));
281 continue;
282 }
283
284 if (quader::geometry::has_degenerate_area(positions)) {
285 report.add_issue(MeshValidationCode::DegenerateFace,
286 slot_message("face", index, "cycle polygon area is degenerate"));
287 }
288 }
289
290 return report;
291}
