20TEST(IoDiagnostic, IoDiagnosticsPreserveSourceLocationPathAndSubject) {
21 quader::io::IoDiagnosticList diagnostics;
22 diagnostics.add(quader::io::IoDiagnostic{
23 quader::io::IoSeverity::Warning,
24 quader::io::IoDiagnosticCode::UnsupportedFeaturePreserved,
25 "preserved unsupported extension",
26 std::filesystem::path{ "scene.gltf" },
27 quader::io::SourceLocation{ 12U, 7U, "/materials/0/extensions" },
28 "MaterialA",
29 });
30 diagnostics.add_error(quader::io::IoDiagnosticCode::ParseFailed, "invalid token");
31
32 EXPECT_FALSE(diagnostics.empty());
33 EXPECT_TRUE(diagnostics.has_errors());
34 EXPECT_EQ(diagnostics.size(), 2U);
35 EXPECT_EQ(diagnostics[0].severity, quader::io::IoSeverity::Warning);
36 EXPECT_EQ(diagnostics[0].code, quader::io::IoDiagnosticCode::UnsupportedFeaturePreserved);
37 EXPECT_EQ(diagnostics[0].source_path, std::filesystem::path{ "scene.gltf" });
38 EXPECT_EQ(diagnostics[0].location.line, 12U);
39 EXPECT_EQ(diagnostics[0].location.column, 7U);
40 EXPECT_EQ(diagnostics[0].location.json_pointer, std::string("/materials/0/extensions"));
41 EXPECT_EQ(diagnostics[0].subject, std::string("MaterialA"));
42 EXPECT_EQ(diagnostics[1].severity, quader::io::IoSeverity::Error);
43}

45TEST(IoDiagnostic, IoDiagnosticsConvertToFoundationDiagnostics) {
46 quader::io::IoDiagnosticList diagnostics;
47 diagnostics.add(quader::io::IoSeverity::Info,
48 quader::io::IoDiagnosticCode::Unknown,
49 "loaded optional metadata");
50 diagnostics.add(quader::io::IoSeverity::Warning,
51 quader::io::IoDiagnosticCode::UnsupportedFeaturePreserved,
52 "preserved extension");
53 diagnostics.add_error(quader::io::IoDiagnosticCode::ValidationFailed, "invalid mesh");
54
55 const auto kConverted = quader::io::to_foundation_diagnostics(diagnostics);
56 EXPECT_EQ(kConverted.size(), 3U);
57 EXPECT_TRUE(kConverted.has_errors());
58 EXPECT_EQ(kConverted[0].severity, quader::foundation::Severity::Info);
59 EXPECT_EQ(kConverted[1].severity, quader::foundation::Severity::Warning);
60 EXPECT_EQ(kConverted[2].severity, quader::foundation::Severity::Error);
61 EXPECT_EQ(kConverted[2].message, std::string("invalid mesh"));
62}
