26[[nodiscard]] QString first_diagnostic_message(const quader::commands::CommandResult &result) {

56quader::commands::CommandResult DocumentUiController::execute_command(
57 std::unique_ptr<quader::commands::ICommand> command,
58 CommandFeedback feedback) {
59 auto result = command_history_.execute(std::move(command), document_);
60 if (result.is_applied()) {
61 drain_pending_changes();
62 Q_EMIT command_history_changed();
63 } else if (feedback == CommandFeedback::NotifyOnReject) {
64 report_rejected_command(result);
65 }
66
67 refresh_actions();
68 return result;
69}

102void DocumentUiController::drain_pending_changes() {
103 const auto kChanges = document_.take_pending_changes();
104 if (kChanges.empty()) {
105 return;
106 }
107
108 for (const auto &change : kChanges) {
109 emit_change(change);
110 }
111
112 Q_EMIT document_events_drained();
113}

115void DocumentUiController::emit_change(const quader::document::DocumentChange &change) {
116 using quader::document::DocumentChangeKind;
117
118 switch (change.kind) {
119 case DocumentChangeKind::ObjectCreated:
120 case DocumentChangeKind::ObjectDeleted:
121 Q_EMIT object_list_changed();
122 break;
123 case DocumentChangeKind::ObjectRenamed:
124 case DocumentChangeKind::TransformChanged:
125 case DocumentChangeKind::MeshTopologyChanged:
126 case DocumentChangeKind::MeshGeometryChanged:
127 Q_EMIT object_changed(change.object);
128 break;
129 case DocumentChangeKind::SelectionChanged:
130 Q_EMIT selection_changed();
131 break;
132 case DocumentChangeKind::DirtyStateChanged:
133 Q_EMIT dirty_state_changed(change.dirty_flags);
134 break;
135 }
136}
