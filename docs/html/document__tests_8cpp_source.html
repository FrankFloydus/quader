23bool contains_change(const std::vector<quader::document::DocumentChange> &changes,
24 quader::document::DocumentChangeKind kind) {
25 for (const auto &change : changes) {
26 if (change.kind == kind) {
27 return true;
28 }
29 }
30
31 return false;
32}

34void discard_pending_changes(quader::document::Document &document) {

39TEST(Document, CreateMeshObjectStoresMeshTransformDirtyFlagsAndEvents) {
40 quader::document::Document document;
41 auto mesh_fixture = quader::tests::document_fixtures::make_triangle_mesh();
42
43 quader::document::Transform transform;
44 transform.translation = quader::math::Vec3{ 2.0F, 3.0F, 4.0F };
45 transform.rotation_euler = quader::math::Vec3{ 0.0F, 90.0F, 0.0F };
46 transform.scale = quader::math::Vec3{ 1.0F, 2.0F, 3.0F };
47
48 const auto kObject = document.create_mesh_object("Triangle", std::move(mesh_fixture.mesh), transform);
49 EXPECT_TRUE(kObject);
50 EXPECT_TRUE(kObject.value().is_valid());
51 EXPECT_EQ(document.object_count(), 1U);
52
53 const auto *stored = document.find_mesh_object(kObject.value());
54 EXPECT_TRUE(stored != nullptr);
55 EXPECT_EQ(stored->name, std::string("Triangle"));
56 EXPECT_TRUE(stored->transform == transform);
57 EXPECT_EQ(stored->material, quader::document::default_box_material());
58 EXPECT_EQ(stored->mesh.vertex_count(), 3U);
59 EXPECT_EQ(stored->mesh.face_count(), 1U);
60 EXPECT_TRUE(quader::mesh::validate_mesh(stored->mesh).ok());
61
62 EXPECT_TRUE(document.is_dirty());
63 EXPECT_TRUE(document.has_dirty_flag(quader::document::DocumentDirtyFlag::Objects));
64
65 const auto kChanges = document.take_pending_changes();
66 EXPECT_EQ(kChanges.size(), 2U);
67 EXPECT_EQ(kChanges[0].kind, quader::document::DocumentChangeKind::ObjectCreated);
68 EXPECT_EQ(kChanges[0].object, kObject.value());
69 EXPECT_EQ(kChanges[1].kind, quader::document::DocumentChangeKind::DirtyStateChanged);
70 EXPECT_TRUE(quader::document::has_dirty_flag(kChanges[1].dirty_flags,
71 quader::document::DocumentDirtyFlag::Objects));
72}

140 fixture.document.clear_dirty();
141 discard_pending_changes(fixture.document);
142
143 quader::document::Transform transform;
144 transform.translation = quader::math::Vec3{ 4.0F, 5.0F, 6.0F };
145 transform.scale = quader::math::Vec3{ 2.0F, 2.0F, 2.0F };
146
147 EXPECT_TRUE(fixture.document.set_transform(fixture.object, transform));
148 const auto *object = fixture.document.find_mesh_object(fixture.object);
149 EXPECT_TRUE(object != nullptr);
150 EXPECT_TRUE(object->transform == transform);
151
152 auto changes = fixture.document.take_pending_changes();
153 EXPECT_EQ(changes.size(), 2U);
154 EXPECT_EQ(changes[0].kind, quader::document::DocumentChangeKind::TransformChanged);
155 EXPECT_EQ(changes[1].kind, quader::document::DocumentChangeKind::DirtyStateChanged);
156 EXPECT_TRUE(fixture.document.has_dirty_flag(quader::document::DocumentDirtyFlag::Transforms));
157
158 fixture.document.clear_dirty();
159 discard_pending_changes(fixture.document);
160 const auto kBefore = quader::tests::document_fixtures::capture_document_state(fixture.document);
161
162 quader::document::Transform invalid_transform = transform;
163 invalid_transform.translation.x = std::numeric_limits<float>::infinity();
164 const auto kInvalidResult = fixture.document.set_transform(fixture.object, invalid_transform);
165 EXPECT_FALSE(kInvalidResult);
166 EXPECT_EQ(kInvalidResult.error().code, quader::document::DocumentErrorCode::InvalidTransform);
167 EXPECT_EQ(quader::tests::document_fixtures::capture_document_state(fixture.document), kBefore);
168 EXPECT_TRUE(fixture.document.take_pending_changes().empty());
169
170 const auto kInvalidObjectResult = fixture.document.set_transform(
171 quader::document::ObjectId{ 999U, 1U },
172 transform);
173 EXPECT_FALSE(kInvalidObjectResult);
174 EXPECT_EQ(kInvalidObjectResult.error().code, quader::document::DocumentErrorCode::InvalidObject);
175}

195 },
196 }));
197 const auto kInvalidComponent = fixture.document.set_selection(invalid_component_selection);
198 EXPECT_FALSE(kInvalidComponent);
199 EXPECT_EQ(kInvalidComponent.error().code, quader::document::DocumentErrorCode::InvalidVertex);
200 EXPECT_TRUE(fixture.document.selection().empty());
201
202 quader::document::Selection component_selection;
203 EXPECT_TRUE(component_selection.set_components({
204 quader::document::ComponentRef{ fixture.object, fixture.face },
205 quader::document::ComponentRef{ fixture.object, fixture.vertices[0] },
206 quader::document::ComponentRef{ fixture.object, fixture.edge },
207 quader::document::ComponentRef{ fixture.object, fixture.edge },
208 }));
209 EXPECT_TRUE(fixture.document.set_selection(component_selection));
210 EXPECT_TRUE(fixture.document.selection().selected_objects().empty());
211 EXPECT_EQ(fixture.document.selection().selected_components().size(), 3U);
212 EXPECT_TRUE(fixture.document.has_dirty_flag(quader::document::DocumentDirtyFlag::Selection));
213
214 const auto kChanges = fixture.document.take_pending_changes();
215 EXPECT_TRUE(contains_change(kChanges, quader::document::DocumentChangeKind::SelectionChanged));
216 EXPECT_TRUE(contains_change(kChanges, quader::document::DocumentChangeKind::DirtyStateChanged));
217
218 quader::document::Selection object_selection;
219 EXPECT_TRUE(object_selection.set_objects({ fixture.object, fixture.object }));
220 EXPECT_TRUE(fixture.document.set_selection(object_selection));
221 EXPECT_EQ(fixture.document.selection().selected_objects().size(), 1U);
222 EXPECT_TRUE(fixture.document.selection().selected_components().empty());
223}
