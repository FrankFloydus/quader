128TEST(FrameStatsDiagnostics, RendererCountersAreTypedForDiagnosticsSnapshots) {
129 const crimson::FrameStats kStats = crimson::make_frame_stats(crimson::FrameStatsInput{
130 .frame_index = 7,
131 .graph_pass_count = 12,
132 .queues = crimson::FrameQueueStats{ .draw_packet_count = 2, .opaque_draw_count = 1 },
133 .culling = crimson::FrameCullingStats{ .input_object_count = 2, .visible_object_count = 1 },
134 .uploads = crimson::FrameUploadStats{ .texture_upload_count = 5, .uploaded_vertex_bytes = 96 },
135 .timings = crimson::FrameTimingStats{ .cpu_frame_us = 5 },
136 });
137 const std::vector<crimson::RendererCounter> kCounters = crimson::make_renderer_counters(kStats);
138
139 bool saw_frame_index = false;
140 bool saw_upload_bytes = false;
141 bool saw_texture_uploads = false;
142 for (const crimson::RendererCounter &counter : kCounters) {
143 if (counter.domain == crimson::RendererCounterDomain::Frame && counter.name == "frame_index") {
144 saw_frame_index = counter.value == 7 && counter.unit == crimson::RendererMetricUnit::Frames;
145 }
146 if (counter.domain == crimson::RendererCounterDomain::Upload && counter.name == "uploaded_vertex_bytes") {
147 saw_upload_bytes = counter.value == 96 && counter.unit == crimson::RendererMetricUnit::Bytes;
148 }
149 if (counter.domain == crimson::RendererCounterDomain::Upload && counter.name == "texture_upload_count") {
150 saw_texture_uploads = counter.value == 5 && counter.unit == crimson::RendererMetricUnit::Count;
151 }
152 }
153
154 expect_true(saw_frame_index, "typed counters expose frame index with frame units");
155 expect_true(saw_upload_bytes, "typed counters expose upload bytes with byte units");
156 expect_true(saw_texture_uploads, "typed counters expose live texture upload counts");
157 expect_true(
158 crimson::renderer_counter_domain_name(crimson::RendererCounterDomain::Instancing) == "Instancing",
159 "counter domain names are stable");
160 expect_true(
161 crimson::renderer_metric_unit_name(crimson::RendererMetricUnit::Microseconds) == "Microseconds",
162 "metric unit names are stable");
163}

165TEST(FrameStatsDiagnostics, DiagnosticRingCapsAndPreservesNewestEntries) {
166 crimson::DiagnosticRing ring(3);
167 for (std::uint64_t index = 0; index < 5; ++index) {
168 ring.push(crimson::RendererDiagnostic{
169 .severity = crimson::RendererDiagnosticSeverity::Warning,

174 }
175

179 expect_true(kDiagnostics[2].frame_index == 4, "diagnostic ring keeps newest entry last");

182TEST(FrameStatsDiagnostics, DiagnosticsStatusSnapshotUsesBoundedRecentEntries) {
183 crimson::RendererStatus status;
184 status.backend_name = "test";
185 status.initialized = true;
186 for (std::uint64_t index = 0; index < 5; ++index) {
187 status.diagnostics.push_back(crimson::RendererDiagnostic{
188 .severity = crimson::RendererDiagnosticSeverity::Warning,
189 .code = crimson::RendererDiagnosticCode::PerformanceCounterInvalid,
190 .message = "diagnostic",
191 .frame_index = index,
192 });
193 }
194
195 const std::vector<crimson::RendererDiagnostic> kRecent =
196 crimson::make_diagnostic_ring_from_recent(status.diagnostics, 2).snapshot();
197 const crimson::RendererStatus kSnapshot =
198 crimson::make_bounded_renderer_status_snapshot(status, kRecent);
199
200 expect_true(kSnapshot.backend_name == "test", "bounded status keeps backend identity");
201 expect_true(kSnapshot.initialized, "bounded status keeps initialization state");
202 expect_true(kSnapshot.diagnostics.size() == 2, "bounded status avoids copying unbounded diagnostics");
203 expect_true(kSnapshot.diagnostics[0].frame_index == 3, "bounded status keeps oldest retained diagnostic first");
204 expect_true(kSnapshot.diagnostics[1].frame_index == 4, "bounded status keeps newest diagnostic last");
205}
