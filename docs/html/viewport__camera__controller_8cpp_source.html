40[[nodiscard]] Vec3 forward_from_angles(float yaw_radians, float pitch_radians) {
41 const float kCp = std::cos(pitch_radians);
42 return normalized(Vec3{
43 std::sin(yaw_radians) * kCp,
44 std::sin(pitch_radians),
45 -std::cos(yaw_radians) * kCp,
46 });
47}

49[[nodiscard]] Vec3 fallback_up_for_forward(Vec3 forward) {
50 if (std::abs(forward.y) > 0.999F) {
51 return forward.y < 0.0F ? Vec3{ 0.0F, 0.0F, 1.0F } : Vec3{ 0.0F, 0.0F, -1.0F };
52 }
53 return { 0.0F, 1.0F, 0.0F };
54}

56struct CameraBasis {
57 Vec3 forward{ 0.0F, 0.0F, -1.0F };
58 Vec3 up{ 0.0F, 1.0F, 0.0F };

61struct CameraTransform {

63 Vec3 y_axis{ 0.0F, 1.0F, 0.0F };
64 Vec3 z_axis{ 0.0F, 0.0F, 1.0F };

68[[nodiscard]] CameraBasis make_camera_basis(
69 Vec3 forward_direction,
70 Vec3 up_direction,
71 Vec3 fallback_forward = { 0.0F, 0.0F, -1.0F }) {
72 Vec3 forward = normalized(forward_direction);
73 if (length(forward) <= 0.000001F) {
74 forward = normalized(fallback_forward);
75 }
76 if (length(forward) <= 0.000001F) {
77 forward = { 0.0F, 0.0F, -1.0F };
78 }
79
80 Vec3 candidate_up = normalized(up_direction);
81 if (length(candidate_up) <= 0.000001F || std::abs(dot(candidate_up, forward)) > 0.999F) {
82 candidate_up = fallback_up_for_forward(forward);
83 }
84
85 Vec3 right = normalized(cross(forward, candidate_up));
86 if (length(right) <= 0.000001F) {
87 right = { 1.0F, 0.0F, 0.0F };
88 }
89
90 Vec3 up = normalized(cross(right, forward));
91 if (length(up) <= 0.000001F) {
92 up = fallback_up_for_forward(forward);
93 }
94
95 return { forward, up };
96}

98void sync_angles_for_forward(ViewportCameraState &camera, Vec3 forward) {
99 const Vec3 kNormalizedForward = normalized(forward);
100 if (length(kNormalizedForward) <= 0.000001F) {
101 return;
102 }
103
104 camera.pitch_radians = std::asin(std::clamp(kNormalizedForward.y, -1.0F, 1.0F));
105 camera.yaw_radians = std::atan2(kNormalizedForward.x, -kNormalizedForward.z);
106}

108[[nodiscard]] Vec3 transform_vector(const CameraTransform &transform, Vec3 value) {
109 return transform.x_axis * value.x + transform.y_axis * value.y + transform.z_axis * value.z;
110}

112[[nodiscard]] Vec3 rotate_vector_around_axis(Vec3 value, Vec3 axis, float angle_radians) {
113 const Vec3 kNormal = normalized(axis);
114 if (length(kNormal) <= 0.000001F) {
115 return value;
116 }
117
118 const float kC = std::cos(angle_radians);
119 const float kS = std::sin(angle_radians);
120 const float kOneMinusC = 1.0F - kC;
121 return value * kC + cross(kNormal, value) * kS + kNormal * (dot(kNormal, value) * kOneMinusC);
122}

124[[nodiscard]] CameraTransform orthonormalized_basis(CameraTransform basis) {
125 Vec3 right = normalized(basis.x_axis);
126 if (length(right) <= 0.000001F) {
127 right = { 1.0F, 0.0F, 0.0F };
128 }
129
130 Vec3 up = basis.y_axis - right * dot(basis.y_axis, right);
131 up = normalized(up);
132 if (length(up) <= 0.000001F) {
133 const Vec3 kFallback = std::abs(right.y) > 0.9F ? Vec3{ 0.0F, 0.0F, 1.0F } : Vec3{ 0.0F, 1.0F, 0.0F };
134 up = normalized(kFallback - right * dot(kFallback, right));
135 }
136
137 Vec3 back = normalized(cross(right, up));
138 if (length(back) <= 0.000001F) {
139 back = { 0.0F, 0.0F, 1.0F };
140 }
141
142 up = normalized(cross(back, right));
143 return { right, up, back, basis.origin };
144}

146[[nodiscard]] CameraTransform rotation_basis(Vec3 axis, float angle_radians) {
147 const Vec3 kNormal = normalized(axis);
148 if (length(kNormal) <= 0.000001F) {
149 return {};
150 }
151
152 return {
153 rotate_vector_around_axis({ 1.0F, 0.0F, 0.0F }, kNormal, angle_radians),
154 rotate_vector_around_axis({ 0.0F, 1.0F, 0.0F }, kNormal, angle_radians),
155 rotate_vector_around_axis({ 0.0F, 0.0F, 1.0F }, kNormal, angle_radians),
156 {},
157 };
158}

160[[nodiscard]] CameraTransform basis_multiply(const CameraTransform &left, const CameraTransform &right) {
161 return {
162 transform_vector(left, right.x_axis),
163 transform_vector(left, right.y_axis),
164 transform_vector(left, right.z_axis),
165 {},

169[[nodiscard]] CameraTransform rotated_camera_basis(
170 const CameraTransform &camera_transform,
171 const CameraTransform &rotation) {
172 return orthonormalized_basis({
173 transform_vector(rotation, camera_transform.x_axis),
174 transform_vector(rotation, camera_transform.y_axis),
175 transform_vector(rotation, camera_transform.z_axis),
176 camera_transform.origin,
177 });

180[[nodiscard]] CameraTransform clamped_rotation_to_upright(
181 const CameraTransform &camera_transform,
182 const CameraTransform &rotation) {
183 const Vec3 kWorldUp{ 0.0F, 1.0F, 0.0F };
184 const Vec3 kNewUp = normalized(transform_vector(rotation, camera_transform.y_axis));
185 if (dot(kNewUp, kWorldUp) >= 0.0F) {
186 return orthonormalized_basis(rotation);
187 }
188
189 const Vec3 kNewRight = normalized(transform_vector(rotation, camera_transform.x_axis));
190 const Vec3 kNewForward = normalized(transform_vector(rotation, camera_transform.z_axis * -1.0F));
191 Vec3 clamped_up = kNewUp - kWorldUp * dot(kNewUp, kWorldUp);
192 clamped_up = length(clamped_up) <= 0.000001F ? Vec3{ 1.0F, 0.0F, 0.0F } : normalized(clamped_up);
193
194 const float kCosAngle = std::clamp(dot(clamped_up, kNewUp), -1.0F, 1.0F);
195 const float kCorrectionAngle = std::acos(kCosAngle);
196 const float kCorrectionSign = dot(kNewForward, kWorldUp) > 0.0F ? -1.0F : 1.0F;
197 const CameraTransform kCorrection = rotation_basis(kNewRight, kCorrectionAngle * kCorrectionSign);
198 return orthonormalized_basis(basis_multiply(kCorrection, rotation));
199}

202 const Vec3 kForward = normalized(camera_transform.z_axis * -1.0F);
203 return length(kForward) <= 0.000001F ? Vec3{ 0.0F, 0.0F, -1.0F } : kForward;

213 return length(kUp) <= 0.000001F ? Vec3{ 0.0F, 1.0F, 0.0F } : kUp;

216[[nodiscard]] float viewport_distance_to_pivot(const CameraTransform &camera_transform, Vec3 pivot) {
217 const Vec3 kToPivot = pivot - camera_transform.origin;
218 const float kForwardDistance = dot(kToPivot, viewport_camera_forward(camera_transform));
219 if (kForwardDistance > kMinCameraDistance) {
220 return kForwardDistance;
221 }
222
223 return std::max(kMinCameraDistance, length(kToPivot));
224}

226[[nodiscard]] CameraTransform transform_from_pose(const ViewportCameraPose &pose) {
227 const CameraBasis kBasis = make_camera_basis(pose.target - pose.eye, pose.up);
228 Vec3 right = normalized(cross(kBasis.forward, kBasis.up));
229 if (length(right) <= 0.000001F) {
230 right = { 1.0F, 0.0F, 0.0F };
231 }
232
233 const Vec3 kUp = normalized(cross(right, kBasis.forward));
234 return {
235 right,
236 length(kUp) <= 0.000001F ? kBasis.up : kUp,
237 kBasis.forward * -1.0F,
238 pose.eye,
239 };
240}

242void apply_navigation_transform(
243 ViewportCameraState &camera,
244 const CameraTransform &transform,
245 Vec3 pivot,
246 bool has_camera_size = false,
247 float camera_size = 0.0F,
248 bool preserve_view_distance = false) {
249 const Vec3 kForward = viewport_camera_forward(transform);
250 const CameraBasis kBasis = make_camera_basis(kForward, transform.y_axis, camera.forward);
251 const float kRequestedDistance = preserve_view_distance
252 ? camera.distance
253 : viewport_distance_to_pivot(transform, pivot);
254 const float kViewDistance = std::clamp(kRequestedDistance, kMinCameraDistance, kMaxCameraDistance);
255
256 camera.pivot = pivot;
257 camera.forward = kBasis.forward;
258 camera.up = kBasis.up;
259 camera.target = transform.origin + camera.forward * kViewDistance;
260 camera.distance = kViewDistance;
261 if (has_camera_size) {
262 camera.orthographic_size = std::max(0.0001F, camera_size);
263 }
264 sync_angles_for_forward(camera, camera.forward);
265}
266
267void apply_orbit_transform(ViewportCameraState &camera, const CameraTransform &transform, Vec3 pivot) {
268 Vec3 forward = normalized(pivot - transform.origin);
269 if (length(forward) <= 0.000001F) {
270 forward = viewport_camera_forward(transform);
271 }
272
273 const float kRadius = std::clamp(length(pivot - transform.origin), kMinCameraDistance, kMaxCameraDistance);
274 const CameraBasis kBasis = make_camera_basis(forward, transform.y_axis, camera.forward);
275
276 camera.pivot = pivot;
277 camera.forward = kBasis.forward;
278 camera.up = kBasis.up;
279 camera.target = pivot;
280 camera.distance = kRadius;
281 sync_angles_for_forward(camera, camera.forward);
282}

284[[nodiscard]] float capped_orthographic_zoom_size(float current_size, float factor, float steps) {
285 const float kSafeCurrentSize = std::max(0.0001F, current_size);
286 const float kRequestedSize = kSafeCurrentSize / factor;
287 if (steps < 0.0F) {
288 if (kSafeCurrentSize >= kViewportGridDefaultPlaneSize) {
289 return kSafeCurrentSize;
290 }
291 return std::clamp(kRequestedSize, 0.0001F, kViewportGridDefaultPlaneSize);
292 }
293
294 return std::max(0.0001F, kRequestedSize);
295}

369bool ViewportCameraController::begin_navigation(int camera_index, NavigationMode mode, ViewportPoint position) {
370 set_active_camera_index(camera_index);
371 ViewportCameraState &camera = active_camera();
372 if (mode == NavigationMode::Orbit && camera.locked_orthographic_view) {
373 return false;
374 }
375
376 if (mode == NavigationMode::Orbit) {
377 camera.pivot = camera.target;
378 }
379
380 navigation_mode_ = mode;
381 last_navigation_mouse_ = position;
382
383 if (mode == NavigationMode::Fly) {
384 camera.projection = CameraProjection::Perspective;
385 camera.locked_orthographic_view = false;
386 camera.pivot = camera.target;
387 }
388
389 return true;
390}

392void ViewportCameraController::update_navigation(ViewportPoint position, ViewportPixelSize viewport_size) {
393 if (navigation_mode_ == NavigationMode::None) {
394 return;
395 }
396
397 ViewportCameraState &camera = active_camera();
398 const ViewportPoint kDelta{ position.x - last_navigation_mouse_.x, position.y - last_navigation_mouse_.y };
399 last_navigation_mouse_ = position;
400 const CameraTransform kCameraTransform = transform_from_pose(camera_pose(active_camera_index_));
401
402 if (navigation_mode_ == NavigationMode::Orbit) {
403 if (camera.locked_orthographic_view) {
404 return;
405 }
406
407 camera.projection = CameraProjection::Perspective;
408
409 const float kYaw = static_cast<float>(kDelta.x) * kOrbitRadiansPerPixel;
410 const float kPitch = -static_cast<float>(kDelta.y) * kOrbitRadiansPerPixel;
411 const CameraTransform kYawRotation = rotation_basis({ 0.0F, 1.0F, 0.0F }, kYaw);
412 Vec3 right = normalized(transform_vector(kYawRotation, viewport_camera_right(kCameraTransform)));
413 if (length(right) <= 0.000001F) {
414 right = { 1.0F, 0.0F, 0.0F };
415 }
416 const CameraTransform kPitchRotation = rotation_basis(right, kPitch);
417 const CameraTransform kRotation = clamped_rotation_to_upright(
418 kCameraTransform,
419 basis_multiply(kPitchRotation, kYawRotation));
420
421 CameraTransform result_transform = kCameraTransform;
422 result_transform.origin = camera.pivot + transform_vector(kRotation, kCameraTransform.origin - camera.pivot);
423 const CameraTransform kRotatedBasis = rotated_camera_basis(kCameraTransform, kRotation);
424 result_transform.x_axis = kRotatedBasis.x_axis;
425 result_transform.y_axis = kRotatedBasis.y_axis;
426 result_transform.z_axis = kRotatedBasis.z_axis;
427 apply_orbit_transform(camera, result_transform, camera.pivot);
428 } else if (navigation_mode_ == NavigationMode::Pan) {
429 const ViewportPixelSize kSafeSize = safe_viewport_size(viewport_size);
430 const float kViewportHeight = static_cast<float>(kSafeSize.height);
431 const float kVisibleHeight = camera.projection == CameraProjection::Orthographic
432 ? std::max(0.0001F, camera.orthographic_size)
433 : std::max(0.0001F, 2.0F * viewport_distance_to_pivot(kCameraTransform, camera.pivot) * std::tan(camera.fov_degrees * 3.14159265359F / 180.0F * 0.5F));
434 const float kWorldPerPixel = kVisibleHeight / kViewportHeight;
435 const Vec3 kMove = viewport_camera_right(kCameraTransform) * (static_cast<float>(kDelta.x) * kWorldPerPixel) + viewport_camera_up(kCameraTransform) * (static_cast<float>(kDelta.y) * kWorldPerPixel);
436 CameraTransform result_transform = kCameraTransform;
437 result_transform.origin = result_transform.origin + kMove;
438 apply_navigation_transform(camera, result_transform, camera.pivot + kMove);
439 } else if (navigation_mode_ == NavigationMode::Fly) {
440 const float kYaw = static_cast<float>(kDelta.x) * kFlyLookRadiansPerPixel;
441 const float kPitch = -static_cast<float>(kDelta.y) * kFlyLookRadiansPerPixel;
442 const CameraTransform kYawRotation = rotation_basis({ 0.0F, 1.0F, 0.0F }, kYaw);
443 Vec3 right = normalized(transform_vector(kYawRotation, viewport_camera_right(kCameraTransform)));
444 if (length(right) <= 0.000001F) {
445 right = { 1.0F, 0.0F, 0.0F };
446 }
447 const CameraTransform kPitchRotation = rotation_basis(right, kPitch);
448 const CameraTransform kRotation = clamped_rotation_to_upright(
449 kCameraTransform,
450 basis_multiply(kPitchRotation, kYawRotation));
451 const CameraTransform kRotatedBasis = rotated_camera_basis(kCameraTransform, kRotation);
452
453 CameraTransform result_transform = kCameraTransform;
454 result_transform.x_axis = kRotatedBasis.x_axis;
455 result_transform.y_axis = kRotatedBasis.y_axis;
456 result_transform.z_axis = kRotatedBasis.z_axis;
457 apply_navigation_transform(camera, result_transform, camera.pivot, false, 0.0F, true);
458 camera.pivot = camera.target;
459 camera.projection = CameraProjection::Perspective;
460 camera.locked_orthographic_view = false;
461 }
462}

473void ViewportCameraController::apply_wheel_zoom(
474 int camera_index,
475 float wheel_steps,
476 ViewportPixelSize viewport_size) {
477 (void)viewport_size;
478 set_active_camera_index(camera_index);
479
480 ViewportCameraState &camera = active_camera();
481 const float kZoomFactor = std::pow(kNavigationZoomStepFactor, wheel_steps);
482 const CameraTransform kCameraTransform = transform_from_pose(camera_pose(active_camera_index_));
483 if (camera.projection == CameraProjection::Orthographic) {
484 const float kNewSize = capped_orthographic_zoom_size(camera.orthographic_size, kZoomFactor, wheel_steps);
485 apply_navigation_transform(camera, kCameraTransform, camera.pivot, true, kNewSize);
486 } else {
487 const float kDistance = viewport_distance_to_pivot(kCameraTransform, camera.pivot);
488 const float kNewDistance = std::max(kMinCameraDistance, kDistance / kZoomFactor);
489 CameraTransform result_transform = kCameraTransform;
490 result_transform.origin = result_transform.origin + viewport_camera_forward(kCameraTransform) * (kDistance - kNewDistance);
491 apply_navigation_transform(camera, result_transform, camera.pivot);
492 }
493}

523void ViewportCameraController::tick_fly_navigation(float delta_seconds) {
524 if (navigation_mode_ != NavigationMode::Fly || delta_seconds <= 0.0F) {
525 return;
526 }
527
528 ViewportCameraState &camera = active_camera();
529 Vec3 local_direction{};
530 if (fly_forward_) {
531 local_direction.z += 1.0F;
532 }
533 if (fly_backward_) {
534 local_direction.z -= 1.0F;
535 }
536 if (fly_right_) {
537 local_direction.x -= 1.0F;
538 }
539 if (fly_left_) {
540 local_direction.x += 1.0F;
541 }
542
543 if (length(local_direction) <= 0.000001F) {
544 return;
545 }
546
547 const CameraTransform kCameraTransform = transform_from_pose(camera_pose(active_camera_index_));
548 Vec3 world_direction = viewport_camera_right(kCameraTransform) * local_direction.x + viewport_camera_up(kCameraTransform) * local_direction.y + viewport_camera_forward(kCameraTransform) * local_direction.z;
549 world_direction = normalized(world_direction);
550 if (length(world_direction) <= 0.000001F) {
551 return;
552 }
553
554 const Vec3 kMove = world_direction * (fly_speed_ * std::clamp(delta_seconds, 0.0F, 0.25F));
555 CameraTransform result_transform = kCameraTransform;
556 result_transform.origin = result_transform.origin + kMove;
557 apply_navigation_transform(camera, result_transform, camera.pivot + kMove, false, 0.0F, true);
558 camera.pivot = camera.target;
559}

561ViewportCameraPose ViewportCameraController::camera_pose(int camera_index) const {
562 const ViewportCameraState &camera = this->camera(camera_index);
563 Vec3 forward = camera.projection == CameraProjection::Orthographic && camera.locked_orthographic_view
564 ? normalized(camera.locked_forward)
565 : normalized(camera.forward);
566 if (length(forward) <= 0.000001F) {
567 forward = forward_from_angles(camera.yaw_radians, camera.pitch_radians);
568 }
569 if (length(forward) <= 0.000001F) {
570 forward = { 0.0F, 0.0F, -1.0F };
571 }
572
573 Vec3 up_candidate = camera.projection == CameraProjection::Orthographic && camera.locked_orthographic_view
574 ? normalized(camera.locked_up)
575 : normalized(camera.up);
576 if (length(up_candidate) <= 0.000001F || std::abs(dot(up_candidate, forward)) > 0.999F) {
577 up_candidate = fallback_up_for_forward(forward);
578 }
579
580 const CameraBasis kBasis = make_camera_basis(forward, up_candidate);
581
582 const Vec3 kEye = camera.target - kBasis.forward * std::max(kMinCameraDistance, camera.distance);
583 return { kEye, camera.target, kBasis.up, kBasis.forward };
584}
