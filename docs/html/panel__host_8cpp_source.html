26QDockWidget &PanelHost::add_panel(PanelSpec spec, QWidget *panel_widget) {
27 QUADER_ASSERT(panel_widget != nullptr);
28 QUADER_ASSERT(docks_.find(spec.id) == docks_.end());
29
30 auto *dock = new QDockWidget(spec.title, &main_window_);
31 dock->setObjectName(panel_object_name(spec.id));
32 dock->setAllowedAreas(spec.allowed_areas);
33 dock->setWidget(panel_widget);
34
35 main_window_.addDockWidget(spec.default_area, dock);
36 dock->setVisible(spec.default_visible);
37 docks_.emplace(spec.id, dock);
38 specs_.emplace(spec.id, std::move(spec));
39 return *dock;
40}

60void PanelHost::bind_panel_visibility_action(PanelId panel, ActionRegistry &actions, ActionId action_id) {
61 auto *panel_dock = dock(panel);
62 QUADER_ASSERT(panel_dock != nullptr);
63
64 QAction &action = actions.action(action_id);
65 action.setCheckable(true);
66 {
67 const QSignalBlocker kBlocker(&action);
68 action.setChecked(!panel_dock->isHidden());
69 }
70
71 connect(&action, &QAction::toggled, panel_dock, [this, panel](bool checked) {
72 if (checked) {
73 show_panel(panel);
74 } else {
75 hide_panel(panel);
76 }
77 });
78
79 connect(panel_dock, &QDockWidget::visibilityChanged, &action, [&action](bool visible) {
80 const QSignalBlocker kBlocker(&action);
81 action.setChecked(visible);
82 });
83}

93void PanelHost::restore_workspace(SettingsService &settings) {
94 for (const auto &[panel, panel_dock] : docks_) {
95 if (panel_dock == nullptr) {
96 continue;
97 }
98
99 const auto kSpec = specs_.find(panel);
100 const bool kFallbackVisible = kSpec == specs_.end() ? !panel_dock->isHidden() : kSpec->second.default_visible;
101 const bool kVisible = settings.panel_visible(panel, kFallbackVisible);
102 if (kVisible) {
103 panel_dock->show();
104 } else {
105 panel_dock->hide();
106 }
107 }
108}

110void PanelHost::reset_to_default_layout() {
111 auto *scene = dock(PanelId::Scene);
112 auto *properties = dock(PanelId::Properties);
113 auto *diagnostics = dock(PanelId::Diagnostics);
114
115 if (scene != nullptr) {
116 main_window_.addDockWidget(Qt::RightDockWidgetArea, scene);
117 scene->show();
118 scene->raise();
119 }
120
121 if (properties != nullptr) {
122 main_window_.addDockWidget(Qt::RightDockWidgetArea, properties);
123 if (scene != nullptr) {
124 main_window_.splitDockWidget(scene, properties, Qt::Vertical);
125 }
126 properties->show();
127 }
128
129 if (diagnostics != nullptr) {
130 main_window_.addDockWidget(Qt::BottomDockWidgetArea, diagnostics);
131 diagnostics->hide();
132 }
133}
