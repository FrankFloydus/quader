44void ActionStateUpdater::refresh_from_snapshot(const EditorStateSnapshot &snapshot) {
45 set_enabled(actions_, ActionId::NewScene, snapshot.new_scene_available);
46 set_enabled(actions_, ActionId::OpenScene, snapshot.document_services_available && snapshot.import_available);
47 set_enabled(actions_,
48 ActionId::SaveScene,
49 snapshot.has_active_document && snapshot.document_dirty && snapshot.export_available);
50 set_enabled(actions_, ActionId::SaveSceneAs, snapshot.has_active_document && snapshot.export_available);
51
52 set_enabled(actions_, ActionId::Exit, true);
53
54 set_enabled(actions_, ActionId::Undo, snapshot.can_undo);
55 set_enabled(actions_, ActionId::Redo, snapshot.can_redo);
56 set_text(actions_, ActionId::Undo, snapshot.undo_text.isEmpty() ? QStringLiteral("&Undo") : snapshot.undo_text);
57 set_text(actions_, ActionId::Redo, snapshot.redo_text.isEmpty() ? QStringLiteral("&Redo") : snapshot.redo_text);
58
59 set_enabled(actions_, ActionId::DuplicateSelection, snapshot.can_duplicate_selection);
60 set_enabled(actions_, ActionId::DeleteSelection, snapshot.can_delete_selection);
61
62 const bool kToolsEnabled = snapshot.tools_available;
63 set_enabled(actions_, ActionId::SelectTool, kToolsEnabled);
64 set_enabled(actions_, ActionId::MoveTool, kToolsEnabled);
65 set_enabled(actions_, ActionId::RotateTool, kToolsEnabled);
66 set_enabled(actions_, ActionId::ScaleTool, kToolsEnabled);
67 set_enabled(actions_, ActionId::BoxTool, kToolsEnabled);
68
69 set_checked(actions_, ActionId::SelectTool, kToolsEnabled && snapshot.active_tool == ActionId::SelectTool);
70 set_checked(actions_, ActionId::MoveTool, kToolsEnabled && snapshot.active_tool == ActionId::MoveTool);
71 set_checked(actions_, ActionId::RotateTool, kToolsEnabled && snapshot.active_tool == ActionId::RotateTool);
72 set_checked(actions_, ActionId::ScaleTool, kToolsEnabled && snapshot.active_tool == ActionId::ScaleTool);
73 set_checked(actions_, ActionId::BoxTool, kToolsEnabled && snapshot.active_tool == ActionId::BoxTool);
74
75 const bool kCreationEnabled = snapshot.has_active_document && snapshot.creation_available;
76 set_enabled(actions_, ActionId::CreateCube, kCreationEnabled);
77 set_enabled(actions_, ActionId::CreateLight, kCreationEnabled);
78 set_enabled(actions_, ActionId::CreateCamera, kCreationEnabled);
79
80 if (snapshot.viewport_state_known) {
81 set_enabled(actions_, ActionId::ViewPerspective, snapshot.viewport_available);
82 set_enabled(actions_, ActionId::ViewShaded, snapshot.viewport_available);
83 set_enabled(actions_, ActionId::ToggleQuadViewports, snapshot.viewport_available);
84 set_checked(actions_, ActionId::ToggleQuadViewports, snapshot.quad_viewports_enabled);
85 set_enabled(actions_, ActionId::FocusViewport, snapshot.viewport_available);
86 }
87
88 set_enabled(actions_, ActionId::ResetWorkspaceLayout, true);
89 set_enabled(actions_, ActionId::ShowScenePanel, true);
90 set_enabled(actions_, ActionId::ShowPropertiesPanel, true);
91 set_enabled(actions_, ActionId::ShowDiagnosticsPanel, true);
92}
