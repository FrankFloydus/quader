125RenderMeshHandle GpuMeshCache::create_unit_box(RendererStatus &status) {
126 GpuMeshResource resource;
127 resource.vertex_layout
128 .begin()
129 .add(bgfx::Attrib::Position, 3, bgfx::AttribType::Float)
130 .add(bgfx::Attrib::Normal, 3, bgfx::AttribType::Float)
131 .end();
132 resource.vertex_buffer.reset(bgfx::createVertexBuffer(
133 bgfx::makeRef(kCubeVertices, sizeof(kCubeVertices)),
134 resource.vertex_layout));
135 resource.index_buffer.reset(bgfx::createIndexBuffer(bgfx::makeRef(kCubeIndices, sizeof(kCubeIndices))));
136
137 if (!resource.valid()) {
138 push_mesh_creation_error(status);
139 return {};
140 }
141
142 const RenderMeshHandle kHandle = meshes_.create(std::move(resource));
143 ++upload_stats_.mesh_create_count;
144 upload_stats_.uploaded_vertex_bytes += sizeof(kCubeVertices);
145 upload_stats_.uploaded_index_bytes += sizeof(kCubeIndices);
146 return kHandle;
147}

149bool GpuMeshCache::upload_mesh(const RenderMeshUploadDesc &desc, RendererStatus &status) {
150 auto validated = validate_render_mesh_upload_desc(desc);
151 if (!validated) {
152 push_mesh_upload_error(status, std::move(validated).error());
153 return false;
154 }
155
156 GpuMeshResource resource;
157 resource.vertex_layout
158 .begin()
159 .add(bgfx::Attrib::Position, 3, bgfx::AttribType::Float)
160 .add(bgfx::Attrib::Normal, 3, bgfx::AttribType::Float)
161 .end();
162 resource.vertex_buffer.reset(bgfx::createVertexBuffer(
163 bgfx::copy(
164 desc.position_normal_interleaved.data(),
165 static_cast<std::uint32_t>(desc.position_normal_interleaved.size_bytes())),
166 resource.vertex_layout));
167 resource.index_buffer.reset(bgfx::createIndexBuffer(
168 bgfx::copy(desc.indices.data(), static_cast<std::uint32_t>(desc.indices.size_bytes())),
169 BGFX_BUFFER_INDEX32));
170
171 if (!resource.valid()) {
172 push_mesh_upload_creation_error(status, desc.handle);
173 return false;
174 }
175
176 const bool kUpdated = get(desc.handle) != nullptr;
177 if (!meshes_.upsert(desc.handle, std::move(resource))) {
178 push_mesh_upload_creation_error(status, desc.handle);
179 return false;
180 }
181
182 const RenderMeshUploadRecord kRecord = validated.value();
183 if (kUpdated) {
184 ++upload_stats_.mesh_update_count;
185 } else {
186 ++upload_stats_.mesh_create_count;
187 }
188 upload_stats_.uploaded_vertex_bytes += kRecord.vertex_bytes;
189 upload_stats_.uploaded_index_bytes += kRecord.index_bytes;
190 return true;
191}
