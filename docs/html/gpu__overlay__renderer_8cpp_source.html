150std::uint32_t GpuOverlayRenderer::submit_grid(
151 bgfx::ViewId view_id,
152 const RenderView &view,
153 const PreparedGridOverlayCommand &prepared,
154 bgfx::ProgramHandle program) const noexcept {
155 if (!impl_->ready || !bgfx::isValid(program)) {
156 return 0;
157 }
158
159 const GridOverlayCommand &grid = prepared.grid;
160 float transform[16];
161 bx::mtxSRT(
162 transform,
163 grid.plane_width,
164 1.0F,
165 grid.plane_height,
166 grid.rotation_x,
167 grid.rotation_y,
168 grid.rotation_z,
169 grid.origin.x,
170 grid.origin.y,
171 grid.origin.z);
172
173 const float kPlaneSize[4] = { grid.plane_width, grid.plane_height, 0.0F, 0.0F };
174 const float kCameraPosition[4] = { view.camera.eye.x, view.camera.eye.y, view.camera.eye.z, 0.0F };
175 const float kGridUAxis[4] = { grid.u_axis.x, grid.u_axis.y, grid.u_axis.z, 0.0F };
176 const float kGridVAxis[4] = { grid.v_axis.x, grid.v_axis.y, grid.v_axis.z, 0.0F };
177 const float kParams0[4] = {
178 grid.minor_spacing_m,
179 std::max(grid.major_spacing_m, grid.minor_spacing_m),
180 grid.minor_line_scale,
181 grid.major_line_scale,
182 };
183 const float kParams1[4] = {
184 grid.axis_line_scale,
185 grid.fade_start_m,
186 grid.fade_end_m,
187 std::max(grid.plane_width, grid.plane_height) * 0.5F,
188 };
189 const float kParams2[4] = {
190 std::max(0.0001F, grid.orthographic_height_m),
191 std::max(1.0F, grid.viewport_height_px),
192 grid.minor_spacing_m,
193 grid.edge_softness_m,
194 };
195
196 bgfx::setTransform(transform);
197 bgfx::setUniform(impl_->grid_plane_size_uniform.get(), kPlaneSize);
198 bgfx::setUniform(impl_->grid_color_uniform.get(), prepared.minor_color_linear_sdr.data());
199 bgfx::setUniform(impl_->major_grid_color_uniform.get(), prepared.major_color_linear_sdr.data());
200 bgfx::setUniform(impl_->origin_u_color_uniform.get(), prepared.axis_u_color_linear_sdr.data());
201 bgfx::setUniform(impl_->origin_v_color_uniform.get(), prepared.axis_v_color_linear_sdr.data());
202 bgfx::setUniform(impl_->camera_position_uniform.get(), kCameraPosition);
203 bgfx::setUniform(impl_->grid_u_axis_uniform.get(), kGridUAxis);
204 bgfx::setUniform(impl_->grid_v_axis_uniform.get(), kGridVAxis);
205 bgfx::setUniform(impl_->grid_params0_uniform.get(), kParams0);
206 bgfx::setUniform(impl_->grid_params1_uniform.get(), kParams1);
207 bgfx::setUniform(impl_->grid_params2_uniform.get(), kParams2);
208 bgfx::setVertexBuffer(0, impl_->grid_vertex_buffer.get());
209 bgfx::setIndexBuffer(impl_->grid_index_buffer.get());
210 bgfx::setState(BGFX_STATE_WRITE_RGB | BGFX_STATE_WRITE_A | BGFX_STATE_DEPTH_TEST_LESS | BGFX_STATE_MSAA | BGFX_STATE_BLEND_FUNC(BGFX_STATE_BLEND_ONE, BGFX_STATE_BLEND_INV_SRC_ALPHA));
211 bgfx::submit(view_id, program);
212 return 1;
213}

215std::uint32_t GpuOverlayRenderer::submit_lines(
216 bgfx::ViewId view_id,
217 const PreparedLineOverlayCommand &lines,
218 bgfx::ProgramHandle program) const noexcept {
219 if (!impl_->ready || !bgfx::isValid(program) || lines.segments.empty()) {
220 return 0;
221 }
222
223 const std::uint32_t kVertexCount = static_cast<std::uint32_t>(std::min<std::size_t>(
224 lines.segments.size() * 2U,
225 std::numeric_limits<std::uint32_t>::max()));
226 if (kVertexCount < 2U || bgfx::getAvailTransientVertexBuffer(kVertexCount, impl_->line_vertex_layout) < kVertexCount) {
227 return 0;
228 }
229
230 bgfx::TransientVertexBuffer vertices{};
231 bgfx::allocTransientVertexBuffer(&vertices, kVertexCount, impl_->line_vertex_layout);
232 auto *data = reinterpret_cast<LineVertex *>(vertices.data);
233 std::uint32_t vertex_index = 0;
234 for (const LineOverlaySegment &segment : lines.segments) {
235 if (vertex_index + 1U >= kVertexCount) {
236 break;
237 }
238 data[vertex_index++] = LineVertex{ segment.start.x, segment.start.y, segment.start.z };
239 data[vertex_index++] = LineVertex{ segment.end.x, segment.end.y, segment.end.z };
240 }
241
242 float transform[16];
243 bx::mtxIdentity(transform);
244 std::uint64_t state = BGFX_STATE_WRITE_RGB | BGFX_STATE_WRITE_A | BGFX_STATE_PT_LINES | BGFX_STATE_MSAA | BGFX_STATE_BLEND_FUNC(BGFX_STATE_BLEND_SRC_ALPHA, BGFX_STATE_BLEND_INV_SRC_ALPHA);
245 if (lines.command.depth_mode == OverlayDepthMode::DepthTested) {
246 state |= BGFX_STATE_DEPTH_TEST_LESS;
247 } else if (lines.command.depth_mode == OverlayDepthMode::XRay) {
248 state |= BGFX_STATE_DEPTH_TEST_LEQUAL;
249 }
250
251 bgfx::setTransform(transform);
252 bgfx::setUniform(impl_->line_color_uniform.get(), lines.color_linear_sdr.data());
253 bgfx::setVertexBuffer(0, &vertices, 0, vertex_index);
254 bgfx::setState(state);
255 bgfx::submit(view_id, program);
256 return 1;
257}
