39TEST(Picking, RawIdRgba8EncodingRoundTripsWithoutColorManagement) {
40 const crimson::PackedPickingIdRgba8 kZero = crimson::encode_picking_id_rgba8(0);
41 expect_true(kZero.bytes == std::array<std::uint8_t, 4>{ 0, 0, 0, 0 }, "raw ID 0 encodes to no-hit bytes");
42 expect_true(crimson::decode_picking_id_rgba8(kZero) == 0, "raw ID 0 decodes as no-hit");
43
44 const crimson::PackedPickingIdRgba8 kEncoded = crimson::encode_picking_id_rgba8(0x12345678U);
45 expect_true(
46 kEncoded.bytes == std::array<std::uint8_t, 4>{ 0x78, 0x56, 0x34, 0x12 },
47 "RGBA8 fallback stores little-endian data bytes");
48 expect_true(crimson::decode_picking_id_rgba8(kEncoded) == 0x12345678U, "RGBA8 fallback decodes raw ID");
49
50 const std::array<float, 4> kUnorm = crimson::picking_id_rgba8_to_unorm(kEncoded);
51 expect_true(std::fabs(kUnorm[0] - (120.0F / 255.0F)) < 0.0001F, "fallback byte becomes direct UNORM red");
52 expect_true(std::fabs(kUnorm[3] - (18.0F / 255.0F)) < 0.0001F, "fallback byte becomes direct UNORM alpha");
53}

55TEST(Picking, AllocatorScopesIdsToOneRequest) {
56 crimson::PickingIdAllocator allocator;
57 allocator.begin_request(11);
58 const std::uint32_t kFirstId = allocator.allocate(crimson::PickingPayload{
59 .object_id = 100,
60 .submesh_index = 3,
61 .element_kind = crimson::PickingElementKind::Object,
62 });
63
64 const std::optional<crimson::PickingPayload> kFirstPayload = allocator.resolve(kFirstId);
65 expect_true(kFirstId == 1, "request allocator starts raw IDs at 1");
66 expect_true(kFirstPayload.has_value() && kFirstPayload->object_id == 100, "first request resolves its payload");
67
68 allocator.begin_request(12);
69 expect_true(allocator.request_id() == 12, "allocator records the active request");
70 expect_true(!allocator.resolve(kFirstId).has_value(), "new request invalidates previous raw ID mapping");
71
72 const std::uint32_t kSecondId = allocator.allocate(crimson::PickingPayload{
73 .object_id = 200,
74 .submesh_index = 0,
75 .element_kind = crimson::PickingElementKind::Object,
76 });
77 const std::optional<crimson::PickingPayload> kSecondPayload = allocator.resolve(kSecondId);
78 expect_true(kSecondId == 1, "request-scoped allocator reuses raw IDs per request");
79 expect_true(kSecondPayload.has_value() && kSecondPayload->object_id == 200, "second request resolves new payload");
80}

92TEST(Picking, PickingPassStaysOutOfColorManagedTargets) {
93 const crimson::RenderGraph kGraph = crimson::make_v1_correctness_render_graph(
94 crimson::ViewportExtent{ 640, 480, 1.0F },
95 crimson::PickingIdTargetFormat::Rgba8Data);
96 const crimson::RenderPass *picking = find_pass(kGraph, "PickingPass");
97 expect_true(picking != nullptr, "picking pass exists in V1 graph");
98 if (picking == nullptr) {
99 return;
100 }
101
102 for (const crimson::ResourceUse &use : picking->resources) {
103 expect_true(!crimson::is_color_managed_frame_target(use.resource_name), "picking pass avoids color targets");
104 }
105
106 const crimson::RenderResourceRecord *target = kGraph.resources().find(crimson::kPickingIdTargetName);
107 expect_true(target != nullptr, "picking ID target exists");
108 expect_true(
109 target != nullptr && target->desc.format == crimson::RenderResourceFormat::Rgba8Unorm,
110 "RGBA8 fallback remains a data target");
111 expect_true(crimson::is_data_frame_target(crimson::kPickingIdTargetName), "picking ID target is data");
112}
