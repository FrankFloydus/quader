182 parameters.push_back(enum_param("alpha_source_channel", 3, 0, 3));
183

186 ui_schema.push_back(parameter_field("Alpha Source Channel", MaterialUiControl::Combo, "alpha_source_channel"));
187
188 return BaseShaderDefinition{
189 .id = BaseShaderId::AlphaCutoutPbr,
190 .name = "AlphaCutoutPbr",
191 .domain = RenderDomain::LitSurface,
192 .default_queue = RenderQueue::AlphaCutout,
193 .depth_mode = DepthMode::TestWrite,
194 .blend_mode = BlendMode::Off,
195 .cull_mode = CullMode::Back,
196 .shadow_mode = ShadowMode::AlphaTestedCastAndReceive,
197 .required_attributes = kPbrVertexAttributes,
198 .parameters = std::move(parameters),
199 .texture_slots = opaque_pbr_texture_slots(),
200 .ui_schema = std::move(ui_schema),

202 };

285const BaseShaderDefinition *BaseShaderRegistry::find(BaseShaderId id) const noexcept {
286 for (const BaseShaderDefinition &definition : definitions_) {
287 if (definition.id == id) {
288 return &definition;
289 }
290 }
291
292 return nullptr;
293}

303BaseShaderRegistry make_v1_base_shader_registry() {
304 std::vector<BaseShaderDefinition> definitions;
305 definitions.reserve(4);
306 definitions.push_back(make_opaque_pbr());
307 definitions.push_back(make_alpha_cutout_pbr());
308 definitions.push_back(make_transparent_pbr());
309 definitions.push_back(make_overlay_unlit());
310 return BaseShaderRegistry(std::move(definitions));
311}
