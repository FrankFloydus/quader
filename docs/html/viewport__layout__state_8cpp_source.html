64ViewportPaneArray ViewportLayoutState::panes_for(ViewportPixelSize size) const {
65 const int kWidth = safe_extent(size.width);
66 const int kHeight = safe_extent(size.height);
67
68 if (mode_ != ViewportLayoutMode::Quad) {
69 return { {
70 ViewportPane{ { 0, 0, kWidth, kHeight }, 0, QStringLiteral("Perspective") },
71 ViewportPane{},
72 ViewportPane{},
73 ViewportPane{},
74 } };
75 }
76
77 const int kGap = active_splitter_width(size);
78 const int kSplitX = clamped_split_start(kWidth, vertical_split_, kGap);
79 const int kSplitY = clamped_split_start(kHeight, horizontal_split_, kGap);
80 const int kLeftWidth = std::max(1, kSplitX);
81 const int kRightX = std::min(kWidth - 1, kSplitX + kGap);
82 const int kRightWidth = std::max(1, kWidth - kRightX);
83 const int kTopHeight = std::max(1, kSplitY);
84 const int kBottomY = std::min(kHeight - 1, kSplitY + kGap);
85 const int kBottomHeight = std::max(1, kHeight - kBottomY);
86
87 return { {
88 ViewportPane{ { 0, 0, kLeftWidth, kTopHeight }, 0, QStringLiteral("Perspective") },
89 ViewportPane{ { kRightX, 0, kRightWidth, kTopHeight }, 1, QStringLiteral("Top") },
90 ViewportPane{ { 0, kBottomY, kLeftWidth, kBottomHeight }, 2, QStringLiteral("Front") },
91 ViewportPane{ { kRightX, kBottomY, kRightWidth, kBottomHeight }, 3, QStringLiteral("Right") },
92 } };
93}

99int ViewportLayoutState::pane_at(ViewportPixelSize size, ViewportPoint point) const {
100 const ViewportPaneArray kPanes = panes_for(size);
101 for (int index = 0; index < pane_count(); ++index) {
102 const ViewportRect &rect = kPanes[static_cast<std::size_t>(index)].rect;
103 if (point.x >= rect.x && point.y >= rect.y && point.x < rect.x + rect.width && point.y < rect.y + rect.height) {
104 return kPanes[static_cast<std::size_t>(index)].camera_index;
105 }
106 }
107
108 return -1;
109}

111ViewportSplitHandle ViewportLayoutState::splitter_at(ViewportPixelSize size, ViewportPoint point) const {
112 if (mode_ != ViewportLayoutMode::Quad) {
113 return ViewportSplitHandle::None;
114 }
115
116 const int kGap = active_splitter_width(size);
117 const int kVerticalStart = clamped_split_start(safe_extent(size.width), vertical_split_, kGap);
118 const int kHorizontalStart = clamped_split_start(safe_extent(size.height), horizontal_split_, kGap);
119 const int kVerticalCenter = kVerticalStart + kGap / 2;
120 const int kHorizontalCenter = kHorizontalStart + kGap / 2;
121 const int kVerticalDistance = std::abs(static_cast<int>(std::round(point.x)) - kVerticalCenter);
122 const int kHorizontalDistance = std::abs(static_cast<int>(std::round(point.y)) - kHorizontalCenter);
123
124 if (kVerticalDistance <= kViewportSplitterHitTolerance && kVerticalDistance <= kHorizontalDistance) {
125 return ViewportSplitHandle::Vertical;
126 }
127 if (kHorizontalDistance <= kViewportSplitterHitTolerance) {
128 return ViewportSplitHandle::Horizontal;
129 }
130
131 return ViewportSplitHandle::None;
132}

134void ViewportLayoutState::set_split_from_position(
135 ViewportPixelSize size,
136 ViewportSplitHandle handle,
137 ViewportPoint point) noexcept {
138 const int kGap = active_splitter_width(size);
139
140 if (handle == ViewportSplitHandle::Vertical) {
141 const float kFraction = static_cast<float>(point.x) / static_cast<float>(safe_extent(size.width));
142 const int kStart = clamped_split_start(safe_extent(size.width), kFraction, kGap);
143 vertical_split_ = static_cast<float>(kStart) / static_cast<float>(std::max(1, safe_extent(size.width) - kGap));
144 } else if (handle == ViewportSplitHandle::Horizontal) {
145 const float kFraction = static_cast<float>(point.y) / static_cast<float>(safe_extent(size.height));
146 const int kStart = clamped_split_start(safe_extent(size.height), kFraction, kGap);
147 horizontal_split_ = static_cast<float>(kStart) / static_cast<float>(std::max(1, safe_extent(size.height) - kGap));
148 }
149}

163int ViewportLayoutState::clamped_split_start(int size, float fraction, int splitter_width) const noexcept {
164 const int kAvailable = std::max(1, size - splitter_width);
165 const int kMinExtent = std::min(kMinimumViewportExtent, std::max(1, kAvailable / 2));
166 const int kMaxStart = std::max(kMinExtent, kAvailable - kMinExtent);
167 const int kRequested = static_cast<int>(std::round(kAvailable * clamped_fraction(fraction)));
168 return std::clamp(kRequested, kMinExtent, kMaxStart);
169}
