57quader::math::Vec3 average_points(std::span<const quader::math::Vec3> points) {
58 quader::math::Vec3 average;
59 for (quader::math::Vec3 point : points) {
60 average = average + point;
61 }
62 return points.empty() ? average : average / static_cast<float>(points.size());
63}

65std::vector<quader::math::Vec3> face_points(
66 const quader::mesh::Polyhedron &mesh,
67 quader::mesh::FaceId face) {
68 auto vertices = quader::mesh::face_vertices(mesh, face);
69 EXPECT_TRUE(vertices);
70 std::vector<quader::math::Vec3> points;
71 if (!vertices) {
72 return points;
73 }
74
75 points.reserve(vertices.value().size());
76 for (const quader::mesh::VertexId kVertex : vertices.value()) {
77 auto position = mesh.vertex_position(kVertex);
78 EXPECT_TRUE(position);
79 if (position) {
80 points.push_back(position.value());
81 }
82 }
83 return points;
84}

86quader::math::Vec3 mesh_center(const quader::mesh::Polyhedron &mesh) {
87 std::vector<quader::math::Vec3> points;
88 points.reserve(mesh.vertex_count());
89 for (const quader::mesh::VertexId kVertex : mesh.vertex_ids()) {
90 auto position = mesh.vertex_position(kVertex);
91 EXPECT_TRUE(position);
92 if (position) {
93 points.push_back(position.value());
94 }
95 }
96 return average_points(points);
97}

99void expect_ground_box_faces_oriented_outward(const quader::mesh::Polyhedron &mesh) {
100 const quader::math::Vec3 kCenter = mesh_center(mesh);
101 int up_faces = 0;
102 int down_faces = 0;
103 for (const quader::mesh::FaceId kFace : mesh.face_ids()) {
104 const std::vector<quader::math::Vec3> kPoints = face_points(mesh, kFace);
105 ASSERT_GE(kPoints.size(), 3U);
106 const quader::math::Vec3 kNormal = quader::geometry::polygon_normal(kPoints);
107 const quader::math::Vec3 kFaceCenter = average_points(kPoints);
108 EXPECT_GT(quader::math::dot(kNormal, kFaceCenter - kCenter), 0.0F);
109
110 const quader::math::Vec3 kUnitNormal = quader::math::normalized(kNormal);
111 if (quader::math::dot(kUnitNormal, { 0.0F, 1.0F, 0.0F }) > 0.999F) {
112 ++up_faces;
113 }
114 if (quader::math::dot(kUnitNormal, { 0.0F, -1.0F, 0.0F }) > 0.999F) {
115 ++down_faces;
116 }
117 }
118 EXPECT_EQ(up_faces, 1);
119 EXPECT_EQ(down_faces, 1);
120}

172class RecordingTool final : public quader::tools::ITool {
173public:
174 explicit RecordingTool(quader::tools::ToolId tool_id,
175 std::optional<quader::document::ObjectId> command_object = std::nullopt) : tool_id_(tool_id),
176 command_object_(command_object) {
177 }
178
179 [[nodiscard]] quader::tools::ToolId id() const noexcept override {
180 return tool_id_;
181 }
182
183 void activate(quader::tools::ToolContext &context) override {
184 (void)context;
185 ++activations;
186 }
187
188 void deactivate(quader::tools::ToolContext &context) override {
189 (void)context;
190 ++deactivations;
191 preview_data.clear();
192 }
193
194 void cancel(quader::tools::ToolContext &context) override {
195 (void)context;
196 ++cancellations;
197 preview_data.clear();
198 }
199
200 [[nodiscard]] bool on_pointer_event(const quader::tools::PointerEvent &event,
201 quader::tools::ToolContext &context) override {
202 (void)context;
203 ++pointer_events;
204 last_pointer = event;
205 preview_data.clear();
206 preview_data.active = true;
207 preview_data.status_text = "pointer";
208 preview_data.points.push_back(event.position);
209 return true;
210 }
211
212 [[nodiscard]] bool on_key_event(const quader::tools::KeyEvent &event,
213 quader::tools::ToolContext &context) override {
214 ++key_events;
215 last_key = event;
216 preview_data.clear();
217 preview_data.active = true;
218 preview_data.status_text = "key";
219
220 if (command_object_.has_value() && event.pressed && !event.auto_repeat) {
221 auto result = context.execute_command(
222 std::make_unique<quader::commands::RenameObjectCommand>(*command_object_,
223 "Renamed By Tool"));
224 last_command_status = result.status;
225 return result.is_applied();
226 }
227
228 return true;
229 }
230
231 [[nodiscard]] quader::tools::ToolPreview preview() const override {
232 return preview_data;
233 }
234
235 int activations = 0;
236 int deactivations = 0;
237 int cancellations = 0;
238 int pointer_events = 0;
239 int key_events = 0;
240 std::optional<quader::tools::PointerEvent> last_pointer;
241 std::optional<quader::tools::KeyEvent> last_key;
242 std::optional<quader::commands::CommandStatus> last_command_status;
243 quader::tools::ToolPreview preview_data;
244
245private:
246 quader::tools::ToolId tool_id_;
247 std::optional<quader::document::ObjectId> command_object_;
248};
