123
124 EXPECT_EQ(registry.size(), kStandardActions.size());
125 for (const quader::ui::ActionId kId : kStandardActions) {
126 EXPECT_TRUE(registry.contains(kId));
127 EXPECT_NE(quader::ui::action_id_name(kId), std::string_view("Unknown"));

129
130 QAction &first_lookup = registry.action(quader::ui::ActionId::ToggleQuadViewports);
131 QAction &second_lookup = registry.action(quader::ui::ActionId::ToggleQuadViewports);

241TEST(UiShell, NotificationServiceEmitsTypedMessages) {
242 quader::ui::NotificationService notifications;
243 quader::ui::NotificationMessage last_message;
244 int posted_count = 0;
245
246 QObject::connect(&notifications,
247 &quader::ui::NotificationService::notification_posted,
248 &notifications,
249 [&last_message, &posted_count](const quader::ui::NotificationMessage &message) {
250 last_message = message;
251 ++posted_count;
252 });
253
254 notifications.show_status(QStringLiteral("Ready"), 1200);
255 EXPECT_EQ(posted_count, 1);
256 EXPECT_EQ(last_message.severity, quader::ui::NotificationSeverity::Status);
257 EXPECT_EQ(last_message.summary, QStringLiteral("Ready"));
258 EXPECT_EQ(last_message.timeout_ms, 1200);
259
260 notifications.show_error(quader::ui::NotificationMessage{
261 quader::ui::NotificationSeverity::Status,
262 QStringLiteral("Renderer failed"),
263 QStringLiteral("No adapter"),
264 -1,
265 });
266 EXPECT_EQ(posted_count, 2);
267 EXPECT_EQ(last_message.severity, quader::ui::NotificationSeverity::Error);
268 EXPECT_EQ(last_message.timeout_ms, 0);
269}

290TEST(UiShell, PanelHostCreatesStableDocksAndVisibilityActions) {
291 UiFixture fixture;
292 QMainWindow host_window;
293 quader::ui::PanelHost panel_host(host_window, quader::ui::PanelContext{ fixture.context }, &host_window);
294
295 auto *scene_panel = new QWidget;
296 QDockWidget &scene_dock = panel_host.add_panel(
297 quader::ui::PanelSpec{
298 quader::ui::PanelId::Scene,
299 QStringLiteral("Scene"),
300 Qt::RightDockWidgetArea,
301 Qt::LeftDockWidgetArea | Qt::RightDockWidgetArea,
302 },
303 scene_panel);
304 auto *properties_panel = new QWidget;
305 QDockWidget &properties_dock = panel_host.add_panel(
306 quader::ui::PanelSpec{
307 quader::ui::PanelId::Properties,
308 QStringLiteral("Properties"),
309 Qt::RightDockWidgetArea,
310 Qt::LeftDockWidgetArea | Qt::RightDockWidgetArea,
311 },
312 properties_panel);
313 auto *diagnostics_panel = new QWidget;
314 QDockWidget &diagnostics_dock = panel_host.add_panel(
315 quader::ui::PanelSpec{
316 quader::ui::PanelId::Diagnostics,
317 QStringLiteral("Diagnostics"),
318 Qt::BottomDockWidgetArea,
319 Qt::BottomDockWidgetArea | Qt::LeftDockWidgetArea | Qt::RightDockWidgetArea,
320 false,
321 },
322 diagnostics_panel);
323
324 panel_host.bind_panel_visibility_action(
325 quader::ui::PanelId::Scene, fixture.actions, quader::ui::ActionId::ShowScenePanel);
326 panel_host.bind_panel_visibility_action(
327 quader::ui::PanelId::Properties, fixture.actions, quader::ui::ActionId::ShowPropertiesPanel);
328 panel_host.bind_panel_visibility_action(
329 quader::ui::PanelId::Diagnostics, fixture.actions, quader::ui::ActionId::ShowDiagnosticsPanel);
330
331 EXPECT_TRUE(panel_host.dock(quader::ui::PanelId::Scene) == &scene_dock);
332 EXPECT_TRUE(panel_host.dock(quader::ui::PanelId::Properties) == &properties_dock);
333 EXPECT_TRUE(panel_host.dock(quader::ui::PanelId::Diagnostics) == &diagnostics_dock);
334 EXPECT_EQ(scene_dock.objectName(), QStringLiteral("quader.panel.scene"));
335 EXPECT_EQ(properties_dock.objectName(), QStringLiteral("quader.panel.properties"));
336 EXPECT_EQ(diagnostics_dock.objectName(), QStringLiteral("quader.panel.diagnostics"));
337 EXPECT_TRUE(scene_dock.widget() == scene_panel);
338 EXPECT_TRUE(properties_dock.widget() == properties_panel);
339 EXPECT_TRUE(diagnostics_dock.widget() == diagnostics_panel);
340 EXPECT_TRUE(scene_dock.parent() == &host_window);
341 EXPECT_TRUE(diagnostics_dock.isHidden());
342 EXPECT_FALSE(fixture.actions.action(quader::ui::ActionId::ShowDiagnosticsPanel).isChecked());
343
344 QAction &scene_action = fixture.actions.action(quader::ui::ActionId::ShowScenePanel);
345 EXPECT_TRUE(scene_action.isCheckable());
346 EXPECT_TRUE(scene_action.isChecked());
347 scene_action.trigger();
348 EXPECT_TRUE(scene_dock.isHidden());
349 EXPECT_FALSE(scene_action.isChecked());
350 scene_action.trigger();
351 EXPECT_FALSE(scene_dock.isHidden());
352 EXPECT_TRUE(scene_action.isChecked());
353
354 panel_host.hide_panel(quader::ui::PanelId::Properties);
355 EXPECT_TRUE(properties_dock.isHidden());
356 EXPECT_FALSE(fixture.actions.action(quader::ui::ActionId::ShowPropertiesPanel).isChecked());
357
358 panel_host.reset_to_default_layout();
359 EXPECT_FALSE(scene_dock.isHidden());
360 EXPECT_FALSE(properties_dock.isHidden());
361 EXPECT_TRUE(diagnostics_dock.isHidden());
362
363 panel_host.save_workspace(fixture.settings);
364 EXPECT_TRUE(fixture.settings.panel_visible(quader::ui::PanelId::Scene, false));
365 EXPECT_TRUE(fixture.settings.panel_visible(quader::ui::PanelId::Properties, false));
366 EXPECT_FALSE(fixture.settings.panel_visible(quader::ui::PanelId::Diagnostics, true));
367
368 panel_host.hide_panel(quader::ui::PanelId::Scene);
369 fixture.settings.set_panel_visible(quader::ui::PanelId::Scene, false);
370 fixture.settings.set_panel_visible(quader::ui::PanelId::Diagnostics, true);
371 panel_host.restore_workspace(fixture.settings);
372 EXPECT_TRUE(scene_dock.isHidden());
373 EXPECT_FALSE(diagnostics_dock.isHidden());
374}
