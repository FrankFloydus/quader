29[[nodiscard]] const crimson::BaseShaderDefinition *find_shader(
30 const crimson::BaseShaderRegistry &registry,
31 crimson::BaseShaderId id) {
32 const crimson::BaseShaderDefinition *definition = registry.find(id);
33 expect_true(definition != nullptr, "expected base shader definition exists");
34 return definition;
35}

37[[nodiscard]] bool has_parameter(const crimson::BaseShaderDefinition &definition, std::string_view name) {
38 return crimson::find_parameter_desc(definition, name) != nullptr;
39}
40
41[[nodiscard]] bool has_texture_slot(
42 const crimson::BaseShaderDefinition &definition,
43 std::string_view name,
44 crimson::TextureColorSpace color_space) {
45 const crimson::MaterialTextureSlotDesc *slot = crimson::find_texture_slot_desc(definition, name);
46 return slot != nullptr && slot->color_space == color_space;
47}
48
49[[nodiscard]] const crimson::MaterialParameter *find_material_parameter(
50 const crimson::MaterialInstance &material,
51 std::string_view name) {
52 for (const crimson::MaterialParameter &parameter : material.parameters) {
53 if (parameter.name == name) {
54 return &parameter;
55 }
56 }
57
58 return nullptr;
59}

61TEST(MaterialSystem, V1RegistryContainsExpectedBaseShadersAndQueues) {
62 const crimson::BaseShaderRegistry kRegistry = crimson::make_v1_base_shader_registry();
63 expect_true(kRegistry.definitions().size() == 4, "V1 registry contains exactly four base shaders");
64
65 const crimson::BaseShaderDefinition *opaque = find_shader(kRegistry, crimson::BaseShaderId::OpaquePbr);
66 const crimson::BaseShaderDefinition *cutout = find_shader(kRegistry, crimson::BaseShaderId::AlphaCutoutPbr);
67 const crimson::BaseShaderDefinition *transparent = find_shader(kRegistry, crimson::BaseShaderId::TransparentPbr);

75 expect_true(opaque->depth_mode == crimson::DepthMode::TestWrite, "OpaquePbr writes depth");
76 expect_true(opaque->blend_mode == crimson::BlendMode::Off, "OpaquePbr disables blending");

80 expect_true(cutout->blend_mode == crimson::BlendMode::Off, "AlphaCutoutPbr disables blending");
81 expect_true(
82 cutout->shadow_mode == crimson::ShadowMode::AlphaTestedCastAndReceive,
83 "AlphaCutoutPbr declares alpha-tested shadows");
84
85 expect_true(

87 "TransparentPbr is a transparent surface");
88 expect_true(transparent->default_queue == crimson::RenderQueue::Transparent, "TransparentPbr uses transparent queue");

94 expect_true(
95 overlay->default_queue == crimson::RenderQueue::OverlayDepthTested,

97 expect_true(overlay->depth_mode == crimson::DepthMode::OverlayCommand, "OverlayUnlit depth comes from overlay data");
98 expect_true(overlay->blend_mode == crimson::BlendMode::AlphaBlend, "OverlayUnlit alpha blends");
99 expect_true(overlay->shadow_mode == crimson::ShadowMode::None, "OverlayUnlit has no shadows");

180TEST(MaterialSystem, MaterialDefaultsAreDeterministicAndNormalized) {
181 crimson::MaterialSystem materials;
182 const crimson::BaseShaderDefinition *opaque = find_shader(materials.registry(), crimson::BaseShaderId::OpaquePbr);
183 if (opaque == nullptr) {
184 return;
185 }
186
187 const crimson::MaterialInstance kFirstDefault = crimson::make_default_material_instance(*opaque);
188 const crimson::MaterialInstance kSecondDefault = crimson::make_default_material_instance(*opaque);
189 expect_true(
190 kFirstDefault.parameters.size() == opaque->parameters.size(),
191 "default material contains one parameter value per schema parameter");
192 expect_true(kFirstDefault.textures.size() == opaque->texture_slots.size(), "default material binds every texture slot");
193 expect_true(
194 kFirstDefault.parameters.size() == kSecondDefault.parameters.size(),
195 "default material parameter count is deterministic");
196 expect_true(
197 kFirstDefault.parameters[0].name == kSecondDefault.parameters[0].name,
198 "default material parameter order is deterministic");
199 expect_true(
200 kFirstDefault.parameters[0].value == kSecondDefault.parameters[0].value,
201 "default material parameter values are deterministic");
202
203 crimson::MaterialInstance partial;
204 partial.debug_name = "PartialOpaque";
205 partial.base_shader_id = crimson::BaseShaderId::OpaquePbr;
206 partial.parameters.push_back(crimson::MaterialParameter{
207 .name = "roughness",
208 .value = 0.8F,
209 });
210
211 auto handle = materials.create_material(partial);
212 expect_true(handle.has_value(), "partial material with declared override is accepted");
213 if (!handle) {
214 return;
215 }
216
217 const crimson::MaterialInstance *stored = materials.get(handle.value());
218 expect_true(stored != nullptr, "created material resolves from public handle");
219 if (stored == nullptr) {
220 return;
221 }
222
223 expect_true(stored->parameters.size() == opaque->parameters.size(), "stored material is normalized with defaults");
224 const crimson::MaterialParameter *roughness = find_material_parameter(*stored, "roughness");
225 const crimson::MaterialParameter *metallic = find_material_parameter(*stored, "metallic");
226 expect_true(
227 roughness != nullptr && std::get<float>(roughness->value) == 0.8F,
228 "stored material keeps supplied roughness override");
229 expect_true(
230 metallic != nullptr && std::get<float>(metallic->value) == 0.0F,
231 "stored material fills missing metallic default");
232}

307 expect_true(handle.has_value(), "default opaque material is created for GPU packing");

333TEST(MaterialSystem, PbrPacketPreparationBucketsAndSortsDrawPackets) {
334 crimson::MaterialSystem materials;
335 const crimson::RenderMaterialHandle kOpaque = materials.create_default_material(crimson::BaseShaderId::OpaquePbr).value();
336 const crimson::RenderMaterialHandle kCutout =
337 materials.create_default_material(crimson::BaseShaderId::AlphaCutoutPbr).value();
338 const crimson::RenderMaterialHandle kTransparent =
339 materials.create_default_material(crimson::BaseShaderId::TransparentPbr).value();
340
341 crimson::RenderObject opaque_object;
342 opaque_object.object_id = 1;
343 opaque_object.material = kOpaque;
344 opaque_object.base_shader = crimson::BaseShaderId::OpaquePbr;
345 opaque_object.queue = crimson::RenderQueue::Opaque;
346
347 crimson::RenderObject cutout_object;
348 cutout_object.object_id = 2;
349 cutout_object.material = kCutout;
350 cutout_object.base_shader = crimson::BaseShaderId::AlphaCutoutPbr;
351 cutout_object.queue = crimson::RenderQueue::AlphaCutout;
352
353 crimson::RenderObject transparent_a;
354 transparent_a.object_id = 4;
355 transparent_a.material = kTransparent;
356 transparent_a.base_shader = crimson::BaseShaderId::TransparentPbr;
357 transparent_a.queue = crimson::RenderQueue::Transparent;
358
359 crimson::RenderObject transparent_b = transparent_a;
360 transparent_b.object_id = 3;
361
362 const std::array<crimson::RenderObject, 4> kObjects = {
363 transparent_a,
364 opaque_object,
365 transparent_b,
366 cutout_object,
367 };
368 const crimson::RenderCamera kCamera{
369 .eye = { 0.0F, 0.0F, 0.0F },
370 .target = { 0.0F, 0.0F, -1.0F },
371 .forward = { 0.0F, 0.0F, -1.0F },
372 };
373
374 const crimson::gpu::PbrPassPackets kPackets = crimson::gpu::prepare_pbr_pass_packets(
375 kObjects,
376 materials.registry(),
377 materials,
378 kCamera,
379 crimson::RenderMeshHandle{ 9, 1 },
380 kOpaque,
381 1.0F);
382
383 expect_true(kPackets.opaque.size() == 1, "opaque PBR packets are bucketed");
384 expect_true(kPackets.alpha_cutout.size() == 1, "alpha cutout PBR packets are bucketed");
385 expect_true(kPackets.transparent.size() == 2, "transparent PBR packets are bucketed");
386 expect_true(
387 kPackets.transparent[0].object_id == 3 && kPackets.transparent[1].object_id == 4,
388 "same-depth transparent packets sort by stable object id tie-breaker");
389 expect_true(
390 kPackets.opaque[0].program == crimson::ShaderProgramId::OpaquePbr,
391 "PBR packet uses the base shader program, not prototype object program data");
392}
