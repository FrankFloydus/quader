86QVariant DocumentItemModel::data(const QModelIndex &model_index,
87 int role) const {
88 const auto *node = node_from_index(model_index);
89 if (node == nullptr) {
90 return {};
91 }
92
93 const auto *object = object_for_node(*node);
94 if (object == nullptr) {
95 return {};
96 }
97
98 switch (role) {
99 case Qt::DisplayRole:
100 case Qt::EditRole:
101 if (is_name_column(model_index.column())) {
102 return QString::fromStdString(object->name);
103 }
104 if (is_kind_column(model_index.column())) {
105 return QStringLiteral("Mesh");
106 }
107 return {};
108 case document_item_roles::kUiNodeIdRole:
109 return QVariant::fromValue(node->ui_node_id);
110 case document_item_roles::kObjectIdRole:
111 return QVariant::fromValue(node->object);
112 case document_item_roles::kKindRole:
113 return QVariant::fromValue(DocumentItemKind::MeshObject);
114 case document_item_roles::kSelectedRole:
115 return documents_.document().selection().references_object(node->object);
116 default:
117 return {};
118 }
119}

121bool DocumentItemModel::setData(const QModelIndex &model_index,
122 const QVariant &value, int role) {
123 if (role != Qt::EditRole || !model_index.isValid() ||
124 !is_name_column(model_index.column())) {
125 return false;
126 }
127
128 const auto *node = node_from_index(model_index);
129 if (node == nullptr) {
130 return false;
131 }
132
133 const auto *object = object_for_node(*node);
134 if (object == nullptr) {
135 return false;
136 }
137
138 const QString kNewName = value.toString();
139 if (kNewName.isEmpty() || kNewName.toStdString() == object->name) {
140 return false;
141 }
142
143 auto result = documents_.execute_command(
144 std::make_unique<quader::commands::RenameObjectCommand>(
145 node->object, kNewName.toStdString()));
146 return result.is_applied();
147}

190void DocumentItemModel::rebuild_from_document() {
191 beginResetModel();
192 roots_.clear();
193
194 const auto kObjectIds = documents_.document().object_ids();
195 roots_.reserve(kObjectIds.size());
196 for (int row = 0; row < static_cast<int>(kObjectIds.size()); ++row) {
197 auto node = std::make_unique<Node>();
198 node->object = kObjectIds[static_cast<std::size_t>(row)];
199 node->ui_node_id = ui_node_id_for_object(node->object);
200 node->row = row;
201 roots_.push_back(std::move(node));
202 }
203
204 remove_stale_ui_node_ids();
205 endResetModel();
206}

208void DocumentItemModel::refresh_object(quader::document::ObjectId object) {
209 const int kRow = row_for_object(object);
210 if (kRow < 0) {
211 return;

214 Q_EMIT dataChanged(

216 { Qt::DisplayRole, Qt::EditRole, document_item_roles::kObjectIdRole,
217 document_item_roles::kKindRole, document_item_roles::kSelectedRole });

245int DocumentItemModel::row_for_object(
246 quader::document::ObjectId object) const noexcept {
247 const auto kIt =
248 std::find_if(roots_.begin(), roots_.end(), [object](const auto &node) {
249 return node->object == object;
250 });
251
252 if (kIt == roots_.end()) {
253 return -1;
254 }
255
256 return (*kIt)->row;
257}

259UiNodeId
260DocumentItemModel::ui_node_id_for_object(quader::document::ObjectId object) {
261 const auto kIt = ui_node_ids_.find(object);
262 if (kIt != ui_node_ids_.end()) {
263 return kIt->second;
264 }
265
266 UiNodeId id{ next_ui_node_id_++ };
267 ui_node_ids_.emplace(object, id);
268 return id;
269}

271void DocumentItemModel::remove_stale_ui_node_ids() {
272 for (auto it = ui_node_ids_.begin(); it != ui_node_ids_.end();) {
273 if (row_for_object(it->first) < 0) {
274 it = ui_node_ids_.erase(it);
275 } else {
276 ++it;
277 }
278 }
