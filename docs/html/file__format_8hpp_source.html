45[[nodiscard]] inline std::string normalize_extension(std::string_view extension) {
46 if (!extension.empty() && extension.front() == '.') {
47 extension.remove_prefix(1);
48 }
49
50 std::string normalized;
51 normalized.reserve(extension.size());
52 for (const char kCharacter : extension) {
53 normalized.push_back(static_cast<char>(std::tolower(static_cast<unsigned char>(kCharacter))));
54 }
55 return normalized;
56}

75[[nodiscard]] inline bool descriptor_supports_extension(const FileFormatDescriptor &descriptor,
76 const std::filesystem::path &path) {
77 const auto kExtension = path_extension_without_dot(path);
78 if (kExtension.empty()) {
79 return false;
80 }
81
82 return std::any_of(descriptor.extensions.begin(),
83 descriptor.extensions.end(),
84 [&kExtension](const std::string &candidate) {
85 return normalize_extension(candidate) == kExtension;
