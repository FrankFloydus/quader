103void DiagnosticsPanel::refresh_from_service() {
104 const ViewportDiagnosticsSummary kSummary = context_.ui.viewport_diagnostics.summary();
105 summary_label_->setText(kSummary.status_text);
106 summary_label_->setToolTip(kSummary.tooltip_text);
107
108 const auto &snapshot = context_.ui.viewport_diagnostics.latest_snapshot();
109 frame_graph_view_->setPlainText(snapshot.has_value() ? snapshot->frame_graph_dump : QString());
110}

112void DiagnosticsPanel::copy_selected_rows() {
113 const QModelIndexList kRows = tree_view_->selectionModel()->selectedRows(static_cast<int>(DiagnosticsItemColumn::Name));
114 if (kRows.isEmpty()) {
115 context_.ui.notifications.show_status(QStringLiteral("No diagnostics rows selected"), 2000);
116 return;
117 }
118
119 QStringList lines;
120 for (const QModelIndex &row : kRows) {
121 lines.push_back(model_->copy_text_for_index(row));
122 }
123
124 QApplication::clipboard()->setText(lines.join(QStringLiteral("\n")));
125 context_.ui.notifications.show_status(QStringLiteral("Copied selected diagnostics"), 2000);
126}
127
128void DiagnosticsPanel::copy_all_rows() {
129 const QString kText = model_->copy_text_for_all();
130 if (kText.isEmpty()) {
131 context_.ui.notifications.show_status(QStringLiteral("No diagnostics to copy"), 2000);
132 return;
133 }
134
135 QApplication::clipboard()->setText(kText);
136 context_.ui.notifications.show_status(QStringLiteral("Copied diagnostics"), 2000);
137}
