85QModelIndex DiagnosticsItemModel::index(int row, int column, const QModelIndex &parent_index) const {
86 if (row < 0 || column < 0 || column >= columnCount()) {
87 return {};
88 }
89 if (parent_index.isValid() && parent_index.model() != this) {
90 return {};
91 }
92
93 const Node *parent_node = node_from_index(parent_index);
94 const auto &siblings = parent_node == nullptr ? roots_ : parent_node->children;
95 const auto kRowIndex = static_cast<std::size_t>(row);
96 if (kRowIndex >= siblings.size()) {
97 return {};
98 }
99
100 return createIndex(row, column, siblings[kRowIndex].get());
101}

134QVariant DiagnosticsItemModel::data(const QModelIndex &model_index, int role) const {
135 const Node *node = node_from_index(model_index);
136 if (node == nullptr) {
137 return {};
138 }
139
140 switch (role) {
141 case Qt::DisplayRole:
142 if (is_name_column(model_index.column())) {
143 return display_or_dash(node->name);
144 }
145 if (is_value_column(model_index.column())) {
146 return node->kind == DiagnosticsNodeKind::Section ? QVariant{} : display_or_dash(node->value);
147 }
148 if (is_detail_column(model_index.column())) {
149 return node->kind == DiagnosticsNodeKind::Section ? QVariant{} : display_or_dash(node->detail);
150 }
151 return {};
152 case Qt::ToolTipRole:
153 return node->detail.isEmpty() ? copy_text_for_node(*node) : node->detail;
154 case diagnostics_item_roles::kNodeKindRole:
155 return QVariant::fromValue(node->kind);
156 case diagnostics_item_roles::kSeverityRole:
157 return node->severity;
158 case diagnostics_item_roles::kCategoryRole:
159 return node->category;
160 case diagnostics_item_roles::kFrameIndexRole:
161 return node->frame_index.has_value() ? QVariant::fromValue(*node->frame_index) : QVariant{};
162 case diagnostics_item_roles::kCopyTextRole:
163 return copy_text_for_node(*node);
164 default:
165 return {};
166 }
167}

225void DiagnosticsItemModel::rebuild_from_snapshot(const std::optional<ViewportDiagnosticsSnapshot> &snapshot) {
226 roots_.clear();
227
228 auto summary = make_node(DiagnosticsNodeKind::Section, QStringLiteral("Summary"));
229 auto render_passes = make_node(DiagnosticsNodeKind::Section, QStringLiteral("Render Passes"));
230 auto counters = make_node(DiagnosticsNodeKind::Section, QStringLiteral("Counters"));
231 auto diagnostics = make_node(DiagnosticsNodeKind::Section, QStringLiteral("Diagnostics"));
232
233 add_summary_rows(*summary, snapshot);
234
235 if (snapshot.has_value()) {
236 for (const ViewportRenderPassStats &pass : snapshot->passes) {
237 auto node = make_node(
238 DiagnosticsNodeKind::RenderPass,
239 pass.pass_name,
240 QStringLiteral("%1 calls / %2 packets").arg(pass.draw_call_count).arg(pass.draw_packet_count),
241 pass_detail(pass));
242 node->category = QStringLiteral("RenderPass");
243 append_node(*render_passes, std::move(node));
244 }
245
246 for (const ViewportRendererCounter &counter : snapshot->counters) {
247 auto node = make_node(
248 DiagnosticsNodeKind::Counter,
249 counter.name,
250 counter.unit.isEmpty()
251 ? QString::number(counter.value)
252 : QStringLiteral("%1 %2").arg(counter.value).arg(counter.unit),
253 counter.domain);
254 node->category = counter.domain;
255 append_node(*counters, std::move(node));
256 }
257
258 for (const ViewportRendererDiagnosticRow &row : snapshot->diagnostics) {
259 auto node = make_node(DiagnosticsNodeKind::Diagnostic, row.message, row.code, row.detail);
260 node->severity = row.severity;
261 node->category = row.subsystem;
262 node->frame_index = row.frame_index;
263 node->copy_text = diagnostic_copy_text(row);
264 if (!row.resource_name.isEmpty()) {
265 node->detail = row.detail.isEmpty()
266 ? QStringLiteral("resource=%1").arg(row.resource_name)
267 : QStringLiteral("%1; resource=%2").arg(row.detail, row.resource_name);
268 }
269 append_node(*diagnostics, std::move(node));
270 }
271 }
272
273 append_root(std::move(summary));
274 append_root(std::move(render_passes));
275 append_root(std::move(counters));
276 append_root(std::move(diagnostics));
277}

279void DiagnosticsItemModel::add_summary_rows(
280 Node &section,
281 const std::optional<ViewportDiagnosticsSnapshot> &snapshot) {
282 if (!snapshot.has_value()) {
283 append_node(section,
284 make_node(
285 DiagnosticsNodeKind::SummaryField,
286 QStringLiteral("Status"),
287 QStringLiteral("Diagnostics unavailable"),
288 QStringLiteral("No viewport diagnostics snapshot has been produced.")));
289 return;
290 }
291
292 int warning_count = 0;
293 int error_count = 0;
294 for (const ViewportRendererDiagnosticRow &row : snapshot->diagnostics) {
295 if (row.severity.compare(QStringLiteral("Warning"), Qt::CaseInsensitive) == 0) {
296 ++warning_count;
297 } else if (row.severity.compare(QStringLiteral("Error"), Qt::CaseInsensitive) == 0) {
298 ++error_count;
299 }
300 }
301
302 append_node(section, make_node(DiagnosticsNodeKind::SummaryField, QStringLiteral("Renderer"), snapshot->renderer_name));
303 append_node(section, make_node(DiagnosticsNodeKind::SummaryField, QStringLiteral("Frame Size"), frame_size_value(snapshot->frame)));
304 append_node(section,
305 make_node(
306 DiagnosticsNodeKind::SummaryField,
307 QStringLiteral("Viewports"),
308 QString::number(snapshot->frame.viewport_count)));
309 append_node(section, make_node(DiagnosticsNodeKind::SummaryField, QStringLiteral("FPS"), fps_value(snapshot->frame.fps)));
310 append_node(
311 section, make_node(DiagnosticsNodeKind::SummaryField, QStringLiteral("Passes"), QString::number(snapshot->passes.size())));
312 append_node(section,
313 make_node(
314 DiagnosticsNodeKind::SummaryField,
315 QStringLiteral("Counters"),
316 QString::number(snapshot->counters.size())));
317 append_node(section,
318 make_node(
319 DiagnosticsNodeKind::SummaryField,
320 QStringLiteral("Warnings"),
321 QString::number(warning_count)));
322 append_node(
323 section, make_node(DiagnosticsNodeKind::SummaryField, QStringLiteral("Errors"), QString::number(error_count)));
324}

351QString DiagnosticsItemModel::copy_text_for_node(const Node &node) const {
352 if (!node.copy_text.isEmpty()) {
353 return node.copy_text;
354 }
355
356 QStringList fields;
357 fields.push_back(node.name);
358 if (!node.value.isEmpty()) {
359 fields.push_back(node.value);
360 }
361 if (!node.detail.isEmpty()) {
362 fields.push_back(node.detail);
363 }
364 return fields.join(QStringLiteral("\t"));

367void DiagnosticsItemModel::append_copy_text_recursive(const Node &node, QStringList &lines) const {
368 lines.push_back(copy_text_for_node(node));
369 for (const auto &child : node.children) {
