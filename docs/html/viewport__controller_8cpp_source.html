31[[nodiscard]] quader::math::Vec3 scale(quader::math::Vec3 value, float factor) noexcept {
32 return quader::math::Vec3{ value.x * factor, value.y * factor, value.z * factor };
33}

35[[nodiscard]] quader::math::Vec3 normalized_or(quader::math::Vec3 value,
36 quader::math::Vec3 fallback) noexcept {
37 const auto kNormalized = quader::math::normalized(value, kRayEpsilon);
38 return quader::math::length_squared(kNormalized) <= kRayEpsilon * kRayEpsilon ? fallback : kNormalized;
39}

41[[nodiscard]] quader::math::Vec3 rotate_x(quader::math::Vec3 value, float radians) noexcept {
42 const float kSin = std::sin(radians);
43 const float kCos = std::cos(radians);
44 return { value.x, value.y * kCos - value.z * kSin, value.y * kSin + value.z * kCos };
45}
46
47[[nodiscard]] quader::math::Vec3 rotate_y(quader::math::Vec3 value, float radians) noexcept {
48 const float kSin = std::sin(radians);
49 const float kCos = std::cos(radians);
50 return { value.x * kCos + value.z * kSin, value.y, -value.x * kSin + value.z * kCos };
51}
52
53[[nodiscard]] quader::math::Vec3 rotate_z(quader::math::Vec3 value, float radians) noexcept {
54 const float kSin = std::sin(radians);
55 const float kCos = std::cos(radians);
56 return { value.x * kCos - value.y * kSin, value.x * kSin + value.y * kCos, value.z };
57}

59[[nodiscard]] float radians_from_degrees(float degrees) noexcept {
60 return degrees * kPi / 180.0F;
61}

63[[nodiscard]] quader::math::Vec3 rotate_euler(quader::math::Vec3 value,
64 quader::math::Vec3 degrees) noexcept {
65 value = rotate_x(value, radians_from_degrees(degrees.x));
66 value = rotate_y(value, radians_from_degrees(degrees.y));
67 return rotate_z(value, radians_from_degrees(degrees.z));
68}

70[[nodiscard]] quader::math::Vec3 transform_point(quader::math::Vec3 point,
71 quader::document::Transform transform) noexcept {
72 point = quader::math::Vec3{
73 point.x * transform.scale.x,
74 point.y * transform.scale.y,
75 point.z * transform.scale.z,
76 };
77 return rotate_euler(point, transform.rotation_euler) + transform.translation;

80struct RayTriangleHit {
81 bool hit = false;
82 float distance = 0.0F;
83 quader::math::Vec3 position;
84};

86[[nodiscard]] RayTriangleHit intersect_triangle(const quader::tools::ViewportRay &ray,
87 quader::math::Vec3 a,
88 quader::math::Vec3 b,
89 quader::math::Vec3 c) noexcept {
90 const quader::math::Vec3 kDirection = normalized_or(ray.direction, {});
91 if (quader::math::length_squared(kDirection) <= kRayEpsilon * kRayEpsilon) {
92 return {};
93 }
94
95 const quader::math::Vec3 kEdge1 = b - a;
96 const quader::math::Vec3 kEdge2 = c - a;
97 const quader::math::Vec3 kP = quader::math::cross(kDirection, kEdge2);
98 const float kDet = quader::math::dot(kEdge1, kP);
99 if (std::abs(kDet) <= kRayEpsilon) {
100 return {};
101 }
102
103 const float kInvDet = 1.0F / kDet;
104 const quader::math::Vec3 kT = ray.origin - a;
105 const float kU = quader::math::dot(kT, kP) * kInvDet;
106 if (kU < 0.0F || kU > 1.0F) {
107 return {};
108 }
109
110 const quader::math::Vec3 kQ = quader::math::cross(kT, kEdge1);
111 const float kV = quader::math::dot(kDirection, kQ) * kInvDet;
112 if (kV < 0.0F || kU + kV > 1.0F) {
113 return {};
114 }
115
116 const float kDistance = quader::math::dot(kEdge2, kQ) * kInvDet;
117 if (kDistance < 0.0F || !quader::math::is_finite(kDistance)) {
118 return {};
119 }
120
121 return RayTriangleHit{
122 .hit = true,
123 .distance = kDistance,
124 .position = ray.origin + scale(kDirection, kDistance),
125 };
126}

128[[nodiscard]] std::optional<std::size_t> pane_array_index_at(
129 const ViewportLayoutState &layout,
130 ViewportPixelSize size,
131 ViewportPoint point) {
132 const ViewportPaneArray kPanes = layout.panes_for(size);
133 for (int index = 0; index < layout.pane_count(); ++index) {
134 const ViewportRect &rect = kPanes[static_cast<std::size_t>(index)].rect;
135 if (point.x >= rect.x && point.y >= rect.y && point.x < rect.x + rect.width && point.y < rect.y + rect.height) {
136 return static_cast<std::size_t>(index);
137 }
138 }
139 return std::nullopt;
140}

181 request_frame();

237void ViewportController::render_frame(double elapsed_seconds, float delta_seconds) {
238 if (!surface_initialized_) {
239 return;
240 }
241
242 cameras_.tick_fly_navigation(delta_seconds);
243
244 const ViewportPaneArray kPanes = layout_.panes_for(surface_size_);
245 const ViewportCameraSnapshotArray kCameras = cameras_.camera_snapshots();
246 const int kPaneCount = layout_.pane_count();
247 const ViewportRenderRequest kRequest{
248 .surface_size = surface_size_,
249 .device_pixel_ratio = device_pixel_ratio_,
250 .layout_mode = layout_.mode(),
251 .panes = std::span<const ViewportPane>(kPanes.data(), static_cast<std::size_t>(kPaneCount)),
252 .cameras = std::span<const ViewportCameraSnapshot>(kCameras.data(), kCameras.size()),
253 .prototype_animation_enabled = prototype_animation_enabled_,
254 .elapsed_seconds = elapsed_seconds,
255 };
256
257 const ViewportRenderResult kResult = render_host_.render_frame(kRequest);
258 if (!kResult.ok) {
259 emit_result_error(kResult);
260 return;
261 }
262
263 if (const std::optional<ViewportFrameStats> kStats = render_host_.latest_frame_stats()) {
264 Q_EMIT frame_stats_changed(*kStats);
265 }
266}

284bool ViewportController::handle_mouse_press(
285 ViewportMouseButton button,
286 ViewportPoint point,
287 bool shift_modifier,
288 ViewportPixelSize size) {
289 const auto kPaneIndex = pane_array_index_at(layout_, size, point);
290 if (kPaneIndex.has_value()) {
291 const ViewportPaneArray kPanes = layout_.panes_for(size);
292 cameras_.set_active_camera_index(kPanes[*kPaneIndex].camera_index);
293 }
294
295 if (button == ViewportMouseButton::Middle) {
296 const NavigationMode kMode = shift_modifier ? NavigationMode::Pan : NavigationMode::Orbit;
297 if (cameras_.begin_navigation(cameras_.active_camera_index(), kMode, point)) {
298 request_frame();
299 return true;
300 }
301 } else if (button == ViewportMouseButton::Right) {
302 const NavigationMode kMode = cameras_.camera_projection(cameras_.active_camera_index()) == CameraProjection::Orthographic
303 ? NavigationMode::Pan
304 : NavigationMode::Fly;
305 if (cameras_.begin_navigation(cameras_.active_camera_index(), kMode, point)) {
306 request_frame();
307 return true;
308 }
309 }
310
311 if (button == ViewportMouseButton::Left) {
312 return dispatch_tool_pointer(button, point, size, quader::tools::PointerPhase::Press, true);
313 }
314
315 return false;
316}

349void ViewportController::handle_wheel(float wheel_steps, ViewportPoint point, ViewportPixelSize size) {
350 if (std::abs(wheel_steps) <= 0.000001F) {
351 return;
352 }
353
354 if (cameras_.navigation_mode() == NavigationMode::Fly) {
355 cameras_.adjust_fly_speed(wheel_steps);
356 return;
357 }
358
359 const auto kPaneIndex = pane_array_index_at(layout_, size, point);
360 const ViewportPaneArray kPanes = layout_.panes_for(size);
361 const int kCameraIndex = kPaneIndex.has_value() ? kPanes[*kPaneIndex].camera_index : cameras_.active_camera_index();
362 cameras_.apply_wheel_zoom(kCameraIndex, wheel_steps, pane_size_for_camera(size, kCameraIndex));
363 request_frame();
364}

366bool ViewportController::handle_key(ViewportKey key, bool pressed, bool auto_repeat) {
367 if (cameras_.navigation_mode() == NavigationMode::Fly) {
368 if (!auto_repeat) {
369 switch (key) {
370 case ViewportKey::W:
371 cameras_.set_fly_key(ViewportFlyKey::Forward, pressed);
372 break;
373 case ViewportKey::A:
374 cameras_.set_fly_key(ViewportFlyKey::Left, pressed);
375 break;
376 case ViewportKey::S:
377 cameras_.set_fly_key(ViewportFlyKey::Backward, pressed);
378 break;
379 case ViewportKey::D:
380 cameras_.set_fly_key(ViewportFlyKey::Right, pressed);
381 break;
382 case ViewportKey::Escape:
383 case ViewportKey::Other:
384 break;
385 }
386 }
387 return true;
388 }
389
390 if (key == ViewportKey::Escape && pressed && !auto_repeat && tool_manager_ != nullptr) {
391 const bool kCanceled = tool_manager_->cancel_active_tool();
392 if (kCanceled) {
393 request_frame();
394 }
395 return kCanceled;
396 }
397
398 return false;

401std::optional<quader::tools::ViewportRay> ViewportController::ray_for_point(
402 ViewportPoint point,
403 ViewportPixelSize size) const {
404 const auto kPaneIndex = pane_array_index_at(layout_, size, point);
405 if (!kPaneIndex.has_value()) {
406 return std::nullopt;
407 }
408
409 const ViewportPaneArray kPanes = layout_.panes_for(size);
410 const ViewportPane &pane = kPanes[*kPaneIndex];
411 const ViewportCameraSnapshotArray kSnapshots = cameras_.camera_snapshots();
412 const ViewportCameraSnapshot &camera = kSnapshots[static_cast<std::size_t>(std::clamp(pane.camera_index, 0, 3))];
413 const crimson::RenderCamera kCamera{
414 .eye = camera.eye,
415 .target = camera.target,
416 .up = camera.up,
417 .forward = camera.forward,
418 .projection = camera.projection == CameraProjection::Orthographic
419 ? crimson::CameraProjection::Orthographic
420 : crimson::CameraProjection::Perspective,
421 .vertical_fov_degrees = camera.fov_degrees,
422 .orthographic_height_m = camera.orthographic_size,
423 };
424 const crimson::RenderCameraRay kRay = crimson::render_camera_ray_from_viewport_point(
425 kCamera,
426 crimson::RenderCameraViewportSize{
427 static_cast<float>(std::max(1, pane.rect.width)),
428 static_cast<float>(std::max(1, pane.rect.height)),
429 },
430 quader::math::Vec2{
431 static_cast<float>(point.x - pane.rect.x),
432 static_cast<float>(point.y - pane.rect.y),
433 },
434 crimson::current_render_homogeneous_depth());
435 return quader::tools::ViewportRay{
436 .origin = kRay.origin,
437 .direction = kRay.direction,
438 };
439}

441std::optional<quader::tools::SurfaceHit> ViewportController::surface_hit_for_ray(
442 const quader::tools::ViewportRay &ray) const {
443 if (tool_manager_ == nullptr) {
444 return std::nullopt;
445 }
446
447 const auto &document = tool_manager_->context().document();
448 float best_distance = std::numeric_limits<float>::infinity();
449 std::optional<quader::tools::SurfaceHit> best_hit;
450 for (const auto kObjectId : document.object_ids()) {
451 const auto *object = document.find_mesh_object(kObjectId);
452 if (object == nullptr) {
453 continue;
454 }
455
456 for (const auto kFaceId : object->mesh.face_ids()) {
457 auto face_vertices = quader::mesh::face_vertices(object->mesh, kFaceId);
458 if (!face_vertices || face_vertices.value().size() < 3U) {
459 continue;
460 }
461
462 std::vector<quader::math::Vec3> points;
463 points.reserve(face_vertices.value().size());
464 for (const auto kVertex : face_vertices.value()) {
465 auto position = object->mesh.vertex_position(kVertex);
466 if (!position) {
467 points.clear();
468 break;
469 }
470 points.push_back(transform_point(position.value(), object->transform));
471 }
472 if (points.size() < 3U) {
473 continue;
474 }
475
476 const quader::math::Vec3 kNormal = normalized_or(quader::geometry::polygon_normal(points),
477 { 0.0F, 1.0F, 0.0F });
478 for (std::size_t index = 1U; index + 1U < points.size(); ++index) {
479 const RayTriangleHit kHit = intersect_triangle(ray, points[0], points[index], points[index + 1U]);
480 if (!kHit.hit || kHit.distance >= best_distance) {
481 continue;
482 }
483
484 best_distance = kHit.distance;
485 best_hit = quader::tools::SurfaceHit{
486 .position = kHit.position,
487 .normal = kNormal,
488 .object_id = (static_cast<std::uint64_t>(kObjectId.generation()) << 32U) | kObjectId.index(),
489 .component_index = kFaceId.index(),
490 .kind = quader::tools::SurfaceHitKind::Face,
491 };
492 }
493 }
494 }
495
496 return best_hit;
497}

499bool ViewportController::dispatch_tool_pointer(ViewportMouseButton button,
500 ViewportPoint point,
501 ViewportPixelSize size,
502 quader::tools::PointerPhase phase,
503 bool pressed) {
504 if (tool_manager_ == nullptr || tool_manager_->active_tool() == nullptr) {
505 return false;
506 }
507
508 const auto kViewIndex = pane_array_index_at(layout_, size, point);
509 if (!kViewIndex.has_value()) {
510 return false;
511 }
512
513 const ViewportPaneArray kPanes = layout_.panes_for(size);
514 const ViewportPane &pane = kPanes[*kViewIndex];
515 const quader::math::Vec2 kLocalPosition{
516 static_cast<float>(point.x - pane.rect.x),
517 static_cast<float>(point.y - pane.rect.y),
518 };
519 const auto kRay = ray_for_point(point, size);
520 quader::tools::PointerEvent event{
521 .position = kLocalPosition,
522 .button = button == ViewportMouseButton::Left ? quader::tools::PointerButton::Left
523 : button == ViewportMouseButton::Middle
524 ? quader::tools::PointerButton::Middle
525 : button == ViewportMouseButton::Right ? quader::tools::PointerButton::Right
526 : quader::tools::PointerButton::None,
527 .pressed = pressed,
528 .phase = phase,
529 .navigation_active = cameras_.navigation_mode() != NavigationMode::None,
530 .snap_to_grid = true,
531 .grid_size = 1.0F,
532 .ray = kRay,
533 .view_index = static_cast<std::uint32_t>(*kViewIndex),
534 };
535 if (kRay.has_value()) {
536 event.surface_hit = surface_hit_for_ray(*kRay);
537 }
538
539 const bool kHandled = tool_manager_->dispatch_pointer_event(event);
540 if (kHandled) {
541 request_frame();
542 }
543 return kHandled;
544}
