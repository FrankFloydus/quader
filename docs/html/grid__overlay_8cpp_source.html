37 XZ,
38 YZ,
39 XY,
40};

42[[nodiscard]] GridPlane grid_plane_for_camera(const RenderCamera &camera) {
43 if (camera.projection != CameraProjection::Orthographic) {
44 return GridPlane::XZ;
45 }
46
47 Vec3 forward = normalized(camera.target - camera.eye);
48 if (length(forward) <= quader::math::kDefaultEpsilon) {
49 forward = normalized(camera.forward);
50 }
51 if (length(forward) <= quader::math::kDefaultEpsilon) {
52 forward = { 0.0F, 0.0F, 1.0F };
53 }
54
55 const float kAbsX = std::abs(forward.x);
56 const float kAbsY = std::abs(forward.y);
57 const float kAbsZ = std::abs(forward.z);
58 if (kAbsX >= kAbsY && kAbsX >= kAbsZ) {
59 return GridPlane::YZ;
60 }
61 if (kAbsZ >= kAbsX && kAbsZ >= kAbsY) {
62 return GridPlane::XY;
63 }
64 return GridPlane::XZ;
65}

69GridOverlayCommand make_grid_overlay_for_view(const RenderView &view) {
70 GridOverlayCommand command;
71 command.view_index = view.view_index;
72
73 const RenderCamera &camera = view.camera;
74 const GridPlane kPlane = grid_plane_for_camera(camera);
75 const float kAspect = std::max(0.0001F, static_cast<float>(std::max<std::uint16_t>(1, view.rect.width)) / static_cast<float>(std::max<std::uint16_t>(1, view.rect.height)));
76
77 command.plane_width = kViewportGridDefaultPlaneSize;
78 command.plane_height = kViewportGridDefaultPlaneSize;
79 if (camera.projection == CameraProjection::Orthographic) {
80 const float kPlaneHeight = std::max(0.01F, camera.orthographic_height_m) * kViewportGridOrthographicMargin;
81 if (kPlane == GridPlane::YZ) {
82 command.plane_width = kPlaneHeight;
83 command.plane_height = kPlaneHeight * kAspect;
84 } else {
85 command.plane_width = kPlaneHeight * kAspect;
86 command.plane_height = kPlaneHeight;
87 }
88 }
89
90 if (kPlane == GridPlane::YZ) {
91 command.origin = { kViewportGridPlaneOffset, camera.eye.y, camera.eye.z };
92 command.rotation_z = -kHalfPi;
93 command.u_axis = { 0.0F, 0.0F, -1.0F };
94 command.v_axis = { 0.0F, 1.0F, 0.0F };
95 command.axis_u_color = { 0.059F, 0.612F, 1.0F, 0.72F };
96 command.axis_v_color = { 0.29F, 0.58F, 0.0F, 0.0F };
97 } else if (kPlane == GridPlane::XY) {
98 command.origin = { camera.eye.x, camera.eye.y, kViewportGridPlaneOffset };
99 command.rotation_x = kHalfPi;
100 command.u_axis = { 1.0F, 0.0F, 0.0F };
101 command.v_axis = { 0.0F, 1.0F, 0.0F };
102 command.axis_u_color = { 1.0F, 0.239F, 0.0F, 0.72F };
103 command.axis_v_color = { 0.29F, 0.58F, 0.0F, 0.0F };
104 } else {
105 command.origin = { camera.eye.x, kViewportGridPlaneOffset, camera.eye.z };
106 }
107
108 command.minor_spacing_m = kGridWorldSpacing;
109 command.major_spacing_m = std::max(kParentGridWorldSize, kGridWorldSpacing);
110 command.fade_start_m = kGroundGridFadeStart;
111 command.fade_end_m = kGroundGridFadeEnd;
112 command.minor_line_scale = kDefaultViewportGridMinorLineSize;
113 command.major_line_scale = kDefaultViewportGridMajorLineSize;
114 command.axis_line_scale = kDefaultViewportGridAxisLineSize;
115 command.orthographic_height_m = std::max(0.0001F, camera.orthographic_height_m);
116 command.viewport_height_px = static_cast<float>(std::max<std::uint16_t>(1, view.rect.height));
117 command.edge_softness_m = 0.001F;
118 return command;
119}
