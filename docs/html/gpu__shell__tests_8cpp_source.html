31TEST(GpuShell, ResourceTableRejectsStaleHandlesAfterDestroyAndReuse) {
32 crimson::gpu::GpuResourceTable<TestResource, crimson::RenderMeshHandle> table;
33
34 const crimson::RenderMeshHandle kFirst = table.create(TestResource{ 42 });
35 expect_true(crimson::is_valid_handle(kFirst), "created resource handle is valid");
36 expect_true(table.live_count() == 1, "created resource increments live count");
37 expect_true(table.get(kFirst) != nullptr && table.get(kFirst)->value == 42, "created resource is retrievable");
38
39 expect_true(table.destroy(kFirst), "destroying a live resource succeeds");
40 expect_true(table.live_count() == 0, "destroyed resource decrements live count");
41 expect_true(table.get(kFirst) == nullptr, "destroyed handle no longer resolves");
42 expect_true(!table.destroy(kFirst), "destroying a stale handle fails");
43
44 const crimson::RenderMeshHandle kSecond = table.create(TestResource{ 84 });
45 expect_true(kSecond.index == kFirst.index, "resource table reuses freed slots");
46 expect_true(kSecond.generation != kFirst.generation, "reused slot advances generation");
47 expect_true(table.get(kFirst) == nullptr, "old generation does not resolve after reuse");
48 expect_true(table.get(kSecond) != nullptr && table.get(kSecond)->value == 84, "new generation resolves after reuse");
49}

51TEST(GpuShell, BackendSelectionUsesPlatformPriorityAndPreferences) {
52 using crimson::GraphicsBackendPreference;
53 using crimson::NativeSurfacePlatform;
54 using crimson::gpu::GraphicsBackend;
55
56 const std::vector<GraphicsBackend> kAllWindows = {
57 GraphicsBackend::Direct3D11,
58 GraphicsBackend::Vulkan,
59 GraphicsBackend::Direct3D12,
60 };
61
62 expect_true(
63 crimson::gpu::choose_backend(NativeSurfacePlatform::Windows, GraphicsBackendPreference::Auto, kAllWindows) == GraphicsBackend::Vulkan,
64 "Windows auto preference chooses Vulkan first");
65 expect_true(
66 crimson::gpu::choose_backend(NativeSurfacePlatform::Windows, GraphicsBackendPreference::Direct3D11, kAllWindows) == GraphicsBackend::Direct3D11,
67 "explicit Direct3D11 preference is honored when available");
68
69 const std::vector<GraphicsBackend> kWindowsWithoutVulkan = {
70 GraphicsBackend::Direct3D11,
71 GraphicsBackend::Direct3D12,
72 };
73 expect_true(
74 crimson::gpu::choose_backend(
75 NativeSurfacePlatform::Windows,
76 GraphicsBackendPreference::Auto,
77 kWindowsWithoutVulkan) == GraphicsBackend::Direct3D12,
78 "Windows auto preference falls back to Direct3D12");
79
80 const std::vector<GraphicsBackend> kMetalOnly = { GraphicsBackend::Metal };
81 expect_true(
82 crimson::gpu::choose_backend(NativeSurfacePlatform::MacOS, GraphicsBackendPreference::Auto, kMetalOnly) == GraphicsBackend::Metal,
83 "macOS auto preference chooses Metal");
84
85 const std::vector<GraphicsBackend> kVulkanOnly = { GraphicsBackend::Vulkan };
86 expect_true(
87 crimson::gpu::choose_backend(NativeSurfacePlatform::LinuxX11, GraphicsBackendPreference::Auto, kVulkanOnly) == GraphicsBackend::Vulkan,
88 "Linux auto preference chooses Vulkan");
89
90 expect_true(
91 !crimson::gpu::choose_backend(NativeSurfacePlatform::MacOS, GraphicsBackendPreference::Vulkan, kMetalOnly),
92 "unsupported explicit preference returns no backend");
93 expect_true(
94 !crimson::gpu::choose_backend(NativeSurfacePlatform::Unknown, GraphicsBackendPreference::Auto, kAllWindows),
95 "unknown platform returns no backend");
96}
