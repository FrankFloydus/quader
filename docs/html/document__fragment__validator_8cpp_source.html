86FragmentValidationResult validate_document_fragment(const DocumentFragment &fragment) {
87 FragmentValidationResult result;
88
89 if (fragment.mesh_objects.empty()) {
90 add_fragment_diagnostic(result.diagnostics,
91 IoSeverity::Warning,
92 IoDiagnosticCode::ValidationFailed,
93 "document fragment does not contain mesh objects",
94 "fragment");
95 }
96
97 for (std::size_t index = 0; index < fragment.mesh_objects.size(); ++index) {
98 const auto &object = fragment.mesh_objects[index];
99 const auto kSubject = object_subject(index, object);
100
101 if (!quader::document::is_finite(object.transform)) {
102 add_fragment_diagnostic(result.diagnostics,
103 IoSeverity::Error,
104 IoDiagnosticCode::ValidationFailed,
105 "imported mesh object has a non-finite transform",
106 kSubject);
107 }
108
109 if (object.material_index && *object.material_index >= fragment.materials.size()) {
110 add_fragment_diagnostic(result.diagnostics,
111 IoSeverity::Error,
112 IoDiagnosticCode::ValidationFailed,
113 "imported mesh object references a missing material",
114 kSubject);
115 }
116
117 const auto kMeshValidation = quader::mesh::validate_mesh(object.mesh);
118 for (const auto &issue : kMeshValidation.issues()) {
119 add_fragment_diagnostic(result.diagnostics,
120 IoSeverity::Error,
121 IoDiagnosticCode::MeshValidationFailed,
122 issue.diagnostic.message,
123 kSubject);
124 }
125 }
126
127 return result;
128}

130quader::foundation::Result<quader::document::Document, IoError> build_document_from_import(
131 ImportedDocument imported) {
132 IoDiagnosticList diagnostics;
133 append_import_diagnostics(imported, diagnostics);
134
135 if (imported.root_fragment.mesh_objects.empty()) {
136 diagnostics.add(IoDiagnostic{
137 IoSeverity::Error,
138 IoDiagnosticCode::ValidationFailed,
139 "cannot build a document from an empty import",
140 {},
141 {},
142 "document",
143 });
144 }
145
146 auto validation = validate_document_fragment(imported.root_fragment);
147 diagnostics.append(validation.diagnostics);
148 if (diagnostics.has_errors()) {
149 return quader::foundation::Result<quader::document::Document, IoError>::failure(
150 validation_error_from_diagnostics(diagnostics));
151 }
152
153 quader::document::Document document;
154 for (auto &object : imported.root_fragment.mesh_objects) {
155 auto created = document.create_mesh_object(std::move(object.name),
156 std::move(object.mesh),
157 object.transform);
158 if (!created) {
159 diagnostics.add(IoDiagnostic{
160 IoSeverity::Error,
161 IoDiagnosticCode::DocumentValidationFailed,
162 created.error().diagnostic.message,
163 {},
164 {},
165 "document",
166 });
167 return quader::foundation::Result<quader::document::Document, IoError>::failure(
168 validation_error_from_diagnostics(diagnostics));
169 }
170 }
171
172 return quader::foundation::Result<quader::document::Document, IoError>::success(
173 std::move(document));
174}
