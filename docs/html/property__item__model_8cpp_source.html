46[[nodiscard]] bool value_to_finite_float(const QVariant &value,
47 float &out_value) {
48 bool ok = false;
49 const double kParsed = value.toDouble(&ok);
50 if (!ok || !std::isfinite(kParsed)) {
51 return false;
52 }
53
54 out_value = static_cast<float>(kParsed);
55 return std::isfinite(out_value);
56}

58[[nodiscard]] bool
59replace_transform_component(quader::document::Transform &transform,
60 PropertyField field, float value) {
61 float *target = nullptr;
62 switch (field) {
63 case PropertyField::TranslationX:
64 target = &transform.translation.x;
65 break;
66 case PropertyField::TranslationY:
67 target = &transform.translation.y;
68 break;
69 case PropertyField::TranslationZ:
70 target = &transform.translation.z;
71 break;
72 case PropertyField::RotationX:
73 target = &transform.rotation_euler.x;
74 break;
75 case PropertyField::RotationY:
76 target = &transform.rotation_euler.y;
77 break;
78 case PropertyField::RotationZ:
79 target = &transform.rotation_euler.z;
80 break;
81 case PropertyField::ScaleX:
82 target = &transform.scale.x;
83 break;
84 case PropertyField::ScaleY:
85 target = &transform.scale.y;
86 break;
87 case PropertyField::ScaleZ:
88 target = &transform.scale.z;
89 break;
90 default:
91 return false;
92 }
93
94 if (*target == value) {
95 return false;
96 }
97
98 *target = value;
99 return true;
100}

131QVariant PropertyItemModel::data(const QModelIndex &model_index,
132 int role) const {
133 const auto *descriptor = descriptor_for_row(model_index.row());
134 if (descriptor == nullptr || !model_index.isValid()) {
135 return {};
136 }
137
138 switch (role) {
139 case Qt::DisplayRole:
140 if (is_label_column(model_index.column())) {
141 return descriptor->label;
142 }
143 if (is_value_column(model_index.column())) {
144 return descriptor->value;
145 }
146 return {};
147 case Qt::EditRole:
148 if (is_value_column(model_index.column())) {
149 return descriptor->value;
150 }
151 return {};
152 case Qt::ToolTipRole:
153 return descriptor->tooltip;
154 case property_item_roles::kPropertyPathRole:
155 return QVariant::fromValue(descriptor->path);
156 case property_item_roles::kValueKindRole:
157 return QVariant::fromValue(descriptor->kind);
158 case property_item_roles::kEditableRole:
159 return descriptor->editable;
160 case property_item_roles::kDescriptorRole:
161 return QVariant::fromValue(*descriptor);
162 default:
163 return {};
164 }
165}

167bool PropertyItemModel::setData(const QModelIndex &model_index,
168 const QVariant &value, int role) {
169 if (role != Qt::EditRole || !model_index.isValid() ||
170 !is_value_column(model_index.column())) {
171 return false;
172 }
173
174 const auto *descriptor = descriptor_for_row(model_index.row());
175 if (descriptor == nullptr || !descriptor->editable) {
176 return false;
177 }
178
179 return apply_descriptor_edit(*descriptor, value);
180}

182Qt::ItemFlags PropertyItemModel::flags(const QModelIndex &model_index) const {
183 if (!model_index.isValid()) {
184 return Qt::NoItemFlags;
185 }
186
187 auto item_flags = Qt::ItemIsEnabled | Qt::ItemIsSelectable;
188 const auto *descriptor = descriptor_for_row(model_index.row());
189 if (descriptor != nullptr && descriptor->editable &&
190 is_value_column(model_index.column())) {
191 item_flags |= Qt::ItemIsEditable;
192 }
193
194 return item_flags;
195}

220void PropertyItemModel::refresh_object(quader::document::ObjectId object) {
221 const bool kSelectedObjectChanged =
222 documents_.document().selection().references_object(object);
223 const bool kCurrentDescriptorsReferenceObject =
224 std::any_of(descriptors_.begin(), descriptors_.end(),
225 [object](const auto &descriptor) {
226 return descriptor_matches_object(descriptor, object);
227 });
228
229 if (kSelectedObjectChanged || kCurrentDescriptorsReferenceObject) {
230 rebuild_from_document();
231 }

243bool PropertyItemModel::apply_descriptor_edit(
244 const PropertyDescriptor &descriptor, const QVariant &value) {
245 const auto *object =
246 documents_.document().find_mesh_object(descriptor.path.object);
247 if (object == nullptr) {
248 return false;
249 }
250
251 if (descriptor.path.field == PropertyField::ObjectName) {
252 const QString kNewName = value.toString();
253 if (kNewName.isEmpty() || kNewName.toStdString() == object->name) {
254 return false;
255 }
256
257 auto result = documents_.execute_command(
258 std::make_unique<quader::commands::RenameObjectCommand>(
259 descriptor.path.object, kNewName.toStdString()));
260 return result.is_applied();
261 }
262
263 float numeric_value = 0.0F;
264 if (!value_to_finite_float(value, numeric_value)) {
265 return false;
266 }
267
268 auto transform = object->transform;
269 if (!replace_transform_component(transform, descriptor.path.field,
270 numeric_value)) {
271 return false;
272 }
273
274 auto result = documents_.execute_command(
275 std::make_unique<quader::commands::SetObjectTransformCommand>(
276 descriptor.path.object, transform));
277 return result.is_applied();
278}
