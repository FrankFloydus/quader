60std::vector<InstanceBatch> build_instance_batches(std::span<const DrawPacket> packets) {
61 std::vector<DrawPacket> sorted_packets;
62 sorted_packets.reserve(packets.size());
63 for (const DrawPacket &packet : packets) {
64 if (render_queue_supports_instancing(packet.queue)) {
65 sorted_packets.push_back(packet);
66 }
67 }
68 std::sort(sorted_packets.begin(), sorted_packets.end(), [](const DrawPacket &left, const DrawPacket &right) {
69 return instance_batch_key_less(instance_batch_key_for(left), instance_batch_key_for(right));
70 });
71
72 std::vector<InstanceBatch> batches;
73 for (const DrawPacket &packet : sorted_packets) {
74 const InstanceBatchKey kEy = instance_batch_key_for(packet);
75 if (batches.empty() || !(batches.back().key == kEy)) {
76 batches.push_back(InstanceBatch{ .key = kEy });
77 }
78 batches.back().instances.push_back(packet);
79 }
80 return batches;
81}

83FrameInstancingStats make_instancing_stats(std::span<const InstanceBatch> batches) noexcept {
84 FrameInstancingStats stats;
85 stats.batch_count = static_cast<std::uint32_t>(batches.size());
86 for (const InstanceBatch &batch : batches) {
87 const std::uint32_t kInstanceCount = static_cast<std::uint32_t>(batch.instances.size());
88 stats.submitted_instance_count += kInstanceCount;
89 if (kInstanceCount > 1) {
90 ++stats.instanced_batch_count;
91 stats.saved_draw_call_count += kInstanceCount - 1;
92 } else {
93 ++stats.single_draw_batch_count;
94 }
95 }
96 return stats;
97}
