24[[nodiscard]] crimson::BaseShaderId base_shader_for_alpha_mode(GltfAlphaMode mode) noexcept {
25 switch (mode) {
26 case GltfAlphaMode::Opaque:
27 return crimson::BaseShaderId::OpaquePbr;
28 case GltfAlphaMode::Mask:
29 return crimson::BaseShaderId::AlphaCutoutPbr;
30 case GltfAlphaMode::Blend:
31 return crimson::BaseShaderId::TransparentPbr;
32 }
33
34 return crimson::BaseShaderId::OpaquePbr;
35}

80void set_parameter_if_declared(
81 crimson::MaterialInstance &material,
82 const crimson::BaseShaderDefinition &definition,
83 std::string_view name,
84 crimson::MaterialParameterValue value) {
85 if (crimson::find_parameter_desc(definition, name) == nullptr) {
86 return;
87 }
88
89 for (crimson::MaterialParameter &parameter : material.parameters) {
90 if (parameter.name == name) {
91 parameter.value = value;
92 return;
93 }
94 }
95
96 material.parameters.push_back(crimson::MaterialParameter{
97 .name = std::string(name),
98 .value = value,
99 });
100}

102[[nodiscard]] const std::optional<GltfTextureRef> *texture_source_for_slot(
103 const GltfMaterialSource &source,
104 std::string_view slot_name) noexcept {
105 if (slot_name == "base_color") {
106 return &source.pbr.base_color_texture;
107 }
108 if (slot_name == "metallic_roughness") {
109 return &source.pbr.metallic_roughness_texture;
110 }
111 if (slot_name == "normal") {
112 return &source.normal_texture;
113 }
114 if (slot_name == "occlusion") {
115 return &source.occlusion_texture;
116 }
117 if (slot_name == "emissive") {
118 return &source.emissive_texture;
119 }
120
121 return nullptr;
122}

124void append_texture_slot_mappings(
125 GltfMaterialMapping &mapping,
126 const GltfMaterialSource &source,
127 const crimson::BaseShaderDefinition &definition) {
128 for (const crimson::MaterialTextureSlotDesc &slot : definition.texture_slots) {
129 const std::optional<GltfTextureRef> *texture = texture_source_for_slot(source, slot.name);
130 if (texture == nullptr || !texture->has_value()) {
131 continue;
132 }
133
134 mapping.texture_slots.push_back(TextureSlotMapping{
135 .slot_name = slot.name,
136 .source = texture->value(),
137 .expected_color_space = slot.color_space,
138 });
139 }
140}

142void preserve_unsupported_extensions(GltfMaterialMapping &mapping, const GltfMaterialSource &source) {
143 mapping.metadata.preserved_extensions = source.unsupported_extensions;
144 std::sort(mapping.metadata.preserved_extensions.begin(), mapping.metadata.preserved_extensions.end());
145 mapping.metadata.preserved_extensions.erase(
146 std::unique(mapping.metadata.preserved_extensions.begin(), mapping.metadata.preserved_extensions.end()),
147 mapping.metadata.preserved_extensions.end());
148
149 for (const std::string &extension : mapping.metadata.preserved_extensions) {
150 add_nonfatal_warning(
151 mapping,
152 "Preserved unsupported glTF material extension '" + extension + "'.",
153 extension);
154 }
155}

157void preserve_unsupported_occlusion(GltfMaterialMapping &mapping, const GltfMaterialSource &source) {
158 const bool kHasUnsupportedOcclusion = source.occlusion_texture.has_value() || source.occlusion_strength != 1.0F;
159 if (!kHasUnsupportedOcclusion) {
160 return;
161 }
162
163 add_nonfatal_warning(
164 mapping,
165 "Preserved glTF occlusion data because the selected Crimson base shader does not declare an occlusion slot.",
166 "occlusion");
167}

171quader::foundation::Result<GltfMaterialMapping, IoError> map_gltf_material_to_crimson(
172 const GltfMaterialSource &source,
173 const crimson::BaseShaderRegistry &registry) {
174 const crimson::BaseShaderId kBaseShaderId = base_shader_for_alpha_mode(source.alpha_mode);
175 const crimson::BaseShaderDefinition *definition = registry.find(kBaseShaderId);
176 if (definition == nullptr) {
177 return quader::foundation::Result<GltfMaterialMapping, IoError>::failure(make_mapping_error(
178 "Crimson base shader schema is unavailable for glTF material mapping.",
179 std::string(crimson::base_shader_id_name(kBaseShaderId)),
180 source.name));
181 }
182
183 GltfMaterialMapping mapping;
184 mapping.material = crimson::make_default_material_instance(*definition);
185 mapping.material.debug_name = source.name;
186 mapping.material.overrides.double_sided = source.double_sided;
187 mapping.metadata.source_name = source.name;
188 mapping.metadata.source_format = std::string(kGltfSourceFormat);
189 mapping.metadata.source_material_id = source.name;
190
191 set_parameter_if_declared(mapping.material, *definition, "base_color", source.pbr.base_color_factor);
192 set_parameter_if_declared(mapping.material, *definition, "metallic", source.pbr.metallic_factor);
193 set_parameter_if_declared(mapping.material, *definition, "roughness", source.pbr.roughness_factor);
194 set_parameter_if_declared(mapping.material, *definition, "normal_strength", source.normal_scale);
195 set_parameter_if_declared(mapping.material, *definition, "emissive_color", source.emissive_factor);
196 set_parameter_if_declared(mapping.material, *definition, "emissive_strength", 1.0F);
197 set_parameter_if_declared(mapping.material, *definition, "double_sided", source.double_sided);
198
199 if (crimson::find_parameter_desc(*definition, "occlusion_strength") != nullptr) {
200 set_parameter_if_declared(mapping.material, *definition, "occlusion_strength", source.occlusion_strength);
201 } else {
202 preserve_unsupported_occlusion(mapping, source);
203 }
204
205 if (source.alpha_mode == GltfAlphaMode::Mask) {
206 set_parameter_if_declared(mapping.material, *definition, "alpha_cutoff", source.alpha_cutoff);
207 }
208 if (source.alpha_mode == GltfAlphaMode::Blend) {
209 set_parameter_if_declared(mapping.material, *definition, "opacity", source.pbr.base_color_factor.a);
210 }
211
212 append_texture_slot_mappings(mapping, source, *definition);
213 preserve_unsupported_extensions(mapping, source);
214
215 const auto kValidation = crimson::validate_material_instance(registry, mapping.material);
216 if (!kValidation) {
217 IoError error = make_mapping_error(
218 "Crimson rejected mapped glTF material.",
219 kValidation.error().message + " " + kValidation.error().detail,
220 source.name);
221 error.related.append(mapping.diagnostics);
222 return quader::foundation::Result<GltfMaterialMapping, IoError>::failure(std::move(error));
223 }
224
225 return quader::foundation::Result<GltfMaterialMapping, IoError>::success(std::move(mapping));
226}
