34[[nodiscard]] const crimson::RenderPass *find_pass(const crimson::RenderGraph &graph, std::string_view name) {
35 for (const crimson::RenderPass &pass : graph.passes()) {

62TEST(Overlay, PrototypeGridIsOverlayCommandNotLitObject) {
63 const std::array<crimson::PrototypeCamera, 1> kCameras = { make_camera() };
64 const std::array<crimson::PrototypeViewportView, 1> kViews = {
65 crimson::PrototypeViewportView{
66 .rect = crimson::PrototypeViewportRect{ 0, 0, 320, 240 },
67 .camera_index = 0,
68 .debug_name = "Perspective",
69 },
70 };
71 const crimson::PrototypeViewportFrame kFrame{
72 .target_extent = crimson::ViewportExtent{ 640, 480, 1.0F },
73 .views = kViews,
74 .cameras = kCameras,
75 };
76
77 const crimson::FrameBuilder kBuilder;
78 auto built = kBuilder.build_prototype_snapshot(kFrame);
79 expect_true(built.has_value(), "valid prototype frame builds for overlay tests");
80 if (!built) {
81 return;
82 }
83
84 const crimson::FrameSnapshot kSnapshot = std::move(built).value();
85 expect_true(kSnapshot.objects().empty(), "empty document viewport has no default lit render object");
86 expect_true(kSnapshot.overlays().size() == 1, "prototype grid is emitted through overlay commands");
87 expect_true(kSnapshot.grid_overlay_payloads().size() == 1, "prototype grid payload is snapshot-owned");
88 expect_true(kSnapshot.line_overlay_payloads().empty(), "empty document viewport has no preview line payload");
89 expect_true(
90 kSnapshot.overlays()[0].primitive == crimson::OverlayPrimitive::Grid,
91 "prototype overlay command is a grid primitive");
92 expect_true(
93 kSnapshot.overlays()[0].base_shader == crimson::BaseShaderId::OverlayUnlit,
94 "prototype grid uses OverlayUnlit schema");
95 for (const crimson::RenderObject &object : kSnapshot.objects()) {
96 expect_true(
97 object.queue != crimson::RenderQueue::OverlayDepthTested && object.queue != crimson::RenderQueue::OverlayXRay && object.queue != crimson::RenderQueue::OverlayAlwaysOnTop,
98 "overlay queues are not represented as lit render objects");
99 }
100}

119 .view_index = 7,
120 .primitive = crimson::OverlayPrimitive::Grid,
121 .depth_mode = crimson::OverlayDepthMode::DepthTested,
122 .payload_offset = 0,
123 .payload_count = 1,
124 },
125 crimson::OverlayCommand{
126 .view_index = 7,
127 .primitive = crimson::OverlayPrimitive::Grid,
128 .depth_mode = crimson::OverlayDepthMode::XRay,
129 .opacity = 0.5F,
130 .payload_offset = 1,
131 .payload_count = 1,
132 },

135 .primitive = crimson::OverlayPrimitive::Grid,
136 .depth_mode = crimson::OverlayDepthMode::AlwaysOnTop,

138 .payload_count = 1,

144 expect_true(kLists.xray.grid_commands.size() == 1, "XRay grid command is bucketed");

146 expect_true(kLists.command_count() == 6, "overlay draw lists retain command and grid payload records");
147 expect_true(
148 near(kLists.xray.grid_commands[0].minor_color_linear_sdr[3], 0.5F),
149 "prepared grid payload alpha honors overlay command opacity");

178TEST(Overlay, OverlayPassesStayAfterToneMapAndOutOfHdrSceneColor) {
179 const crimson::RenderGraph kGraph = crimson::make_v1_correctness_render_graph(
180 crimson::ViewportExtent{ 800, 600, 1.0F });
181 const crimson::RenderPass *tone_map = find_pass(kGraph, "ToneMapPass");
182 const crimson::RenderPass *depth_tested = find_pass(kGraph, "OverlayDepthTestedPass");
183 const crimson::RenderPass *xray = find_pass(kGraph, "OverlayXRayPass");
184 const crimson::RenderPass *always_on_top = find_pass(kGraph, "OverlayAlwaysOnTopPass");
185 expect_true(tone_map != nullptr, "tone-map pass exists before overlays");
186 expect_true(depth_tested != nullptr && xray != nullptr && always_on_top != nullptr, "all overlay passes exist");
187
188 bool seen_tone_map = false;
189 for (const crimson::RenderPass &pass : kGraph.passes()) {
190 if (pass.name == "ToneMapPass") {
191 seen_tone_map = true;
192 }
193 if (pass.name == "OverlayDepthTestedPass" || pass.name == "OverlayXRayPass" || pass.name == "OverlayAlwaysOnTopPass") {
194 expect_true(seen_tone_map, "overlay pass is ordered after ToneMapPass");
195 expect_true(
196 !pass_uses_resource(pass, crimson::kHdrSceneColorTargetName),
197 "overlay pass never touches HdrSceneColor");
198 expect_true(
199 pass_uses_resource(pass, crimson::kToneMappedColorTargetName),
200 "overlay pass composites into ToneMappedColor");
201 }
202 }
