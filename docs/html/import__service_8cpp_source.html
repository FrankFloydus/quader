64void append_payload_diagnostics(const ImportResult &result, IoDiagnosticList &diagnostics) {
65 diagnostics.append(result.diagnostics);
66 std::visit(
67 [&diagnostics](const auto &payload) {
68 using Payload = std::decay_t<decltype(payload)>;
69 if constexpr (std::is_same_v<Payload, ImportedDocument>) {
70 append_imported_document_diagnostics(payload, diagnostics);
71 } else {
72 append_fragment_diagnostics(payload, diagnostics);
73 }
74 },
75 result.payload);
76}

95void validate_payload(const ImportRequest &request, ImportResult &result, IoDiagnosticList &diagnostics) {
96 std::visit(
97 [&request, &diagnostics](const auto &payload) {
98 using Payload = std::decay_t<decltype(payload)>;
99 if constexpr (std::is_same_v<Payload, ImportedDocument>) {
100 auto validation = validate_document_fragment(payload.root_fragment);
101 diagnostics.append(validation.diagnostics);
102 if (request.options.mode == DocumentImportMode::ReplaceDocument) {
103 append_empty_replace_error(request.path, payload.root_fragment, diagnostics);
104 }
105 } else {
106 auto validation = validate_document_fragment(payload);
107 diagnostics.append(validation.diagnostics);
108 if (request.options.mode == DocumentImportMode::ReplaceDocument) {
109 append_empty_replace_error(request.path, payload, diagnostics);
110 }
111 }
112 },
113 result.payload);
114}

121quader::foundation::Result<ImportResult, IoError> ImportService::import_file(
122 const ImportRequest &request) const {
123 const IImporter *importer = registry_.importer_for_path(request.path);
124 if (importer == nullptr) {
125 return quader::foundation::Result<ImportResult, IoError>::failure(make_io_error(
126 IoErrorCode::UnsupportedFormat,
127 IoDiagnosticCode::UnsupportedFormat,
128 "no importer is registered for '" + path_extension_without_dot(request.path) + "' files",
129 request.path));
130 }
131
132 auto imported = importer->import_file(request);
133 if (!imported) {
134 return imported;
135 }
136
137 ImportResult result = std::move(imported).value();
138 IoDiagnosticList diagnostics;
139 append_payload_diagnostics(result, diagnostics);
140 validate_payload(request, result, diagnostics);
141 result.diagnostics = diagnostics;
142
143 if (diagnostics.has_errors()) {
144 return quader::foundation::Result<ImportResult, IoError>::failure(
145 validation_error_from_diagnostics(diagnostics, request.path));
146 }
147
148 return quader::foundation::Result<ImportResult, IoError>::success(std::move(result));
149}
