41RenderProgramHandle GpuProgramCache::load_program(
42 const ShaderLibrary &library,
43 ShaderProgramId program,
44 ShaderTarget target,
45 RendererStatus &status) {
46 if (const auto kCached = handles_by_program_.find(program); kCached != handles_by_program_.end()) {
47 return kCached->second;
48 }
49
50 bgfx::ShaderHandle vertex = load_shader(library, program, ShaderStage::Vertex, target, status.diagnostics);
51 bgfx::ShaderHandle fragment = load_shader(library, program, ShaderStage::Fragment, target, status.diagnostics);
52 if (!bgfx::isValid(vertex) || !bgfx::isValid(fragment)) {
53 destroy_shader(vertex);
54 destroy_shader(fragment);
55 return {};
56 }
57
58 bgfx::ProgramHandle bgfx_program = bgfx::createProgram(vertex, fragment, true);
59 if (!bgfx::isValid(bgfx_program)) {
60 push_program_creation_error(status, program);
61 return {};
62 }
63
64 const RenderProgramHandle kHandle = programs_.create(GpuProgramResource{
65 .program = UniqueProgramHandle(bgfx_program),
66 });
67 handles_by_program_.emplace(program, kHandle);
68 return kHandle;
69}
