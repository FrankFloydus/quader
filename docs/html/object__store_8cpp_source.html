38quader::foundation::Result<void, DocumentError> ObjectStore::restore_mesh_object(MeshObject object) {
39 if (!object.id.is_valid() || object.id.index() >= slots_.size()) {
40 return quader::foundation::Result<void, DocumentError>::failure(
41 make_document_error(DocumentErrorCode::InvalidObject,
42 "cannot restore a mesh object with an invalid object id"));
43 }
44
45 auto &slot = slots_[object.id.index()];
46 if (slot.alive) {
47 return quader::foundation::Result<void, DocumentError>::failure(
48 make_document_error(DocumentErrorCode::ObjectAlreadyExists,
49 "cannot restore a mesh object over a live object slot"));
50 }
51
52 const auto kFreeIt = std::find(free_slots_.begin(), free_slots_.end(), object.id.index());
53 if (kFreeIt == free_slots_.end()) {
54 return quader::foundation::Result<void, DocumentError>::failure(
55 make_document_error(DocumentErrorCode::InvalidObject,
56 "cannot restore a mesh object into an unavailable object slot"));
57 }
58
59 free_slots_.erase(kFreeIt);
60 slot.generation = object.id.generation();
61 slot.alive = true;
62 slot.object = std::move(object);
63
64 return quader::foundation::Result<void, DocumentError>::success();
65}

67quader::foundation::Result<MeshObject, DocumentError> ObjectStore::remove_mesh_object(ObjectId id) {
68 auto *object = find_mesh_object(id);
69 if (object == nullptr) {
70 return quader::foundation::Result<MeshObject, DocumentError>::failure(
71 make_document_error(DocumentErrorCode::InvalidObject,
72 "cannot remove an unknown mesh object id"));
73 }
74
75 MeshObject removed = std::move(*object);
76 auto &slot = slots_[id.index()];
77 slot.object = MeshObject{};
78 slot.alive = false;
79 slot.generation = next_generation(slot.generation);
80 free_slots_.push_back(id.index());
81
82 return quader::foundation::Result<MeshObject, DocumentError>::success(std::move(removed));
83}

85MeshObject *ObjectStore::find_mesh_object(ObjectId id) noexcept {
86 if (!id.is_valid() || id.index() >= slots_.size()) {
87 return nullptr;
88 }
89
90 auto &slot = slots_[id.index()];
91 if (!slot.alive || slot.generation != id.generation()) {
92 return nullptr;
93 }
94
95 return &slot.object;
96}

115std::vector<ObjectId> ObjectStore::object_ids() const {
116 std::vector<ObjectId> ids;
117 ids.reserve(size());
118
119 for (quader::foundation::IdIndex index = 0; index < slots_.size(); ++index) {
120 const auto &slot = slots_[index];
121 if (slot.alive) {
122 ids.push_back(ObjectId{ index, slot.generation });
123 }
124 }
125

133quader::foundation::IdIndex ObjectStore::allocate_slot() {
134 if (!free_slots_.empty()) {
135 const auto kIndex = free_slots_.back();
136 free_slots_.pop_back();
137 slots_[kIndex].object = MeshObject{};
138 return kIndex;
139 }
140
141 const auto kIndex = static_cast<quader::foundation::IdIndex>(slots_.size());
142 slots_.push_back(ObjectSlot{});
143 return kIndex;
144}

146quader::foundation::IdGeneration ObjectStore::next_generation(

148 auto next = static_cast<quader::foundation::IdGeneration>(generation + 1U);
