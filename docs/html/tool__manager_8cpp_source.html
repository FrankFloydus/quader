53bool ToolManager::set_active_tool(ToolId id) {
54 const auto kIterator = tools_.find(id);
55 if (kIterator == tools_.end()) {
56 return false;
57 }
58
59 auto *next_tool = kIterator->second.get();
60 if (active_tool_ == next_tool) {
61 return true;
62 }
63
64 if (active_tool_ != nullptr) {
65 active_tool_->deactivate(context_);
66 }
67
68 active_tool_ = next_tool;
69 active_tool_->activate(context_);
70 return true;
71}
