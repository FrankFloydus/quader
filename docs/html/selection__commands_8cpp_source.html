30CommandResult SetSelectionCommand::execute(quader::document::Document &document) {
31 previous_selection_ = document.selection();
32 if (*previous_selection_ == selection_) {
33 previous_selection_.reset();
34 return CommandResult::rejected("selection command would not change the active selection");
35 }
36
37 auto selected = document.set_selection(selection_);
38 if (!selected) {
39 previous_selection_.reset();
40 return rejected_from_document_error(std::move(selected).error());
41 }
42
43 return CommandResult::applied();
44}
