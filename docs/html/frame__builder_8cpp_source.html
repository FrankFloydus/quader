54quader::foundation::Result<FrameSnapshot, RendererDiagnostic> FrameBuilder::build_prototype_snapshot(
55 const PrototypeViewportFrame &frame) const {
56 if (!is_valid_extent(frame.target_extent)) {
57 return quader::foundation::Result<FrameSnapshot, RendererDiagnostic>::failure(
58 invalid_snapshot_diagnostic("Prototype frames require a valid target extent."));
59 }
60
61 if (frame.views.empty()) {
62 return quader::foundation::Result<FrameSnapshot, RendererDiagnostic>::failure(
63 invalid_snapshot_diagnostic("Prototype frames require at least one view."));
64 }
65
66 if (frame.cameras.empty()) {
67 return quader::foundation::Result<FrameSnapshot, RendererDiagnostic>::failure(
68 invalid_snapshot_diagnostic("Prototype frames require at least one camera."));
69 }
70
71 std::vector<RenderView> views;
72 views.reserve(frame.views.size());
73 for (std::size_t index = 0; index < frame.views.size(); ++index) {
74 const PrototypeViewportView &source = frame.views[index];
75 if (!is_valid_prototype_view(source) || source.camera_index >= frame.cameras.size()) {
76 return quader::foundation::Result<FrameSnapshot, RendererDiagnostic>::failure(
77 invalid_snapshot_diagnostic("A prototype view has an invalid rectangle or camera index."));
78 }
79
80 const char *debug_name = source.debug_name == nullptr ? "Viewport" : source.debug_name;
81 views.push_back(RenderView{
82 .view_index = static_cast<std::uint32_t>(index),
83 .rect = RenderViewportRect{
84 .x = source.rect.x,
85 .y = source.rect.y,
86 .width = source.rect.width,
87 .height = source.rect.height,
88 },
89 .camera = render_camera_from(frame.cameras[source.camera_index]),
90 .debug_name = debug_name,
91 });
92 }
93
94 std::vector<OverlayCommand> overlays;
95 overlays.reserve(views.size() + frame.overlays.size());
96 std::vector<GridOverlayCommand> grid_overlay_payloads;
97 grid_overlay_payloads.reserve(views.size());
98 for (const RenderView &view : views) {
99 GridOverlayCommand grid = make_grid_overlay_for_view(view);
100 overlays.push_back(make_grid_overlay_command(
101 grid,
102 static_cast<std::uint32_t>(grid_overlay_payloads.size())));
103 grid_overlay_payloads.push_back(grid);
104 }
105 overlays.insert(overlays.end(), frame.overlays.begin(), frame.overlays.end());
106
107 std::vector<RenderMeshUpload> mesh_uploads(frame.mesh_uploads.begin(), frame.mesh_uploads.end());
108 std::vector<RenderObject> objects(frame.objects.begin(), frame.objects.end());
109 std::vector<LineOverlaySegment> line_overlay_payloads(
110 frame.line_overlay_payloads.begin(),
111 frame.line_overlay_payloads.end());
112
113 std::vector<RenderLight> lights;
114 lights.push_back(RenderLight{});
115
116 std::vector<PickingRequest> picking_requests(frame.picking_requests.begin(), frame.picking_requests.end());
117
118 ViewportSettings viewport_settings;
119 viewport_settings.exposure_ev100 = 0.0F;
120
121 return quader::foundation::Result<FrameSnapshot, RendererDiagnostic>::success(FrameSnapshot{
122 0,
123 frame.elapsed_seconds,
124 frame.target_extent,
125 std::move(views),
126 std::move(mesh_uploads),
127 std::move(objects),
128 std::move(lights),
129 RenderEnvironment{},
130 std::move(overlays),
131 std::move(grid_overlay_payloads),
132 std::move(line_overlay_payloads),
133 std::move(picking_requests),
134 viewport_settings,
135 DebugViewMode::FinalColor,
136 });
137}
