39[[nodiscard]] inline quader::math::Vec3 polygon_normal(std::span<const quader::math::Vec3> points) noexcept {
40 quader::math::Vec3 normal;
41
42 if (points.size() < 3) {
43 return normal;
44 }
45
46 for (std::size_t index = 0; index < points.size(); ++index) {
47 const auto kCurrent = points[index];
48 const auto kNext = points[(index + 1U) % points.size()];
49
50 normal.x += (kCurrent.y - kNext.y) * (kCurrent.z + kNext.z);
51 normal.y += (kCurrent.z - kNext.z) * (kCurrent.x + kNext.x);
52 normal.z += (kCurrent.x - kNext.x) * (kCurrent.y + kNext.y);
53 }
54
55 return normal;
56}
