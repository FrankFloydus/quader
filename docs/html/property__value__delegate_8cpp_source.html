44QWidget *PropertyValueDelegate::createEditor(QWidget *parent,
45 const QStyleOptionViewItem &option,
46 const QModelIndex &index) const {
47 (void)option;
48 if (!editable_for_index(index)) {
49 return nullptr;
50 }
51
52 switch (value_kind_for_index(index)) {
53 case PropertyValueKind::String: {
54 auto *editor = new QLineEdit(parent);
55 editor->setFrame(false);
56 return editor;
57 }
58 case PropertyValueKind::Float: {
59 const auto kDescriptor = descriptor_for_index(index);
60 auto *editor = new QDoubleSpinBox(parent);
61 editor->setFrame(false);
62 editor->setKeyboardTracking(false);
63 editor->setDecimals(kDescriptor.decimals);
64 editor->setMinimum(kDescriptor.minimum.isValid()
65 ? kDescriptor.minimum.toDouble()
66 : -std::numeric_limits<double>::max());
67 editor->setMaximum(kDescriptor.maximum.isValid()
68 ? kDescriptor.maximum.toDouble()
69 : std::numeric_limits<double>::max());
70 return editor;
71 }
72 case PropertyValueKind::Int:
73 case PropertyValueKind::Vector3:
74 case PropertyValueKind::ReadOnlyText:
75 return nullptr;
76 }
77
78 return nullptr;
79}

81void PropertyValueDelegate::setEditorData(QWidget *editor,
82 const QModelIndex &index) const {
83 if (auto *line_edit = qobject_cast<QLineEdit *>(editor)) {
84 line_edit->setText(index.data(Qt::EditRole).toString());
85 return;
86 }
87
88 if (auto *spin_box = qobject_cast<QDoubleSpinBox *>(editor)) {
89 spin_box->setValue(index.data(Qt::EditRole).toDouble());
90 return;
91 }
92
93 QStyledItemDelegate::setEditorData(editor, index);
94}

96void PropertyValueDelegate::setModelData(QWidget *editor,

98 const QModelIndex &index) const {
99 if (auto *line_edit = qobject_cast<QLineEdit *>(editor)) {
100 model->setData(index, line_edit->text(), Qt::EditRole);
101 return;
102 }
103
104 if (auto *spin_box = qobject_cast<QDoubleSpinBox *>(editor)) {
105 model->setData(index, spin_box->value(), Qt::EditRole);
106 return;
107 }
108
109 QStyledItemDelegate::setModelData(editor, model, index);
110}

112QString PropertyValueDelegate::displayText(const QVariant &value,
113 const QLocale &locale) const {
114 if (value.metaType().id() == QMetaType::Double ||
115 value.canConvert<double>()) {
116 bool ok = false;
117 const double kNumber = value.toDouble(&ok);
118 if (ok) {
119 return locale.toString(kNumber, 'f', 3);
120 }
121 }
122
123 return QStyledItemDelegate::displayText(value, locale);
124}
