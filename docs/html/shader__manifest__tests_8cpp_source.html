24 const crimson::gpu::ShaderLibrary &library,

26 crimson::gpu::ShaderStage stage,
27 crimson::gpu::ShaderTarget target,
28 std::string_view expected_path,

126TEST(ShaderManifest, RuntimeLookupNeverPointsToShaderSources) {
127 crimson::gpu::ShaderLibrary library("runtime/shaders");
128 const auto kBinary = library.shader_binary(
129 crimson::ShaderProgramId::PrototypeLitCube,
130 crimson::gpu::ShaderStage::Vertex,
131 crimson::gpu::ShaderTarget::Vulkan);
132
133 expect_true(kBinary.has_value(), "shader binary ref exists");
134 if (!kBinary.has_value()) {
135 return;
136 }
137 const auto &binary_ref = kBinary.value();
138 expect_true(binary_ref.relative_path.extension() == ".bin", "shader lookup returns compiled binaries only");
139 expect_true(binary_ref.relative_path.extension() != ".sc", "shader lookup never returns shader source files");
140}
