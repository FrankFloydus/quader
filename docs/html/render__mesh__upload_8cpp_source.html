47quader::foundation::Result<RenderMeshUploadRecord, RendererDiagnostic> validate_render_mesh_upload_desc(
48 const RenderMeshUploadDesc &desc) {
49 if (!is_valid_handle(desc.handle)) {
50 return quader::foundation::Result<RenderMeshUploadRecord, RendererDiagnostic>::failure(
51 upload_diagnostic("Prepared render mesh uploads require a valid public render mesh handle.", desc.handle));
52 }
53 if (desc.position_normal_interleaved.empty()) {
54 return quader::foundation::Result<RenderMeshUploadRecord, RendererDiagnostic>::failure(
55 upload_diagnostic("Prepared render mesh uploads require vertex data.", desc.handle));
56 }
57 if (desc.position_normal_interleaved.size() % 6 != 0) {
58 return quader::foundation::Result<RenderMeshUploadRecord, RendererDiagnostic>::failure(
59 upload_diagnostic("Position/normal upload data must be interleaved as 6 floats per vertex.", desc.handle));
60 }
61 if (desc.indices.empty()) {
62 return quader::foundation::Result<RenderMeshUploadRecord, RendererDiagnostic>::failure(
63 upload_diagnostic("Prepared render mesh uploads require index data.", desc.handle));
64 }
65 if ((desc.attributes & VertexAttributePosition) == 0) {
66 return quader::foundation::Result<RenderMeshUploadRecord, RendererDiagnostic>::failure(
67 upload_diagnostic("Prepared render mesh uploads require position attributes.", desc.handle));
68 }
69 if (!valid_bounds(desc.bounds)) {
70 return quader::foundation::Result<RenderMeshUploadRecord, RendererDiagnostic>::failure(
71 upload_diagnostic("Prepared render mesh uploads require finite non-empty bounds.", desc.handle));
72 }
73
74 return quader::foundation::Result<RenderMeshUploadRecord, RendererDiagnostic>::success(RenderMeshUploadRecord{
75 .handle = desc.handle,
76 .uploaded_revision = desc.revision,
77 .vertex_bytes = byte_size(desc.position_normal_interleaved),
78 .index_bytes = byte_size(desc.indices),
79 });
80}

82quader::foundation::Result<FrameUploadStats, RendererDiagnostic> RenderMeshUploadTracker::process_uploads(
83 std::span<const RenderMeshUploadDesc> uploads) {
84 FrameUploadStats stats;
85 for (const RenderMeshUploadDesc &desc : uploads) {
86 auto validated = validate_render_mesh_upload_desc(desc);
87 if (!validated) {
88 return quader::foundation::Result<FrameUploadStats, RendererDiagnostic>::failure(
89 std::move(validated).error());
90 }
91
92 RenderMeshUploadRecord record = validated.value();
93 auto existing = std::find_if(records_.begin(), records_.end(), [record](const RenderMeshUploadRecord &current) {
94 return current.handle == record.handle;
95 });
96
97 if (existing == records_.end()) {
98 records_.push_back(record);
99 ++stats.mesh_create_count;
100 stats.uploaded_vertex_bytes += record.vertex_bytes;
101 stats.uploaded_index_bytes += record.index_bytes;
102 continue;
103 }
104
105 if (existing->uploaded_revision == record.uploaded_revision) {
106 ++stats.skipped_clean_resource_count;
107 continue;
108 }
109
110 *existing = record;
111 ++stats.mesh_update_count;
112 stats.uploaded_vertex_bytes += record.vertex_bytes;
113 stats.uploaded_index_bytes += record.index_bytes;
114 }
115
116 return quader::foundation::Result<FrameUploadStats, RendererDiagnostic>::success(stats);
117}

119bool RenderMeshUploadTracker::destroy(RenderMeshHandle handle) noexcept {
120 const auto kExisting = std::find_if(records_.begin(), records_.end(), [handle](const RenderMeshUploadRecord &current) {
121 return current.handle == handle;
122 });
123 if (kExisting == records_.end()) {
124 return false;
125 }
126 records_.erase(kExisting);
127 return true;
128}
