281[[nodiscard]] bool create_post_process_resources(
282 PostProcessResources &resources,
283 RendererStatus &status) {
284 destroy_post_process_resources(resources);
285
286 resources.fullscreen_vertex_layout
287 .begin()
288 .add(bgfx::Attrib::Position, 3, bgfx::AttribType::Float)
289 .add(bgfx::Attrib::TexCoord0, 2, bgfx::AttribType::Float)
290 .end();
291 const std::array<FullscreenVertex, 3> kFullscreenTriangle =
292 make_fullscreen_triangle(runtime_screen_origin());
293 resources.fullscreen_vertex_buffer.reset(bgfx::createVertexBuffer(
294 bgfx::copy(kFullscreenTriangle.data(), sizeof(FullscreenVertex) * kFullscreenTriangle.size()),
295 resources.fullscreen_vertex_layout));
296 resources.hdr_scene_sampler.reset(bgfx::createUniform("s_hdrSceneColor", bgfx::UniformType::Sampler));
297 resources.tone_mapped_sampler.reset(bgfx::createUniform("s_toneMappedColor", bgfx::UniformType::Sampler));
298 resources.tone_map_params_uniform.reset(bgfx::createUniform("u_toneMapParams", bgfx::UniformType::Vec4));
299 resources.present_params_uniform.reset(bgfx::createUniform("u_presentParams", bgfx::UniformType::Vec4));
300
301 resources.ready = resources.fullscreen_vertex_buffer.valid() && resources.hdr_scene_sampler.valid() && resources.tone_mapped_sampler.valid() && resources.tone_map_params_uniform.valid() && resources.present_params_uniform.valid();
302 if (!resources.ready) {
303 push_diagnostic(status, RendererDiagnostic{
304 .severity = RendererDiagnosticSeverity::Error,
305 .code = RendererDiagnosticCode::ResourceCreationFailed,
306 .message = "Crimson failed to create post-process GPU resources.",
307 .subsystem = "gpu.post",
308 .resource_name = "PostProcessResources",
309 });
310 destroy_post_process_resources(resources);
311 }
312
313 return resources.ready;
314}

395[[nodiscard]] std::uint64_t state_for_packet(const PbrDrawPacket &packet, PbrSubmitMode mode) noexcept {
396 if (mode == PbrSubmitMode::DepthOnly) {
397 return BGFX_STATE_WRITE_Z | BGFX_STATE_DEPTH_TEST_LESS | BGFX_STATE_MSAA;
398 }
399
400 std::uint64_t state = BGFX_STATE_WRITE_RGB | BGFX_STATE_WRITE_A | BGFX_STATE_DEPTH_TEST_LEQUAL | BGFX_STATE_MSAA;
401 if (packet.queue == RenderQueue::Opaque || packet.queue == RenderQueue::AlphaCutout) {
402 state |= BGFX_STATE_WRITE_Z;
403 }
404 if (packet.queue == RenderQueue::Transparent) {
405 state |= BGFX_STATE_BLEND_FUNC(BGFX_STATE_BLEND_SRC_ALPHA, BGFX_STATE_BLEND_INV_SRC_ALPHA);
406 }
407 return state;
408}

410void configure_scene_view(
411 bgfx::ViewId view_id,
412 const RenderView &view,
413 const std::string &view_name,
414 ViewClearMode clear_mode,
415 bgfx::FrameBufferHandle framebuffer = BGFX_INVALID_HANDLE) {

428 bgfx::touch(view_id);
429}

464std::uint32_t submit_fullscreen_pass(
465 bgfx::ViewId view_id,
466 const std::string &view_name,
467 ViewportExtent extent,
468 bgfx::FrameBufferHandle framebuffer,
469 bgfx::TextureHandle source_texture,
470 bgfx::ProgramHandle program,
471 bgfx::UniformHandle sampler,
472 bgfx::UniformHandle params_uniform,
473 const std::array<float, 4> &params,
474 const PostProcessResources &resources) {
475 if (!resources.ready || !bgfx::isValid(source_texture) || !bgfx::isValid(program) || !bgfx::isValid(sampler) || !bgfx::isValid(params_uniform)) {
476 return 0;
477 }
478
479 bgfx::setViewName(view_id, view_name.c_str());
480 bgfx::setViewMode(view_id, bgfx::ViewMode::Sequential);
481 bgfx::setViewRect(
482 view_id,
483 0,
484 0,
485 safe_view_dimension(extent.width_px),
486 safe_view_dimension(extent.height_px));
487 bgfx::setViewClear(view_id, BGFX_CLEAR_NONE, 0x00000000, 1.0F, 0);
488 bgfx::setViewFrameBuffer(view_id, framebuffer);
489 float ortho[16];
490 bx::mtxOrtho(
491 ortho,
492 0.0F,
493 1.0F,
494 1.0F,
495 0.0F,
496 0.0F,
497 100.0F,
498 0.0F,
499 runtime_homogeneous_depth());
500 bgfx::setViewTransform(view_id, nullptr, ortho);
501 bgfx::touch(view_id);
502
503 float model[16];
504 bx::mtxIdentity(model);
505 bgfx::discard(BGFX_DISCARD_ALL);
506 bgfx::setTexture(0, sampler, source_texture);
507 bgfx::setUniform(params_uniform, params.data());
508 bgfx::setTransform(model);
509 bgfx::setVertexBuffer(0, resources.fullscreen_vertex_buffer.get(), 0, 3);
510 bgfx::setState(BGFX_STATE_WRITE_RGB | BGFX_STATE_WRITE_A);
511 bgfx::submit(view_id, program);
512 return 1;
513}

515std::uint32_t submit_pbr_packets(
516 bgfx::ViewId view_id,
517 std::span<const PbrDrawPacket> packets,
518 const BaseShaderRegistry &registry,
519 const MaterialSystem &materials,
520 const GpuMeshCache &mesh_cache,
521 const GpuProgramCache &program_cache,
522 const GpuMaterialCache &material_cache,
523 const PrototypeResources &resources,
524 PbrSubmitMode mode = PbrSubmitMode::Lit) {
525 std::uint32_t draw_calls = 0;
526 for (const PbrDrawPacket &packet : packets) {
527 const GpuMeshResource *mesh = mesh_cache.get(packet.mesh);
528 const BaseShaderDefinition *definition = registry.find(packet.base_shader);
529 const bgfx::ProgramHandle kProgram = program_cache.program(program_cache.cached_handle(packet.program));
530 if (mesh == nullptr || !mesh->valid() || definition == nullptr || !bgfx::isValid(kProgram)) {
531 continue;
532 }
533
534 bgfx::setTransform(packet.world_from_object.data());
535 bgfx::setUniform(resources.pbr_model_uniform, packet.world_from_object.data());
536 material_cache.bind_pbr_material(material_cache.material_block(materials, packet.material, *definition));
537 bgfx::setVertexBuffer(0, mesh->vertex_buffer.get());
538 bgfx::setIndexBuffer(mesh->index_buffer.get());
539 bgfx::setState(state_for_packet(packet, mode));
540 bgfx::submit(view_id, kProgram);
541 ++draw_calls;
542 }
543 return draw_calls;
544}

546std::uint32_t submit_grid_overlay_bucket(
547 const OverlayDrawBucket &bucket,
548 OverlayDepthMode depth_mode,
549 std::span<const RenderView> views,
550 const GpuOverlayRenderer &overlay_renderer,
551 const GpuProgramCache &program_cache,
552 RenderProgramHandle overlay_program,
553 bgfx::FrameBufferHandle framebuffer) {
554 std::uint32_t draw_calls = 0;
555 const bgfx::ProgramHandle kProgram = program_cache.program(overlay_program);
556 const std::uint32_t kQueueOffset = overlay_queue_offset(depth_mode);
557 for (std::size_t view_index = 0; view_index < views.size(); ++view_index) {
558 const RenderView &view = views[view_index];
559 const bgfx::ViewId kViewId = overlay_view_id(view_index, kQueueOffset);
560 const std::string kViewName = std::string(render_queue_name(render_queue_for_overlay_depth_mode(depth_mode))) + ":" + view.debug_name;
561 configure_scene_view(kViewId, view, kViewName, ViewClearMode::None, framebuffer);
562
563 for (const PreparedGridOverlayCommand &grid : bucket.grid_commands) {
564 if (grid.grid.view_index != view.view_index) {
565 continue;
566 }
567 draw_calls += overlay_renderer.submit_grid(kViewId, view, grid, kProgram);
568 }
569 }
570 return draw_calls;
571}

582 const bgfx::ProgramHandle kProgram = program_cache.program(line_program);
583 const std::uint32_t kQueueOffset = overlay_queue_offset(depth_mode);
584 for (std::size_t view_index = 0; view_index < views.size(); ++view_index) {
585 const RenderView &view = views[view_index];
586 const bgfx::ViewId kViewId = overlay_view_id(view_index, kQueueOffset);
587 const std::string kViewName = std::string(render_queue_name(render_queue_for_overlay_depth_mode(depth_mode))) + ":" + view.debug_name;
588 configure_scene_view(kViewId, view, kViewName, ViewClearMode::None, framebuffer);
589
590 for (const PreparedLineOverlayCommand &lines : bucket.line_commands) {
591 if (lines.command.view_index != view.view_index) {
592 continue;
593 }
594 draw_calls += overlay_renderer.submit_lines(kViewId, lines, kProgram);
595 }
596 }
597 return draw_calls;
598}

1031 FrameTimingStats timing_stats;

1033 if (!status_.initialized) {
1034 return quader::foundation::Result<FrameRenderResult, RendererDiagnostic>::failure(RendererDiagnostic{
1035 .severity = RendererDiagnosticSeverity::Error,
1036 .code = RendererDiagnosticCode::InvalidFrameSnapshot,
1037 .message = "Cannot render before Crimson is initialized.",
1038 .subsystem = "gpu",
1039 .resource_name = "GpuDevice",
1040 .frame_index = snapshot.frame_index(),
1041 });
1042 }
1043
1044 if (!is_valid_extent(snapshot.target_extent()) || snapshot.views().empty()) {
1045 return quader::foundation::Result<FrameRenderResult, RendererDiagnostic>::failure(
1046 invalid_snapshot_diagnostic(
1047 "Frame snapshots require a valid extent and at least one render view.",
1048 snapshot.frame_index()));

1050
1051 const ViewportExtent kTargetExtent = snapshot.target_extent();
1052 if (kTargetExtent.width_px != extent_.width_px || kTargetExtent.height_px != extent_.height_px || kTargetExtent.device_pixel_ratio != extent_.device_pixel_ratio) {
1053 if (!resize(kTargetExtent)) {
1054 const RendererStatus kCurrentStatus = status();

1056 .severity = RendererDiagnosticSeverity::Error,
1057 .code = RendererDiagnosticCode::ResizeFailed,
1058 .message = "Crimson failed to resize before rendering.",
1059 };
1060 if (!kCurrentStatus.diagnostics.empty()) {
1061 diagnostic = kCurrentStatus.diagnostics.back();

1064 }
1065 }
1066
1067 for (const RenderView &view : snapshot.views()) {
1068 if (!is_valid_render_view(view)) {
1069 return quader::foundation::Result<FrameRenderResult, RendererDiagnostic>::failure(
1070 invalid_snapshot_diagnostic("A render view has an invalid rectangle.", snapshot.frame_index()));
1071 }
1072 }
1073 const auto kUploadStart = std::chrono::steady_clock::now();
1074 if (!impl_->frame_resources.synchronize(graph, status_)) {
1075 RendererDiagnostic diagnostic{
1076 .severity = RendererDiagnosticSeverity::Error,
1077 .code = RendererDiagnosticCode::ResourceLifetimeError,
1078 .message = "Crimson failed to synchronize graph GPU frame resources.",
1079 .subsystem = "gpu.frame_resources",
1080 .resource_name = "V1FrameTargets",
1081 .frame_index = snapshot.frame_index(),
1082 };
1083 if (!status_.diagnostics.empty()) {
1084 diagnostic = status_.diagnostics.back();
1085 diagnostic.frame_index = snapshot.frame_index();
1086 }
1087 return quader::foundation::Result<FrameRenderResult, RendererDiagnostic>::failure(std::move(diagnostic));
1088 }
1089 timing_stats.cpu_resource_upload_us += elapsed_us_since(kUploadStart);
1090
1091 const auto kWidthPx = safe_view_dimension(kTargetExtent.width_px);
1092 const auto kHeightPx = safe_view_dimension(kTargetExtent.height_px);
1093 bgfx::setViewRect(kBackgroundView, 0, 0, kWidthPx, kHeightPx);
1094 bgfx::setViewClear(kBackgroundView, BGFX_CLEAR_NONE, kViewportClearColor, 1.0F, 0);

1098 const bgfx::FrameBufferHandle kSceneDepthFramebuffer = impl_->frame_resources.scene_depth_framebuffer();
1099 const bgfx::FrameBufferHandle kToneMappedColorFramebuffer =
1100 impl_->frame_resources.tone_mapped_color_framebuffer();
1101 const bgfx::FrameBufferHandle kToneMappedFramebuffer = impl_->frame_resources.tone_mapped_framebuffer();
1102 const bgfx::TextureHandle kHdrSceneTexture = impl_->frame_resources.texture(kHdrSceneColorTargetName);
1103 const bgfx::TextureHandle kToneMappedTexture = impl_->frame_resources.texture(kToneMappedColorTargetName);
1104
1105 GpuPickingFrameResult picking_frame;
1106 if (impl_->picking.ready()) {
1107 picking_frame = impl_->picking.submit_frame_requests(
1108 snapshot,
1109 impl_->mesh_cache,
1110 impl_->program_cache,
1111 impl_->unit_box_mesh,
1112 impl_->picking_program,
1113 impl_->completed_bgfx_frame);
1114 }
1115
1116 FrameQueueStats queue_stats;
1117 FrameCullingStats culling_stats;
1118 FramePacketStats packet_stats;
1119 FrameInstancingStats instancing_stats;
1120 FrameUploadStats upload_stats;
1121 for (const RenderMeshUpload &upload : snapshot.mesh_uploads()) {
1122 if (!impl_->mesh_cache.upload_mesh(upload.desc(), status_)) {
1123 RendererDiagnostic diagnostic{
1124 .severity = RendererDiagnosticSeverity::Error,
1125 .code = RendererDiagnosticCode::ResourceCreationFailed,
1126 .message = "Crimson failed to upload a frame mesh.",
1127 .subsystem = "gpu.mesh",
1128 .resource_name = "FrameMeshUpload",
1129 .frame_index = snapshot.frame_index(),
1130 };
1131 if (!status_.diagnostics.empty()) {
1132 diagnostic = status_.diagnostics.back();
1133 diagnostic.frame_index = snapshot.frame_index();
1134 }
1135 return quader::foundation::Result<FrameRenderResult, RendererDiagnostic>::failure(std::move(diagnostic));
1136 }
1137 }
1138 if (impl_->prototype_resources.ready) {
1139 const auto kSubmitStart = std::chrono::steady_clock::now();
1140 for (std::size_t index = 0; index < snapshot.views().size(); ++index) {

1144 kDepthPrepassViewId,
1145 view,

1148 kSceneDepthFramebuffer);

1153 impl_->material_system.registry(),
1154 impl_->material_system,
1155 view.camera,

1157 impl_->default_opaque_material,
1158 view_aspect_ratio(view));

1162 accumulate_packet_stats(packet_stats, packets.packets);
1163 const std::vector<DrawPacket> kLitPackets = collect_lit_packets(packets);
1164 const std::vector<InstanceBatch> kBatches = build_instance_batches(kLitPackets);
1165 accumulate_instancing_stats(instancing_stats, make_instancing_stats(kBatches));
1166
1167 const bgfx::ViewId kOpaqueViewId = scene_view_id(index, 0);
1168 configure_scene_view(
1169 kOpaqueViewId,
1170 view,
