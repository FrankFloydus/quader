33struct FakeRenderHost final : quader::ui::IViewportRenderHost {
34 int initialize_calls = 0;
35 int resize_calls = 0;
36 int render_calls = 0;
37 quader::ui::ViewportPixelSize initialized_size;
38 quader::ui::ViewportPixelSize resized_size;
39 quader::ui::ViewportLayoutMode last_layout = quader::ui::ViewportLayoutMode::Single;
40 std::size_t last_pane_count = 0;
41 std::size_t last_camera_count = 0;
42 bool last_animation_enabled = false;
43 std::optional<quader::ui::ViewportFrameStats> stats;
44 std::optional<quader::ui::ViewportDiagnosticsSnapshot> diagnostics;
45
46 quader::ui::ViewportRenderResult initialize_surface(
47 const quader::ui::NativeViewportSurface &,
48 quader::ui::ViewportPixelSize size,
49 double) override {
50 ++initialize_calls;
51 initialized_size = size;
52 stats = quader::ui::ViewportFrameStats{ size.width, size.height, 1, 60.0 };
53 return quader::ui::ViewportRenderResult::success(QStringLiteral("FakeRenderer"));
54 }
55
56 quader::ui::ViewportRenderResult resize_surface(quader::ui::ViewportPixelSize size, double) override {
57 ++resize_calls;
58 resized_size = size;
59 stats = quader::ui::ViewportFrameStats{ size.width, size.height, 1, 60.0 };
60 return quader::ui::ViewportRenderResult::success(QStringLiteral("FakeRenderer"));
61 }
62
63 quader::ui::ViewportRenderResult render_frame(const quader::ui::ViewportRenderRequest &request) override {
64 ++render_calls;
65 last_layout = request.layout_mode;
66 last_pane_count = request.panes.size();
67 last_camera_count = request.cameras.size();
68 last_animation_enabled = request.prototype_animation_enabled;
69 stats = quader::ui::ViewportFrameStats{
70 request.surface_size.width,
71 request.surface_size.height,
72 static_cast<int>(request.panes.size()),
73 60.0,
74 };
75 return quader::ui::ViewportRenderResult::success(QStringLiteral("FakeRenderer"));
76 }
77
78 void shutdown_surface() override {}
79
80 std::optional<quader::ui::ViewportFrameStats> latest_frame_stats() const override {
81 return stats;
82 }
83
84 std::optional<quader::ui::ViewportDiagnosticsSnapshot> latest_diagnostics() const override {
85 return diagnostics;
86 }
87
88 QString renderer_name() const override {
89 return QStringLiteral("FakeRenderer");
90 }
91};
