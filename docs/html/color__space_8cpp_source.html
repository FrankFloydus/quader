26[[nodiscard]] float srgb_channel_to_linear(float channel) noexcept {
27 const float kSrgb = clamp01(channel);
28 if (kSrgb <= 0.04045F) {
29 return kSrgb / 12.92F;
30 }
31
32 return std::pow((kSrgb + 0.055F) / 1.055F, 2.4F);
33}
34
35[[nodiscard]] float linear_channel_to_srgb(float channel) noexcept {
36 const float kLinear = clamp01(channel);
37 if (kLinear <= 0.0031308F) {
38 return kLinear * 12.92F;
39 }
40
41 return 1.055F * std::pow(kLinear, 1.0F / 2.4F) - 0.055F;
42}

64TextureColorEncoding texture_color_encoding(TextureDataRole role) noexcept {
65 switch (role) {
66 case TextureDataRole::BaseColor:
67 case TextureDataRole::Emissive:
68 return TextureColorEncoding::Srgb;
69 case TextureDataRole::MetallicRoughness:
70 case TextureDataRole::Normal:
71 case TextureDataRole::Occlusion:
72 case TextureDataRole::PickingId:
73 case TextureDataRole::Depth:
74 case TextureDataRole::ShadowLike:
75 return TextureColorEncoding::LinearData;
76 }
77
78 return TextureColorEncoding::LinearData;
