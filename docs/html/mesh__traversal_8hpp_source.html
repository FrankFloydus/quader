29[[nodiscard]] inline quader::foundation::Result<std::vector<VertexId>, MeshError> face_vertices(
30 const Polyhedron &mesh,
31 FaceId face) {
32 auto halfedges = mesh.face_halfedges(face);
33 if (!halfedges) {
34 return quader::foundation::Result<std::vector<VertexId>, MeshError>::failure(
35 std::move(halfedges).error());
36 }
37
38 std::vector<VertexId> vertices;
39 vertices.reserve(halfedges.value().size());
40
41 for (const auto kHalfedge : halfedges.value()) {
42 auto origin = mesh.halfedge_origin(kHalfedge);
43 if (!origin) {
44 return quader::foundation::Result<std::vector<VertexId>, MeshError>::failure(
45 std::move(origin).error());
46 }
47
48 vertices.push_back(origin.value());
49 }
50
51 return quader::foundation::Result<std::vector<VertexId>, MeshError>::success(std::move(vertices));
52}

61[[nodiscard]] inline quader::foundation::Result<std::vector<EdgeId>, MeshError> face_edges(

93[[nodiscard]] inline HalfedgeId find_halfedge(const Polyhedron &mesh,
94 VertexId origin,
95 VertexId target) noexcept {
96 if (!mesh.is_valid(origin) || !mesh.is_valid(target)) {
97 return HalfedgeId::invalid();
98 }
99
100 for (const auto kHalfedge : mesh.halfedge_ids()) {
101 auto halfedge_origin = mesh.halfedge_origin(kHalfedge);
102 if (!halfedge_origin || halfedge_origin.value() != origin) {
103 continue;
104 }
105
106 auto halfedge_target = mesh.halfedge_target(kHalfedge);
107 if (halfedge_target && halfedge_target.value() == target) {
108 return kHalfedge;
109 }
110 }
111
112 return HalfedgeId::invalid();
113}
