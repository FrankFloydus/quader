25[[nodiscard]] const crimson::RendererCapabilityStatus *find_status(
26 const std::vector<crimson::RendererCapabilityStatus> &statuses,
27 crimson::RendererCapability capability) noexcept {
28 for (const crimson::RendererCapabilityStatus &status : statuses) {
29 if (status.capability == capability) {
30 return &status;
31 }
32 }
33
34 return nullptr;
35}

54TEST(BackendCapability, BackendSelectionUsesV1PlatformPolicy) {
55 using crimson::GraphicsBackendPreference;
56 using crimson::NativeSurfacePlatform;
57 using crimson::gpu::GraphicsBackend;
58
59 const std::vector<GraphicsBackend> kAllWindows = {
60 GraphicsBackend::Direct3D11,
61 GraphicsBackend::Vulkan,
62 GraphicsBackend::Direct3D12,
63 };
64
65 expect_true(
66 crimson::gpu::choose_backend(NativeSurfacePlatform::Windows, GraphicsBackendPreference::Auto, kAllWindows) == GraphicsBackend::Vulkan,
67 "Windows Auto chooses Vulkan first");
68
69 const std::vector<GraphicsBackend> kWindowsWithoutVulkan = {
70 GraphicsBackend::Direct3D11,
71 GraphicsBackend::Direct3D12,
72 };
73 expect_true(
74 crimson::gpu::choose_backend(
75 NativeSurfacePlatform::Windows,
76 GraphicsBackendPreference::Auto,
77 kWindowsWithoutVulkan) == GraphicsBackend::Direct3D12,
78 "Windows Auto falls back to Direct3D12 before Direct3D11");
79
80 const std::vector<GraphicsBackend> kMetalOnly = { GraphicsBackend::Metal };
81 expect_true(
82 !crimson::gpu::choose_backend(NativeSurfacePlatform::MacOS, GraphicsBackendPreference::Vulkan, kMetalOnly),
83 "explicit unsupported backend preference fails");
84}

86TEST(BackendCapability, FullCapabilitySetPassesWithManualSrgbFallback) {
87 const crimson::DisplayConversionSettings kDisplay;
88 const crimson::gpu::GpuCapabilityValidation kValidation =
89 crimson::gpu::validate_v1_capabilities(full_capabilities(), kDisplay);
90
91 expect_true(kValidation.ok(), "full V1 capability set passes validation");
92 expect_true(kValidation.statuses.size() == 12, "validation reports every public capability");
93
94 const crimson::RendererCapabilityStatus *srgb_backbuffer = find_status(
95 kValidation.statuses,
96 crimson::RendererCapability::SrgbBackbuffer);
97 const crimson::RendererCapabilityStatus *manual_srgb = find_status(
98 kValidation.statuses,
99 crimson::RendererCapability::ManualSrgbFinalConversion);
100 expect_true(
101 srgb_backbuffer != nullptr && !srgb_backbuffer->supported,
102 "sRGB backbuffer may be unavailable when manual fallback is present");
103 expect_true(
104 manual_srgb != nullptr && manual_srgb->supported,
105 "manual final sRGB conversion fallback satisfies display conversion alternative");
106}

108TEST(BackendCapability, MissingRequiredCapabilityReportsFatalDiagnostic) {
109 crimson::gpu::GpuCapabilityInput input = full_capabilities();
110 input.instancing = false;
111
112 const crimson::gpu::GpuCapabilityValidation kValidation =
113 crimson::gpu::validate_v1_capabilities(input, crimson::DisplayConversionSettings{});
114
115 expect_true(!kValidation.ok(), "missing instancing fails V1 capability validation");
116 expect_true(!kValidation.diagnostics.empty(), "missing capability produces a diagnostic");
117 if (!kValidation.diagnostics.empty()) {
118 const crimson::RendererDiagnostic &diagnostic = kValidation.diagnostics.front();
119 expect_true(
120 diagnostic.severity == crimson::RendererDiagnosticSeverity::Fatal,
121 "missing required capability is fatal");
122 expect_true(
123 diagnostic.code == crimson::RendererDiagnosticCode::CapabilityMissing,
124 "missing required capability uses CapabilityMissing code");
125 expect_true(diagnostic.subsystem == "gpu", "capability diagnostic records gpu subsystem");
126 expect_true(
127 diagnostic.resource_name == "Instancing",
128 "capability diagnostic records missing capability name");
129 }
130}
