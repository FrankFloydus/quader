272inline bool has_error_diagnostic(const RendererStatus &status) {
273 for (const RendererDiagnostic &diagnostic : status.diagnostics) {
274 if (diagnostic.severity == RendererDiagnosticSeverity::Error || diagnostic.severity == RendererDiagnosticSeverity::Fatal) {
275 return true;
276 }
277 }
278
279 return false;
280}
