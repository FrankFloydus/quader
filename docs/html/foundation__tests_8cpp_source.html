35TEST(Foundation, StrongIdValidityEqualityAndHashing) {
36 using quader::foundation::Id;
37 using quader::foundation::kInvalidIdIndex;
38
39 static_assert(!std::is_convertible_v<Id<ObjectTag>, quader::foundation::IdIndex>);
40 static_assert(!std::is_same_v<Id<ObjectTag>, Id<OtherTag>>);
41
42 const Id<ObjectTag> kInvalid;
43 const Id<ObjectTag> kAlsoInvalid = Id<ObjectTag>::invalid();
44 const Id<ObjectTag> kFirst{ 7 };
45 const Id<ObjectTag> kFirstCopy{ 7 };
46 const Id<ObjectTag> kSecond{ 8 };
47 const Id<ObjectTag> kSentinel{ kInvalidIdIndex };
48
49 EXPECT_FALSE(kInvalid.is_valid());
50 EXPECT_FALSE(kAlsoInvalid.is_valid());
51 EXPECT_FALSE(kSentinel.is_valid());
52 EXPECT_TRUE(kFirst.is_valid());
53 EXPECT_EQ(kFirst.index(), 7U);
54 EXPECT_EQ(kFirst, kFirstCopy);
55 EXPECT_NE(kFirst, kSecond);
56
57 std::unordered_set<Id<ObjectTag>> ids;
58 ids.insert(kFirst);
59 EXPECT_EQ(ids.count(kFirstCopy), 1U);
60 EXPECT_EQ(ids.count(kSecond), 0U);
61}
