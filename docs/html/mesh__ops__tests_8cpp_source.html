28[[nodiscard]] quader::math::Vec3 average_points(std::span<const quader::math::Vec3> points) noexcept {
29 quader::math::Vec3 average;
30 for (quader::math::Vec3 point : points) {
31 average = average + point;
32 }
33 return points.empty() ? average : average / static_cast<float>(points.size());
34}

36[[nodiscard]] std::vector<quader::math::Vec3> face_points(
37 const quader::mesh::Polyhedron &mesh,
38 quader::mesh::FaceId face) {
39 auto vertices = quader::mesh::face_vertices(mesh, face);
40 EXPECT_TRUE(vertices);
41 std::vector<quader::math::Vec3> points;
42 if (!vertices) {
43 return points;
44 }
45
46 points.reserve(vertices.value().size());
47 for (const quader::mesh::VertexId kVertex : vertices.value()) {
48 auto position = mesh.vertex_position(kVertex);
49 EXPECT_TRUE(position);
50 if (position) {
51 points.push_back(position.value());
52 }
53 }
54 return points;
55}

57[[nodiscard]] quader::math::Vec3 mesh_center(const quader::mesh::Polyhedron &mesh) {
58 std::vector<quader::math::Vec3> points;
59 points.reserve(mesh.vertex_count());
60 for (const quader::mesh::VertexId kVertex : mesh.vertex_ids()) {
61 auto position = mesh.vertex_position(kVertex);
62 EXPECT_TRUE(position);
63 if (position) {
64 points.push_back(position.value());
65 }
66 }
67 return average_points(points);
68}

70void expect_faces_oriented_outward(const quader::mesh::Polyhedron &mesh) {
71 const quader::math::Vec3 kCenter = mesh_center(mesh);
72 for (const quader::mesh::FaceId kFace : mesh.face_ids()) {
73 const std::vector<quader::math::Vec3> kPoints = face_points(mesh, kFace);
74 ASSERT_GE(kPoints.size(), 3U);
75 const quader::math::Vec3 kNormal = quader::geometry::polygon_normal(kPoints);
76 const quader::math::Vec3 kFaceCenter = average_points(kPoints);
77 EXPECT_GT(quader::math::dot(kNormal, kFaceCenter - kCenter), 0.0F);
78 }
79}

160 kStart + kV,
161 kStart + kU + kV,
162 kStart + kU,
163 kStart + kHeight,
164 kStart + kV + kHeight,
165 kStart + kU + kV + kHeight,
166 kStart + kU + kHeight,
167 },
168 });
