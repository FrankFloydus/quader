17[[nodiscard]] std::uint8_t channel_delta(std::uint8_t actual, std::uint8_t expected) noexcept {
18 return actual > expected
19 ? static_cast<std::uint8_t>(actual - expected)
20 : static_cast<std::uint8_t>(expected - actual);

25std::size_t expected_rgba8_byte_count(const GoldenImage &image) noexcept {
26 if (image.width == 0 || image.height == 0) {
27 return 0;
28 }
29
30 const std::uint64_t kPixelCount = static_cast<std::uint64_t>(image.width) * image.height;
31 const std::uint64_t kByteCount = kPixelCount * 4ULL;
32 if (kByteCount > static_cast<std::uint64_t>(std::numeric_limits<std::size_t>::max())) {
33 return 0;
34 }
35
36 return static_cast<std::size_t>(kByteCount);
37}

39bool has_expected_rgba8_byte_count(const GoldenImage &image) noexcept {
40 const std::size_t kExpectedSize = expected_rgba8_byte_count(image);
41 return kExpectedSize != 0 && image.rgba8_srgb.size() == kExpectedSize;

44GoldenImageDiff GoldenImageComparator::compare(
45 const GoldenImage &actual,
46 const GoldenImage &expected,
47 GoldenTolerance tolerance) const {
48 GoldenImageDiff diff;
49
50 if (actual.width != expected.width || actual.height != expected.height) {
51 diff.dimensions_match = false;
52 diff.passed = false;
53 diff.message = "Image dimensions differ.";
54 return diff;
55 }
56
57 if (!has_expected_rgba8_byte_count(actual) || !has_expected_rgba8_byte_count(expected)) {
58 diff.pixel_buffer_sizes_match = false;
59 diff.passed = false;
60 diff.message = "Image RGBA8 buffers do not match their declared dimensions.";
61 return diff;
62 }
63
64 std::uint64_t channel_delta_sum = 0;
65 for (std::size_t pixel_offset = 0; pixel_offset < actual.rgba8_srgb.size(); pixel_offset += 4) {
66 bool pixel_changed = false;
67 for (std::size_t channel = 0; channel < 4; ++channel) {
68 const std::uint8_t kDelta = channel_delta(
69 actual.rgba8_srgb[pixel_offset + channel],
70 expected.rgba8_srgb[pixel_offset + channel]);
71 diff.max_channel_delta = kDelta > diff.max_channel_delta ? kDelta : diff.max_channel_delta;
72 channel_delta_sum += kDelta;
73 pixel_changed = pixel_changed || kDelta != 0;
74 }
75 if (pixel_changed) {
76 ++diff.changed_pixels;
77 }
78 }
79
80 diff.mean_channel_delta = static_cast<double>(channel_delta_sum) / static_cast<double>(actual.rgba8_srgb.size());
81 diff.passed = diff.max_channel_delta <= tolerance.max_channel_delta && diff.mean_channel_delta <= tolerance.max_mean_channel_delta && diff.changed_pixels <= tolerance.max_changed_pixels;
82 diff.message = diff.passed ? "Images are within tolerance." : "Images differ beyond tolerance.";
83 return diff;
84}
