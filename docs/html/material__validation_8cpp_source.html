46quader::foundation::Result<void, RendererDiagnostic> validate_material_instance(
47 const BaseShaderRegistry &registry,
48 const MaterialInstance &instance) {
49 const BaseShaderDefinition *definition = registry.find(instance.base_shader_id);
50 if (definition == nullptr) {
51 return quader::foundation::Result<void, RendererDiagnostic>::failure(material_diagnostic(
52 RendererDiagnosticCode::MaterialSchemaInvalid,
53 "Crimson material references an unknown base shader.",
54 std::string(base_shader_id_name(instance.base_shader_id)),
55 instance.debug_name));
56 }
57
58 std::vector<std::string> seen_parameters;
59 seen_parameters.reserve(instance.parameters.size());
60 for (const MaterialParameter &parameter : instance.parameters) {
61 if (contains_name(seen_parameters, parameter.name)) {
62 return quader::foundation::Result<void, RendererDiagnostic>::failure(material_diagnostic(
63 RendererDiagnosticCode::MaterialValidationFailed,
64 "Crimson material contains a duplicate parameter.",
65 parameter.name,
66 definition->name));
67 }
68 seen_parameters.push_back(parameter.name);
69
70 const MaterialParameterDesc *desc = find_parameter_desc(*definition, parameter.name);
71 if (desc == nullptr) {
72 return quader::foundation::Result<void, RendererDiagnostic>::failure(material_diagnostic(
73 RendererDiagnosticCode::MaterialValidationFailed,
74 "Crimson material contains a parameter that is not declared by its base shader.",
75 parameter.name,
76 definition->name));
77 }
78
79 const MaterialParameterKind kValueKind = material_parameter_value_kind(parameter.value);
80 if (kValueKind != desc->kind) {
81 return quader::foundation::Result<void, RendererDiagnostic>::failure(material_diagnostic(
82 RendererDiagnosticCode::MaterialValidationFailed,
83 "Crimson material parameter has the wrong value kind.",
84 parameter.name + " expected " + material_parameter_kind_name(desc->kind) + " got " + material_parameter_kind_name(kValueKind),
85 definition->name));
86 }
87 }
88
89 std::vector<std::string> seen_textures;
90 seen_textures.reserve(instance.textures.size());
91 for (const MaterialTextureBinding &texture : instance.textures) {
92 if (contains_name(seen_textures, texture.name)) {
93 return quader::foundation::Result<void, RendererDiagnostic>::failure(material_diagnostic(
94 RendererDiagnosticCode::MaterialValidationFailed,
95 "Crimson material contains a duplicate texture slot binding.",
96 texture.name,
97 definition->name));
98 }
99 seen_textures.push_back(texture.name);
100
101 if (find_texture_slot_desc(*definition, texture.name) == nullptr) {
102 return quader::foundation::Result<void, RendererDiagnostic>::failure(material_diagnostic(
103 RendererDiagnosticCode::MaterialValidationFailed,
104 "Crimson material contains a texture slot that is not declared by its base shader.",
105 texture.name,
106 definition->name));
107 }
108 }
109
110 return quader::foundation::Result<void, RendererDiagnostic>::success();
111}
