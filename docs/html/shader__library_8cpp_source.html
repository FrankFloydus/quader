48std::optional<ShaderBinaryRef> ShaderLibrary::shader_binary(
49 ShaderProgramId program,
50 ShaderStage stage,
51 ShaderTarget target) const {
52 const std::string_view kFileName = shader_file_name(program, stage);
53 if (kFileName.empty()) {
54 return std::nullopt;
55 }
56
57 return ShaderBinaryRef{
58 .program = program,
59 .stage = stage,
60 .target = target,
61 .relative_path = std::filesystem::path(shader_target_directory_name(target)) / kFileName,
62 };
63}

65bgfx::ShaderHandle load_shader(
66 const ShaderLibrary &library,
67 ShaderProgramId program,
68 ShaderStage stage,
69 ShaderTarget target,
70 std::vector<RendererDiagnostic> &diagnostics) {
71 const std::optional<ShaderBinaryRef> kBinary = library.shader_binary(program, stage, target);
72 if (!kBinary) {
73 diagnostics.push_back(RendererDiagnostic{
74 .severity = RendererDiagnosticSeverity::Error,
75 .code = RendererDiagnosticCode::ShaderManifestInvalid,
76 .message = "Crimson shader program is not present in the runtime manifest map.",
77 .detail = shader_resource_name(program, stage),
78 .subsystem = "shader",
79 .resource_name = shader_resource_name(program, stage),
80 });
81 return BGFX_INVALID_HANDLE;
82 }
83
84 const std::filesystem::path kPath = library.shader_root() / kBinary->relative_path;
85 if (!std::filesystem::is_directory(kPath.parent_path())) {
86 diagnostics.push_back(RendererDiagnostic{
87 .severity = RendererDiagnosticSeverity::Error,
88 .code = RendererDiagnosticCode::ShaderTargetMissing,
89 .message = "Crimson shader target folder is missing.",
90 .detail = kPath.parent_path().string(),
91 .subsystem = "shader",
92 .resource_name = std::string(shader_target_directory_name(target)),
93 });
94 return BGFX_INVALID_HANDLE;
95 }
96
97 std::ifstream file(kPath, std::ios::binary);
98 if (!file) {
99 diagnostics.push_back(RendererDiagnostic{
100 .severity = RendererDiagnosticSeverity::Error,
101 .code = RendererDiagnosticCode::ShaderFileMissing,
102 .message = "Crimson shader file is missing.",
103 .detail = kPath.string(),
104 .subsystem = "shader",
105 .resource_name = shader_resource_name(program, stage),
106 });
107 return BGFX_INVALID_HANDLE;
108 }
109
110 const std::vector<char> kBytes{
111 std::istreambuf_iterator<char>(file),
112 std::istreambuf_iterator<char>()
113 };
114 if (kBytes.empty()) {
115 diagnostics.push_back(RendererDiagnostic{
116 .severity = RendererDiagnosticSeverity::Error,
117 .code = RendererDiagnosticCode::ShaderFileMissing,
118 .message = "Crimson shader file is empty.",
119 .detail = kPath.string(),
120 .subsystem = "shader",
121 .resource_name = shader_resource_name(program, stage),
122 });
123 return BGFX_INVALID_HANDLE;
124 }
125
126 const bgfx::Memory *memory = bgfx::copy(kBytes.data(), static_cast<std::uint32_t>(kBytes.size()));
127 bgfx::ShaderHandle shader = bgfx::createShader(memory);
128 if (!bgfx::isValid(shader)) {
129 diagnostics.push_back(RendererDiagnostic{
130 .severity = RendererDiagnosticSeverity::Error,
131 .code = RendererDiagnosticCode::ShaderProgramCreationFailed,
132 .message = "Crimson failed to create a runtime shader handle.",
133 .detail = kPath.string(),
134 .subsystem = "gpu.shader",
135 .resource_name = shader_resource_name(program, stage),
136 });
137 return BGFX_INVALID_HANDLE;
138 }
139
140 const std::string kName = kPath.filename().string();
141 bgfx::setName(shader, kName.c_str());
142 return shader;
143}

186std::string_view shader_file_name(ShaderProgramId program, ShaderStage stage) noexcept {
187 switch (program) {
188 case ShaderProgramId::PrototypeLitCube:
189 return stage == ShaderStage::Vertex ? "vs_cube.bin" : "fs_cube.bin";
190 case ShaderProgramId::PrototypeGridOverlay:
191 return stage == ShaderStage::Vertex ? "vs_grid.bin" : "fs_grid.bin";
192 case ShaderProgramId::ToneMap:
193 return stage == ShaderStage::Vertex ? "fullscreen_triangle.bin" : "tone_map.bin";
194 case ShaderProgramId::Present:
195 return stage == ShaderStage::Vertex ? "fullscreen_triangle.bin" : "present.bin";
196 case ShaderProgramId::OpaquePbr:
197 return stage == ShaderStage::Vertex ? "opaque_pbr.vs.bin" : "opaque_pbr.fs.bin";
198 case ShaderProgramId::AlphaCutoutPbr:
199 return stage == ShaderStage::Vertex ? "alpha_cutout_pbr.vs.bin" : "alpha_cutout_pbr.fs.bin";
200 case ShaderProgramId::TransparentPbr:
201 return stage == ShaderStage::Vertex ? "transparent_pbr.vs.bin" : "transparent_pbr.fs.bin";
202 case ShaderProgramId::OverlayUnlit:
203 return stage == ShaderStage::Vertex ? "overlay_unlit.vs.bin" : "overlay_unlit.fs.bin";
204 case ShaderProgramId::OverlayLine:
205 return stage == ShaderStage::Vertex ? "overlay_line.vs.bin" : "overlay_line.fs.bin";
206 case ShaderProgramId::Picking:
207 return stage == ShaderStage::Vertex ? "picking.vs.bin" : "picking.fs.bin";
208 }
209
210 return "";
211}
