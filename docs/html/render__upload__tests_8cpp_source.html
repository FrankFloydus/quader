63TEST(RenderUpload, UploadTrackerCountsCreateCleanSkipAndUpdate) {
64 crimson::RenderMeshUploadTracker tracker;
65
66 const std::array kFirstUploads = { upload_desc(crimson::RenderMeshRevision{ 1, 1, 1 }) };
67 const auto kFirst = tracker.process_uploads(kFirstUploads);
68 expect_true(kFirst.has_value(), "first prepared upload is accepted");
69 expect_true(kFirst && kFirst.value().mesh_create_count == 1, "new revision creates an upload record");
70 expect_true(kFirst && kFirst.value().uploaded_vertex_bytes == 72, "vertex upload bytes are counted");
71 expect_true(kFirst && kFirst.value().uploaded_index_bytes == 12, "index upload bytes are counted");
72
73 const std::array kCleanUploads = { upload_desc(crimson::RenderMeshRevision{ 1, 1, 1 }) };
74 const auto kClean = tracker.process_uploads(kCleanUploads);
75 expect_true(kClean.has_value(), "clean repeated upload is accepted");
76 expect_true(kClean && kClean.value().skipped_clean_resource_count == 1, "same revision increments clean skip counter");
77 expect_true(kClean && kClean.value().uploaded_vertex_bytes == 0, "clean upload does not count bytes");
78
79 const std::array kChangedUploads = { upload_desc(crimson::RenderMeshRevision{ 2, 1, 1 }) };
80 const auto kChanged = tracker.process_uploads(kChangedUploads);
81 expect_true(kChanged.has_value(), "changed prepared upload is accepted");
82 expect_true(kChanged && kChanged.value().mesh_update_count == 1, "changed revision schedules an update");
83 expect_true(tracker.find(crimson::RenderMeshHandle{ 1, 1 }) != nullptr, "upload tracker stores latest record");
84}

86TEST(RenderUpload, InvalidUploadsReturnStructuredDiagnostics) {
87 crimson::RenderMeshUploadDesc invalid = upload_desc(crimson::RenderMeshRevision{ 1, 1, 1 });
88 invalid.position_normal_interleaved = {};
89
90 const auto kResult = crimson::validate_render_mesh_upload_desc(invalid);
91 expect_true(!kResult, "invalid upload is rejected");
92 expect_true(
93 !kResult && kResult.error().code == crimson::RendererDiagnosticCode::UploadSkippedInvalidResource,
94 "invalid upload uses upload-specific diagnostic code");
95 expect_true(!kResult && crimson::has_structured_context(kResult.error()), "invalid upload diagnostic is structured");
96}

98TEST(RenderUpload, DestroyRemovesTrackedRecord) {
99 crimson::RenderMeshUploadTracker tracker;
100 const std::array kUploads = { upload_desc(crimson::RenderMeshRevision{ 1, 1, 1 }) };
101 (void)tracker.process_uploads(kUploads);
102 expect_true(tracker.live_record_count() == 1, "tracker has one record before destroy");
103 expect_true(tracker.destroy(crimson::RenderMeshHandle{ 1, 1 }), "tracked record can be destroyed");
104 expect_true(tracker.live_record_count() == 0, "destroy removes tracked record");
105 expect_true(!tracker.destroy(crimson::RenderMeshHandle{ 1, 1 }), "destroying a stale record returns false");
