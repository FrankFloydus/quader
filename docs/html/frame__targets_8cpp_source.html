88std::vector<RenderResourceDesc> make_v1_frame_target_resource_descs(
89 ViewportExtent extent,
90 PickingIdTargetFormat picking_format) {
91 std::vector<RenderResourceDesc> resources;
92 const auto kTargets = v1_frame_target_descriptors(picking_format);
93 resources.reserve(kTargets.size());
94 for (const FrameTargetDescriptor &target : kTargets) {
95 resources.push_back(to_resource_desc(target, extent));
96 }
97 return resources;
98}
