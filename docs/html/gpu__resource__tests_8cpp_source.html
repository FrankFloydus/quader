145TEST(GpuResource, FrameResourcesFollowRenderGraphTargetGenerations) {
146 crimson::RenderGraph graph = crimson::make_v1_correctness_render_graph(
147 crimson::ViewportExtent{ 640, 480, 1.0F });
148 crimson::gpu::GpuFrameResources resources;
149
150 resources.synchronize_descriptors(graph);
151 const crimson::gpu::GpuFrameTargetRecord *hdr = resources.find(crimson::kHdrSceneColorTargetName);
152 expect_true(hdr != nullptr, "GPU frame resources track HDR scene target");
153 expect_true(hdr != nullptr && hdr->recreate_count == 1, "initial target sync creates the HDR target");
154 expect_true(hdr != nullptr && !hdr->external(), "HDR scene target is GPU-owned");
155 expect_true(
156 hdr != nullptr && !hdr->runtime_ready(),
157 "descriptor-only frame resource sync does not claim runtime handles");
158 const crimson::gpu::GpuFrameTargetRecord *backbuffer = resources.find(crimson::kBackbufferColorTargetName);
159 expect_true(backbuffer != nullptr && backbuffer->external(), "backbuffer color remains external");
160 expect_true(backbuffer != nullptr && backbuffer->runtime_ready(), "external backbuffer target is runtime-ready by contract");
161 expect_true(resources.graph_resize_generation() == graph.resize_generation(), "frame resources remember graph generation");
162
163 resources.synchronize_descriptors(graph);
164 hdr = resources.find(crimson::kHdrSceneColorTargetName);
165 expect_true(hdr != nullptr && hdr->recreate_count == 1, "unchanged graph does not recreate targets");
166
167 graph.resize(crimson::ViewportExtent{ 1280, 720, 1.0F });
168 resources.synchronize_descriptors(graph);
169 hdr = resources.find(crimson::kHdrSceneColorTargetName);
170 expect_true(hdr != nullptr && hdr->recreate_count == 2, "resize generation recreates dependent targets");
171 expect_true(
172 hdr != nullptr && hdr->desc.extent.width_px == 1280,
173 "GPU frame target descriptor follows resized graph extent");
174}
