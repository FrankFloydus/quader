22[[nodiscard]] std::uint32_t parse_uint_arg(
23 int argc,
24 char **argv,
25 std::string_view name,
26 std::uint32_t fallback) {
27 const std::string kPrefix = std::string(name) + "=";
28 for (int index = 1; index < argc; ++index) {
29 const std::string_view kArg(argv[index]);
30 if (kArg.rfind(kPrefix, 0) != 0) {
31 continue;
32 }
33 try {
34 return static_cast<std::uint32_t>(std::stoul(std::string(kArg.substr(kPrefix.size()))));
35 } catch (...) {
36 return fallback;
37 }
38 }
39 return fallback;
40}

42[[nodiscard]] std::string escape_json(std::string_view value) {
43 std::string escaped;
44 escaped.reserve(value.size());
45 for (const char kC : value) {
46 if (kC == '"' || kC == '\\') {
47 escaped.push_back('\\');
48 }
49 escaped.push_back(kC);
50 }
51 return escaped;
52}

56BenchmarkRunConfig parse_benchmark_config(int argc, char **argv) {

66BenchmarkResult run_benchmark(std::string name, const BenchmarkRunConfig &config, const BenchmarkBody &body) {
67 for (std::uint32_t iteration = 0; iteration < config.warmup_iterations; ++iteration) {
68 (void)body();
69 }
70
71 std::vector<std::uint64_t> samples;
72 samples.reserve(config.measured_iterations);
73 std::vector<crimson::RendererCounter> counters;
74 for (std::uint32_t iteration = 0; iteration < config.measured_iterations; ++iteration) {
75 const auto kStart = std::chrono::steady_clock::now();
76 counters = body();
77 const auto kElapsed = std::chrono::steady_clock::now() - kStart;
78 samples.push_back(static_cast<std::uint64_t>(
79 std::chrono::duration_cast<std::chrono::microseconds>(kElapsed).count()));
80 }
81
82 std::sort(samples.begin(), samples.end());
83 return BenchmarkResult{
84 .name = std::move(name),
85 .iterations = config.measured_iterations,
86 .median_us = samples[samples.size() / 2],
87 .min_us = samples.front(),
88 .max_us = samples.back(),
89 .counters = std::move(counters),
90 };
91}

93void print_jsonl(const BenchmarkResult &result) {
94 std::cout << "{\"name\":\"" << escape_json(result.name)
95 << "\",\"iterations\":" << result.iterations
96 << ",\"median_us\":" << result.median_us
97 << ",\"min_us\":" << result.min_us
98 << ",\"max_us\":" << result.max_us
99 << ",\"counters\":{";
100 for (std::size_t index = 0; index < result.counters.size(); ++index) {
101 const crimson::RendererCounter &counter = result.counters[index];
102 if (index != 0) {
103 std::cout << ',';
104 }
105 std::cout << "\"" << escape_json(std::string(crimson::renderer_counter_domain_name(counter.domain)))
106 << "." << escape_json(counter.name) << "\":" << counter.value;
107 }
108 std::cout << "}}\n";
109}
