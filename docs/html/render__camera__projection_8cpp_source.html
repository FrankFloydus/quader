46RenderCameraMatrices render_camera_matrices(
47 const RenderCamera &camera,
48 float aspect_ratio,
49 bool homogeneous_depth) {
50 RenderCameraMatrices matrices;
51 const float kAspect = std::max(0.0001F, aspect_ratio);
52
53 bx::mtxLookAt(matrices.view.data(), to_bx(camera.eye), to_bx(camera.target), to_bx(camera.up));
54 if (camera.projection == CameraProjection::Orthographic) {
55 const float kExtent = std::max(0.01F, camera.orthographic_height_m) * 0.5F;
56 bx::mtxOrtho(
57 matrices.projection.data(),
58 -kExtent * kAspect,
59 kExtent * kAspect,
60 -kExtent,
61 kExtent,
62 camera.near_plane_m,
63 camera.far_plane_m,
64 0.0F,
65 homogeneous_depth);
66 return matrices;
67 }
68
69 bx::mtxProj(
70 matrices.projection.data(),
71 camera.vertical_fov_degrees,
72 kAspect,
73 camera.near_plane_m,
74 camera.far_plane_m,
75 homogeneous_depth);
76 return matrices;
77}

79RenderCameraRay render_camera_ray_from_viewport_point(
80 const RenderCamera &camera,
81 RenderCameraViewportSize viewport_size,
82 quader::math::Vec2 local_point_px,
83 bool homogeneous_depth) {
84 const float kWidth = std::max(1.0F, viewport_size.width_px);
85 const float kHeight = std::max(1.0F, viewport_size.height_px);
86 const float kAspect = kWidth / kHeight;
87 const RenderCameraMatrices kMatrices = render_camera_matrices(camera, kAspect, homogeneous_depth);
88
89 float view_projection[16];
90 bx::mtxMul(view_projection, kMatrices.view.data(), kMatrices.projection.data());
91
92 float inverse_view_projection[16];
93 bx::mtxInverse(inverse_view_projection, view_projection);
94
95 const float kNdcX = (local_point_px.x / kWidth) * 2.0F - 1.0F;
96 const float kNdcY = ((kHeight - local_point_px.y) / kHeight) * 2.0F - 1.0F;
97 const bx::Vec3 kPickEye = bx::mulH({ kNdcX, kNdcY, 0.0F }, inverse_view_projection);
98 const bx::Vec3 kPickAt = bx::mulH({ kNdcX, kNdcY, 1.0F }, inverse_view_projection);
99 const quader::math::Vec3 kOrigin = from_bx(kPickEye);
100 const quader::math::Vec3 kDirection = normalized_or(from_bx(kPickAt) - kOrigin, camera.forward);
101 if (camera.projection == CameraProjection::Orthographic) {
102 return RenderCameraRay{
103 .origin = kOrigin,
104 .direction = normalized_or(camera.forward, { 0.0F, 0.0F, -1.0F }),
105 };
106 }
107
108 return RenderCameraRay{
109 .origin = camera.eye,
110 .direction = kDirection,
111 };
112}
