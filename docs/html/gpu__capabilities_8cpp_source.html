34[[nodiscard]] bool contains_backend(
35 std::span<const GraphicsBackend> supported_backends,
36 GraphicsBackend backend) noexcept {
37 return std::find(supported_backends.begin(), supported_backends.end(), backend) != supported_backends.end();
38}

40[[nodiscard]] std::optional<GraphicsBackend> backend_for_preference(GraphicsBackendPreference preference) noexcept {
41 switch (preference) {
42 case GraphicsBackendPreference::Auto:
43 return std::nullopt;
44 case GraphicsBackendPreference::Vulkan:
45 return GraphicsBackend::Vulkan;
46 case GraphicsBackendPreference::Metal:
47 return GraphicsBackend::Metal;
48 case GraphicsBackendPreference::Direct3D12:
49 return GraphicsBackend::Direct3D12;
50 case GraphicsBackendPreference::Direct3D11:
51 return GraphicsBackend::Direct3D11;
52 }
53
54 return std::nullopt;
55}

57[[nodiscard]] std::span<const GraphicsBackend> priority_for_platform(NativeSurfacePlatform platform) noexcept {
58 switch (platform) {
59 case NativeSurfacePlatform::Windows:
60 return kWindowsBackendPriority;
61 case NativeSurfacePlatform::LinuxX11:
62 case NativeSurfacePlatform::LinuxWayland:
63 return kLinuxBackendPriority;
64 case NativeSurfacePlatform::MacOS:
65 return kMacBackendPriority;
66 case NativeSurfacePlatform::Unknown:
67 return {};
68 }
69
70 return {};
71}

88 return RendererDiagnostic{

98void push_required_status(
99 GpuCapabilityValidation &validation,
100 RendererCapability capability,
101 bool supported,
102 std::string supported_detail,
103 std::string missing_detail) {
104 push_status(
105 validation.statuses,
106 capability,
107 supported,
108 supported ? std::move(supported_detail) : missing_detail);
109
110 if (!supported) {
111 validation.diagnostics.push_back(capability_missing_diagnostic(capability, std::move(missing_detail)));

121std::optional<GraphicsBackend> choose_backend(
122 NativeSurfacePlatform platform,
123 GraphicsBackendPreference preference,
124 std::span<const GraphicsBackend> supported_backends) noexcept {
125 if (std::optional<GraphicsBackend> preferred_backend = backend_for_preference(preference)) {
126 return contains_backend(supported_backends, *preferred_backend)
127 ? preferred_backend
128 : std::nullopt;
129 }
130
131 for (GraphicsBackend backend : priority_for_platform(platform)) {
132 if (contains_backend(supported_backends, backend)) {
133 return backend;
134 }
135 }
136
137 return std::nullopt;
138}

153GpuCapabilityValidation validate_v1_capabilities(
154 const GpuCapabilityInput &input,
155 const DisplayConversionSettings &display_conversion) {
156 GpuCapabilityValidation validation;
157 validation.statuses.reserve(12);
158
159 push_required_status(
160 validation,
161 RendererCapability::Instancing,
162 input.instancing,
163 "Runtime reports instanced rendering support.",
164 "Instancing is required for V1 draw packet batching.");
165 push_required_status(
166 validation,
167 RendererCapability::Texture2D,
168 input.texture_2d,
169 "Runtime reports 2D texture support.",
170 "2D textures are required for V1 materials and framebuffer resources.");
171 push_required_status(
172 validation,
173 RendererCapability::TextureCube,
174 input.texture_cube,
175 "Runtime reports cube texture support.",
176 "Cube textures are required by the Crimson V1 renderer contract.");
177 push_required_status(
178 validation,
179 RendererCapability::FloatingPointRenderTarget,
180 input.floating_point_render_target,
181 "Runtime reports RGBA16F framebuffer support.",
182 "Floating-point render targets are required for the linear HDR scene target.");
183 push_required_status(
184 validation,
185 RendererCapability::RenderToTexture,
186 input.render_to_texture,
187 "Runtime reports texture framebuffer support.",
188 "Render-to-texture is required for V1 scene, picking, and final targets.");
189 push_required_status(
190 validation,
191 RendererCapability::DepthTexture,
192 input.depth_texture,
193 "Runtime reports a usable depth framebuffer format.",
194 "Depth texture framebuffer support is required by V1 depth-aware passes.");
195 push_required_status(
196 validation,
197 RendererCapability::SrgbTextureSampling,
198 input.srgb_texture_sampling,
199 "Runtime reports sRGB 2D texture sampling support.",
200 "sRGB texture sampling is required for base color and emissive texture slots.");
201
202 const bool kSrgbBackbuffer = display_conversion.prefer_srgb_backbuffer && input.srgb_backbuffer;
203 push_status(
204 validation.statuses,
205 RendererCapability::SrgbBackbuffer,
206 kSrgbBackbuffer,
207 kSrgbBackbuffer
208 ? "Runtime reports sRGB backbuffer support."
209 : "sRGB backbuffer support is unavailable or not requested.");
210
211 const bool kManualSrgb = display_conversion.allow_manual_final_srgb && input.manual_srgb_final_conversion;
212 push_status(
213 validation.statuses,
214 RendererCapability::ManualSrgbFinalConversion,
215 kManualSrgb,
216 kManualSrgb
217 ? "Manual final linear-to-sRGB conversion fallback is enabled."
218 : "Manual final linear-to-sRGB conversion fallback is disabled.");
219 if (!kSrgbBackbuffer && !kManualSrgb) {
220 validation.diagnostics.push_back(capability_missing_diagnostic(
221 RendererCapability::ManualSrgbFinalConversion,
222 "Crimson requires either an sRGB backbuffer or the manual final conversion fallback."));
223 }
224
225 push_status(
226 validation.statuses,
227 RendererCapability::IntegerPickingTarget,
228 input.integer_picking_target,
229 input.integer_picking_target
230 ? "Runtime reports R32U framebuffer support for picking IDs."
231 : "R32U picking target is unavailable; RGBA8 data fallback may be used.");
232 push_status(
233 validation.statuses,
234 RendererCapability::Rgba8PickingTarget,
235 input.rgba8_picking_target,
236 input.rgba8_picking_target
237 ? "Runtime reports RGBA8 framebuffer support for picking fallback."
238 : "RGBA8 picking fallback target is unavailable.");
239 if (!input.integer_picking_target && !input.rgba8_picking_target) {
240 validation.diagnostics.push_back(capability_missing_diagnostic(
241 RendererCapability::IntegerPickingTarget,
242 "Crimson requires either R32U picking IDs or RGBA8 data-encoded picking IDs."));
243 }
244
245 push_required_status(
246 validation,
247 RendererCapability::Readback,
248 input.readback,
249 "Runtime reports texture readback support.",
250 "Texture readback is required for request-scoped picking results.");
251
252 return validation;
253}
