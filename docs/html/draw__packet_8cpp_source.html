25[[nodiscard]] const BaseShaderDefinition *definition_for_object(
26 const BaseShaderRegistry &registry,
27 const MaterialSystem &materials,
28 const RenderObject &object,
29 RenderMaterialHandle resolved_material) noexcept {
30 if (const MaterialInstance *material = materials.get(resolved_material)) {
31 if (const BaseShaderDefinition *definition = registry.find(material->base_shader_id)) {
32 return definition;
33 }
34 }
35 return registry.find(object.base_shader);

38[[nodiscard]] DrawPacket packet_from_object(
39 const RenderObject &object,
40 const BaseShaderDefinition &definition,
41 RenderMaterialHandle material,
42 RenderMeshHandle unit_box_mesh,
43 const RenderCamera &camera) {
44 const RenderQueue kQueue = object.queue == RenderQueue::PrototypeOpaque
45 ? RenderQueue::Opaque
46 : object.queue;
47 const RenderMeshHandle kMesh = is_valid_handle(object.mesh) ? object.mesh
48 : object.built_in_mesh == BuiltInRenderMesh::UnitBox ? unit_box_mesh
49 : RenderMeshHandle{};
50
51 return DrawPacket{
52 .object_id = object.object_id,
53 .mesh = kMesh,
54 .material = material,
55 .base_shader = definition.id,
56 .queue = kQueue,
57 .program = definition.program,
58 .world_from_object = object.world_from_object,
59 .world_bounds = object.world_bounds,
60 .submesh_index = object.submesh_index,
61 .camera_distance_sq = object_camera_distance_sq(object, camera),
62 .double_sided = object.pickable && definition.cull_mode == CullMode::None,
63 };
64}

72DrawPacketBuildResult build_draw_packets(
73 std::span<const RenderObject> objects,
74 const BaseShaderRegistry &registry,
75 const MaterialSystem &materials,
76 const RenderCamera &camera,
77 RenderMeshHandle unit_box_mesh,
78 RenderMaterialHandle fallback_material,
79 float view_aspect_ratio) {
80 DrawPacketBuildResult result;
81 result.opaque.reserve(objects.size());
82 result.alpha_cutout.reserve(objects.size());
83 result.transparent.reserve(objects.size());
84 result.culling.input_object_count = static_cast<std::uint32_t>(objects.size());
85
86 const CameraFrustum kFrustum = make_camera_frustum(camera, view_aspect_ratio);
87 for (const RenderObject &object : objects) {
88 if (!object.visible) {
89 ++result.culling.culled_object_count;
90 continue;
91 }
92
93 const CullingDecision kCulling = cull_object_bounds(kFrustum, object.world_bounds);
94 if (kCulling.invalid_bounds) {
95 ++result.culling.invalid_bounds_count;
96 }
97 if (!kCulling.visible) {
98 ++result.culling.culled_object_count;
99 continue;
100 }
101
102 const RenderMaterialHandle kMaterial = material_or_fallback(materials, object.material, fallback_material);
103 const BaseShaderDefinition *definition = definition_for_object(registry, materials, object, kMaterial);
104 if (definition == nullptr) {
105 continue;
106 }
107 if (definition->domain != RenderDomain::LitSurface && definition->domain != RenderDomain::TransparentSurface) {
108 continue;
109 }
110 if (!is_valid_handle(object.mesh) && object.built_in_mesh == BuiltInRenderMesh::None) {
111 continue;
112 }
113
114 DrawPacket packet = packet_from_object(object, *definition, kMaterial, unit_box_mesh, camera);
115 switch (packet.queue) {
116 case RenderQueue::Opaque:
117 result.opaque.push_back(packet);
118 break;
119 case RenderQueue::AlphaCutout:
120 result.alpha_cutout.push_back(packet);
121 break;
122 case RenderQueue::Transparent:
123 result.transparent.push_back(packet);
124 break;
125 case RenderQueue::PrototypeOpaque:
126 packet.queue = RenderQueue::Opaque;
127 result.opaque.push_back(packet);
128 break;
129 case RenderQueue::OverlayDepthTested:
130 case RenderQueue::OverlayXRay:
131 case RenderQueue::OverlayAlwaysOnTop:
132 case RenderQueue::Picking:
133 break;
134 }
135 }
136
137 sort_opaque_draw_packets(result.opaque);
138 sort_opaque_draw_packets(result.alpha_cutout);
139 sort_transparent_draw_packets(result.transparent);
140
141 result.packets.opaque_packet_count = static_cast<std::uint32_t>(result.opaque.size());
142 result.packets.alpha_cutout_packet_count = static_cast<std::uint32_t>(result.alpha_cutout.size());
143 result.packets.transparent_packet_count = static_cast<std::uint32_t>(result.transparent.size());
144 result.packets.draw_packet_count = static_cast<std::uint32_t>(result.draw_packet_count());
145 result.packets.sorted_packet_count = result.packets.draw_packet_count;
146 result.culling.visible_object_count = result.packets.draw_packet_count;
147 return result;
148}
