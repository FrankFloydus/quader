45void DocumentSelectionAdapter::sync_from_document() {
46 syncing_ = true;
47 const QSignalBlocker kBlocker(&selection_model_);
48 selection_model_.clearSelection();
49
50 QItemSelection selection;
51 for (const auto kObject :
52 documents_.document().selection().selected_objects()) {
53 const QModelIndex kLeft = model_.index_for_object(
54 kObject, static_cast<int>(DocumentItemColumn::Name));
55 if (!kLeft.isValid()) {
56 continue;
57 }
58
59 const QModelIndex kRight = model_.index_for_object(
60 kObject, static_cast<int>(DocumentItemColumn::Kind));
61 selection.select(kLeft, kRight.isValid() ? kRight : kLeft);
62 }
63
64 if (!selection.isEmpty()) {
65 selection_model_.select(selection, QItemSelectionModel::ClearAndSelect | QItemSelectionModel::Rows);
66 selection_model_.setCurrentIndex(selection.indexes().front(),
67 QItemSelectionModel::NoUpdate);
68 }
69
70 syncing_ = false;
71}

73void DocumentSelectionAdapter::on_view_selection_changed(
74 const QItemSelection &selected, const QItemSelection &deselected) {
75 (void)selected;
76 (void)deselected;
77 if (syncing_) {
78 return;
79 }
80
81 auto selected_ids = selected_object_ids_from_view();
82 quader::document::Selection selection;
83 auto selected_result = selection.set_objects(std::move(selected_ids));
84 if (!selected_result) {
85 return;
86 }
87
88 (void)documents_.execute_command(
89 std::make_unique<quader::commands::SetSelectionCommand>(
90 std::move(selection)),
91 CommandFeedback::Silent);
92}

94std::vector<quader::document::ObjectId>
95DocumentSelectionAdapter::selected_object_ids_from_view() const {
96 std::vector<quader::document::ObjectId> object_ids;
97 const auto kIndexes = selection_model_.selectedIndexes();
98 object_ids.reserve(static_cast<std::size_t>(kIndexes.size()));
99
100 for (const auto &index : kIndexes) {
101 const auto kObject = index.data(document_item_roles::kObjectIdRole)
102 .value<quader::document::ObjectId>();
103 if (kObject.is_valid()) {
104 object_ids.push_back(kObject);
105 }
106 }
107
108 std::sort(object_ids.begin(), object_ids.end());
109 object_ids.erase(std::unique(object_ids.begin(), object_ids.end()),
110 object_ids.end());
111 return object_ids;
112}
