20[[nodiscard]] quader::foundation::Result<void, DocumentError> invalid_component_result(
21 ComponentKind kind,
22 const char *message) {
23 switch (kind) {
24 case ComponentKind::Vertex:
25 return quader::foundation::Result<void, DocumentError>::failure(
26 make_document_error(DocumentErrorCode::InvalidVertex, message));
27 case ComponentKind::Edge:
28 return quader::foundation::Result<void, DocumentError>::failure(
29 make_document_error(DocumentErrorCode::InvalidEdge, message));
30 case ComponentKind::Face:
31 return quader::foundation::Result<void, DocumentError>::failure(
32 make_document_error(DocumentErrorCode::InvalidFace, message));
33 }
34
35 return quader::foundation::Result<void, DocumentError>::failure(
36 make_document_error(DocumentErrorCode::InvalidSelection, message));
37}

41quader::foundation::Result<ObjectId, DocumentError> Document::create_mesh_object(
42 std::string name,
43 quader::mesh::Polyhedron mesh,
44 Transform transform,
45 PbrMaterial material) {
46 if (!is_finite(transform)) {
47 return quader::foundation::Result<ObjectId, DocumentError>::failure(
48 make_document_error(DocumentErrorCode::InvalidTransform,
49 "cannot create a mesh object with a non-finite transform"));
50 }
51
52 if (!is_valid(material)) {
53 return quader::foundation::Result<ObjectId, DocumentError>::failure(
54 make_document_error(DocumentErrorCode::InvalidMaterial,
55 "cannot create a mesh object with invalid material values"));
56 }
57
58 auto mesh_validation = validate_mesh(mesh);
59 if (!mesh_validation) {
60 return quader::foundation::Result<ObjectId, DocumentError>::failure(
61 std::move(mesh_validation).error());
62 }
63
64 auto created = objects_.create_mesh_object(std::move(name), std::move(mesh), transform, material);
65 if (!created) {
66 return created;
67 }
68
69 emit_change(DocumentChangeKind::ObjectCreated, created.value());
70 mark_dirty(to_dirty_flags(DocumentDirtyFlag::Objects));
71 return created;
72}

82 if (!restored) {
83 return restored;

85
86 emit_change(DocumentChangeKind::ObjectCreated, kId);
87 mark_dirty(to_dirty_flags(DocumentDirtyFlag::Objects));
88 return quader::foundation::Result<void, DocumentError>::success();

100quader::foundation::Result<MeshObject, DocumentError> Document::extract_mesh_object(ObjectId id) {
101 auto removed = objects_.remove_mesh_object(id);
102 if (!removed) {
103 return removed;
104 }
105
106 const bool kSelectionChanged = selection_.references_object(id);
107 if (kSelectionChanged) {
108 selection_.remove_object(id);
109 }
110
111 emit_change(DocumentChangeKind::ObjectDeleted, id);
112 if (kSelectionChanged) {
113 emit_change(DocumentChangeKind::SelectionChanged, id);
114 }
115
116 auto dirty = to_dirty_flags(DocumentDirtyFlag::Objects);
117 if (kSelectionChanged) {
118 dirty |= to_dirty_flags(DocumentDirtyFlag::Selection);
119 }
120 mark_dirty(dirty);
121
122 return removed;
123}

187quader::foundation::Result<void, DocumentError> Document::validate_component_ref(
188 const ComponentRef &component) const {
189 const auto *object = find_mesh_object(component.object);
190 if (object == nullptr) {
191 return quader::foundation::Result<void, DocumentError>::failure(
192 make_document_error(DocumentErrorCode::InvalidObject,
193 "component reference points at an unknown object id"));
194 }
195
196 if (!component_id_is_valid(component)) {
197 return invalid_component_result(component_kind(component),
198 "component reference contains an invalid component id");
199 }
200
201 return std::visit(
202 [&object](auto id) -> quader::foundation::Result<void, DocumentError> {
203 using ComponentId = std::decay_t<decltype(id)>;
204 if constexpr (std::is_same_v<ComponentId, quader::mesh::VertexId>) {
205 if (!object->mesh.is_valid(id)) {
206 return quader::foundation::Result<void, DocumentError>::failure(
207 make_document_error(DocumentErrorCode::InvalidVertex,
208 "component reference points at an unknown vertex id"));
209 }
210 } else if constexpr (std::is_same_v<ComponentId, quader::mesh::EdgeId>) {
211 if (!object->mesh.is_valid(id)) {
212 return quader::foundation::Result<void, DocumentError>::failure(
213 make_document_error(DocumentErrorCode::InvalidEdge,
214 "component reference points at an unknown edge id"));
215 }
216 } else if constexpr (std::is_same_v<ComponentId, quader::mesh::FaceId>) {
217 if (!object->mesh.is_valid(id)) {
218 return quader::foundation::Result<void, DocumentError>::failure(
219 make_document_error(DocumentErrorCode::InvalidFace,
220 "component reference points at an unknown face id"));
221 }
222 }
223
224 return quader::foundation::Result<void, DocumentError>::success();
225 },
226 component.component);
227}

316quader::foundation::Result<void, DocumentError> Document::validate_mesh_object(
317 const MeshObject &object) const {
318 if (!object.id.is_valid()) {
319 return quader::foundation::Result<void, DocumentError>::failure(
320 make_document_error(DocumentErrorCode::InvalidObject,
321 "mesh object has an invalid object id"));
322 }
323
324 if (!is_finite(object.transform)) {
325 return quader::foundation::Result<void, DocumentError>::failure(
326 make_document_error(DocumentErrorCode::InvalidTransform,
327 "mesh object has a non-finite transform"));
328 }
329
330 if (!is_valid(object.material)) {
331 return quader::foundation::Result<void, DocumentError>::failure(
332 make_document_error(DocumentErrorCode::InvalidMaterial,
333 "mesh object has invalid material values"));
334 }
335
336 return validate_mesh(object.mesh);
337}

339quader::foundation::Result<void, DocumentError> Document::validate_mesh(
340 const quader::mesh::Polyhedron &mesh) const {
341 const auto kReport = quader::mesh::validate_mesh(mesh);
342 if (kReport.ok()) {
343 return quader::foundation::Result<void, DocumentError>::success();
344 }
345
346 auto message = std::string("mesh object failed validation");
347 if (!kReport.issues().empty()) {
348 message += ": ";
349 message += kReport.issues().front().diagnostic.message;
350 }
351
352 return quader::foundation::Result<void, DocumentError>::failure(
353 make_document_error(DocumentErrorCode::InvalidMesh, std::move(message)));
354}

356quader::foundation::Result<void, DocumentError> Document::validate_selection(
357 const Selection &selection) const {
358 for (const auto kObject : selection.selected_objects()) {
359 auto validation = validate_object_id(kObject);
360 if (!validation) {
361 return validation;
362 }
363 }
364
365 for (const auto &component : selection.selected_components()) {
366 auto validation = validate_component_ref(component);
367 if (!validation) {
368 return validation;
369 }
370 }
371
372 return quader::foundation::Result<void, DocumentError>::success();
373}

383void Document::mark_dirty(DocumentDirtyFlags flags) {
384 const auto kNewFlags = dirty_flags_ | flags;

387 }
388

390 emit_change(DocumentChangeKind::DirtyStateChanged, ObjectId::invalid(), dirty_flags_);
