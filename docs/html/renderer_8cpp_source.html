45[[nodiscard]] quader::foundation::Result<void, RendererDiagnostic> validate_snapshot(
46 const FrameSnapshot &snapshot) {
47 if (!is_valid_extent(snapshot.target_extent())) {
48 return quader::foundation::Result<void, RendererDiagnostic>::failure(
49 invalid_snapshot_diagnostic("Frame snapshots require a valid target extent.", snapshot.frame_index()));
50 }
51 if (snapshot.views().empty()) {
52 return quader::foundation::Result<void, RendererDiagnostic>::failure(
53 invalid_snapshot_diagnostic("Frame snapshots require at least one render view.", snapshot.frame_index()));
54 }
55 for (const RenderView &view : snapshot.views()) {
56 if (!is_valid_render_view(view)) {
57 return quader::foundation::Result<void, RendererDiagnostic>::failure(
58 invalid_snapshot_diagnostic(
59 "A frame snapshot render view has an invalid rectangle.",
60 snapshot.frame_index()));
61 }
62 }
63
64 return quader::foundation::Result<void, RendererDiagnostic>::success();
65}

81quader::foundation::Result<std::unique_ptr<Renderer>, RendererDiagnostic> Renderer::create(
82 RendererConfig config,
83 const NativeSurfaceDescriptor &surface) {
84 auto impl = std::make_unique<Impl>();
85 impl->config = std::move(config);
86 if (!impl->device.initialize(impl->config, surface)) {
87 const RendererStatus kStatus = impl->device.status();
88 RendererDiagnostic diagnostic{
89 .severity = RendererDiagnosticSeverity::Fatal,
90 .code = RendererDiagnosticCode::RuntimeInitializationFailed,
91 .message = "Crimson renderer initialization failed.",
92 .subsystem = "gpu",
93 .resource_name = "GpuDevice",
94 };
95 if (!kStatus.diagnostics.empty()) {
96 diagnostic = kStatus.diagnostics.back();
97 }
98 return quader::foundation::Result<std::unique_ptr<Renderer>, RendererDiagnostic>::failure(
99 std::move(diagnostic));
100 }
101
102 impl->frame_graph = make_v1_correctness_render_graph(surface.extent);
103 impl->current_extent = surface.extent;
104
105 return quader::foundation::Result<std::unique_ptr<Renderer>, RendererDiagnostic>::success(
106 std::unique_ptr<Renderer>(new Renderer(std::move(impl))));
107}

109quader::foundation::Result<void, RendererDiagnostic> Renderer::resize(const ViewportExtent &extent) {
110 if (extent.width_px == impl_->current_extent.width_px && extent.height_px == impl_->current_extent.height_px && extent.device_pixel_ratio == impl_->current_extent.device_pixel_ratio) {
111 return quader::foundation::Result<void, RendererDiagnostic>::success();
112 }
113
114 if (!impl_->device.resize(extent)) {
115 const RendererStatus kStatus = impl_->device.status();
116 RendererDiagnostic diagnostic{
117 .severity = RendererDiagnosticSeverity::Error,
118 .code = RendererDiagnosticCode::ResizeFailed,
119 .message = "Crimson renderer resize failed.",
120 .subsystem = "gpu",
121 .resource_name = "Backbuffer",
122 };
123 if (!kStatus.diagnostics.empty()) {
124 diagnostic = kStatus.diagnostics.back();
125 }
126 return quader::foundation::Result<void, RendererDiagnostic>::failure(std::move(diagnostic));
127 }
128
129 impl_->frame_graph.resize(extent);
130 impl_->current_extent = extent;
131 return quader::foundation::Result<void, RendererDiagnostic>::success();
132}

134quader::foundation::Result<FrameRenderResult, RendererDiagnostic> Renderer::render(const FrameSnapshot &snapshot) {
135 auto snapshot_validation = validate_snapshot(snapshot);
136 if (!snapshot_validation) {
137 return quader::foundation::Result<FrameRenderResult, RendererDiagnostic>::failure(
138 std::move(snapshot_validation).error());
139 }
140
141 const ViewportExtent kExtent = snapshot.target_extent();
142 auto resized = resize(kExtent);
143 if (!resized) {
144 return quader::foundation::Result<FrameRenderResult, RendererDiagnostic>::failure(std::move(resized).error());
145 }
146
147 auto graph_validation = impl_->frame_graph.validate();
148 if (!graph_validation) {
149 return quader::foundation::Result<FrameRenderResult, RendererDiagnostic>::failure(
150 frame_graph_diagnostic(std::move(graph_validation).error()));
151 }
152
153 return impl_->device.render_frame(snapshot, impl_->frame_graph);
154}

164RendererDiagnosticsSnapshot Renderer::diagnostics_snapshot() const {
165 RendererDiagnosticsSnapshot snapshot;
166 const RendererStatus &status = impl_->device.status();
167 snapshot.recent_diagnostics = make_diagnostic_ring_from_recent(status.diagnostics).snapshot();
168 snapshot.status = make_bounded_renderer_status_snapshot(status, snapshot.recent_diagnostics);
169 snapshot.latest_frame_stats = impl_->device.latest_frame_stats();
170 snapshot.pass_stats = impl_->device.latest_pass_stats();
171 if (snapshot.pass_stats.empty()) {
172 snapshot.pass_stats = make_render_pass_stats(
173 impl_->frame_graph,
174 snapshot.latest_frame_stats ? &*snapshot.latest_frame_stats : nullptr);
175 }
176 if (snapshot.latest_frame_stats) {
177 snapshot.counters = make_renderer_counters(*snapshot.latest_frame_stats);
178 }
179 snapshot.frame_graph_dump = impl_->frame_graph.debug_dump();
180 return snapshot;
181}
