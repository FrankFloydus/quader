30TEST(ColorSpace, SrgbReferenceConversionMatchesKnownMidpoint) {
31 const crimson::ColorLinear kLinear = crimson::srgb_to_linear(crimson::ColorSrgb{ 0.5F, 0.5F, 0.5F, 1.0F });
32 expect_near(kLinear.r, 0.214041F, 0.0005F, "sRGB 0.5 decodes to reference linear value");
33 expect_near(kLinear.g, kLinear.r, 0.00001F, "sRGB conversion is channel-stable for equal input");
34
35 const crimson::ColorSrgb kSrgb = crimson::linear_to_srgb(kLinear);
36 expect_near(kSrgb.r, 0.5F, 0.0005F, "linear midpoint round-trips back to sRGB");
37}

64TEST(ColorSpace, ToneMappersAndExposureAreFinite) {
65 const float kEv12 = crimson::exposure_multiplier_from_ev100(12.0F, 0.0F);
66 const float kEv12PlusOne = crimson::exposure_multiplier_from_ev100(12.0F, 1.0F);
67 expect_true(std::isfinite(kEv12) && kEv12 > 0.0F, "EV100 exposure multiplier is finite and positive");
68 expect_near(kEv12PlusOne / kEv12, 2.0F, 0.0005F, "one exposure compensation EV doubles exposure");
69
70 const crimson::ColorLinear kHdr{ 4.0F, 2.0F, 0.5F, 1.0F };
71 for (const crimson::ToneMapper kMapper : {
72 crimson::ToneMapper::AcesFitted,
73 crimson::ToneMapper::AgxApprox,
74 crimson::ToneMapper::Reinhard,
75 crimson::ToneMapper::Linear,
76 }) {
77 const crimson::ColorLinear kMapped = crimson::apply_tone_mapper(kMapper, kHdr);
78 expect_true(std::isfinite(kMapped.r) && kMapped.r >= 0.0F && kMapped.r <= 1.0F, "tone mapper red is finite SDR");
79 expect_true(std::isfinite(kMapped.g) && kMapped.g >= 0.0F && kMapped.g <= 1.0F, "tone mapper green is finite SDR");
80 expect_true(std::isfinite(kMapped.b) && kMapped.b >= 0.0F && kMapped.b <= 1.0F, "tone mapper blue is finite SDR");
81 }
82}
