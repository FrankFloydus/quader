106[[nodiscard]] int row_for_field(const quader::ui::PropertyItemModel &model,
107 quader::ui::PropertyField field) {
108 for (int row = 0; row < model.rowCount(); ++row) {
109 const auto kPath =
110 model
111 .index(row, static_cast<int>(quader::ui::PropertyItemColumn::Value))
112 .data(quader::ui::property_item_roles::kPropertyPathRole)
113 .value<quader::ui::PropertyPath>();
114 if (kPath.field == field) {
115 return row;
116 }
117 }
118
119 return -1;
120}
121
122[[nodiscard]] QModelIndex value_index(quader::ui::PropertyItemModel &model,
123 quader::ui::PropertyField field) {
124 const int kRow = row_for_field(model, field);
125 if (kRow < 0) {
126 return {};
127 }
128
129 return model.index(kRow,
130 static_cast<int>(quader::ui::PropertyItemColumn::Value));
131}
