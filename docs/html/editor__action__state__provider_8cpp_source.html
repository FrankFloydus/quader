26[[nodiscard]] QString command_action_text(const QString &verb, std::string_view command_name) {
27 if (command_name.empty()) {

30
31 return QStringLiteral("%1 %2").arg(verb, QString::fromStdString(std::string(command_name)));

34[[nodiscard]] bool has_any_registered_tool(const quader::tools::ToolManager &tool_manager) {
35 return tool_manager.has_tool(quader::tools::ToolId::Select) || tool_manager.has_tool(quader::tools::ToolId::Move) || tool_manager.has_tool(quader::tools::ToolId::Rotate) || tool_manager.has_tool(quader::tools::ToolId::Scale) || tool_manager.has_tool(quader::tools::ToolId::Box);
36}

38[[nodiscard]] ActionId action_for_tool(quader::tools::ToolId id) noexcept {

68EditorStateSnapshot EditorActionStateProvider::editor_state_snapshot() const {
69 EditorStateSnapshot snapshot;
70 snapshot.document_services_available = true;
71 snapshot.import_available = !io_registry_.import_formats().empty();
72 snapshot.export_available = !io_registry_.export_formats().empty();
73 snapshot.file_io_available = snapshot.import_available || snapshot.export_available;
74 snapshot.has_active_document = true;
75 snapshot.document_dirty = document_.is_dirty();
76 snapshot.has_selection = !document_.selection().empty();
77 snapshot.can_delete_selection = document_.selection().selected_objects().size() == 1U && document_.selection().selected_components().empty();
78
79 snapshot.can_undo = command_history_.can_undo();
80 snapshot.undo_text = command_action_text(QStringLiteral("Undo"), command_history_.undo_name());
81 snapshot.can_redo = command_history_.can_redo();
82 snapshot.redo_text = command_action_text(QStringLiteral("Redo"), command_history_.redo_name());
83
84 snapshot.tools_available = has_any_registered_tool(tool_manager_);
85 if (const auto kActiveTool = tool_manager_.active_tool_id()) {
86 snapshot.active_tool = action_for_tool(*kActiveTool);
87 }
88
89 return snapshot;
90}
