123GpuPbrMaterialBlock pack_pbr_material_block(
124 const MaterialInstance &material,
125 const BaseShaderDefinition &definition) noexcept {
126 const MaterialColorSrgb kBaseColor = color_parameter(
127 material,
128 "base_color",
129 MaterialColorSrgb{ 1.0F, 1.0F, 1.0F, 1.0F });
130 const MaterialColorSrgb kEmissiveColor = color_parameter(
131 material,
132 "emissive_color",
133 MaterialColorSrgb{ 0.0F, 0.0F, 0.0F, 1.0F });
134 const float kEmissiveStrength = float_parameter(material, "emissive_strength", 0.0F);
135 const float kMetallic = std::clamp(float_parameter(material, "metallic", 0.0F), 0.0F, 1.0F);
136 const float kRoughness = std::clamp(float_parameter(material, "roughness", 0.5F), 0.0F, 1.0F);
137 const float kNormalStrength = std::max(0.0F, float_parameter(material, "normal_strength", 1.0F));
138 const float kOcclusionStrength = std::clamp(float_parameter(material, "occlusion_strength", 1.0F), 0.0F, 1.0F);
139 const float kOpacity = std::clamp(float_parameter(material, "opacity", kBaseColor.a), 0.0F, 1.0F);
140 const float kAlphaCutoff = std::clamp(float_parameter(material, "alpha_cutoff", 0.5F), 0.0F, 1.0F);
141 const MaterialVec2 kUvTiling = vec2_parameter(material, "uv_tiling", MaterialVec2{ 1.0F, 1.0F });
142 const MaterialVec2 kUvOffset = vec2_parameter(material, "uv_offset", MaterialVec2{ 0.0F, 0.0F });
143 const bool kDoubleSided = bool_parameter(material, "double_sided", material.overrides.double_sided);
144
145 GpuPbrMaterialBlock block;
146 block.base_color_factor_linear = linear_color_array(kBaseColor);
147 if (definition.id == BaseShaderId::TransparentPbr) {
148 block.base_color_factor_linear[3] = kOpacity;
149 }
150 block.emissive_factor_linear = linear_color_array(kEmissiveColor);
151 block.emissive_factor_linear[3] = std::max(0.0F, kEmissiveStrength);
152 block.factors = { kMetallic, kRoughness, kNormalStrength, kOcclusionStrength };
153 if (definition.id == BaseShaderId::AlphaCutoutPbr) {
154 block.factors[3] = kAlphaCutoff;
155 }
156 if (definition.id == BaseShaderId::TransparentPbr) {
157 block.factors[3] = kOpacity;
158 }
159 block.uv_transform_0 = { kUvTiling.x, kUvTiling.y, kUvOffset.x, kUvOffset.y };
160 block.flags = {
161 kDoubleSided ? 1.0F : 0.0F,
162 definition.id == BaseShaderId::AlphaCutoutPbr ? 1.0F : 0.0F,
163 definition.id == BaseShaderId::TransparentPbr ? 1.0F : 0.0F,
164 0.0F,
165 };
166 return block;
167}

174bool GpuMaterialCache::initialize(RendererStatus &status) {
175 shutdown();
176
177 impl_->base_color_uniform.reset(bgfx::createUniform("u_pbrBaseColor", bgfx::UniformType::Vec4));
178 impl_->emissive_uniform.reset(bgfx::createUniform("u_pbrEmissive", bgfx::UniformType::Vec4));
179 impl_->factors_uniform.reset(bgfx::createUniform("u_pbrFactors", bgfx::UniformType::Vec4));
180 impl_->uv_transform_uniform.reset(bgfx::createUniform("u_pbrUvTransform0", bgfx::UniformType::Vec4));
181 impl_->flags_uniform.reset(bgfx::createUniform("u_pbrFlags", bgfx::UniformType::Vec4));
182 impl_->default_textures[0] = DefaultTexture{
183 create_default_texture(0xffffffffU, TextureColorSpace::Srgb),
184 TextureColorSpace::Srgb,
185 };
186 impl_->default_textures[1] = DefaultTexture{
187 create_default_texture(0xff8080ffU, TextureColorSpace::Linear),
188 TextureColorSpace::Linear,
189 };
190 impl_->default_textures[2] = DefaultTexture{
191 create_default_texture(0xffff8080U, TextureColorSpace::Data),
192 TextureColorSpace::Data,
193 };
194 impl_->default_textures[3] = DefaultTexture{
195 create_default_texture(0xffffffffU, TextureColorSpace::Linear),
196 TextureColorSpace::Linear,
197 };
198 impl_->default_textures[4] = DefaultTexture{
199 create_default_texture(0xff000000U, TextureColorSpace::Srgb),
200 TextureColorSpace::Srgb,
201 };
202
203 impl_->initialized = impl_->base_color_uniform.valid() && impl_->emissive_uniform.valid() && impl_->factors_uniform.valid() && impl_->uv_transform_uniform.valid() && impl_->flags_uniform.valid();
204 for (const DefaultTexture &texture : impl_->default_textures) {
205 impl_->initialized = impl_->initialized && texture.texture.valid();
206 }
207
208 if (!impl_->initialized) {
209 push_resource_error(status, "Crimson failed to create default PBR material GPU resources.", "GpuMaterialCache");
210 shutdown();
211 return false;
212 }
213
214 impl_->upload_stats.texture_upload_count += static_cast<std::uint32_t>(impl_->default_textures.size());
215 impl_->upload_stats.uploaded_texture_bytes += impl_->default_textures.size() * sizeof(std::uint32_t);
216 return true;
217}

219void GpuMaterialCache::shutdown() noexcept {
220 for (DefaultTexture &texture : impl_->default_textures) {
221 texture.texture.reset();
222 texture.color_space = TextureColorSpace::Linear;
223 }
224 impl_->base_color_uniform.reset();
225 impl_->emissive_uniform.reset();
226 impl_->factors_uniform.reset();
227 impl_->uv_transform_uniform.reset();
228 impl_->flags_uniform.reset();
229 impl_->upload_stats = {};
230 impl_->initialized = false;
231}

243void GpuMaterialCache::bind_pbr_material(const GpuPbrMaterialBlock &block) const noexcept {
244 if (!impl_->initialized) {
245 return;
246 }
247
248 bgfx::setUniform(impl_->base_color_uniform.get(), block.base_color_factor_linear.data());
249 bgfx::setUniform(impl_->emissive_uniform.get(), block.emissive_factor_linear.data());
250 bgfx::setUniform(impl_->factors_uniform.get(), block.factors.data());
251 bgfx::setUniform(impl_->uv_transform_uniform.get(), block.uv_transform_0.data());
252 bgfx::setUniform(impl_->flags_uniform.get(), block.flags.data());
253}
