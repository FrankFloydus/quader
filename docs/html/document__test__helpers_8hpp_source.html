89inline TriangleMeshFixture make_triangle_mesh() {
90 quader::mesh::Polyhedron mesh;
91 const std::array<quader::mesh::VertexId, 3> kVertices{
92 mesh.create_vertex(quader::math::Vec3{ 0.0F, 0.0F, 0.0F }),
93 mesh.create_vertex(quader::math::Vec3{ 1.0F, 0.0F, 0.0F }),
94 mesh.create_vertex(quader::math::Vec3{ 0.0F, 1.0F, 0.0F }),
95 };
96
97 auto face = mesh.create_face(kVertices);
98 QUADER_ASSERT(face);
99
100 auto edges = quader::mesh::face_edges(mesh, face.value());
101 QUADER_ASSERT(edges);
102 QUADER_ASSERT(!edges.value().empty());
103
104 return TriangleMeshFixture{
105 std::move(mesh),
106 kVertices,
107 edges.value().front(),
108 face.value(),
109 };

130inline DocumentSemanticState capture_document_state(const quader::document::Document &document) {
131 DocumentSemanticState state;
132 state.dirty = document.is_dirty();
133 state.dirty_flags = document.dirty_flags();
134
135 for (const auto kObjectId : document.object_ids()) {
136 const auto *object = document.find_mesh_object(kObjectId);
137 QUADER_ASSERT(object != nullptr);
138
139 state.objects.push_back(SemanticMeshObjectState{
140 object->id,
141 object->name,
142 object->transform,
143 object->material,
144 object->mesh.vertex_count(),
145 object->mesh.edge_count(),
146 object->mesh.face_count(),
147 });
148 }
149
150 state.selected_objects.assign(document.selection().selected_objects().begin(),
151 document.selection().selected_objects().end());
152 state.selected_components.assign(document.selection().selected_components().begin(),
153 document.selection().selected_components().end());
154 return state;
155}
