63[[nodiscard]] quader::math::Vec3 average_points(std::span<const quader::math::Vec3> points) noexcept {
64 quader::math::Vec3 average;
65 for (quader::math::Vec3 point : points) {
66 average = average + point;
67 }
68 const float kScale = points.empty() ? 0.0F : 1.0F / static_cast<float>(points.size());
69 return average * kScale;
70}

83[[nodiscard]] quader::foundation::Result<std::array<std::size_t, 4>, quader::mesh::MeshError>
84orient_face_outward(
85 const BoxCorners &corners,
86 std::array<std::size_t, 4> face,
87 quader::math::Vec3 box_center) {
88 const std::array<quader::math::Vec3, 4> kPoints = face_points(corners, face);
89 const quader::math::Vec3 kFaceCenter = average_points(kPoints);
90 const quader::math::Vec3 kOutward = kFaceCenter - box_center;
91 const quader::math::Vec3 kArea = quader::geometry::polygon_normal(kPoints);
92 if (quader::math::length_squared(kArea) <= kMinimumBoxExtent * kMinimumBoxExtent ||
93 quader::math::length_squared(kOutward) <= kMinimumBoxExtent * kMinimumBoxExtent) {
94 return quader::foundation::Result<std::array<std::size_t, 4>, quader::mesh::MeshError>::failure(
95 invalid_box_error("cannot create a box with a degenerate face"));
96 }
97
98 if (quader::math::dot(kArea, kOutward) < 0.0F) {
99 std::reverse(face.begin(), face.end());
100 }
101 return quader::foundation::Result<std::array<std::size_t, 4>, quader::mesh::MeshError>::success(face);
102}

104[[nodiscard]] quader::foundation::Result<void, quader::mesh::MeshError> validate_corners(
105 const BoxCorners &corners) {
106 if (!quader::geometry::all_points_finite(corners.points)) {
107 return quader::foundation::Result<void, quader::mesh::MeshError>::failure(
108 quader::mesh::make_mesh_error(quader::mesh::MeshErrorCode::NonFinitePosition,
109 "cannot create a box from non-finite corner positions"));
110 }
111
112 for (const auto &face : box_face_indices()) {
113 const std::array<quader::math::Vec3, 4> kPoints{
114 corners.points[face[0]],
115 corners.points[face[1]],
116 corners.points[face[2]],
117 corners.points[face[3]],
118 };
119 if (quader::geometry::has_degenerate_area(kPoints, kMinimumBoxExtent)) {
120 return quader::foundation::Result<void, quader::mesh::MeshError>::failure(
121 quader::mesh::make_mesh_error(quader::mesh::MeshErrorCode::DegenerateFace,
122 "cannot create a box with a degenerate face"));
123 }
124 }
125
126 return quader::foundation::Result<void, quader::mesh::MeshError>::success();
127}

131quader::foundation::Result<quader::mesh::Polyhedron, quader::mesh::MeshError> make_box_from_bounds(
132 BoxDimensions bounds) {
133 const quader::math::Vec3 kMin = component_min(bounds.min, bounds.max);
134 const quader::math::Vec3 kMax = component_max(bounds.min, bounds.max);
135 if (!extent_is_valid(kMin.x, kMax.x) || !extent_is_valid(kMin.y, kMax.y) || !extent_is_valid(kMin.z, kMax.z)) {
136 return quader::foundation::Result<quader::mesh::Polyhedron, quader::mesh::MeshError>::failure(
137 invalid_box_error("cannot create a box with non-finite or degenerate bounds"));
138 }
139
140 return make_box_from_corners(BoxCorners{
141 .points = {
142 quader::math::Vec3{ kMin.x, kMin.y, kMin.z },
143 quader::math::Vec3{ kMax.x, kMin.y, kMin.z },
144 quader::math::Vec3{ kMax.x, kMax.y, kMin.z },
145 quader::math::Vec3{ kMin.x, kMax.y, kMin.z },
146 quader::math::Vec3{ kMin.x, kMin.y, kMax.z },
147 quader::math::Vec3{ kMax.x, kMin.y, kMax.z },
148 quader::math::Vec3{ kMax.x, kMax.y, kMax.z },
149 quader::math::Vec3{ kMin.x, kMax.y, kMax.z },
150 },
151 });
152}

162quader::foundation::Result<quader::mesh::Polyhedron, quader::mesh::MeshError> make_box_from_corners(
163 const BoxCorners &corners) {
164 auto validation = validate_corners(corners);
165 if (!validation) {
166 return quader::foundation::Result<quader::mesh::Polyhedron, quader::mesh::MeshError>::failure(
167 std::move(validation).error());
168 }
169
170 quader::mesh::Polyhedron mesh;
171 std::array<quader::mesh::VertexId, 8> vertices{};
172 for (std::size_t index = 0; index < corners.points.size(); ++index) {
173 vertices[index] = mesh.create_vertex(corners.points[index]);
174 }
175
176 const quader::math::Vec3 kBoxCenter = average_points(corners.points);
177 for (const auto &face : box_face_indices()) {
178 auto oriented_face = orient_face_outward(corners, face, kBoxCenter);
179 if (!oriented_face) {
180 return quader::foundation::Result<quader::mesh::Polyhedron, quader::mesh::MeshError>::failure(
181 std::move(oriented_face).error());
182 }
183
184 const std::array<quader::mesh::VertexId, 4> kFaceVertices{
185 vertices[oriented_face.value()[0]],
186 vertices[oriented_face.value()[1]],
187 vertices[oriented_face.value()[2]],
188 vertices[oriented_face.value()[3]],
189 };
190 auto created_face = mesh.create_face(kFaceVertices);
191 if (!created_face) {
192 return quader::foundation::Result<quader::mesh::Polyhedron, quader::mesh::MeshError>::failure(
193 std::move(created_face).error());
194 }
195 }
196
197 return quader::foundation::Result<quader::mesh::Polyhedron, quader::mesh::MeshError>::success(
198 std::move(mesh));
199}
