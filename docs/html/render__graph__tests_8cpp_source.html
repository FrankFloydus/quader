36TEST(RenderGraph, ValidationRejectsMissingUnwrittenAndAmbiguousResources) {
37 crimson::RenderGraph missing;
38 missing.add_pass(crimson::make_pass("ReadsMissing", {
39 crimson::ResourceUse{ .resource_name = "Missing", .access = crimson::RenderResourceAccess::Read },
40 }));
41 expect_true(!missing.validate(), "reading an undeclared resource fails validation");
42
43 crimson::RenderGraph unwritten;
44 (void)unwritten.add_resource(crimson::RenderResourceDesc{
45 .name = "TransientColor",
46 .format = crimson::RenderResourceFormat::Rgba8Unorm,
47 .extent = crimson::ViewportExtent{ 32, 32, 1.0F },
48 .external = false,
49 });
50 unwritten.add_pass(crimson::make_pass("ReadsBeforeWrite", {
51 crimson::ResourceUse{ .resource_name = "TransientColor", .access = crimson::RenderResourceAccess::Read },
52 }));
53 expect_true(!unwritten.validate(), "reading an unwritten non-external resource fails validation");
54
55 crimson::RenderGraph double_write;

57 .name = "TransientColor",

83TEST(RenderGraph, ResourceRegistryResizeUpdatesGenerations) {
84 crimson::RenderGraph graph = crimson::make_minimal_prototype_render_graph(crimson::ViewportExtent{ 640, 480, 1.0F });
85 const crimson::RenderResourceRecord *before = graph.resources().find("BackbufferColor");
86 expect_true(before != nullptr, "backbuffer color resource exists");
87 const std::uint64_t kGenerationBefore = before == nullptr ? 0 : before->generation;
88
89 graph.resize(crimson::ViewportExtent{ 1920, 1080, 1.0F });
90 const crimson::RenderResourceRecord *after = graph.resources().find("BackbufferColor");
91 expect_true(after != nullptr, "backbuffer color resource still exists after resize");
92 expect_true(after != nullptr && after->generation != kGenerationBefore, "resize advances resource generation");
93 expect_true(after != nullptr && after->desc.extent.width_px == 1920, "resize updates resource width");
94 expect_true(graph.resize_generation() == 2, "graph resize generation advances once");
95}
