31class StyledSplitterHandle final : public QWidget {
32public:
33 explicit StyledSplitterHandle(Qt::Orientation orientation, QWidget *parent = nullptr) : QWidget(parent), orientation_(orientation) {
34 setCursor(orientation_ == Qt::Vertical ? Qt::SizeHorCursor : Qt::SizeVerCursor);
35 setMouseTracking(true);
36 }
37
38 [[nodiscard]] QSize sizeHint() const override {
39 return orientation_ == Qt::Vertical
40 ? QSize(kViewportSplitterWidth, kMinimumViewportExtent)
41 : QSize(kMinimumViewportExtent, kViewportSplitterWidth);
42 }
43
44protected:
45 void paintEvent(QPaintEvent *) override {
46 QPainter painter(this);
47 painter.fillRect(rect(), palette().color(QPalette::Mid));
48
49 QStyleOption option;
50 option.initFrom(this);
51 option.rect = rect();
52 option.state |= QStyle::State_Raised;
53 style()->drawControl(QStyle::CE_Splitter, &option, &painter, this);
54 }
55
56private:
57 Qt::Orientation orientation_ = Qt::Vertical;
58};

62ViewportWidget::ViewportWidget(ViewportController &controller, QWidget *parent) : QWidget(parent), controller_(controller) {
63 setAttribute(Qt::WA_NativeWindow);
64 setAttribute(Qt::WA_PaintOnScreen);
65 setAttribute(Qt::WA_NoSystemBackground);
66 setAttribute(Qt::WA_OpaquePaintEvent);
67 setFocusPolicy(Qt::StrongFocus);
68 setMouseTracking(true);
69
70 vertical_splitter_handle_ = new StyledSplitterHandle(Qt::Vertical, this);
71 vertical_splitter_handle_->installEventFilter(this);
72 vertical_splitter_handle_->hide();
73
74 horizontal_splitter_handle_ = new StyledSplitterHandle(Qt::Horizontal, this);
75 horizontal_splitter_handle_->installEventFilter(this);
76 horizontal_splitter_handle_->hide();
77
78 frame_timer_.setTimerType(Qt::PreciseTimer);
79 frame_timer_.setInterval(16);
80 connect(&frame_timer_, &QTimer::timeout, this, &ViewportWidget::render_now);
81 connect(&controller_, &ViewportController::frame_requested, this, qOverload<>(&ViewportWidget::update));
82 connect(&controller_, &ViewportController::viewport_layout_changed, this, [this](bool, const QString &) {
83 update_splitter_cursor(mapFromGlobal(QCursor::pos()));
84 update_splitter_handles();
85 });
86}

143void ViewportWidget::mousePressEvent(QMouseEvent *event) {
144 setFocus(Qt::MouseFocusReason);
145
146 if (event->button() == Qt::LeftButton && controller_.quad_viewports_enabled()) {
147 active_split_handle_ = controller_.layout_state().splitter_at(
148 pixel_size(),
149 viewport_point_from(event->position() * devicePixelRatioF()));
150 if (active_split_handle_ != ViewportSplitHandle::None) {
151 set_split_from_position(event->pos());
152 event->accept();
153 return;
154 }
155 }
156
157 const bool kHandled = controller_.handle_mouse_press(
158 mouse_button_from(event->button()),
159 viewport_point_from(event->position() * devicePixelRatioF()),
160 event->modifiers().testFlag(Qt::ShiftModifier),
161 pixel_size());
162
163 if (kHandled || event->button() == Qt::MiddleButton || event->button() == Qt::RightButton) {
164 event->accept();
165 return;
166 }
167
168 QWidget::mousePressEvent(event);
169}

171void ViewportWidget::mouseMoveEvent(QMouseEvent *event) {
172 if (active_split_handle_ != ViewportSplitHandle::None) {
173 set_split_from_position(event->pos());
174 event->accept();
175 return;
176 }
177
178 if (controller_.handle_mouse_move(viewport_point_from(event->position() * devicePixelRatioF()), pixel_size())) {
179 event->accept();
180 return;

184 QWidget::mouseMoveEvent(event);

188 if (event->button() == Qt::LeftButton && active_split_handle_ != ViewportSplitHandle::None) {
189 active_split_handle_ = ViewportSplitHandle::None;

197 viewport_point_from(event->position() * devicePixelRatioF()),

199 event->accept();

241bool ViewportWidget::eventFilter(QObject *watched, QEvent *event) {
242 const bool kIsVerticalHandle = watched == vertical_splitter_handle_;
243 const bool kIsHorizontalHandle = watched == horizontal_splitter_handle_;
244
245 if (!kIsVerticalHandle && !kIsHorizontalHandle) {
246 return QWidget::eventFilter(watched, event);
247 }
248
249 const ViewportSplitHandle kHandle = kIsVerticalHandle
250 ? ViewportSplitHandle::Vertical
251 : ViewportSplitHandle::Horizontal;
252
253 switch (event->type()) {
254 case QEvent::MouseButtonPress: {
255 auto *mouse_event = static_cast<QMouseEvent *>(event);
256 if (mouse_event->button() == Qt::LeftButton && controller_.quad_viewports_enabled()) {
257 active_split_handle_ = kHandle;
258 set_split_from_position(mapFromGlobal(mouse_event->globalPosition().toPoint()));
259 return true;
260 }
261 break;
262 }
263 case QEvent::MouseMove: {
264 auto *mouse_event = static_cast<QMouseEvent *>(event);
265 if (active_split_handle_ == kHandle) {
266 set_split_from_position(mapFromGlobal(mouse_event->globalPosition().toPoint()));
267 return true;
268 }
269 break;
270 }
271 case QEvent::MouseButtonRelease: {
272 auto *mouse_event = static_cast<QMouseEvent *>(event);
273 if (mouse_event->button() == Qt::LeftButton && active_split_handle_ == kHandle) {
274 set_split_from_position(mapFromGlobal(mouse_event->globalPosition().toPoint()));
275 active_split_handle_ = ViewportSplitHandle::None;
276 update_splitter_cursor(mapFromGlobal(QCursor::pos()));
277 return true;
278 }
279 break;
280 }
281 default:
282 break;
283 }
284
285 return QWidget::eventFilter(watched, event);
286}

288void ViewportWidget::render_now() {
289 if (!isVisible()) {
290 return;
291 }
292
293 const double kElapsedSeconds = elapsed_.isValid() ? elapsed_.elapsed() / 1000.0 : 0.0;
294 const float kDeltaSeconds = static_cast<float>(std::clamp(kElapsedSeconds - last_frame_seconds_, 0.0, 0.25));
295 last_frame_seconds_ = kElapsedSeconds;
296 controller_.render_frame(kElapsedSeconds, kDeltaSeconds);
297}

333void ViewportWidget::update_splitter_cursor(const QPoint &position) {
334 if (active_split_handle_ == ViewportSplitHandle::Vertical) {
335 setCursor(Qt::SizeHorCursor);
336 return;
337 }
338
339 if (active_split_handle_ == ViewportSplitHandle::Horizontal) {
340 setCursor(Qt::SizeVerCursor);
341 return;
342 }
343
344 switch (controller_.layout_state().splitter_at(pixel_size(), viewport_point_from(position * devicePixelRatioF()))) {
345 case ViewportSplitHandle::Vertical:
346 setCursor(Qt::SizeHorCursor);
347 break;
348 case ViewportSplitHandle::Horizontal:
349 setCursor(Qt::SizeVerCursor);
350 break;
351 case ViewportSplitHandle::None:
352 unsetCursor();
353 break;
354 }
355}

366void ViewportWidget::update_splitter_handles() {
367 if (vertical_splitter_handle_ == nullptr || horizontal_splitter_handle_ == nullptr) {
368 return;
369 }
370
371 if (!controller_.quad_viewports_enabled()) {
372 vertical_splitter_handle_->hide();
373 horizontal_splitter_handle_->hide();
374 return;
375 }
376
377 const double kDpr = std::max(0.01, devicePixelRatioF());
378 const ViewportRect kVertical = controller_.layout_state().vertical_splitter_rect(pixel_size());
379 const ViewportRect kHorizontal = controller_.layout_state().horizontal_splitter_rect(pixel_size());
380
381 vertical_splitter_handle_->setGeometry(
382 static_cast<int>(std::round(kVertical.x / kDpr)),
383 static_cast<int>(std::round(kVertical.y / kDpr)),
384 std::max(1, static_cast<int>(std::round(kVertical.width / kDpr))),
385 std::max(1, static_cast<int>(std::round(kVertical.height / kDpr))));
386 horizontal_splitter_handle_->setGeometry(
387 static_cast<int>(std::round(kHorizontal.x / kDpr)),
388 static_cast<int>(std::round(kHorizontal.y / kDpr)),
389 std::max(1, static_cast<int>(std::round(kHorizontal.width / kDpr))),
390 std::max(1, static_cast<int>(std::round(kHorizontal.height / kDpr))));
391
392 vertical_splitter_handle_->show();
393 horizontal_splitter_handle_->show();
394 vertical_splitter_handle_->raise();
395 horizontal_splitter_handle_->raise();
396}
