56[[nodiscard]] quader::math::Vec3 project_to_plane(quader::math::Vec3 point,
57 const BoxConstructionPlane &plane) noexcept {
58 const float kDistance = quader::math::dot(point - plane.origin, plane.normal);
59 return point - scale(plane.normal, kDistance);

62[[nodiscard]] std::optional<quader::math::Vec3> ray_plane_point(
63 const BoxConstructionPlane &plane,
64 const ViewportRay &ray) noexcept {
65 const quader::math::Vec3 kDirection = normalized_axis(ray.direction, {});
66 if (tiny(kDirection)) {
67 return std::nullopt;
68 }
69
70 const float kDenom = quader::math::dot(kDirection, plane.normal);
71 if (std::abs(kDenom) <= kMinimumBoxSize) {
72 return std::nullopt;
73 }
74
75 const float kT = quader::math::dot(plane.origin - ray.origin, plane.normal) / kDenom;
76 if (kT < 0.0F || !quader::math::is_finite(kT)) {
77 return std::nullopt;
78 }
79
80 return project_to_plane(ray.origin + scale(kDirection, kT), plane);
81}

83[[nodiscard]] ConstructionPoint construction_point_from_event(const BoxConstructionPlane &plane,
84 const PointerEvent &event,
85 SurfaceHitFallback surface_hit_fallback) noexcept {
86 if (event.ray.has_value()) {
87 if (const auto kPoint = ray_plane_point(plane, *event.ray)) {
88 return ConstructionPoint{
89 .ok = true,
90 .point = *kPoint,
91 };
92 }
93 }
94
95 if (surface_hit_fallback == SurfaceHitFallback::Enabled && event.surface_hit.has_value()) {
96 return ConstructionPoint{
97 .ok = true,
98 .point = project_to_plane(event.surface_hit->position, plane),
99 };
100 }
101
102 return {};

105[[nodiscard]] std::array<float, 2> directional_snap_pair(float start_value,
106 float end_value,
107 float grid_size) noexcept {
108 const float kGrid = grid_or_default(grid_size);
109 if (end_value >= start_value) {
110 const float kStart = std::floor(start_value / kGrid) * kGrid;
111 float end = std::ceil(end_value / kGrid) * kGrid;
112 if (end - kStart <= kMinimumBoxSize) {
113 end = kStart + kGrid;
114 }
115 return { kStart, end };
116 }
117
118 const float kStart = std::ceil(start_value / kGrid) * kGrid;
119 float end = std::floor(end_value / kGrid) * kGrid;
120 if (kStart - end <= kMinimumBoxSize) {
121 end = kStart - kGrid;
122 }
123 return { kStart, end };
124}

138[[nodiscard]] std::array<quader::math::Vec3, 4> footprint_points(quader::math::Vec3 start,
139 quader::math::Vec3 end,
140 quader::math::Vec3 axis_u,
141 quader::math::Vec3 axis_v) noexcept {
142 const quader::math::Vec3 kDelta = end - start;
143 const quader::math::Vec3 kU = scale(axis_u, quader::math::dot(kDelta, axis_u));
144 const quader::math::Vec3 kV = scale(axis_v, quader::math::dot(kDelta, axis_v));
145 return {
146 start,
147 start + kV,
148 start + kU + kV,
149 start + kU,
150 };
151}

153[[nodiscard]] bool footprint_has_area(const std::array<quader::math::Vec3, 4> &footprint) noexcept {
154 return !tiny(quader::math::cross(footprint[1] - footprint[0], footprint[3] - footprint[0]));

158 const quader::math::Vec3 kNormal = state.height >= 0.0F ? state.plane.normal : scale(state.plane.normal, -1.0F);

174BoxConstructionPlane make_box_construction_plane(quader::math::Vec3 origin,
175 quader::math::Vec3 normal) {
176 BoxConstructionPlane plane;
177 plane.origin = origin;
178 plane.normal = normalized_axis(normal);
179 plane.snap_origin = project_to_plane({}, plane);
180
181 quader::math::Vec3 preferred{ 1.0F, 0.0F, 0.0F };
182 if (std::abs(quader::math::dot(preferred, plane.normal)) > 0.9F) {
183 preferred = { 0.0F, 0.0F, 1.0F };
184 }
185
186 const quader::math::Vec3 kProjected = preferred - scale(plane.normal, quader::math::dot(preferred, plane.normal));
187 plane.axis_u = normalized_axis(kProjected, { 1.0F, 0.0F, 0.0F });
188 plane.axis_v = normalized_axis(quader::math::cross(plane.normal, plane.axis_u),
189 { 0.0F, 0.0F, 1.0F });
190 return plane;
191}

212bool BoxTool::on_pointer_event(const PointerEvent &event, ToolContext &context) {
213 if (event.navigation_active) {
214 return false;
215 }
216
217 if (event.phase == PointerPhase::Hover || event.phase == PointerPhase::Move) {
218 return state_.stage == BoxToolStage::Idle ? update_hover(event)
219 : update_footprint(event);
220 }
221
222 if (event.button != PointerButton::Left) {
223 return false;
224 }
225
226 if (event.phase == PointerPhase::Press && event.pressed) {
227 if (state_.stage == BoxToolStage::Idle) {
228 return begin_footprint(event);
229 }
230 return finish_footprint_and_commit(event, context);
231 }
232
233 if (event.phase == PointerPhase::Release && state_.stage == BoxToolStage::Footprint) {
234 return finish_footprint_and_commit(event, context);
235 }
236
237 return false;
238}

258bool BoxTool::update_hover(const PointerEvent &event) {
259 const quader::math::Vec3 kOrigin = event.surface_hit.has_value() ? event.surface_hit->position
260 : quader::math::Vec3{};
261 const quader::math::Vec3 kNormal = event.surface_hit.has_value() ? event.surface_hit->normal
262 : quader::math::Vec3{ 0.0F, 1.0F, 0.0F };
263 state_.plane = make_box_construction_plane(kOrigin, kNormal);
264 const ConstructionPoint kPoint = construction_point_from_event(
265 state_.plane,
266 event,
267 SurfaceHitFallback::Enabled);
268 if (!kPoint.ok) {
269 preview_.clear();
270 return false;
271 }
272
273 state_.raw_start = kPoint.point;
274 state_.raw_end = kPoint.point;
275 state_.view_index = event.view_index;
276 if (!update_footprint(PointerEvent{
277 .position = event.position,
278 .button = PointerButton::None,
279 .pressed = false,
280 .phase = PointerPhase::Move,
281 .navigation_active = false,
282 .snap_to_grid = event.snap_to_grid,
283 .grid_size = event.grid_size,
284 .ray = event.ray,
285 .surface_hit = event.surface_hit,
286 .view_index = event.view_index,
287 })) {
288 preview_.clear();
289 return false;
290 }
291 state_.hover_cell = state_.footprint;
292 state_.stage = BoxToolStage::Idle;
293 rebuild_preview();
294 return true;
295}

297bool BoxTool::begin_footprint(const PointerEvent &event) {
298 const quader::math::Vec3 kOrigin = event.surface_hit.has_value() ? event.surface_hit->position
299 : quader::math::Vec3{};
300 const quader::math::Vec3 kNormal = event.surface_hit.has_value() ? event.surface_hit->normal
301 : quader::math::Vec3{ 0.0F, 1.0F, 0.0F };
302 state_.plane = make_box_construction_plane(kOrigin, kNormal);
303 const ConstructionPoint kPoint = construction_point_from_event(
304 state_.plane,
305 event,
306 SurfaceHitFallback::Enabled);
307 if (!kPoint.ok) {
308 return false;
309 }
310
311 state_.stage = BoxToolStage::Footprint;
312 state_.raw_start = kPoint.point;
313 state_.raw_end = kPoint.point;
314 state_.height = 0.0F;
315 state_.view_index = event.view_index;
316 state_.preview_valid = false;
317 return update_footprint(event);
318}

320bool BoxTool::update_footprint(const PointerEvent &event) {
321 const ConstructionPoint kPoint = construction_point_from_event(
322 state_.plane,
323 event,
324 state_.stage == BoxToolStage::Footprint ? SurfaceHitFallback::Disabled : SurfaceHitFallback::Enabled);
325 if (!kPoint.ok) {
326 return false;
327 }
328
329 state_.raw_end = kPoint.point;
330 state_.view_index = event.view_index;
331 if (event.snap_to_grid) {
332 const float kStartU = plane_coordinate(state_.raw_start, state_.plane.snap_origin, state_.plane.axis_u);
333 const float kEndU = plane_coordinate(state_.raw_end, state_.plane.snap_origin, state_.plane.axis_u);
334 const float kStartV = plane_coordinate(state_.raw_start, state_.plane.snap_origin, state_.plane.axis_v);
335 const float kEndV = plane_coordinate(state_.raw_end, state_.plane.snap_origin, state_.plane.axis_v);
336 const auto kUPair = directional_snap_pair(kStartU, kEndU, event.grid_size);
337 const auto kVPair = directional_snap_pair(kStartV, kEndV, event.grid_size);
338 state_.start = plane_point(state_.plane, kUPair[0], kVPair[0]);
339 state_.end = plane_point(state_.plane, kUPair[1], kVPair[1]);
340 } else {
341 state_.start = project_to_plane(state_.raw_start, state_.plane);
342 state_.end = project_to_plane(state_.raw_end, state_.plane);
343 }
344
345 state_.height = grid_or_default(event.grid_size);
346 state_.footprint = footprint_points(state_.start, state_.end, state_.plane.axis_u, state_.plane.axis_v);
347 state_.preview_valid = footprint_has_area(state_.footprint);
348 rebuild_preview();
349 return true;
350}

361bool BoxTool::commit(ToolContext &context) {
362 if (state_.stage != BoxToolStage::Footprint || !state_.preview_valid) {
363 return false;
364 }
365
366 const auto kCorners = preview_corners(state_);
367 auto mesh = quader::mesh::ops::make_box_from_corners(quader::mesh::ops::BoxCorners{ kCorners });
368 if (!mesh) {
369 reset();
370 return false;
371 }
372
373 auto result = context.execute_command(std::make_unique<quader::commands::CreateMeshObjectCommand>(
374 "Box",
375 std::move(mesh).value(),
376 quader::document::Transform{},
377 quader::document::default_box_material()));
378 const bool kApplied = result.is_applied();
379 reset();
380 return kApplied;
381}

388void BoxTool::rebuild_preview() {
389 preview_.clear();
390
391 if (state_.stage == BoxToolStage::Idle && !footprint_has_area(state_.hover_cell)) {
392 return;
393 }
394
395 const std::array<quader::math::Vec3, 4> &footprint = state_.stage == BoxToolStage::Idle
396 ? state_.hover_cell
397 : state_.footprint;
398 if (!footprint_has_area(footprint)) {
399 return;
400 }
401
402 preview_.active = true;
403 preview_.overlay_only = true;
404 preview_.status_text = "Box footprint";
405 preview_.view_index = state_.view_index;
406 for (std::size_t index = 0; index < footprint.size(); ++index) {
407 preview_.world_points.push_back(footprint[index]);
408 preview_.world_segments.push_back(ToolPreviewWorldSegment{
409 .start = footprint[index],
410 .end = footprint[(index + 1U) % footprint.size()],
411 });
412 }
413
414 if (state_.stage == BoxToolStage::Footprint && state_.preview_valid) {
415 preview_.boxes.push_back(ToolPreviewBox{
416 .corners = preview_corners(state_),
417 .active = true,
418 });
419 }

421}
