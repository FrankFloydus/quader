58TEST(FrameSnapshot, BuilderFreezesPrototypeFrameIntoImmutableSnapshot) {
59 std::array<crimson::PrototypeCamera, 1> cameras = { make_camera() };
60 std::array<crimson::PrototypeViewportView, 1> views = {
61 crimson::PrototypeViewportView{
62 .rect = crimson::PrototypeViewportRect{ 4, 8, 320, 200 },
63 .camera_index = 0,
64 .debug_name = "Perspective",
65 },
66 };
67 std::array<crimson::PickingRequest, 1> picking_requests = {
68 crimson::PickingRequest{ .request_id = 77, .view_index = 0, .x_px = 12, .y_px = 24 },
69 };
70 crimson::PrototypeViewportFrame frame{
71 .target_extent = crimson::ViewportExtent{ 640, 480, 1.0F },
72 .views = views,
73 .cameras = cameras,
74 .picking_requests = picking_requests,
75 .animation_enabled = true,
76 .elapsed_seconds = 2.0,
77 };
78
79 const crimson::FrameBuilder kBuilder;
80 auto built = kBuilder.build_prototype_snapshot(frame);
81 expect_true(built.has_value(), "valid prototype frame builds a snapshot");
82 if (!built) {
83 return;
84 }
85
86 crimson::FrameSnapshot snapshot = std::move(built).value();
87 views[0].rect.width = 1;
88 cameras[0].eye.x = 99.0F;
89 picking_requests[0].x_px = 99;
90 frame.elapsed_seconds = 99.0;
91
92 expect_true(snapshot.target_extent().width_px == 640, "snapshot keeps copied target extent");
93 expect_true(snapshot.views().size() == 1, "snapshot keeps copied view count");
94 expect_true(snapshot.views()[0].rect.width == 320, "snapshot view rect is immutable from source changes");
95 expect_true(snapshot.views()[0].camera.eye.x == 0.0F, "snapshot camera is immutable from source changes");
96 expect_true(snapshot.elapsed_seconds() == 2.0, "snapshot keeps copied elapsed seconds");
97 expect_true(snapshot.objects().empty(), "empty prototype frame has no default render object");
98 expect_true(snapshot.mesh_uploads().empty(), "empty prototype frame has no mesh uploads");
99 expect_true(snapshot.overlays().size() == 1, "prototype snapshot contains one overlay command");
100 expect_true(snapshot.grid_overlay_payloads().size() == 1, "prototype snapshot contains one grid overlay payload");
101 expect_true(snapshot.line_overlay_payloads().empty(), "empty prototype frame has no line overlay payload");
102 expect_true(snapshot.picking_requests().size() == 1, "prototype snapshot copies picking requests");
103 expect_true(snapshot.picking_requests()[0].x_px == 12, "picking request is immutable from source changes");
104 expect_true(
105 snapshot.viewport_settings().exposure_ev100 == 0.0F,
106 "prototype snapshot preserves non-physical exposure until fixture lighting is physical");
107 expect_true(
108 snapshot.overlays()[0].primitive == crimson::OverlayPrimitive::Grid && snapshot.overlays()[0].base_shader == crimson::BaseShaderId::OverlayUnlit,
109 "prototype grid is emitted as an OverlayUnlit grid command");
110 expect_true(
111 snapshot.overlays()[0].payload_offset == 0 && snapshot.overlays()[0].payload_count == 1,
112 "grid overlay command points at immutable grid payload data");
113}

115TEST(FrameSnapshot, BuilderCopiesCallerRenderObjectsAndOverlayPayloads) {
116 std::array<crimson::PrototypeCamera, 1> cameras = { make_camera() };
117 std::array<crimson::PrototypeViewportView, 1> views = {
118 crimson::PrototypeViewportView{
119 .rect = crimson::PrototypeViewportRect{ 0, 0, 320, 200 },
120 .camera_index = 0,
121 .debug_name = "Perspective",
122 },
123 };
124 crimson::RenderMeshUpload upload;
125 upload.handle = crimson::RenderMeshHandle{ 7, 2 };
126 upload.revision = crimson::RenderMeshRevision{ 1, 2, 3 };
127 upload.position_normal_interleaved = {
128 0.0F,
129 0.0F,
130 0.0F,
131 0.0F,
132 1.0F,
133 0.0F,
134 1.0F,
135 0.0F,
136 0.0F,
137 0.0F,
138 1.0F,
139 0.0F,
140 0.0F,
141 1.0F,
142 0.0F,
143 0.0F,
144 1.0F,
145 0.0F,
146 };
147 upload.indices = { 0, 1, 2 };
148 upload.bounds = quader::math::Aabb{ .min = { 0.0F, 0.0F, 0.0F }, .max = { 1.0F, 1.0F, 0.0F } };
149 std::array<crimson::RenderMeshUpload, 1> uploads = { upload };
150 std::array<crimson::RenderObject, 1> objects = {
151 crimson::RenderObject{
152 .object_id = 99,
153 .mesh = crimson::RenderMeshHandle{ 7, 2 },
154 .base_shader = crimson::BaseShaderId::OpaquePbr,
155 .world_bounds = quader::math::Aabb{ .min = { 0.0F, 0.0F, 0.0F }, .max = { 1.0F, 1.0F, 1.0F } },
156 .queue = crimson::RenderQueue::Opaque,
157 },
158 };
159 std::array<crimson::LineOverlaySegment, 1> lines = {
160 crimson::LineOverlaySegment{ .start = { 0.0F, 0.0F, 0.0F }, .end = { 1.0F, 0.0F, 0.0F } },
161 };
162 std::array<crimson::OverlayCommand, 1> overlays = {
163 crimson::OverlayCommand{
164 .view_index = 0,
165 .primitive = crimson::OverlayPrimitive::LineList,
166 .depth_mode = crimson::OverlayDepthMode::AlwaysOnTop,
167 .payload_offset = 0,
168 .payload_count = 1,
169 },
170 };
171 crimson::PrototypeViewportFrame frame{
172 .target_extent = crimson::ViewportExtent{ 640, 480, 1.0F },
173 .views = views,
174 .cameras = cameras,
175 .mesh_uploads = uploads,
176 .objects = objects,
177 .overlays = overlays,
178 .line_overlay_payloads = lines,
179 };
180
181 const crimson::FrameBuilder kBuilder;
182 auto built = kBuilder.build_prototype_snapshot(frame);
183 expect_true(built.has_value(), "valid frame with caller render data builds");
184 if (!built) {
185 return;
186 }
187
188 crimson::FrameSnapshot snapshot = std::move(built).value();
189 objects[0].object_id = 1;
190 uploads[0].indices[0] = 2;
191 lines[0].end.x = 5.0F;
192
193 expect_true(snapshot.mesh_uploads().size() == 1, "snapshot copies mesh upload payloads");
194 expect_true(snapshot.mesh_uploads()[0].indices[0] == 0, "mesh upload indices are immutable from source changes");
195 expect_true(snapshot.objects().size() == 1 && snapshot.objects()[0].object_id == 99, "snapshot copies caller render objects");
196 expect_true(snapshot.line_overlay_payloads().size() == 1, "snapshot copies line overlay payloads");
197 expect_true(snapshot.line_overlay_payloads()[0].end.x == 1.0F, "line overlay payload is immutable from source changes");
198}

200TEST(FrameSnapshot, RenderWorldStoresPreparedObjectsById) {
201 crimson::RenderWorld world;
202 crimson::RenderObject object;
203 object.object_id = 7;
204 object.visible = true;
205
206 const crimson::RenderObjectId kId = world.add_object(object);
207 expect_true(kId == 7, "render world preserves explicit object ids");
208 expect_true(world.objects().size() == 1, "render world stores added objects");
209
210 object.visible = false;
211 expect_true(world.update_object(kId, object), "render world updates existing objects");
212 expect_true(!world.objects()[0].visible, "render world exposes updated prepared object data");
213
214 expect_true(world.remove_object(kId), "render world removes existing objects");
215 expect_true(world.objects().empty(), "render world is empty after removal");
216}
