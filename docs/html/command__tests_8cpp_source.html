47TEST(Command, CreateMeshObjectCommandExecutesUndoesAndRedoesSemanticState) {
48 quader::document::Document document;
49 quader::commands::CommandHistory history;
50 const auto kBefore = capture_command_state(document);
51
52 auto mesh = quader::tests::document_fixtures::make_triangle_mesh();
53 const auto kTransform = translated_transform(2.0F, 3.0F, 4.0F);
54
55 auto result = history.execute(std::make_unique<quader::commands::CreateMeshObjectCommand>(
56 "Triangle",
57 std::move(mesh.mesh),
58 kTransform),
59 document);
60 EXPECT_EQ((result).status, quader::commands::CommandStatus::Applied);
61 EXPECT_TRUE(history.can_undo());
62 EXPECT_FALSE(history.can_redo());
63 EXPECT_EQ(history.undo_name(), std::string_view("Create Mesh Object"));
64
65 const auto kAfter = capture_command_state(document);
66 EXPECT_NE(kAfter, kBefore);
67 EXPECT_EQ(kAfter.objects.size(), 1U);
68 EXPECT_EQ(kAfter.objects.front().name, std::string("Triangle"));
69 EXPECT_TRUE(kAfter.objects.front().transform == kTransform);
70 EXPECT_EQ(kAfter.objects.front().material, quader::document::default_box_material());
71 EXPECT_EQ(kAfter.objects.front().vertex_count, 3U);
72 EXPECT_EQ(kAfter.objects.front().face_count, 1U);
73
74 result = history.undo(document);
75 EXPECT_EQ((result).status, quader::commands::CommandStatus::Applied);
76 EXPECT_EQ(capture_command_state(document), kBefore);
77 EXPECT_FALSE(history.can_undo());
78 EXPECT_TRUE(history.can_redo());
79 EXPECT_EQ(history.redo_name(), std::string_view("Create Mesh Object"));
80
81 result = history.redo(document);
82 EXPECT_EQ((result).status, quader::commands::CommandStatus::Applied);
83 EXPECT_EQ(capture_command_state(document), kAfter);
84 EXPECT_TRUE(history.can_undo());
85 EXPECT_FALSE(history.can_redo());
86}
