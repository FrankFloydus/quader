102TEST(RenderGraphCorrectness, ToneMapOverlayAndPresentResourceIsolationIsExplicit) {
103 const crimson::RenderGraph kGraph = crimson::make_v1_correctness_render_graph(
104 crimson::ViewportExtent{ 800, 600, 1.0F });
105
106 const crimson::RenderPass *depth_prepass = find_pass(kGraph, "DepthPrepass");
107 expect_true(depth_prepass != nullptr, "depth prepass exists");
108 expect_true(
109 depth_prepass != nullptr && depth_prepass->resources.size() == 1 && find_use(*depth_prepass, crimson::kSceneDepthTargetName) != nullptr && find_use(*depth_prepass, crimson::kHdrSceneColorTargetName) == nullptr,
110 "depth prepass writes SceneDepth without touching HDR scene color");
111
112 const crimson::RenderPass *opaque_pbr = find_pass(kGraph, "OpaquePbrPass");
113 expect_true(opaque_pbr != nullptr, "opaque PBR pass exists");
114 expect_true(
115 opaque_pbr != nullptr && find_use(*opaque_pbr, crimson::kSceneDepthTargetName) != nullptr,
116 "opaque PBR pass reads the shared SceneDepth target");
117 expect_true(
118 opaque_pbr != nullptr && find_use(*opaque_pbr, crimson::kHdrSceneColorTargetName) != nullptr,
119 "opaque PBR pass writes HDR scene color");
120
121 const crimson::RenderPass *tone_map = find_pass(kGraph, "ToneMapPass");
122 expect_true(tone_map != nullptr, "tone-map pass exists");
123 expect_true(
124 tone_map != nullptr && find_use(*tone_map, crimson::kHdrSceneColorTargetName) != nullptr,
125 "tone-map pass reads HDR scene color");
126 expect_true(
127 tone_map != nullptr && find_use(*tone_map, crimson::kToneMappedColorTargetName) != nullptr,
128 "tone-map pass writes tone-mapped color");
129 expect_true(
130 tone_map != nullptr && find_use(*tone_map, crimson::kSceneDepthTargetName) == nullptr,
131 "tone-map pass does not bind scene depth");
132
133 for (std::string_view pass_name : {
134 "OverlayDepthTestedPass",
135 "OverlayXRayPass",
136 "OverlayAlwaysOnTopPass",
137 }) {
138 const crimson::RenderPass *overlay = find_pass(kGraph, pass_name);
139 expect_true(overlay != nullptr, "overlay pass exists");
140 expect_true(
141 overlay != nullptr && find_use(*overlay, crimson::kHdrSceneColorTargetName) == nullptr,
142 "overlay pass never touches HDR scene color");
143 expect_true(
144 overlay != nullptr && find_use(*overlay, crimson::kToneMappedColorTargetName) != nullptr,
145 "overlay pass modifies tone-mapped color");
146 }
147
148 const crimson::RenderPass *picking = find_pass(kGraph, "PickingPass");
149 expect_true(picking != nullptr, "picking pass exists");
150 if (picking != nullptr) {
151 for (const crimson::ResourceUse &use : picking->resources) {
152 expect_true(
153 !crimson::is_color_managed_frame_target(use.resource_name),
154 "picking pass avoids color-managed targets");
155 }
156 }
157
158 std::uint32_t backbuffer_users = 0;
159 for (const crimson::RenderPass &pass : kGraph.passes()) {
160 if (find_use(pass, crimson::kBackbufferColorTargetName) != nullptr) {
161 ++backbuffer_users;
162 expect_true(pass.name == "PresentPass", "only PresentPass writes final display target");
163 }
164 }
165 expect_true(backbuffer_users == 1, "display conversion reaches the backbuffer exactly once");
166}

168TEST(RenderGraphCorrectness, ResizeUpdatesV1Targets) {
169 crimson::RenderGraph graph = crimson::make_v1_correctness_render_graph(crimson::ViewportExtent{ 640, 480, 1.0F });
170 const crimson::RenderResourceRecord *before = graph.resources().find(crimson::kHdrSceneColorTargetName);
171 const std::uint64_t kBeforeGeneration = before == nullptr ? 0 : before->generation;
172
173 graph.resize(crimson::ViewportExtent{ 1024, 768, 1.0F });
174 const crimson::RenderResourceRecord *after = graph.resources().find(crimson::kHdrSceneColorTargetName);
175 expect_true(after != nullptr, "HDR scene color target exists after resize");
176 expect_true(after != nullptr && after->generation != kBeforeGeneration, "resize updates target generation");
177 expect_true(after != nullptr && after->desc.extent.width_px == 1024, "resize updates target width");
178}
