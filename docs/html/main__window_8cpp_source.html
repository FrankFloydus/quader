38QString notification_text(const NotificationMessage &message) {
39 if (message.detail.isEmpty()) {
40 return message.summary;
41 }
42
43 return QStringLiteral("%1: %2").arg(message.summary, message.detail);

179void MainWindow::connect_actions() {
180 connect(&context_.actions, &ActionRegistry::action_triggered, this, [this](ActionId id) {
181 switch (id) {
182 case ActionId::Exit:
183 close();
184 break;
185 case ActionId::OpenScene:
186 (void)context_.imports.open_scene(this);
187 break;
188 case ActionId::FocusViewport:
189 if (viewport_ != nullptr) {
190 viewport_->setFocus(Qt::OtherFocusReason);
191 }
192 break;
193 case ActionId::ResetWorkspaceLayout:
194 reset_workspace();
195 break;
196 default:
197 break;
198 }
199 });
200
201 connect(&context_.actions, &ActionRegistry::action_toggled, this, [this](ActionId id, bool checked) {
202 if (id == ActionId::ToggleQuadViewports && viewport_controller_ != nullptr) {
203 viewport_controller_->set_quad_viewports_enabled(checked);
204 }
205 });
206}

208void MainWindow::connect_notifications() {
209 connect(&context_.notifications,
210 &NotificationService::notification_posted,
211 this,
212 [this](const NotificationMessage &message) {
213 const QString kText = notification_text(message);
214 const int kTimeout = message.timeout_ms < 0 ? 0 : message.timeout_ms;
215
216 if (message.severity == NotificationSeverity::Error && renderer_label_ != nullptr) {
217 renderer_label_->setText(QStringLiteral("Renderer: error"));
218 }
219
220 statusBar()->showMessage(kText, kTimeout);
221 });
222}

224void MainWindow::connect_viewport() {
225 connect(viewport_controller_.get(), &ViewportController::renderer_ready, this, [this](const QString &renderer_name) {
226 renderer_label_->setText(QStringLiteral("Renderer: %1").arg(renderer_name));
227 context_.notifications.show_status(QStringLiteral("Crimson initialized with %1").arg(renderer_name), 3000);
228 context_.viewport_diagnostics.refresh();
229 });
230
231 connect(viewport_controller_.get(), &ViewportController::frame_stats_changed, this, [this](const ViewportFrameStats &stats) {
232 update_stats_label(stats);
233 context_.viewport_diagnostics.refresh();
234 });
235
236 connect(viewport_controller_.get(), &ViewportController::render_error, this, [this](const QString &summary, const QString &detail) {
237 context_.notifications.show_error(NotificationMessage{
238 NotificationSeverity::Error,
239 summary,
240 detail,
241 0,
242 });
243 context_.viewport_diagnostics.refresh();
244 });
245
246 connect(viewport_controller_.get(), &ViewportController::viewport_layout_changed, this, [this](bool quad_enabled, const QString &layout_name) {
247 quad_viewports_enabled_ = quad_enabled;
248 context_.action_state_updater.refresh_from_snapshot(editor_state_with_viewport());
249 context_.notifications.show_status(layout_name, 2000);
250 });
251
252 connect(&context_.viewport_diagnostics,
253 &ViewportDiagnosticsService::summary_changed,
254 this,
255 &MainWindow::update_diagnostics_label);
256 connect(&context_.viewport_diagnostics,
257 &ViewportDiagnosticsService::diagnostics_unavailable,
258 this,
259 &MainWindow::update_diagnostics_label);
260}

262void MainWindow::restore_workspace() {
263 const QByteArray kGeometry = context_.settings.main_window_geometry();
264 if (!kGeometry.isEmpty()) {
265 restoreGeometry(kGeometry);
266 }
267
268 const QByteArray kState = context_.settings.main_window_state();
269 bool state_restored = false;
270 if (!kState.isEmpty()) {
271 state_restored = restoreState(kState, SettingsService::kWorkspaceStateVersion);
272 }
273
274 if (!state_restored && panel_host_ != nullptr) {
275 panel_host_->reset_to_default_layout();
276 }
277
278 if (panel_host_ != nullptr) {
279 panel_host_->restore_workspace(context_.settings);
280 }
281}

283void MainWindow::save_workspace() {
284 context_.settings.set_main_window_geometry(saveGeometry());
285 context_.settings.set_main_window_state(saveState(SettingsService::kWorkspaceStateVersion));
286 if (panel_host_ != nullptr) {
287 panel_host_->save_workspace(context_.settings);
288 }
289 context_.settings.sync();
290}

292void MainWindow::reset_workspace() {
293 context_.settings.reset_workspace();
294
295 if (panel_host_ != nullptr) {
296 panel_host_->reset_to_default_layout();
297 }
298 if (viewport_ != nullptr) {
299 viewport_controller_->set_quad_viewports_enabled(false);
300 }
301
302 resize(1280, 800);
303 context_.notifications.show_status(QStringLiteral("Workspace layout reset"), 2000);
304}

306EditorStateSnapshot MainWindow::editor_state_with_viewport() const {
307 EditorStateSnapshot snapshot = context_.editor_state.editor_state_snapshot();
308 snapshot.viewport_available = viewport_ != nullptr;
309 snapshot.viewport_state_known = true;
310 snapshot.quad_viewports_enabled = quad_viewports_enabled_;

314void MainWindow::update_stats_label(const ViewportFrameStats &stats) {
315 if (stats_label_ == nullptr) {
316 return;
317 }
318
319 stats_label_->setText(QStringLiteral("%1x%2 | %3 view%4 | %5 fps")
320 .arg(stats.width)
321 .arg(stats.height)
322 .arg(stats.viewport_count)
323 .arg(stats.viewport_count == 1 ? QString() : QStringLiteral("s"))
324 .arg(stats.fps, 0, 'f', 0));
325}
