107[[nodiscard]] quader::math::Vec3 rotate_x(quader::math::Vec3 value, float radians) noexcept {
108 const float kSin = std::sin(radians);
109 const float kCos = std::cos(radians);
110 return { value.x, value.y * kCos - value.z * kSin, value.y * kSin + value.z * kCos };
111}
112
113[[nodiscard]] quader::math::Vec3 rotate_y(quader::math::Vec3 value, float radians) noexcept {
114 const float kSin = std::sin(radians);
115 const float kCos = std::cos(radians);
116 return { value.x * kCos + value.z * kSin, value.y, -value.x * kSin + value.z * kCos };
117}
118
119[[nodiscard]] quader::math::Vec3 rotate_z(quader::math::Vec3 value, float radians) noexcept {
120 const float kSin = std::sin(radians);
121 const float kCos = std::cos(radians);
122 return { value.x * kCos - value.y * kSin, value.x * kSin + value.y * kCos, value.z };
123}

125[[nodiscard]] quader::math::Vec3 rotate_euler(quader::math::Vec3 value,
126 quader::math::Vec3 degrees) noexcept {
127 value = rotate_x(value, radians_from_degrees(degrees.x));
128 value = rotate_y(value, radians_from_degrees(degrees.y));
129 return rotate_z(value, radians_from_degrees(degrees.z));
130}

132[[nodiscard]] quader::math::Vec3 transform_point(quader::math::Vec3 point,
133 quader::document::Transform transform) noexcept {
134 point = quader::math::Vec3{
135 point.x * transform.scale.x,
136 point.y * transform.scale.y,
137 point.z * transform.scale.z,
138 };
139 return rotate_euler(point, transform.rotation_euler) + transform.translation;
140}

142[[nodiscard]] std::array<float, 16> render_transform_from(
143 quader::document::Transform transform) noexcept {
144 const quader::math::Vec3 kColumn0 = rotate_euler({ transform.scale.x, 0.0F, 0.0F }, transform.rotation_euler);
145 const quader::math::Vec3 kColumn1 = rotate_euler({ 0.0F, transform.scale.y, 0.0F }, transform.rotation_euler);
146 const quader::math::Vec3 kColumn2 = rotate_euler({ 0.0F, 0.0F, transform.scale.z }, transform.rotation_euler);
147 return {
148 kColumn0.x,
149 kColumn0.y,
150 kColumn0.z,
151 0.0F,
152 kColumn1.x,
153 kColumn1.y,
154 kColumn1.z,
155 0.0F,
156 kColumn2.x,
157 kColumn2.y,
158 kColumn2.z,
159 0.0F,
160 transform.translation.x,
161 transform.translation.y,
162 transform.translation.z,
163 1.0F,
164 };

177[[nodiscard]] std::uint64_t hash_combine(std::uint64_t seed, std::uint64_t value) noexcept {
178 return seed ^ (value + 0x9e3779b97f4a7c15ULL + (seed << 6U) + (seed >> 2U));
179}

181[[nodiscard]] std::uint64_t hash_float(float value) noexcept {
182 return std::bit_cast<std::uint32_t>(value);

185void append_revision_point(std::uint64_t &hash, quader::math::Vec3 point) noexcept {
186 hash = hash_combine(hash, hash_float(point.x));
187 hash = hash_combine(hash, hash_float(point.y));
188 hash = hash_combine(hash, hash_float(point.z));

191void append_upload_vertex(crimson::RenderMeshUpload &upload,
192 quader::math::Vec3 position,
193 quader::math::Vec3 normal) {
194 upload.position_normal_interleaved.push_back(position.x);
195 upload.position_normal_interleaved.push_back(position.y);
196 upload.position_normal_interleaved.push_back(position.z);
197 upload.position_normal_interleaved.push_back(normal.x);
198 upload.position_normal_interleaved.push_back(normal.y);
199 upload.position_normal_interleaved.push_back(normal.z);
200 upload.indices.push_back(static_cast<std::uint32_t>(upload.indices.size()));

203[[nodiscard]] quader::math::Vec3 normalized_or(quader::math::Vec3 value,
204 quader::math::Vec3 fallback) noexcept {
205 const quader::math::Vec3 kNormalized = quader::math::normalized(value);
206 return quader::math::length_squared(kNormalized) <= quader::math::kDefaultEpsilon * quader::math::kDefaultEpsilon
207 ? fallback
208 : kNormalized;
209}

211[[nodiscard]] crimson::RenderMeshRevision revision_for_mesh(
212 const quader::mesh::Polyhedron &mesh) {
213 std::uint64_t geometry_hash = 1469598103934665603ULL;
214 for (const auto kVertex : mesh.vertex_ids()) {
215 auto position = mesh.vertex_position(kVertex);
216 if (position) {
217 append_revision_point(geometry_hash, position.value());
218 }
219 }
220
221 std::uint64_t topology_hash = 1099511628211ULL;
222 for (const auto kFace : mesh.face_ids()) {
223 topology_hash = hash_combine(topology_hash, kFace.index());
224 auto vertices = quader::mesh::face_vertices(mesh, kFace);
225 if (vertices) {
226 topology_hash = hash_combine(topology_hash, vertices.value().size());
227 for (const auto kVertex : vertices.value()) {
228 topology_hash = hash_combine(topology_hash, kVertex.index());
229 }
230 }
231 }
232
233 return crimson::RenderMeshRevision{
234 .geometry_version = geometry_hash,
235 .topology_version = topology_hash,
236 .attribute_version = geometry_hash ^ topology_hash,
237 };
238}

240[[nodiscard]] std::optional<crimson::RenderMeshUpload> make_mesh_upload(
241 const quader::document::MeshObject &object) {
242 crimson::RenderMeshUpload upload;
243 upload.handle = render_mesh_handle_for(object.id);
244 upload.revision = revision_for_mesh(object.mesh);
245
246 for (const auto kVertex : object.mesh.vertex_ids()) {
247 auto position = object.mesh.vertex_position(kVertex);
248 if (!position) {
249 return std::nullopt;
250 }
251 quader::math::expand(upload.bounds, position.value());
252 }
253 if (quader::math::empty(upload.bounds)) {
254 return std::nullopt;
255 }
256
257 for (const auto kFace : object.mesh.face_ids()) {
258 auto face_vertices = quader::mesh::face_vertices(object.mesh, kFace);
259 if (!face_vertices || face_vertices.value().size() < 3U) {
260 continue;
261 }
262
263 std::vector<quader::math::Vec3> points;
264 points.reserve(face_vertices.value().size());
265 for (const auto kVertex : face_vertices.value()) {
266 auto position = object.mesh.vertex_position(kVertex);
267 if (!position) {
268 points.clear();
269 break;
270 }
271 points.push_back(position.value());
272 }
273 if (points.size() < 3U) {
274 continue;
275 }
276
277 const quader::math::Vec3 kNormal = normalized_or(
278 quader::geometry::polygon_normal(points),
279 { 0.0F, 1.0F, 0.0F });
280 for (std::size_t index = 1U; index + 1U < points.size(); ++index) {
281 append_upload_vertex(upload, points[0], kNormal);
282 append_upload_vertex(upload, points[index], kNormal);
283 append_upload_vertex(upload, points[index + 1U], kNormal);
284 }
285 }
286
287 if (upload.indices.empty()) {
288 return std::nullopt;
289 }
290 return upload;
291}

329ViewportDiagnosticsSnapshot viewport_diagnostics_from_crimson(
330 const crimson::RendererDiagnosticsSnapshot &snapshot,
331 QString renderer_name) {
332 ViewportDiagnosticsSnapshot result;
333 result.renderer_name = !renderer_name.isEmpty()
334 ? std::move(renderer_name)
335 : QString::fromStdString(snapshot.status.backend_name);
336 if (snapshot.latest_frame_stats) {
337 result.frame = viewport_frame_stats_from(*snapshot.latest_frame_stats);
338 }
339
340 result.passes.reserve(static_cast<qsizetype>(snapshot.pass_stats.size()));
341 for (const crimson::RenderPassStats &pass : snapshot.pass_stats) {
342 result.passes.push_back(ViewportRenderPassStats{
343 .pass_name = QString::fromStdString(pass.pass_name),
344 .resource_read_count = bounded_int(pass.resource_read_count),
345 .resource_write_count = bounded_int(pass.resource_write_count),
346 .draw_call_count = bounded_int(pass.draw_call_count),
347 .draw_packet_count = bounded_int(pass.draw_packet_count),
348 .cpu_time_us = static_cast<quint64>(pass.cpu_time_us),
349 });
350 }
351
352 result.counters.reserve(static_cast<qsizetype>(snapshot.counters.size()));
353 for (const crimson::RendererCounter &counter : snapshot.counters) {
354 result.counters.push_back(ViewportRendererCounter{
355 .domain = qstring_from_ascii(crimson::renderer_counter_domain_name(counter.domain)),
356 .name = QString::fromStdString(counter.name),
357 .value = static_cast<quint64>(counter.value),
358 .unit = qstring_from_ascii(crimson::renderer_metric_unit_name(counter.unit)),
359 });
360 }
361
362 result.diagnostics.reserve(static_cast<qsizetype>(snapshot.recent_diagnostics.size()));
363 for (const crimson::RendererDiagnostic &diagnostic : snapshot.recent_diagnostics) {
364 result.diagnostics.push_back(ViewportRendererDiagnosticRow{
365 .severity = qstring_from_ascii(crimson::renderer_diagnostic_severity_name(diagnostic.severity)),
366 .code = qstring_from_ascii(crimson::renderer_diagnostic_code_name(diagnostic.code)),
367 .subsystem = QString::fromStdString(diagnostic.subsystem),
368 .resource_name = QString::fromStdString(diagnostic.resource_name),
369 .message = QString::fromStdString(diagnostic.message),
370 .detail = QString::fromStdString(diagnostic.detail),
371 .frame_index = static_cast<quint64>(diagnostic.frame_index),
372 });
373 }
374
375 result.frame_graph_dump = QString::fromStdString(snapshot.frame_graph_dump);
376 return result;
377}

441ViewportRenderResult CrimsonViewportRenderHost::initialize_surface(
442 const NativeViewportSurface &surface,
443 ViewportPixelSize size,
444 double device_pixel_ratio) {
445 shutdown_surface();
446
447 crimson::RendererConfig config;
448 config.backend_preference = crimson::GraphicsBackendPreference::Auto;
449 config.shader_root = (QCoreApplication::applicationDirPath() + QStringLiteral("/shaders")).toStdString();
450 config.vsync = true;
451 config.enable_debug_text = true;
452
453 auto created = crimson::Renderer::create(config, make_surface_descriptor(surface, size, device_pixel_ratio));
454 if (!created) {
455 return result_from_diagnostic(created.error());
456 }

459 const crimson::RendererStatus kStatus = renderer_->status();
460 renderer_name_ = QString::fromStdString(kStatus.backend_name);

479ViewportRenderResult CrimsonViewportRenderHost::render_frame(const ViewportRenderRequest &request) {
480 if (renderer_ == nullptr) {
481 return ViewportRenderResult::failure(QStringLiteral("Renderer is not initialized."));
482 }
483
484 const auto kCameras = make_cameras(request.cameras);
485 const auto kViews = make_views(request.panes);
486 const std::vector<crimson::PickingRequest> kPickingRequests = make_picking_requests(request.picking_requests);
487 std::vector<crimson::RenderMeshUpload> mesh_uploads;
488 std::vector<crimson::RenderObject> objects;
489 append_document_render_data(mesh_uploads, objects);
490 std::vector<crimson::OverlayCommand> overlays;
491 std::vector<crimson::LineOverlaySegment> line_overlay_payloads;
492 append_tool_preview_overlays(request.panes.size(), overlays, line_overlay_payloads);
493 const auto kFrame = crimson::PrototypeViewportFrame{
494 .target_extent = make_extent(request.surface_size, request.device_pixel_ratio),
495 .views = std::span<const crimson::PrototypeViewportView>(kViews.data(), request.panes.size()),
496 .cameras = std::span<const crimson::PrototypeCamera>(kCameras.data(), kCameras.size()),
497 .mesh_uploads = std::span<const crimson::RenderMeshUpload>(mesh_uploads.data(), mesh_uploads.size()),
498 .objects = std::span<const crimson::RenderObject>(objects.data(), objects.size()),
499 .overlays = std::span<const crimson::OverlayCommand>(overlays.data(), overlays.size()),
500 .line_overlay_payloads = std::span<const crimson::LineOverlaySegment>(line_overlay_payloads.data(), line_overlay_payloads.size()),
501 .picking_requests = std::span<const crimson::PickingRequest>(kPickingRequests.data(), kPickingRequests.size()),
502 .animation_enabled = request.prototype_animation_enabled,
503 .elapsed_seconds = request.elapsed_seconds,
504 };
505
506 const crimson::FrameBuilder kFrameBuilder;
507 auto snapshot = kFrameBuilder.build_prototype_snapshot(kFrame);
508 if (!snapshot) {
509 return result_from_diagnostic(snapshot.error());
510 }
511
512 auto rendered = renderer_->render(snapshot.value());
513 if (!rendered) {
514 return result_from_diagnostic(rendered.error());
515 }
516
517 crimson::FrameRenderResult frame_result = std::move(rendered).value();
518 const crimson::FrameStats kStats = frame_result.stats;
519 latest_stats_ = viewport_frame_stats_from(kStats);
520 std::vector<ViewportPickResult> completed_pick_results;
521 completed_pick_results.reserve(frame_result.completed_picking_results.size());
522 for (const crimson::PickingResult &result : frame_result.completed_picking_results) {
523 completed_pick_results.push_back(viewport_pick_result_from(result));
524 }
525 return ViewportRenderResult::success(renderer_name_, std::move(completed_pick_results));
526}

587std::array<crimson::PrototypeCamera, 4> CrimsonViewportRenderHost::make_cameras(
588 std::span<const ViewportCameraSnapshot> cameras) const {
589 std::array<crimson::PrototypeCamera, 4> result{};
590 for (std::size_t index = 0; index < result.size(); ++index) {
591 const ViewportCameraSnapshot &source = cameras[index < cameras.size() ? index : 0];
592 result[index] = crimson::PrototypeCamera{
593 .eye = source.eye,
594 .target = source.target,
595 .up = source.up,
596 .forward = source.forward,
597 .projection = projection_from(source.projection),
598 .fov_degrees = source.fov_degrees,
599 .orthographic_size = source.orthographic_size,
600 };

639void CrimsonViewportRenderHost::append_document_render_data(
640 std::vector<crimson::RenderMeshUpload> &mesh_uploads,
641 std::vector<crimson::RenderObject> &objects) const {
642 if (document_ == nullptr) {
643 return;
644 }
645
646 const std::vector<quader::document::ObjectId> kObjectIds = document_->object_ids();
647 mesh_uploads.reserve(mesh_uploads.size() + kObjectIds.size());
648 objects.reserve(objects.size() + kObjectIds.size());
649 for (const quader::document::ObjectId kObjectId : kObjectIds) {
650 const quader::document::MeshObject *object = document_->find_mesh_object(kObjectId);
651 if (object == nullptr) {
652 continue;
653 }
654
655 std::optional<crimson::RenderMeshUpload> upload = make_mesh_upload(*object);
656 if (!upload) {
657 continue;
658 }
659
660 const crimson::RenderMeshHandle kMeshHandle = upload->handle;
661 const quader::math::Aabb kWorldBounds = world_bounds_for(*object);
662 if (quader::math::empty(kWorldBounds)) {
663 continue;
664 }
665
666 mesh_uploads.push_back(std::move(*upload));
667 objects.push_back(crimson::RenderObject{
668 .object_id = render_object_id_for(kObjectId),
669 .mesh = kMeshHandle,
670 .built_in_mesh = crimson::BuiltInRenderMesh::None,
671 .material = {},
672 .base_shader = crimson::BaseShaderId::OpaquePbr,
673 .world_from_object = render_transform_from(object->transform),
674 .world_bounds = kWorldBounds,
675 .queue = crimson::RenderQueue::Opaque,
676 .submesh_index = 0,
677 .visible = true,
678 .pickable = true,
679 });
680 }
681}
