21[[nodiscard]] bool descriptors_match(const RenderResourceDesc &left, const RenderResourceDesc &right) noexcept {
22 return left.name == right.name && left.format == right.format && left.extent.width_px == right.extent.width_px && left.extent.height_px == right.extent.height_px && left.extent.device_pixel_ratio == right.extent.device_pixel_ratio && left.external == right.external && left.resize_dependent == right.resize_dependent && left.sample_count == right.sample_count;
23}

29[[nodiscard]] std::optional<bgfx::TextureFormat::Enum> bgfx_texture_format(RenderResourceFormat format) noexcept {
30 switch (format) {

33 case RenderResourceFormat::Rgba16Float:
34 return bgfx::TextureFormat::RGBA16F;
35 case RenderResourceFormat::D24S8:

37 case RenderResourceFormat::D32Float:
38 return bgfx::TextureFormat::D32F;
39 case RenderResourceFormat::R32Uint:
40 return bgfx::TextureFormat::R32U;

42 case RenderResourceFormat::BackbufferDepth:

104void GpuFrameResources::synchronize_descriptors(const RenderGraph &graph) {
105 const std::uint64_t kGraphGeneration = graph.resize_generation();
106 std::vector<GpuFrameTargetRecord> synchronized;
107 synchronized.reserve(graph.resources().resources().size());
108
109 for (const RenderResourceRecord &graph_resource : graph.resources().resources()) {
110 const auto kExisting = std::find_if(targets_.begin(), targets_.end(), [&](const GpuFrameTargetRecord &record) {
111 return record.desc.name == graph_resource.desc.name;
112 });
113
114 if (kExisting == targets_.end()) {
115 GpuFrameTargetRecord record;
116 record.desc = graph_resource.desc;
117 record.graph_generation = graph_resource.generation;
118 record.recreate_count = 1;
119 synchronized.push_back(std::move(record));
120 continue;
121 }
122
123 GpuFrameTargetRecord record = std::move(*kExisting);
124 if (!descriptors_match(record.desc, graph_resource.desc) || record.graph_generation != graph_resource.generation) {
125 record.desc = graph_resource.desc;
126 record.graph_generation = graph_resource.generation;
127 ++record.recreate_count;
128 record.framebuffer.reset();
129 record.texture.reset();
130 }
131 synchronized.push_back(std::move(record));
132 }
133
134 targets_ = std::move(synchronized);
135 graph_resize_generation_ = kGraphGeneration;
136}

138bool GpuFrameResources::synchronize(const RenderGraph &graph, RendererStatus &status) {
139 const std::uint64_t kPreviousGeneration = graph_resize_generation_;
140 synchronize_descriptors(graph);
141 const bool kGenerationChanged = kPreviousGeneration != graph_resize_generation_;
142
143 if (!create_runtime_targets(status)) {
144 return false;
145 }
146 if (kGenerationChanged) {
147 hdr_scene_framebuffer_.reset();
148 tone_mapped_framebuffer_.reset();
149 picking_framebuffer_.reset();
150 }
151 return recreate_composite_framebuffers(status);
152}

207bool GpuFrameResources::create_runtime_targets(RendererStatus &status) {
208 bool ok = true;
209 for (GpuFrameTargetRecord &record : targets_) {
210 if (record.desc.external) {
211 continue;
212 }
213 if (record.texture.valid() && record.framebuffer.valid()) {
214 continue;
215 }
216
217 const std::optional<bgfx::TextureFormat::Enum> kFormat = bgfx_texture_format(record.desc.format);
218 if (!kFormat) {
219 push_frame_resource_error(
220 status,
221 RendererDiagnosticCode::FrameBufferUnsupported,
222 "Crimson cannot map a graph frame target to a runtime texture format.",
223 render_resource_format_name(record.desc.format),
224 record.desc.name);
225 ok = false;
226 continue;
227 }
228
229 record.framebuffer.reset();
230 record.texture.reset(bgfx::createTexture2D(
231 texture_dimension(record.desc.extent.width_px),
232 texture_dimension(record.desc.extent.height_px),
233 false,
234 1,
235 *kFormat,
236 texture_flags_for(record.desc)));
237 name_texture(record.texture.get(), record.desc.name);
238
239 if (!record.texture.valid()) {
240 push_frame_resource_error(
241 status,
242 RendererDiagnosticCode::ResourceCreationFailed,
243 "Crimson failed to create a graph frame target texture.",
244 render_resource_format_name(record.desc.format),
245 record.desc.name);
246 ok = false;
247 continue;
248 }
249
250 const bgfx::TextureHandle kAttachment = record.texture.get();
251 record.framebuffer.reset(bgfx::createFrameBuffer(1, &kAttachment, false));
252 name_framebuffer(record.framebuffer.get(), record.desc.name.c_str());
253
254 if (!record.framebuffer.valid()) {
255 push_frame_resource_error(
256 status,
257 RendererDiagnosticCode::FrameBufferUnsupported,
258 "Crimson failed to create a graph frame target framebuffer.",
259 record.desc.name,
260 record.desc.name);
261 ok = false;
262 }
263 }
264
265 return ok;
266}

268bool GpuFrameResources::recreate_composite_framebuffers(RendererStatus &status) {
269 GpuFrameTargetRecord *hdr = find_mutable(targets_, kHdrSceneColorTargetName);
270 GpuFrameTargetRecord *depth = find_mutable(targets_, kSceneDepthTargetName);
271 GpuFrameTargetRecord *picking = find_mutable(targets_, kPickingIdTargetName);
272 GpuFrameTargetRecord *tone = find_mutable(targets_, kToneMappedColorTargetName);
273
274 if (hdr == nullptr || depth == nullptr || picking == nullptr || tone == nullptr) {
275 push_frame_resource_error(
276 status,
277 RendererDiagnosticCode::ResourceLifetimeError,
278 "Crimson V1 graph frame targets are incomplete.",
279 "Expected HdrSceneColor, SceneDepth, PickingId, and ToneMappedColor.",
280 "V1FrameTargets");
281 return false;
282 }
283 if (!hdr->texture.valid() || !depth->texture.valid() || !picking->texture.valid() || !tone->texture.valid()) {
284 push_frame_resource_error(
285 status,
286 RendererDiagnosticCode::ResourceLifetimeError,
287 "Crimson cannot create composite framebuffers before graph target textures exist.",
288 "One or more V1 frame target textures are invalid.",
289 "V1FrameTargets");
290 return false;
291 }
292
293 if (!hdr_scene_framebuffer_.valid()) {
294 const std::array<bgfx::TextureHandle, 2> kTargets = { hdr->texture.get(), depth->texture.get() };
295 hdr_scene_framebuffer_.reset(bgfx::createFrameBuffer(
296 static_cast<std::uint8_t>(kTargets.size()),
297 kTargets.data(),
298 false));
299 name_framebuffer(hdr_scene_framebuffer_.get(), "HdrSceneColor+SceneDepth");
300 }
301 if (!tone_mapped_framebuffer_.valid()) {
302 const std::array<bgfx::TextureHandle, 2> kTargets = { tone->texture.get(), depth->texture.get() };
303 tone_mapped_framebuffer_.reset(bgfx::createFrameBuffer(
304 static_cast<std::uint8_t>(kTargets.size()),
305 kTargets.data(),
306 false));
307 name_framebuffer(tone_mapped_framebuffer_.get(), "ToneMappedColor+SceneDepth");
308 }
309 if (!picking_framebuffer_.valid()) {
310 const std::array<bgfx::TextureHandle, 2> kTargets = { picking->texture.get(), depth->texture.get() };
311 picking_framebuffer_.reset(bgfx::createFrameBuffer(
312 static_cast<std::uint8_t>(kTargets.size()),
313 kTargets.data(),
314 false));
315 name_framebuffer(picking_framebuffer_.get(), "PickingId+SceneDepth");
316 }
317
318 const bool kOk = hdr_scene_framebuffer_.valid() && tone_mapped_framebuffer_.valid() && picking_framebuffer_.valid();
319 if (!kOk) {
320 push_frame_resource_error(
321 status,
322 RendererDiagnosticCode::FrameBufferUnsupported,
323 "Crimson failed to create one or more V1 composite graph framebuffers.",
324 "HDR/depth, tone-mapped/depth, or picking/depth framebuffer was invalid.",
325 "V1FrameTargets");
326 }
327 return kOk;
328}
