175 template <class T>
176 [[nodiscard]] quader::foundation::Result<AttributeId, MeshError> create(AttributeDomain domain,
177 std::string name,
178 T default_value) {
179 static_assert(AttributeValueTraits<T>::kSupported, "Unsupported mesh attribute value type.");
180
181 if (has_attribute(domain, name)) {
182 return quader::foundation::Result<AttributeId, MeshError>::failure(
183 make_mesh_error(MeshErrorCode::AttributeNameAlreadyExists,
184 "mesh attribute name already exists in the requested domain"));
185 }
186
187 const AttributeId kId{ static_cast<quader::foundation::IdIndex>(storages_.size()) };
188 auto storage = std::make_unique<TypedAttributeStorage<T>>(
189 AttributeDescriptor{ kId, domain, AttributeValueTraits<T>::kValueType, std::move(name) },
190 std::move(default_value),
191 domain_size(domain));
192 storages_.push_back(std::move(storage));
193
194 return quader::foundation::Result<AttributeId, MeshError>::success(kId);
195 }

228 template <class T>
229 [[nodiscard]] quader::foundation::Result<T *, MeshError> value(AttributeId id,
230 quader::foundation::IdIndex slot) {
231 auto *storage = typed_storage<T>(id);
232 if (storage == nullptr) {
233 return quader::foundation::Result<T *, MeshError>::failure(
234 make_mesh_error(MeshErrorCode::AttributeTypeMismatch,
235 "mesh attribute type does not match requested value type"));
236 }
237 if (slot >= storage->values().size()) {
238 return quader::foundation::Result<T *, MeshError>::failure(
239 make_mesh_error(MeshErrorCode::AttributeIndexOutOfRange,
240 "mesh attribute slot is outside the attribute array"));
241 }
242
243 return quader::foundation::Result<T *, MeshError>::success(&storage->values()[slot]);
244 }

367 [[nodiscard]] bool has_attribute(AttributeDomain domain, std::string_view name) const noexcept {
368 for (const auto &storage : storages_) {
369 const auto &descriptor = storage->descriptor();
370 if (descriptor.domain == domain && descriptor.name == name) {
371 return true;
372 }
