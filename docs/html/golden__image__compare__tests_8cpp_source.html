32TEST(GoldenImageCompare, ExactMatchPassesWithZeroDifferences) {
33 const auto kExpected = make_image(2, 1, { 10, 20, 30, 255, 100, 110, 120, 255 });
34 const auto &actual = kExpected;
35
36 const quader::render_tests::GoldenImageDiff kDiff =
37 quader::render_tests::GoldenImageComparator{}.compare(
38 actual,
39 kExpected,
40 quader::render_tests::GoldenTolerance{});
41
42 EXPECT_TRUE(kDiff.passed);
43 EXPECT_TRUE(kDiff.dimensions_match);
44 EXPECT_TRUE(kDiff.pixel_buffer_sizes_match);
45 EXPECT_EQ(kDiff.max_channel_delta, 0);
46 EXPECT_EQ(kDiff.changed_pixels, 0U);
47 EXPECT_TRUE(kDiff.mean_channel_delta == 0.0);
48}

50TEST(GoldenImageCompare, ControlledDeltaReportsMaxMeanAndChangedPixels) {
51 const auto kExpected = make_image(2, 1, { 10, 20, 30, 255, 100, 110, 120, 255 });
52 const auto kActual = make_image(2, 1, { 12, 21, 30, 255, 100, 110, 119, 255 });
53
54 const quader::render_tests::GoldenImageDiff kStrictDiff =
55 quader::render_tests::GoldenImageComparator{}.compare(
56 kActual,
57 kExpected,
58 quader::render_tests::GoldenTolerance{});
59 EXPECT_FALSE(kStrictDiff.passed);
60 EXPECT_EQ(kStrictDiff.max_channel_delta, 2);
61 EXPECT_EQ(kStrictDiff.changed_pixels, 2U);
62 EXPECT_TRUE(kStrictDiff.mean_channel_delta > 0.0);
63
64 const quader::render_tests::GoldenImageDiff kTolerantDiff =
65 quader::render_tests::GoldenImageComparator{}.compare(
66 kActual,
67 kExpected,
68 quader::render_tests::GoldenTolerance{
69 .max_channel_delta = 2,
70 .max_mean_channel_delta = 0.5,
71 .max_changed_pixels = 2,
72 });
73 EXPECT_TRUE(kTolerantDiff.passed);
74}
