134TEST(UiActionWiring, UndoAndRedoActionsExecuteThroughCommandHistory) {
135 UiActionFixture fixture;
136 const auto kObject = add_triangle_object(fixture.document);
137 fixture.document.clear_dirty();
138 discard_pending_changes(fixture.document);
139
140 const auto kBefore = capture_ui_document_state(fixture.document);
141 auto result = fixture.command_history.execute(
142 std::make_unique<quader::commands::RenameObjectCommand>(kObject, "Renamed"),
143 fixture.document);
144 EXPECT_EQ((result).status, quader::commands::CommandStatus::Applied);
145 const auto kAfter = capture_ui_document_state(fixture.document);
146 discard_pending_changes(fixture.document);
147
148 int command_history_changes = 0;
149 int drained_events = 0;
150 QObject::connect(&fixture.document_ui,
151 &quader::ui::DocumentUiController::command_history_changed,
152 &fixture.document_ui,
153 [&command_history_changes]() {
154 ++command_history_changes;
155 });
156 QObject::connect(&fixture.document_ui,
157 &quader::ui::DocumentUiController::document_events_drained,
158 &fixture.document_ui,
159 [&drained_events]() {
160 ++drained_events;
161 });
162
163 fixture.action_state_updater.refresh();
164 fixture.actions.action(quader::ui::ActionId::Undo).trigger();
165 EXPECT_EQ(capture_ui_document_state(fixture.document), kBefore);
166 EXPECT_EQ(command_history_changes, 1);
167 EXPECT_EQ(drained_events, 1);
168 EXPECT_FALSE(fixture.actions.action(quader::ui::ActionId::Undo).isEnabled());
169 EXPECT_TRUE(fixture.actions.action(quader::ui::ActionId::Redo).isEnabled());
170 EXPECT_EQ(fixture.actions.action(quader::ui::ActionId::Redo).text(),
171 QStringLiteral("Redo Rename Object"));
172
173 fixture.actions.action(quader::ui::ActionId::Redo).trigger();
174 EXPECT_EQ(capture_ui_document_state(fixture.document), kAfter);
175 EXPECT_EQ(command_history_changes, 2);
176 EXPECT_EQ(drained_events, 2);
177 EXPECT_TRUE(fixture.actions.action(quader::ui::ActionId::Undo).isEnabled());
178 EXPECT_FALSE(fixture.actions.action(quader::ui::ActionId::Redo).isEnabled());
179}

181TEST(UiActionWiring, DeleteSelectionActionDeletesOneSelectedObjectThroughCommandHistory) {
182 UiActionFixture fixture;
183 const auto kObject = add_triangle_object(fixture.document);
184
185 quader::document::Selection selection;
186 EXPECT_TRUE(selection.set_objects({ kObject }));
187 EXPECT_TRUE(fixture.document.set_selection(selection));
188 fixture.document.clear_dirty();
189 discard_pending_changes(fixture.document);
190
191 fixture.action_state_updater.refresh();
192 EXPECT_TRUE(fixture.actions.action(quader::ui::ActionId::DeleteSelection).isEnabled());
193 EXPECT_FALSE(fixture.actions.action(quader::ui::ActionId::DuplicateSelection).isEnabled());
194
195 int object_list_changes = 0;
196 int selection_changes = 0;
197 int command_history_changes = 0;
198 QObject::connect(&fixture.document_ui,
199 &quader::ui::DocumentUiController::object_list_changed,
200 &fixture.document_ui,
201 [&object_list_changes]() {
202 ++object_list_changes;
203 });
204 QObject::connect(&fixture.document_ui,
205 &quader::ui::DocumentUiController::selection_changed,
206 &fixture.document_ui,
207 [&selection_changes]() {
208 ++selection_changes;
209 });
210 QObject::connect(&fixture.document_ui,
211 &quader::ui::DocumentUiController::command_history_changed,
212 &fixture.document_ui,
213 [&command_history_changes]() {
214 ++command_history_changes;
215 });
216
217 fixture.actions.action(quader::ui::ActionId::DeleteSelection).trigger();
218 EXPECT_EQ(fixture.document.object_count(), 0U);
219 EXPECT_TRUE(fixture.command_history.can_undo());
220 EXPECT_EQ(fixture.command_history.undo_name(), std::string_view("Delete Object"));
221 EXPECT_EQ(object_list_changes, 1);
222 EXPECT_EQ(selection_changes, 1);
223 EXPECT_EQ(command_history_changes, 1);
224 EXPECT_FALSE(fixture.actions.action(quader::ui::ActionId::DeleteSelection).isEnabled());
225 EXPECT_TRUE(fixture.actions.action(quader::ui::ActionId::Undo).isEnabled());
226
227 fixture.actions.action(quader::ui::ActionId::Undo).trigger();
228 EXPECT_EQ(fixture.document.object_count(), 1U);
229 EXPECT_EQ(fixture.document.selection().selected_objects().size(), 1U);
230 EXPECT_EQ(fixture.document.selection().selected_objects().front(), kObject);
231 EXPECT_EQ(object_list_changes, 2);
232 EXPECT_EQ(selection_changes, 2);
233 EXPECT_EQ(command_history_changes, 2);
234 EXPECT_TRUE(fixture.actions.action(quader::ui::ActionId::DeleteSelection).isEnabled());
235}

237TEST(UiActionWiring, ToolActionsRouteToToolManagerAndRefreshCheckedState) {
238 UiActionFixture fixture;
239 auto select_tool = std::make_unique<CountingTool>(quader::tools::ToolId::Select);
240 auto move_tool = std::make_unique<CountingTool>(quader::tools::ToolId::Move);
241 auto rotate_tool = std::make_unique<CountingTool>(quader::tools::ToolId::Rotate);
242 auto scale_tool = std::make_unique<CountingTool>(quader::tools::ToolId::Scale);
243 auto box_tool = std::make_unique<CountingTool>(quader::tools::ToolId::Box);
244 auto *select = select_tool.get();
245 auto *move = move_tool.get();
246 auto *box = box_tool.get();
247
248 EXPECT_TRUE(fixture.tool_manager.register_tool(std::move(select_tool)));
249 EXPECT_TRUE(fixture.tool_manager.register_tool(std::move(move_tool)));
250 EXPECT_TRUE(fixture.tool_manager.register_tool(std::move(rotate_tool)));
251 EXPECT_TRUE(fixture.tool_manager.register_tool(std::move(scale_tool)));
252 EXPECT_TRUE(fixture.tool_manager.register_tool(std::move(box_tool)));
253 EXPECT_TRUE(fixture.tool_manager.set_active_tool(quader::tools::ToolId::Select));
254
255 fixture.action_state_updater.refresh();
256 EXPECT_TRUE(fixture.actions.action(quader::ui::ActionId::SelectTool).isEnabled());
257 EXPECT_TRUE(fixture.actions.action(quader::ui::ActionId::SelectTool).isChecked());
258 EXPECT_FALSE(fixture.actions.action(quader::ui::ActionId::MoveTool).isChecked());
259 EXPECT_FALSE(fixture.actions.action(quader::ui::ActionId::BoxTool).isChecked());
260
261 fixture.actions.action(quader::ui::ActionId::MoveTool).trigger();
262 const auto kActiveToolId = fixture.tool_manager.active_tool_id();
263 if (!kActiveToolId.has_value()) {
264 ADD_FAILURE() << "move action did not set an active tool";
265 return;
266 }
267 EXPECT_EQ(*kActiveToolId, quader::tools::ToolId::Move);
268 EXPECT_EQ(select->deactivations, 1);
269 EXPECT_EQ(move->activations, 1);
270 EXPECT_FALSE(fixture.actions.action(quader::ui::ActionId::SelectTool).isChecked());
271 EXPECT_TRUE(fixture.actions.action(quader::ui::ActionId::MoveTool).isChecked());
272
273 fixture.actions.action(quader::ui::ActionId::BoxTool).trigger();
274 const auto kBoxToolId = fixture.tool_manager.active_tool_id();
275 ASSERT_TRUE(kBoxToolId.has_value());
276 EXPECT_EQ(*kBoxToolId, quader::tools::ToolId::Box);
277 EXPECT_EQ(box->activations, 1);
278 EXPECT_FALSE(fixture.actions.action(quader::ui::ActionId::MoveTool).isChecked());
279 EXPECT_TRUE(fixture.actions.action(quader::ui::ActionId::BoxTool).isChecked());
280 EXPECT_EQ(fixture.actions.action(quader::ui::ActionId::BoxTool).shortcut(), QKeySequence(Qt::Key_B));
281}
