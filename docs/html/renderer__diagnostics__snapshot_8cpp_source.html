41std::vector<RenderPassStats> make_render_pass_stats(const RenderGraph &graph, const FrameStats *stats) {
42 std::vector<RenderPassStats> pass_stats;
43 pass_stats.reserve(graph.passes().size());
44
45 for (const RenderPass &pass : graph.passes()) {
46 RenderPassStats current;
47 current.pass_name = pass.name;
48 for (const ResourceUse &use : pass.resources) {
49 if (access_reads(use.access)) {
50 ++current.resource_read_count;
51 }
52 if (access_writes(use.access)) {
53 ++current.resource_write_count;
54 }
55 }
56
57 if (stats != nullptr) {
58 if (pass.name == "OpaquePbrPass") {
59 current.draw_call_count = stats->opaque_draw_count;
60 current.draw_packet_count = stats->packets.opaque_packet_count;
61 } else if (pass.name == "AlphaCutoutPbrPass") {
62 current.draw_call_count = stats->alpha_cutout_draw_count;
63 current.draw_packet_count = stats->packets.alpha_cutout_packet_count;
64 } else if (pass.name == "TransparentPbrPass") {
65 current.draw_call_count = stats->transparent_draw_count;
66 current.draw_packet_count = stats->packets.transparent_packet_count;
67 } else if (pass.name == "ToneMapPass" || pass.name == "PresentPass") {
68 current.draw_call_count = 1;
69 current.draw_packet_count = 1;
70 } else if (pass.name.rfind("Overlay", 0) == 0) {
71 current.draw_call_count = stats->overlay_draw_count;
72 current.draw_packet_count = stats->packets.overlay_packet_count;
73 } else if (pass.name == "ResourceUploadPass") {
74 current.cpu_time_us = stats->timings.cpu_resource_upload_us;
75 }
76 }
77
78 pass_stats.push_back(std::move(current));
79 }
80
81 return pass_stats;
82}
