43void append_grid_payloads(
44 OverlayDrawBucket &bucket,
45 const OverlayCommand &command,
46 std::span<const GridOverlayCommand> grid_payloads) {
47 const std::size_t kBegin = static_cast<std::size_t>(command.payload_offset);
48 const std::size_t kCount = static_cast<std::size_t>(command.payload_count);
49 if (kBegin >= grid_payloads.size() || kCount == 0 || kBegin + kCount > grid_payloads.size()) {
50 return;
51 }
52
53 for (std::size_t index = kBegin; index < kBegin + kCount; ++index) {
54 const GridOverlayCommand &grid = grid_payloads[index];
55 if (grid.view_index != command.view_index) {
56 continue;
57 }
58
59 bucket.grid_commands.push_back(PreparedGridOverlayCommand{
60 .command = command,
61 .grid = grid,
62 .minor_color_linear_sdr = to_linear_sdr_array(grid.minor_color, command.opacity),
63 .major_color_linear_sdr = to_linear_sdr_array(grid.major_color, command.opacity),
64 .axis_u_color_linear_sdr = to_linear_sdr_array(grid.axis_u_color, command.opacity),
65 .axis_v_color_linear_sdr = to_linear_sdr_array(grid.axis_v_color, command.opacity),
66 });
67 }
68}

70void append_line_payloads(
71 OverlayDrawBucket &bucket,
72 const OverlayCommand &command,
73 std::span<const LineOverlaySegment> line_payloads) {
74 const std::size_t kBegin = static_cast<std::size_t>(command.payload_offset);
75 const std::size_t kCount = static_cast<std::size_t>(command.payload_count);
76 if (kBegin >= line_payloads.size() || kCount == 0 || kBegin + kCount > line_payloads.size()) {
77 return;
78 }
79
80 PreparedLineOverlayCommand prepared;
81 prepared.command = command;
82 prepared.color_linear_sdr = to_linear_sdr_array(command.color_srgb, command.opacity);
83 prepared.segments.reserve(kCount);
84 for (std::size_t index = kBegin; index < kBegin + kCount; ++index) {
85 prepared.segments.push_back(line_payloads[index]);
86 }
87 bucket.line_commands.push_back(std::move(prepared));
88}

119OverlayDrawLists OverlaySystem::prepare(
120 std::span<const OverlayCommand> commands,
121 std::span<const GridOverlayCommand> grid_payloads,
122 std::span<const LineOverlaySegment> line_payloads) const {
123 OverlayDrawLists lists;
124 for (const OverlayCommand &command : commands) {
125 if (command.base_shader != BaseShaderId::OverlayUnlit) {
126 continue;
127 }
128
129 OverlayDrawBucket &bucket = bucket_for_depth_mode(lists, command.depth_mode);
130 const std::array<float, 4> kColor = to_linear_sdr_array(command.color_srgb, command.opacity);
131 bucket.commands.push_back(PreparedOverlayCommand{
132 .command = command,
133 .color_linear_sdr = ColorLinear{ kColor[0], kColor[1], kColor[2], kColor[3] },
134 });
135
136 if (command.primitive == OverlayPrimitive::Grid) {
137 append_grid_payloads(bucket, command, grid_payloads);
138 } else if (command.primitive == OverlayPrimitive::LineList) {
139 append_line_payloads(bucket, command, line_payloads);
140 }
141 }
142
143 return lists;
144}
