30[[nodiscard]] quader::math::Vec3 normalized_or(quader::math::Vec3 value,
31 quader::math::Vec3 fallback) noexcept {
32 const quader::math::Vec3 kNormalized = quader::math::normalized(value);
33 return quader::math::length_squared(kNormalized) <= 0.000001F ? fallback : kNormalized;
34}

120TEST(RendererPublicTypes, CameraProjectionRaysFollowCrimsonRenderedViewBasis) {
121 const crimson::RenderCamera kCamera{
122 .eye = { 0.0F, 0.0F, 10.0F },
123 .target = { 0.0F, 0.0F, 0.0F },
124 .up = { 0.0F, 1.0F, 0.0F },
125 .forward = { 0.0F, 0.0F, -1.0F },
126 .vertical_fov_degrees = 60.0F,
127 };
128 const crimson::RenderCameraViewportSize kViewport{ 100.0F, 100.0F };
129 const quader::math::Vec3 kRenderedRight = normalized_or(
130 quader::math::cross(kCamera.up, kCamera.forward),
131 { -1.0F, 0.0F, 0.0F });
132
133 const crimson::RenderCameraRay kCenterRay = crimson::render_camera_ray_from_viewport_point(
134 kCamera,
135 kViewport,
136 { 50.0F, 50.0F },
137 false);
138 expect_true(
139 quader::math::dot(kCenterRay.direction, kCamera.forward) > 0.999F,
140 "center pixel ray follows the rendered camera forward");
141
142 const crimson::RenderCameraRay kRightRay = crimson::render_camera_ray_from_viewport_point(
143 kCamera,
144 kViewport,
145 { 90.0F, 50.0F },
146 false);
147 expect_true(
148 quader::math::dot(kRightRay.direction, kRenderedRight) > 0.0F,
149 "right-side viewport ray points toward Crimson's rendered right axis");
150
151 const crimson::RenderCameraRay kLeftRay = crimson::render_camera_ray_from_viewport_point(
152 kCamera,
153 kViewport,
154 { 10.0F, 50.0F },
155 false);
156 expect_true(
157 quader::math::dot(kLeftRay.direction, kRenderedRight) < 0.0F,
158 "left-side viewport ray points away from Crimson's rendered right axis");
159}

161TEST(RendererPublicTypes, OrthographicCameraProjectionOffsetsTopRightAlongRenderedAxes) {
162 const crimson::RenderCamera kCamera{
163 .eye = { 0.0F, 0.0F, 10.0F },
164 .target = { 0.0F, 0.0F, 0.0F },
165 .up = { 0.0F, 1.0F, 0.0F },
166 .forward = { 0.0F, 0.0F, -1.0F },
167 .projection = crimson::CameraProjection::Orthographic,
168 .orthographic_height_m = 10.0F,
169 };
170 const crimson::RenderCameraRay kRay = crimson::render_camera_ray_from_viewport_point(
171 kCamera,
172 { 100.0F, 100.0F },
173 { 100.0F, 0.0F },
174 false);
175 const quader::math::Vec3 kRenderedRight = normalized_or(
176 quader::math::cross(kCamera.up, kCamera.forward),
177 { -1.0F, 0.0F, 0.0F });
178 const quader::math::Vec3 kDelta = kRay.origin - kCamera.target;
179
180 expect_true(
181 quader::math::dot(kDelta, kRenderedRight) > 0.0F,
182 "orthographic top-right ray origin offsets along rendered right");
183 expect_true(
184 quader::math::dot(kDelta, kCamera.up) > 0.0F,
185 "orthographic top-right ray origin offsets along camera up");
186 expect_true(
187 quader::math::dot(kRay.direction, kCamera.forward) > 0.999F,
188 "orthographic ray direction follows camera forward");
189}
