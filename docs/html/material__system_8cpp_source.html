46MaterialInstance make_default_material_instance(const BaseShaderDefinition &definition) {
47 MaterialInstance instance;
48 instance.debug_name = definition.name + "Default";
49 instance.base_shader_id = definition.id;
50 instance.parameters.reserve(definition.parameters.size());
51 for (const MaterialParameterDesc &parameter : definition.parameters) {
52 instance.parameters.push_back(MaterialParameter{
53 .name = parameter.name,
54 .value = parameter.default_value,
55 });
56 }
57
58 instance.textures.reserve(definition.texture_slots.size());
59 for (const MaterialTextureSlotDesc &slot : definition.texture_slots) {
60 instance.textures.push_back(MaterialTextureBinding{
61 .name = slot.name,
62 .texture = slot.default_texture,
63 });
64 }
65
66 return instance;
67}

93quader::foundation::Result<RenderMaterialHandle, RendererDiagnostic> MaterialSystem::create_material(
94 const MaterialInstance &instance) {
95 const auto kNormalized = normalize(instance);
96 if (!kNormalized) {
97 return quader::foundation::Result<RenderMaterialHandle, RendererDiagnostic>::failure(kNormalized.error());
98 }
99
100 std::uint32_t index = 0;
101 if (!free_indices_.empty()) {
102 index = free_indices_.back();
103 free_indices_.pop_back();
104
105 Slot &slot = slots_[static_cast<std::size_t>(index - 1)];
106 slot.material = kNormalized.value();
107 slot.occupied = true;
108 } else {
109 index = static_cast<std::uint32_t>(slots_.size() + 1);
110 slots_.push_back(Slot{
111 .material = kNormalized.value(),
112 .generation = 1,
113 .occupied = true,
114 });
115 }
116
117 ++live_material_count_;
118 const Slot &slot = slots_[static_cast<std::size_t>(index - 1)];
119 return quader::foundation::Result<RenderMaterialHandle, RendererDiagnostic>::success(
120 RenderMaterialHandle{ index, slot.generation });
121}

133bool MaterialSystem::destroy(RenderMaterialHandle handle) noexcept {
134 Slot *slot = slot_for(handle);
135 if (slot == nullptr) {
136 return false;
137 }
138
139 slot->material = MaterialInstance{};
140 slot->generation = next_generation(slot->generation);
141 slot->occupied = false;
142 free_indices_.push_back(handle.index);
143 --live_material_count_;
144 return true;
145}

151quader::foundation::Result<void, RendererDiagnostic> MaterialSystem::set_parameter(
152 RenderMaterialHandle handle,
153 std::string name,
154 MaterialParameterValue value) {
155 MaterialInstance *material = get(handle);
156 if (material == nullptr) {
157 return quader::foundation::Result<void, RendererDiagnostic>::failure(invalid_handle_diagnostic(handle));
158 }
159
160 MaterialInstance candidate = *material;
161 bool replaced = false;
162 for (MaterialParameter &parameter : candidate.parameters) {
163 if (parameter.name == name) {
164 parameter.value = value;
165 replaced = true;
166 break;
167 }
168 }
169 if (!replaced) {
170 candidate.parameters.push_back(MaterialParameter{
171 .name = std::move(name),
172 .value = value,
173 });
174 }
175
176 const auto kNormalized = normalize(candidate);
177 if (!kNormalized) {
178 return quader::foundation::Result<void, RendererDiagnostic>::failure(kNormalized.error());
179 }
180
181 *material = kNormalized.value();
182 return quader::foundation::Result<void, RendererDiagnostic>::success();
183}

189 MaterialInstance *material = get(handle);

192 }
193
194 MaterialInstance candidate = *material;
195 bool replaced = false;
196 for (MaterialTextureBinding &binding : candidate.textures) {
197 if (binding.name == name) {
198 binding.texture = texture;
199 replaced = true;

202 }
203 if (!replaced) {
204 candidate.textures.push_back(MaterialTextureBinding{
205 .name = std::move(name),
206 .texture = texture,
207 });
208 }
209
210 const auto kNormalized = normalize(candidate);
211 if (!kNormalized) {
212 return quader::foundation::Result<void, RendererDiagnostic>::failure(kNormalized.error());
213 }
214
215 *material = kNormalized.value();
216 return quader::foundation::Result<void, RendererDiagnostic>::success();

219MaterialSystem::Slot *MaterialSystem::slot_for(RenderMaterialHandle handle) noexcept {
220 if (!is_valid_handle(handle) || handle.index > slots_.size()) {
221 return nullptr;
222 }
223
224 Slot &slot = slots_[static_cast<std::size_t>(handle.index - 1)];
225 if (!slot.occupied || slot.generation != handle.generation) {
226 return nullptr;
227 }
228
229 return &slot;
230}

245quader::foundation::Result<MaterialInstance, RendererDiagnostic> MaterialSystem::normalize(
246 const MaterialInstance &instance) const {
247 const auto kValidation = validate_material_instance(registry_, instance);
248 if (!kValidation) {
249 return quader::foundation::Result<MaterialInstance, RendererDiagnostic>::failure(kValidation.error());
250 }
251
252 const BaseShaderDefinition *definition = registry_.find(instance.base_shader_id);
253 if (definition == nullptr) {
254 return quader::foundation::Result<MaterialInstance, RendererDiagnostic>::failure(material_system_diagnostic(
255 RendererDiagnosticCode::MaterialSchemaInvalid,
256 "Crimson cannot normalize material for an unknown base shader.",
257 std::string(base_shader_id_name(instance.base_shader_id)),
258 instance.debug_name));
259 }
260
261 MaterialInstance normalized = make_default_material_instance(*definition);
262 normalized.debug_name = instance.debug_name.empty() ? normalized.debug_name : instance.debug_name;
263 normalized.overrides = instance.overrides;
264
265 for (const MaterialParameter &input_parameter : instance.parameters) {
266 for (MaterialParameter &output_parameter : normalized.parameters) {
267 if (output_parameter.name == input_parameter.name) {
268 output_parameter.value = input_parameter.value;
269 break;
270 }
271 }
272 }
273
274 for (const MaterialTextureBinding &input_texture : instance.textures) {
275 for (MaterialTextureBinding &output_texture : normalized.textures) {
276 if (output_texture.name == input_texture.name) {
277 output_texture.texture = input_texture.texture;
278 break;
279 }
280 }
281 }
282
283 return quader::foundation::Result<MaterialInstance, RendererDiagnostic>::success(std::move(normalized));
284}
