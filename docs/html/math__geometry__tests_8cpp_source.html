88TEST(MathGeometry, Vec3OperationsPreserveQuaderContract) {
89 using quader::math::Vec3;
90
91 const Vec3 kXAxis{ 1.0F, 0.0F, 0.0F };
92 const Vec3 kYAxis{ 0.0F, 1.0F, 0.0F };
93 const Vec3 kZAxis = quader::math::cross(kXAxis, kYAxis);
94
95 EXPECT_TRUE(quader::math::nearly_equal(Vec3{ 1.0F, 2.0F, 3.0F } + Vec3{ 4.0F, -5.0F, 6.0F },
96 Vec3{ 5.0F, -3.0F, 9.0F }));
97 EXPECT_TRUE(quader::math::nearly_equal(Vec3{ 1.0F, 2.0F, 3.0F } - Vec3{ 4.0F, -5.0F, 6.0F },
98 Vec3{ -3.0F, 7.0F, -3.0F }));
99 EXPECT_TRUE(quader::math::nearly_equal(Vec3{ 2.0F, -3.0F, 4.0F } * 2.5F,
100 Vec3{ 5.0F, -7.5F, 10.0F }));
101 EXPECT_TRUE(quader::math::nearly_equal(Vec3{ 6.0F, -8.0F, 10.0F } / 2.0F,
102 Vec3{ 3.0F, -4.0F, 5.0F }));
103 EXPECT_TRUE(quader::math::nearly_equal(kZAxis, Vec3{ 0.0F, 0.0F, 1.0F }));
104 EXPECT_EQ(quader::math::dot(kXAxis, kYAxis), 0.0F);
105 EXPECT_FLOAT_EQ(quader::math::dot(Vec3{ 2.0F, 3.0F, 4.0F }, Vec3{ 5.0F, -6.0F, 7.0F }), 20.0F);
106 EXPECT_FLOAT_EQ(quader::math::length_squared(Vec3{ 2.0F, 3.0F, 6.0F }), 49.0F);
107 EXPECT_FLOAT_EQ(quader::math::length(Vec3{ 2.0F, 3.0F, 6.0F }), 7.0F);
108 EXPECT_TRUE(quader::math::nearly_equal(quader::math::normalized(Vec3{ 0.0F, 0.0F, 5.0F }),
109 Vec3{ 0.0F, 0.0F, 1.0F }));
110}

143 Vec3{ 0.0F, 0.0F, 0.0F },

159TEST(MathGeometry, AabbDefaultsEmptyAndExpandsToPoints) {
160 quader::math::Aabb bounds;
161 EXPECT_TRUE(quader::math::empty(bounds));
162 EXPECT_TRUE(std::isinf(bounds.min.x));
163 EXPECT_TRUE(std::isinf(bounds.min.y));
164 EXPECT_TRUE(std::isinf(bounds.min.z));
165 EXPECT_TRUE(std::isinf(bounds.max.x));
166 EXPECT_TRUE(std::isinf(bounds.max.y));
167 EXPECT_TRUE(std::isinf(bounds.max.z));
168 EXPECT_GT(bounds.min.x, bounds.max.x);
169
170 quader::math::expand(bounds, quader::math::Vec3{ 2.0F, -1.0F, 3.0F });
171 quader::math::expand(bounds, quader::math::Vec3{ -2.0F, 4.0F, 1.0F });
172
173 EXPECT_FALSE(quader::math::empty(bounds));
174 EXPECT_TRUE(quader::math::nearly_equal(bounds.min, quader::math::Vec3{ -2.0F, -1.0F, 1.0F }));
175 EXPECT_TRUE(quader::math::nearly_equal(bounds.max, quader::math::Vec3{ 2.0F, 4.0F, 3.0F }));
176 EXPECT_TRUE(quader::math::nearly_equal(quader::math::center(bounds),
177 quader::math::Vec3{ 0.0F, 1.5F, 2.0F }));
178}
