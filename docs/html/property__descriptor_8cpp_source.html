94std::vector<PropertyDescriptor>
95build_property_descriptors(const quader::document::Document &document) {
96 const auto &selection = document.selection();
97 if (selection.empty()) {
98 return {};
99 }
100
101 if (!selection.selected_components().empty()) {
102 return { make_selection_summary(
103 QStringLiteral("%1 component(s) selected")
104 .arg(static_cast<qsizetype>(
105 selection.selected_components().size()))) };
106 }
107
108 if (selection.selected_objects().size() != 1U) {
109 return { make_selection_summary(
110 QStringLiteral("%1 object(s) selected")
111 .arg(static_cast<qsizetype>(selection.selected_objects().size()))) };
112 }
113
114 const auto kObjectId = selection.selected_objects().front();
115 const auto *object = document.find_mesh_object(kObjectId);
116 if (object == nullptr) {
117 return { make_selection_summary(
118 QStringLiteral("Selected object is no longer available")) };
119 }
120
121 std::vector<PropertyDescriptor> descriptors;
122 descriptors.reserve(14U);
123 descriptors.push_back(make_descriptor(
124 kObjectId, PropertyField::ObjectName, QStringLiteral("Name"),
125 PropertyValueKind::String, QString::fromStdString(object->name), true,
126 QStringLiteral("Object name")));
127 descriptors.push_back(make_descriptor(
128 kObjectId, PropertyField::ObjectKind, QStringLiteral("Kind"),
129 PropertyValueKind::ReadOnlyText, QStringLiteral("Mesh")));
130 descriptors.push_back(make_descriptor(
131 kObjectId, PropertyField::VertexCount, QStringLiteral("Vertices"),
132 PropertyValueKind::Int, count_value(object->mesh.vertex_count())));
133 descriptors.push_back(make_descriptor(
134 kObjectId, PropertyField::EdgeCount, QStringLiteral("Edges"),
135 PropertyValueKind::Int, count_value(object->mesh.edge_count())));
136 descriptors.push_back(make_descriptor(
137 kObjectId, PropertyField::FaceCount, QStringLiteral("Faces"),
138 PropertyValueKind::Int, count_value(object->mesh.face_count())));
139
140 append_transform_descriptors(descriptors, kObjectId, object->transform);
141 return descriptors;
142}
