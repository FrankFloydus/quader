32[[nodiscard]] QString pattern_for_extension(const std::string &extension) {
33 const auto kNormalized = quader::io::normalize_extension(extension);
34 if (kNormalized.empty()) {
35 return {};
36 }
37
38 return QStringLiteral("*.%1").arg(QString::fromStdString(kNormalized));

41void append_unique_pattern(QStringList &patterns, QString pattern) {

49[[nodiscard]] QString detail_from_error(const quader::io::IoError &error) {
50 if (!error.diagnostic.message.empty()) {
51 return QString::fromStdString(error.diagnostic.message);
52 }
53
54 for (const auto &diagnostic : error.related) {
55 if (!diagnostic.message.empty()) {
56 return QString::fromStdString(diagnostic.message);
57 }
58 }
59
60 return QStringLiteral("The selected file could not be imported.");
61}

63[[nodiscard]] QString first_warning_detail(const quader::io::IoDiagnosticList &diagnostics) {
64 for (const auto &diagnostic : diagnostics) {
65 if (diagnostic.severity == quader::io::IoSeverity::Warning && !diagnostic.message.empty()) {
66 return QString::fromStdString(diagnostic.message);
67 }
68 }
69
70 return {};
71}

98ImportUiResult ImportUiController::open_scene(QWidget *parent) {
99 if (registry_.import_formats().empty()) {
100 notifications_.show_warning(NotificationMessage{
101 NotificationSeverity::Warning,
102 QStringLiteral("No importer registered"),
103 QStringLiteral("Open is unavailable until a production scene importer is registered."),
104 5000,
105 });
106 return ImportUiResult::NoImporterRegistered;
107 }
108
109 const auto kSelected = dialogs_.choose_open_file(parent, make_open_request());
110 if (!kSelected.has_value()) {
111 return ImportUiResult::Canceled;
112 }
113
114 const std::filesystem::path kSelectedPath = path_from_qstring(*kSelected);
115 auto imported = import_service_.import_file(quader::io::ImportRequest{ kSelectedPath, {} });
116 if (!imported) {
117 present_import_failure(imported.error());
118 return ImportUiResult::ImportFailed;
119 }
120
121 present_import_warnings(imported.value().diagnostics);
122 present_parsed_not_applied(kSelectedPath);
123 return ImportUiResult::ParsedNotApplied;
124}

168QList<FileDialogFilter> import_filters_from_formats(
169 const std::vector<quader::io::FileFormatDescriptor> &formats) {
170 QList<FileDialogFilter> filters;
171 QStringList all_patterns;
172 QList<FileDialogFilter> format_filters;
173
174 for (const auto &format : formats) {
175 QStringList patterns;
176 for (const auto &extension : format.extensions) {
177 const QString kPattern = pattern_for_extension(extension);
178 append_unique_pattern(patterns, kPattern);
179 append_unique_pattern(all_patterns, kPattern);
180 }
181
182 if (patterns.empty()) {
183 continue;
184 }
185
186 format_filters.push_back(FileDialogFilter{
187 display_name_for_format(format),
188 patterns,
189 });
190 }
191
192 if (!all_patterns.empty()) {
193 filters.push_back(FileDialogFilter{
194 QStringLiteral("All Supported Files"),
195 all_patterns,
196 });
197 }
198
199 for (const auto &filter : format_filters) {
200 filters.push_back(filter);
201 }
202
203 return filters;
204}
