64TEST(RenderPacketPipeline, CullingKeepsVisibleRejectsOutsideAndCountsInvalidBounds) {
65 crimson::MaterialSystem materials;
66 const crimson::RenderMaterialHandle kOpaque = materials.create_default_material(crimson::BaseShaderId::OpaquePbr).value();
67
68 std::array<crimson::RenderObject, 3> objects = {
69 object(1, kOpaque, crimson::BaseShaderId::OpaquePbr, crimson::RenderQueue::Opaque, bounds(0.0F, 0.0F, -5.0F)),
70 object(2, kOpaque, crimson::BaseShaderId::OpaquePbr, crimson::RenderQueue::Opaque, bounds(100.0F, 0.0F, -5.0F)),
71 object(3, kOpaque, crimson::BaseShaderId::OpaquePbr, crimson::RenderQueue::Opaque, quader::math::Aabb{}),
72 };
73
74 const crimson::DrawPacketBuildResult kPackets = crimson::build_draw_packets(
75 objects,
76 materials.registry(),
77 materials,
78 camera(),
79 crimson::RenderMeshHandle{ 9, 1 },
80 kOpaque,
81 1.0F);
82
83 expect_true(kPackets.culling.input_object_count == 3, "culling records input object count");
84 expect_true(kPackets.culling.visible_object_count == 2, "inside and invalid-bounds objects remain visible");
85 expect_true(kPackets.culling.culled_object_count == 1, "outside object is culled");
86 expect_true(kPackets.culling.invalid_bounds_count == 1, "invalid bounds are kept and counted");
87 expect_true(kPackets.opaque.size() == 2, "only visible opaque packets are emitted");
88}

198TEST(RenderPacketPipeline, InstancingKeysExcludeTransformAndObjectId) {
199 crimson::DrawPacket first;
200 first.object_id = 1;
201 first.mesh = crimson::RenderMeshHandle{ 1, 1 };
202 first.material = crimson::RenderMaterialHandle{ 2, 1 };
203 first.base_shader = crimson::BaseShaderId::OpaquePbr;
204 first.program = crimson::ShaderProgramId::OpaquePbr;
205 first.queue = crimson::RenderQueue::Opaque;
206
207 crimson::DrawPacket second = first;
208 second.object_id = 2;
209 second.world_from_object[12] = 10.0F;
210
211 crimson::DrawPacket overlay = first;
212 overlay.object_id = 3;
213 overlay.queue = crimson::RenderQueue::OverlayAlwaysOnTop;
214
215 const std::array<crimson::DrawPacket, 3> kPackets = { second, overlay, first };
216 const std::vector<crimson::InstanceBatch> kBatches = crimson::build_instance_batches(kPackets);
217 const crimson::FrameInstancingStats kStats = crimson::make_instancing_stats(kBatches);
218
219 expect_true(kBatches.size() == 1, "overlay packets are excluded from PBR instancing batches");
220 expect_true(kBatches[0].instances.size() == 2, "same mesh/material/program packets group despite transform/id differences");
221 expect_true(kStats.instanced_batch_count == 1, "multi-instance batch is counted");
222 expect_true(kStats.saved_draw_call_count == 1, "saved draw calls are counted for staged instancing");
223}
