27[[nodiscard]] quader::foundation::IdGeneration next_generation(
28 quader::foundation::IdGeneration generation) noexcept {
29 auto next = static_cast<quader::foundation::IdGeneration>(generation + 1U);
30 if (next == 0U) {
31 next = 1U;
32 }
33 return next;
34}

51template <class Handle>
52void set_reverse(std::vector<quader::foundation::IdIndex> &reverse,
53 Handle handle,
54 quader::foundation::IdIndex slot) {
55 const auto kIndex = handle_index(handle);
56 if (kIndex < 0) {
57 return;
58 }
59
60 const auto kReverseIndex = static_cast<std::size_t>(kIndex);
61 if (reverse.size() <= kReverseIndex) {
62 reverse.resize(kReverseIndex + 1U, kNoSlot);
63 }
64 reverse[kReverseIndex] = slot;
65}
66
67template <class Handle>
68void clear_reverse(std::vector<quader::foundation::IdIndex> &reverse, Handle handle) {
69 const auto kIndex = handle_index(handle);
70 if (kIndex < 0) {
71 return;
72 }
73
74 const auto kReverseIndex = static_cast<std::size_t>(kIndex);
75 if (kReverseIndex < reverse.size()) {
76 reverse[kReverseIndex] = kNoSlot;
77 }
78}

80template <class Handle>
81[[nodiscard]] quader::foundation::IdIndex reverse_slot(const std::vector<quader::foundation::IdIndex> &reverse,
82 Handle handle) noexcept {
83 const auto kIndex = handle_index(handle);
84 if (kIndex < 0) {
85 return kNoSlot;
86 }
87
88 const auto kReverseIndex = static_cast<std::size_t>(kIndex);
89 if (kReverseIndex >= reverse.size()) {
90 return kNoSlot;
91 }
92
93 return reverse[kReverseIndex];
94}

104template <class Slot>
105[[nodiscard]] std::size_t alive_count(const std::vector<Slot> &slots) noexcept {
106 return static_cast<std::size_t>(
107 std::count_if(slots.begin(), slots.end(), [](const auto &slot) { return slot.alive; }));
108}

110template <class Id, class Slot>
111[[nodiscard]] std::vector<Id> live_ids(const std::vector<Slot> &slots) {
112 std::vector<Id> ids;
113 ids.reserve(alive_count(slots));
114 for (quader::foundation::IdIndex index = 0; index < slots.size(); ++index) {
115 const auto &slot = slots[index];
116 if (slot.alive) {
117 ids.push_back(Id{ index, slot.generation });
118 }
119 }
120 return ids;
121}

128[[nodiscard]] bool has_duplicate_vertices(std::span<const VertexId> vertices) noexcept {
129 for (std::size_t i = 0; i < vertices.size(); ++i) {
130 for (std::size_t j = i + 1U; j < vertices.size(); ++j) {
131 if (vertices[i] == vertices[j]) {
132 return true;
133 }
134 }
135 }
136 return false;
137}

162VertexId Polyhedron::create_vertex(quader::math::Vec3 position) {
163 const auto kHandle =
164 plain_handle<detail::QuaderOpenMesh::VertexHandle>(storage_->backend.add_vertex(to_openmesh_point(position)));
165
166 quader::foundation::IdIndex slot_index = 0;
167 if (!storage_->free_vertices.empty()) {
168 slot_index = storage_->free_vertices.back();
169 storage_->free_vertices.pop_back();
170 auto &slot = storage_->vertices[slot_index];
171 slot.alive = true;
172 slot.handle = kHandle;
173 } else {
174 slot_index = static_cast<quader::foundation::IdIndex>(storage_->vertices.size());
175 storage_->vertices.push_back(detail::OpenMeshIdSlot<detail::QuaderOpenMesh::VertexHandle>{});
176 auto &slot = storage_->vertices.back();
177 slot.alive = true;
178 slot.handle = kHandle;
179 attributes_.resize_domain(AttributeDomain::Vertex, storage_->vertices.size());
180 }
181
182 set_reverse(storage_->vertex_reverse.handle_to_slot, kHandle, slot_index);
183 attributes_.reset_slot_to_defaults(AttributeDomain::Vertex, slot_index);
184 return VertexId{ slot_index, storage_->vertices[slot_index].generation };
185}

187quader::foundation::Result<void, MeshError> Polyhedron::delete_vertex(VertexId vertex) {
188 if (!is_valid(vertex)) {
189 return quader::foundation::Result<void, MeshError>::failure(
190 make_mesh_error(MeshErrorCode::InvalidVertex, "cannot delete an invalid vertex id"));
191 }
192
193 const auto kHandle = storage_->vertices[vertex.index()].handle;
194 if (!storage_->backend.is_isolated(kHandle)) {
195 return quader::foundation::Result<void, MeshError>::failure(
196 make_mesh_error(MeshErrorCode::VertexStillReferenced,
197 "cannot delete a vertex while live halfedges reference it"));
198 }
199
200 storage_->backend.delete_vertex(kHandle, false);
201 clear_reverse(storage_->vertex_reverse.handle_to_slot, kHandle);
202
203 auto &slot = storage_->vertices[vertex.index()];
204 slot.alive = false;
205 slot.generation = next_generation(slot.generation);
206 slot.handle = invalid_handle<detail::QuaderOpenMesh::VertexHandle>();
207 storage_->free_vertices.push_back(vertex.index());
208
209 return quader::foundation::Result<void, MeshError>::success();
210}

212quader::foundation::Result<FaceId, MeshError> Polyhedron::create_face(std::span<const VertexId> vertices) {
213 if (vertices.size() < 3U) {
214 return quader::foundation::Result<FaceId, MeshError>::failure(
215 make_mesh_error(MeshErrorCode::FaceNeedsAtLeastThreeVertices, "a face requires at least three vertices"));
216 }
217
218 if (has_duplicate_vertices(vertices)) {
219 return quader::foundation::Result<FaceId, MeshError>::failure(
220 make_mesh_error(MeshErrorCode::DegenerateFace, "cannot create a face with duplicate vertices"));
221 }
222
223 std::vector<detail::QuaderOpenMesh::VertexHandle> handles;
224 std::vector<quader::math::Vec3> positions;
225 handles.reserve(vertices.size());
226 positions.reserve(vertices.size());
227
228 for (const auto kVertex : vertices) {
229 if (!is_valid(kVertex)) {
230 return quader::foundation::Result<FaceId, MeshError>::failure(
231 make_mesh_error(MeshErrorCode::InvalidVertex, "cannot create a face with an invalid vertex id"));
232 }
233
234 const auto kHandle = storage_->vertices[kVertex.index()].handle;
235 handles.push_back(kHandle);
236 positions.push_back(from_openmesh_point(storage_->backend.point(kHandle)));
237 }
238
239 if (!quader::geometry::all_points_finite(positions)) {
240 return quader::foundation::Result<FaceId, MeshError>::failure(
241 make_mesh_error(MeshErrorCode::NonFinitePosition,
242 "cannot create a face with non-finite vertex positions"));
243 }
244
245 if (quader::geometry::has_degenerate_area(positions)) {
246 return quader::foundation::Result<FaceId, MeshError>::failure(
247 make_mesh_error(MeshErrorCode::DegenerateFace, "cannot create a face with degenerate polygon area"));
248 }
249
250 const auto kRollback = *storage_;
251 const auto kFaceHandle =
252 plain_handle<detail::QuaderOpenMesh::FaceHandle>(storage_->backend.add_face(handles));
253 if (!kFaceHandle.is_valid()) {
254 *storage_ = kRollback;
255 return quader::foundation::Result<FaceId, MeshError>::failure(
256 make_mesh_error(MeshErrorCode::BackendRejectedFace,
257 "OpenMesh rejected the requested face without committing it"));
258 }
259
260 auto allocate_face_id = [this](detail::QuaderOpenMesh::FaceHandle handle) {
261 const auto kExisting = reverse_slot(storage_->face_reverse.handle_to_slot, handle);
262 if (kExisting != kNoSlot && kExisting < storage_->faces.size() && storage_->faces[kExisting].alive) {
263 return FaceId{ kExisting, storage_->faces[kExisting].generation };
264 }
265
266 quader::foundation::IdIndex slot_index = 0;
267 if (!storage_->free_faces.empty()) {
268 slot_index = storage_->free_faces.back();
269 storage_->free_faces.pop_back();
270 auto &slot = storage_->faces[slot_index];
271 slot.alive = true;
272 slot.handle = handle;
273 } else {
274 slot_index = static_cast<quader::foundation::IdIndex>(storage_->faces.size());
275 storage_->faces.push_back(detail::OpenMeshIdSlot<detail::QuaderOpenMesh::FaceHandle>{});
276 auto &slot = storage_->faces.back();
277 slot.alive = true;
278 slot.handle = handle;
279 attributes_.resize_domain(AttributeDomain::Face, storage_->faces.size());
280 }
281 set_reverse(storage_->face_reverse.handle_to_slot, handle, slot_index);
282 attributes_.reset_slot_to_defaults(AttributeDomain::Face, slot_index);
283 return FaceId{ slot_index, storage_->faces[slot_index].generation };
284 };
285
286 auto allocate_halfedge_id = [this](detail::QuaderOpenMesh::HalfedgeHandle handle) {
287 const auto kExisting = reverse_slot(storage_->halfedge_reverse.handle_to_slot, handle);
288 if (kExisting != kNoSlot && kExisting < storage_->halfedges.size() && storage_->halfedges[kExisting].alive) {
289 return HalfedgeId{ kExisting, storage_->halfedges[kExisting].generation };
290 }
291
292 quader::foundation::IdIndex slot_index = 0;
293 if (!storage_->free_halfedges.empty()) {
294 slot_index = storage_->free_halfedges.back();
295 storage_->free_halfedges.pop_back();
296 auto &slot = storage_->halfedges[slot_index];
297 slot.alive = true;
298 slot.handle = handle;
299 } else {
300 slot_index = static_cast<quader::foundation::IdIndex>(storage_->halfedges.size());
301 storage_->halfedges.push_back(detail::OpenMeshIdSlot<detail::QuaderOpenMesh::HalfedgeHandle>{});
302 auto &slot = storage_->halfedges.back();
303 slot.alive = true;
304 slot.handle = handle;
305 attributes_.resize_domain(AttributeDomain::Halfedge, storage_->halfedges.size());
306 }
307 set_reverse(storage_->halfedge_reverse.handle_to_slot, handle, slot_index);
308 attributes_.reset_slot_to_defaults(AttributeDomain::Halfedge, slot_index);
309 return HalfedgeId{ slot_index, storage_->halfedges[slot_index].generation };
310 };
311
312 auto allocate_edge_id = [this](detail::QuaderOpenMesh::EdgeHandle handle, HalfedgeId representative) {
313 const auto kExisting = reverse_slot(storage_->edge_reverse.handle_to_slot, handle);
314 if (kExisting != kNoSlot && kExisting < storage_->edges.size() && storage_->edges[kExisting].alive) {
315 return EdgeId{ kExisting, storage_->edges[kExisting].generation };
316 }
317
318 quader::foundation::IdIndex slot_index = 0;
319 if (!storage_->free_edges.empty()) {
320 slot_index = storage_->free_edges.back();
321 storage_->free_edges.pop_back();
322 auto &slot = storage_->edges[slot_index];
323 slot.alive = true;
324 slot.handle = handle;
325 slot.representative_halfedge = representative;
326 } else {
327 slot_index = static_cast<quader::foundation::IdIndex>(storage_->edges.size());
328 storage_->edges.push_back(detail::OpenMeshEdgeSlot{});
329 auto &slot = storage_->edges.back();
330 slot.alive = true;
331 slot.handle = handle;
332 slot.representative_halfedge = representative;
333 attributes_.resize_domain(AttributeDomain::Edge, storage_->edges.size());
334 }
335 set_reverse(storage_->edge_reverse.handle_to_slot, handle, slot_index);
336 attributes_.reset_slot_to_defaults(AttributeDomain::Edge, slot_index);
337 return EdgeId{ slot_index, storage_->edges[slot_index].generation };
338 };
339
340 const FaceId kFaceId = allocate_face_id(kFaceHandle);
341
342 for (auto halfedge_it = storage_->backend.fh_iter(kFaceHandle); halfedge_it.is_valid(); ++halfedge_it) {
343 const auto kHalfedgeHandle = plain_handle<detail::QuaderOpenMesh::HalfedgeHandle>(*halfedge_it);
344 const auto kHalfedgeId = allocate_halfedge_id(kHalfedgeHandle);
345 const auto kOppositeHandle = plain_handle<detail::QuaderOpenMesh::HalfedgeHandle>(
346 storage_->backend.opposite_halfedge_handle(kHalfedgeHandle));
347 allocate_halfedge_id(kOppositeHandle);
348 allocate_edge_id(plain_handle<detail::QuaderOpenMesh::EdgeHandle>(storage_->backend.edge_handle(kHalfedgeHandle)),
349 kHalfedgeId);
350 }
351
352 return quader::foundation::Result<FaceId, MeshError>::success(kFaceId);
353}

355quader::foundation::Result<void, MeshError> Polyhedron::delete_face(FaceId face) {
356 if (!is_valid(face)) {
357 return quader::foundation::Result<void, MeshError>::failure(
358 make_mesh_error(MeshErrorCode::InvalidFace, "cannot delete an invalid face id"));
359 }
360
361 auto halfedges = face_halfedges(face);
362 if (!halfedges) {
363 return quader::foundation::Result<void, MeshError>::failure(std::move(halfedges).error());
364 }
365
366 std::vector<HalfedgeId> affected_halfedges;
367 std::vector<EdgeId> affected_edges;
368 affected_halfedges.reserve(halfedges.value().size() * 2U);
369 affected_edges.reserve(halfedges.value().size());
370
371 for (const auto kHalfedge : halfedges.value()) {
372 affected_halfedges.push_back(kHalfedge);
373 auto opposite = opposite_halfedge(kHalfedge);
374 if (opposite) {
375 affected_halfedges.push_back(opposite.value());
376 }
377 auto edge = halfedge_edge(kHalfedge);
378 if (edge) {
379 affected_edges.push_back(edge.value());
380 }
381 }
382
383 const auto kFaceHandle = storage_->faces[face.index()].handle;
384 storage_->backend.delete_face(kFaceHandle, false);
385
386 clear_reverse(storage_->face_reverse.handle_to_slot, kFaceHandle);
387 auto &face_slot = storage_->faces[face.index()];
388 face_slot.alive = false;
389 face_slot.generation = next_generation(face_slot.generation);
390 face_slot.handle = invalid_handle<detail::QuaderOpenMesh::FaceHandle>();
391 storage_->free_faces.push_back(face.index());
392
393 for (const auto kEdge : affected_edges) {
394 if (kEdge.index() >= storage_->edges.size()) {
395 continue;
396 }
397 auto &slot = storage_->edges[kEdge.index()];
398 if (!slot.alive || !storage_->backend.status(slot.handle).deleted()) {
399 continue;
400 }
401 clear_reverse(storage_->edge_reverse.handle_to_slot, slot.handle);
402 slot.alive = false;
403 slot.generation = next_generation(slot.generation);
404 slot.handle = invalid_handle<detail::QuaderOpenMesh::EdgeHandle>();
405 slot.representative_halfedge = HalfedgeId::invalid();
406 storage_->free_edges.push_back(kEdge.index());
407 }
408
409 for (const auto kHalfedge : affected_halfedges) {
410 if (kHalfedge.index() >= storage_->halfedges.size()) {
411 continue;
412 }
413 auto &slot = storage_->halfedges[kHalfedge.index()];
414 if (!slot.alive || !storage_->backend.status(slot.handle).deleted()) {
415 continue;
416 }
417 clear_reverse(storage_->halfedge_reverse.handle_to_slot, slot.handle);
418 slot.alive = false;
419 slot.generation = next_generation(slot.generation);
420 slot.handle = invalid_handle<detail::QuaderOpenMesh::HalfedgeHandle>();
421 storage_->free_halfedges.push_back(kHalfedge.index());
422 }
423
424 return quader::foundation::Result<void, MeshError>::success();
425}

428 if (!id.is_valid() || id.index() >= storage_->vertices.size()) {
429 return false;

432 return slot.alive && slot.generation == id.generation() && live_backend_handle(storage_->backend, slot.handle) && reverse_slot(storage_->vertex_reverse.handle_to_slot, slot.handle) == id.index();
433}

667quader::foundation::Result<std::vector<HalfedgeId>, MeshError> Polyhedron::face_halfedges(FaceId face) const {
668 if (!is_valid(face)) {
669 return quader::foundation::Result<std::vector<HalfedgeId>, MeshError>::failure(
670 make_mesh_error(MeshErrorCode::InvalidFace, "cannot traverse an invalid face id"));
671 }
672
673 std::vector<HalfedgeId> result;
674 const auto kFaceHandle = storage_->faces[face.index()].handle;
675 for (auto halfedge_it = storage_->backend.cfh_iter(kFaceHandle); halfedge_it.is_valid(); ++halfedge_it) {
676 const auto kSlot = reverse_slot(storage_->halfedge_reverse.handle_to_slot, *halfedge_it);
677 if (kSlot == kNoSlot || kSlot >= storage_->halfedges.size()) {
678 return quader::foundation::Result<std::vector<HalfedgeId>, MeshError>::failure(
679 make_mesh_error(MeshErrorCode::InvalidHalfedge, "face cycle references an invalid halfedge"));
680 }
681 result.push_back(HalfedgeId{ kSlot, storage_->halfedges[kSlot].generation });
682 }
683
684 if (result.empty()) {
685 return quader::foundation::Result<std::vector<HalfedgeId>, MeshError>::failure(
686 make_mesh_error(MeshErrorCode::BrokenFaceCycle, "face cycle did not close"));
687 }
688
689 return quader::foundation::Result<std::vector<HalfedgeId>, MeshError>::success(std::move(result));
690}

713quader::foundation::Result<MeshCompactionReport, MeshError> Polyhedron::compact_storage() {
714 MeshCompactionReport report;
715 report.removed_vertices = storage_->backend.n_vertices() > vertex_count()
716 ? storage_->backend.n_vertices() - vertex_count()
717 : 0U;
718 report.removed_halfedges = storage_->backend.n_halfedges() > halfedge_count()
719 ? storage_->backend.n_halfedges() - halfedge_count()
720 : 0U;
721 report.removed_edges = storage_->backend.n_edges() > edge_count() ? storage_->backend.n_edges() - edge_count()
722 : 0U;
723 report.removed_faces = storage_->backend.n_faces() > face_count() ? storage_->backend.n_faces() - face_count()
724 : 0U;
725
726 std::vector<detail::QuaderOpenMesh::VertexHandle *> vertex_handles;
727 std::vector<detail::QuaderOpenMesh::HalfedgeHandle *> halfedge_handles;
728 std::vector<detail::QuaderOpenMesh::FaceHandle *> face_handles;
729 vertex_handles.reserve(vertex_count());
730 halfedge_handles.reserve(halfedge_count());
731 face_handles.reserve(face_count());
732
733 for (auto &slot : storage_->vertices) {
734 if (slot.alive) {
735 vertex_handles.push_back(&slot.handle);
736 }
737 }
738 for (auto &slot : storage_->halfedges) {
739 if (slot.alive) {
740 halfedge_handles.push_back(&slot.handle);
741 }
742 }
743 for (auto &slot : storage_->faces) {
744 if (slot.alive) {
745 face_handles.push_back(&slot.handle);
746 }
747 }
748
749 storage_->backend.garbage_collection(vertex_handles, halfedge_handles, face_handles, true, true, true);
750
751 storage_->vertex_reverse.handle_to_slot.clear();
752 storage_->halfedge_reverse.handle_to_slot.clear();
753 storage_->edge_reverse.handle_to_slot.clear();
754 storage_->face_reverse.handle_to_slot.clear();
755
756 for (quader::foundation::IdIndex index = 0; index < storage_->vertices.size(); ++index) {
757 if (storage_->vertices[index].alive) {
758 set_reverse(storage_->vertex_reverse.handle_to_slot, storage_->vertices[index].handle, index);
759 }
760 }
761 for (quader::foundation::IdIndex index = 0; index < storage_->halfedges.size(); ++index) {
762 if (storage_->halfedges[index].alive) {
763 set_reverse(storage_->halfedge_reverse.handle_to_slot, storage_->halfedges[index].handle, index);
764 }
765 }
766 for (quader::foundation::IdIndex index = 0; index < storage_->edges.size(); ++index) {
767 auto &edge_slot = storage_->edges[index];
768 if (!edge_slot.alive || !is_valid(edge_slot.representative_halfedge)) {
769 continue;
770 }
771 edge_slot.handle = storage_->backend.edge_handle(storage_->halfedges[edge_slot.representative_halfedge.index()].handle);
772 set_reverse(storage_->edge_reverse.handle_to_slot, edge_slot.handle, index);
773 }
774 for (quader::foundation::IdIndex index = 0; index < storage_->faces.size(); ++index) {
775 if (storage_->faces[index].alive) {
776 set_reverse(storage_->face_reverse.handle_to_slot, storage_->faces[index].handle, index);
777 }
778 }
779
780 report.live_vertices = vertex_count();
781 report.live_halfedges = halfedge_count();
782 report.live_edges = edge_count();
783 report.live_faces = face_count();
784
785 const auto kValidation = validate_mesh(*this);
786 if (!kValidation.ok()) {
787 return quader::foundation::Result<MeshCompactionReport, MeshError>::failure(
788 make_mesh_error(MeshErrorCode::CompactionFailed, "mesh validation failed after storage compaction"));
789 }
790
791 return quader::foundation::Result<MeshCompactionReport, MeshError>::success(report);
792}
