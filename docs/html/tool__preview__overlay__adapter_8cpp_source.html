35void append_tool_preview_line_overlays(
36 const quader::tools::ToolPreview &preview,
37 std::size_t view_count,
38 std::vector<crimson::OverlayCommand> &overlays,
39 std::vector<crimson::LineOverlaySegment> &line_payloads) {
40 if (!preview.active || (preview.world_segments.empty() && preview.boxes.empty())) {
41 return;
42 }
43
44 const std::uint32_t kPayloadOffset = static_cast<std::uint32_t>(line_payloads.size());
45 for (const quader::tools::ToolPreviewWorldSegment &segment : preview.world_segments) {
46 line_payloads.push_back(crimson::LineOverlaySegment{
47 .start = segment.start,
48 .end = segment.end,
49 });
50 }
51 for (const quader::tools::ToolPreviewBox &box : preview.boxes) {
52 if (!box.active) {
53 continue;
54 }
55 const std::size_t kFirstBoxEdge = preview.world_segments.empty() ? 0U : 4U;
56 for (std::size_t edge_index = kFirstBoxEdge; edge_index < kBoxEdges.size(); ++edge_index) {
57 const auto &[start_index, end_index] = kBoxEdges[edge_index];
58 line_payloads.push_back(crimson::LineOverlaySegment{
59 .start = box.corners[start_index],
60 .end = box.corners[end_index],
61 });
62 }
63 }
64
65 const std::uint32_t kPayloadCount = static_cast<std::uint32_t>(line_payloads.size()) - kPayloadOffset;
66 if (kPayloadCount == 0U) {
67 return;
68 }
69
70 auto append_overlay = [&](std::uint32_t view_index) {
71 overlays.push_back(crimson::OverlayCommand{
72 .view_index = view_index,
73 .primitive = crimson::OverlayPrimitive::LineList,
74 .depth_mode = crimson::OverlayDepthMode::AlwaysOnTop,
75 .base_shader = crimson::BaseShaderId::OverlayUnlit,
76 .color_srgb = kBoxToolPreviewLineColorSrgb,
77 .opacity = kBoxToolPreviewLineOpacity,
78 .thickness_px = kBoxToolPreviewLineThicknessPx,
79 .payload_offset = kPayloadOffset,
80 .payload_count = kPayloadCount,
81 });
82 };
83
84 if (preview.view_index.has_value()) {
85 if (*preview.view_index < view_count) {
86 append_overlay(*preview.view_index);
87 }
88 return;
89 }
90
91 overlays.reserve(overlays.size() + view_count);
92 for (std::size_t index = 0; index < view_count; ++index) {
93 append_overlay(static_cast<std::uint32_t>(index));
94 }
95}
