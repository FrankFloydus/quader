24PropertiesPanel::PropertiesPanel(PanelContext context, QWidget *parent) : QWidget(parent), context_(context) {
25 model_ = new PropertyItemModel(context_.ui.documents, this);
26 delegate_ = new PropertyValueDelegate(this);
27
28 empty_label_ = new QLabel(QStringLiteral("No selection"), this);
29 empty_label_->setAlignment(Qt::AlignCenter);
30
31 table_view_ = new QTableView(this);
32 table_view_->setModel(model_);
33 table_view_->setItemDelegateForColumn(static_cast<int>(PropertyItemColumn::Value), delegate_);
34 table_view_->setAlternatingRowColors(true);
35 table_view_->setSelectionBehavior(QAbstractItemView::SelectRows);
36 table_view_->setSelectionMode(QAbstractItemView::SingleSelection);
37 table_view_->verticalHeader()->setVisible(false);
38 table_view_->horizontalHeader()->setStretchLastSection(true);
39 table_view_->horizontalHeader()->setMinimumSectionSize(72);
40
41 auto *layout = new QVBoxLayout(this);
42 layout->setContentsMargins(0, 0, 0, 0);
43 layout->addWidget(empty_label_);
44 layout->addWidget(table_view_);
45
46 connect(model_, &QAbstractItemModel::modelReset, this, &PropertiesPanel::update_empty_state);
47 connect(model_, &QAbstractItemModel::rowsInserted, this, &PropertiesPanel::update_empty_state);
48 connect(model_, &QAbstractItemModel::rowsRemoved, this, &PropertiesPanel::update_empty_state);
49 update_empty_state();
50}
