36CommandResult CreateMeshObjectCommand::execute(quader::document::Document &document) {
37 if (created_object_) {
38 if (document.find_mesh_object(created_object_->id) != nullptr) {
39 return CommandResult::rejected("cannot redo create because the object is already live");
40 }
41
42 const auto kId = created_object_->id;
43 auto restored = document.restore_mesh_object(std::move(*created_object_));
44 if (!restored) {
45 created_object_.reset();
46 return rejected_from_document_error(std::move(restored).error());
47 }
48
49 created_object_.reset();
50 object_id_ = kId;
51 return CommandResult::applied();
52 }
53
54 if (!mesh_) {
55 return CommandResult::rejected("create command has no mesh data to execute");
56 }
57
58 auto created = document.create_mesh_object(name_, std::move(*mesh_), transform_, material_);
59 if (!created) {
60 mesh_.reset();
61 return rejected_from_document_error(std::move(created).error());
62 }
63
64 object_id_ = created.value();
65 mesh_.reset();
66 return CommandResult::applied();
67}

69CommandResult CreateMeshObjectCommand::undo(quader::document::Document &document) {
70 if (!object_id_) {
71 return CommandResult::rejected("create command has not created an object");
72 }
73
74 auto removed = document.extract_mesh_object(*object_id_);
75 if (!removed) {
76 return rejected_from_document_error(std::move(removed).error());
77 }
78
79 created_object_ = std::move(removed).value();
80 return CommandResult::applied();
81}

107CommandResult DeleteObjectCommand::undo(quader::document::Document &document) {
108 if (!removed_object_ || !selection_before_) {
109 return CommandResult::rejected("delete command has not removed an object");
110 }
111
112 auto restored = document.restore_mesh_object(std::move(*removed_object_));
113 if (!restored) {
114 removed_object_.reset();
115 return rejected_from_document_error(std::move(restored).error());
116 }
117
118 removed_object_.reset();
119 auto selection_restored = document.set_selection(*selection_before_);
120 if (!selection_restored) {
121 return rejected_from_document_error(std::move(selection_restored).error());
122 }
123
124 return CommandResult::applied();
125}

134CommandResult RenameObjectCommand::execute(quader::document::Document &document) {
135 const auto *object = document.find_mesh_object(object_);
136 if (object == nullptr) {
137 return CommandResult::rejected("cannot rename an unknown object");
138 }
139
140 if (object->name == new_name_) {
141 return CommandResult::rejected("rename command would not change the object name");
142 }
143
144 previous_name_ = object->name;
145 auto renamed = document.rename_object(object_, new_name_);
146 if (!renamed) {
147 previous_name_.reset();
148 return rejected_from_document_error(std::move(renamed).error());
149 }
150
151 return CommandResult::applied();
152}
