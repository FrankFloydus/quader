84quader::foundation::Result<void, std::string> RenderResourceRegistry::add(RenderResourceDesc desc) {
85 if (desc.name.empty()) {
86 return quader::foundation::Result<void, std::string>::failure("Render resource name must not be empty.");
87 }
88 if (!is_valid_extent(desc.extent)) {
89 return quader::foundation::Result<void, std::string>::failure(
90 "Render resource '" + desc.name + "' has an invalid extent.");
91 }
92 if (desc.sample_count == 0) {
93 return quader::foundation::Result<void, std::string>::failure(
94 "Render resource '" + desc.name + "' has an invalid sample count.");
95 }
96 if (find(desc.name) != nullptr) {
97 return quader::foundation::Result<void, std::string>::failure(
98 "Render resource '" + desc.name + "' is already registered.");
99 }
100
101 resources_.push_back(RenderResourceRecord{ .desc = std::move(desc), .generation = resize_generation_ });
102 return quader::foundation::Result<void, std::string>::success();
103}

127void RenderResourceRegistry::resize(ViewportExtent extent) {
128 if (!is_valid_extent(extent)) {
129 return;
130 }
131
132 ++resize_generation_;
133 if (resize_generation_ == 0) {
134 resize_generation_ = 1;
135 }
136
137 for (RenderResourceRecord &record : resources_) {
138 if (record.desc.resize_dependent) {
139 record.desc.extent = extent;
140 record.generation = resize_generation_;
141 }
142 }
143}

158quader::foundation::Result<void, std::string> RenderGraph::validate() const {
159 std::unordered_set<std::string> written_resources;
160 for (const RenderResourceRecord &resource : resources_.resources()) {
161 if (resource.desc.external) {
162 insert_name(written_resources, resource.desc.name);
163 }
164 }
165
166 std::unordered_set<std::string> pass_names;
167 for (const RenderPass &pass : passes_) {
168 if (pass.name.empty()) {
169 return quader::foundation::Result<void, std::string>::failure("Render pass name must not be empty.");
170 }
171 if (contains_name(pass_names, pass.name)) {
172 return quader::foundation::Result<void, std::string>::failure(
173 "Render pass '" + pass.name + "' is registered more than once.");
174 }
175 insert_name(pass_names, pass.name);
176
177 std::unordered_set<std::string> resources_used_by_pass;
178 for (const ResourceUse &use : pass.resources) {
179 const RenderResourceRecord *resource = resources_.find(use.resource_name);
180 if (resource == nullptr) {
181 return quader::foundation::Result<void, std::string>::failure(
182 "Render pass '" + pass.name + "' references missing resource '" + use.resource_name + "'.");
183 }
184 if (contains_name(resources_used_by_pass, use.resource_name)) {
185 return quader::foundation::Result<void, std::string>::failure(
186 "Render pass '" + pass.name + "' references resource '" + use.resource_name + "' more than once.");
187 }
188 insert_name(resources_used_by_pass, use.resource_name);
189
190 const bool kWasWritten = contains_name(written_resources, use.resource_name);
191 if (use.access == RenderResourceAccess::Read && !kWasWritten) {
192 return quader::foundation::Result<void, std::string>::failure(
193 "Render pass '" + pass.name + "' reads resource '" + use.resource_name + "' before it is written.");
194 }
195 if (use.access == RenderResourceAccess::Write && kWasWritten) {
196 return quader::foundation::Result<void, std::string>::failure(
197 "Render pass '" + pass.name + "' writes resource '" + use.resource_name + "' after it was already written; use ReadWrite for ordered modification.");
198 }
199 if (access_writes(use.access)) {
200 insert_name(written_resources, use.resource_name);
201 }
202 }
203 }
204
205 return quader::foundation::Result<void, std::string>::success();
206}

208std::string RenderGraph::debug_dump() const {
209 std::ostringstream output;
210 output << "RenderGraph generation=" << resources_.resize_generation() << '\n';
211 output << "Resources:\n";
212 for (const RenderResourceRecord &resource : resources_.resources()) {
213 output << " - " << resource.desc.name
214 << " format=" << render_resource_format_name(resource.desc.format)
215 << " extent=" << resource.desc.extent.width_px << 'x' << resource.desc.extent.height_px
216 << " external=" << (resource.desc.external ? "true" : "false")
217 << " generation=" << resource.generation << '\n';
218 }
219
220 output << "Passes:\n";
221 for (const RenderPass &pass : passes_) {
222 output << " - " << pass.name;
223 if (pass.resources.empty()) {
224 output << " resources=[]\n";
225 continue;
226 }
227
228 output << " resources=[";
229 for (std::size_t index = 0; index < pass.resources.size(); ++index) {
230 const ResourceUse &use = pass.resources[index];
231 if (index != 0) {
232 output << ", ";
233 }
234 output << use.resource_name << ':' << render_resource_access_name(use.access);
235 }
236 output << "]\n";
237 }
238
239 return output.str();
240}

263RenderGraph make_minimal_prototype_render_graph(ViewportExtent extent) {
264 RenderGraph graph;
265 (void)graph.add_resource(RenderResourceDesc{
266 .name = "BackbufferColor",
267 .format = RenderResourceFormat::BackbufferColor,
268 .extent = extent,
269 .external = true,
270 .resize_dependent = true,
271 });
272 (void)graph.add_resource(RenderResourceDesc{
273 .name = "BackbufferDepth",
274 .format = RenderResourceFormat::BackbufferDepth,
275 .extent = extent,
276 .external = true,
277 .resize_dependent = true,
278 });
279
280 graph.add_pass(make_cpu_pass("FrameSetupPass"));
281 graph.add_pass(make_pass("BackbufferClearPass", {
282 ResourceUse{ .resource_name = "BackbufferColor", .access = RenderResourceAccess::ReadWrite },
283 ResourceUse{ .resource_name = "BackbufferDepth", .access = RenderResourceAccess::ReadWrite },
284 }));
285 graph.add_pass(make_pass("PrototypeOpaquePass", {
286 ResourceUse{ .resource_name = "BackbufferColor", .access = RenderResourceAccess::ReadWrite },
287 ResourceUse{ .resource_name = "BackbufferDepth", .access = RenderResourceAccess::ReadWrite },
288 }));
289 graph.add_pass(make_pass("OverlayDepthTestedPass", {
290 ResourceUse{ .resource_name = "BackbufferDepth", .access = RenderResourceAccess::Read },
291 ResourceUse{ .resource_name = "BackbufferColor", .access = RenderResourceAccess::ReadWrite },
292 }));
293 graph.add_pass(make_pass("PresentPass", {
294 ResourceUse{ .resource_name = "BackbufferColor", .access = RenderResourceAccess::Read },
295 }));
296
297 return graph;
298}
